25     /* Create a directory */
26     void 
27     FC_FUNC_(clib_mkdir,CLIB_MKDIR)
28       (int *ierr, STR_F_TYPE fname STR_ARG1)
29     {
30       struct stat buf;
31       char* cname;
32       TO_C_STR1(fname, cname);
33     
34       *ierr = 1;
35       if (!*cname) return; 
36     
37       if (stat(cname, &buf) == 0) {
38         xfree(cname);
39         *ierr=2; return;
40       }
41     
42       *ierr = mkdir(cname, 0775);
43       if (*ierr == -1) fprintf (stderr, "Cannot create dir %s; %s\n", cname, strerror(errno) );
44       xfree(cname);
45     }

89     /* GNU's getcwd (NULL, 0) using the standard behavior of getcwd */
90     static char* gnu_getcwd(void);
91     
92     static char*
93     gnu_getcwd (void)
94     {
95       size_t size = 100;
96       while (1){
97         char* buffer = (char*) xmalloc (size);
98         if ( (char*) getcwd (buffer, size) == buffer) return buffer;
99         xfree (buffer);
100         if (errno != ERANGE) return NULL;
101         size *= 2;
102       }
103     }

105     /* Return the name of the current working directory */
106     void 
107     FC_FUNC_(clib_getcwd, CLIB_GETCWD)
108       (int* ierr, STR_F_TYPE fname STR_ARG1)
109     {
110       int ln; 
111       char* cname = gnu_getcwd();
112       ln = strlen(cname);
113       *ierr = 0;
114     
115       if (ln > l1){
116         fprintf(stderr, " Dir name %s will be truncated\n", cname);
117         *ierr= 1;
118       }
119       TO_F_STR1(cname, fname);
120       xfree(cname);
121     }

124     
125     static char*
126     gnu_gethostname (void)
127     {
128       size_t size = 100;
129       while (1){
130         char *buffer = (char *) xmalloc(size);
131         if ( gethostname(buffer, size) == 0) return buffer;
132         xfree (buffer);
133         if (errno != ENAMETOOLONG) return NULL;
134         size *= 2;
135       }
136     }

141       (int* ierr, STR_F_TYPE fname STR_ARG1)
142     {
143       int ln;
144       char* cname = gnu_gethostname();

146       *ierr = 0;
