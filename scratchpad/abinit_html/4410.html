61             return [path for path in options.paths if path.endswith(".nc")]
62     
63         else:
64             # Walk directory tree and find netcdf files.

68             if not os.path.isdir(top):
69                raise ValueError("Expecting existenting directory, got %s" % top)

71             paths = []
72             for dirpath, dirnames, filenames in os.walk(top):
