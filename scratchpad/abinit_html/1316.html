131         def toString(self,xlabel,x,ylabel,y):
132     	res = ""
133     	tab = ""
134     	if (len(x)!=0 and len(y)!=0):
135        	    if(len(x) == len(y)):
136     		if (len(xlabel) == 0 and len(ylabel) == 0):
137     		    tab = ' '	    
138     	        elif (len(xlabel) >= 8 and len(str(x[0])) <= 8 ) :
139     		    tab = ' \t\t '
140     	            res+=xlabel+'\t'+ylabel+'\n'
141     		    
142     	        elif (len(xlabel) <= 8 and len(str(x[0])) >= 8 ) :
143     		    tab = ' \t '
144     	            res+=xlabel+'\t\t'+ylabel+'\n'
145     		    		    
146     	        elif (len(xlabel) > 8 and len(str(x[0])) > 8):
147     		    tab = '\t'
148     		    res+=xlabel+' \t '+ylabel+'\n'		    
149     					
150     		else:
151     		    tab = '\t'
152     		    res+=xlabel+'\t'+ylabel+'\n'
153     		    		    			
154     		for i in range(len(x)):
155     		    res+=str(x[i])+tab+str(y[i])+'\n'
156     		    
157     	        return res
158     		
159     	    else:
160     	        self.showError("Datas haven't the same size") 
161     		return ""
162     	else:
163     	    if (len(x)==0):
164     		xlabel = ylabel
165     		x = y
166     	    res+=xlabel+'\n'
167     	    for i in range(len(x)):
168     		res+=str(x[i])+'\n'
169     	    return res

279         def showBirch(self):
280      
281     	vol = self.file2.getVol()
282     	
283     	Energie = self.file2.getE_Tot()
284             
285     	if (len(vol)!=len(Energie) or (len(vol)==0 or len(Energie) ==0)):
286     	        return 
287     	
288     	data = self.toString('',vol,'',Energie)
289     	
290     	try :
291     	    self.value
292     	except:
293     	    self.value = Analysis.birch(data,len(vol)).getValue()
294     	    pass
295     	
296     	V0 = self.value['V0']  
297     	B0 = self.value['B0'] / 29421.033e0
298     	E0 = self.value['E0']
299     	dB0 = self.value['dB0']
300     	RMS = self.value['RMS']
301     	
302     	x = linspace( min(vol), max(vol), 1000) 
303     
304     	y = E0 + ( (B0 * x) / dB0 ) * (  ((V0/x)**(dB0)) / (dB0-1) + 1  ) - ( (B0*V0)/(dB0-1) )
305     	
306     	xlabel ="VOLUME ("+str(self.units['Volume'][0])+")"
307     	
308     	ylabel ="ENERGY ("+str(self.units['Energy'][0])+")"
309     	
310     	x *=  self.units['Volume'][1]
311     	
312     	y *=  self.units['Energy'][1]
313     
314     	#Update the textBox:
315     	self.res  = 'Results of the fit : \n\n'	
316     	
317     	self.res += 'B0   = '  +str(self.value['B0'] * self.units['Pressure'][1]) +' ' +str(self.units['Pressure'][0]) + '\n'
318     	self.res += 'B0\'  = ' +str(dB0)+'\n'
319     	self.res += 'E0   = '  +str(E0 * self.units['Energy'][1])+' ' +str(self.units['Energy'][0]) + '\n'
320     	self.res += 'V0   = '  +str(V0 * self.units['Volume'][1])+'  '+str(self.units['Volume'][0])+'\n'
321     	self.res += 'RMS = '   +str(RMS* self.units['Energy'][1])+' ' +str(self.units['Energy'][0]) + '\n'
322     	     
323     	self.tbro.setText(self.res)
324         	self.isShow = 'BIRCH'
325     
326     	#update the graph:
327     	if self.birchCheckbox.isChecked():
328     	    boxData = self.getBoxData()
329     	    xlabel  = boxData[0]
330     	    ylabel  = boxData[1]
331     	    data1   = boxData[2]
332     	    data2   = boxData[3]
333     	    try :
334     		self.graph.setPlot(data1,data2,xlabel,ylabel,point = True)
335     						
336     		if xlabel == "ENERGY ("+str(self.units['Energy'][0])+")" :	            
337     		    self.graph.addPlot(y,x)
338     		else:
339     		    self.graph.addPlot(x,y)
340     		    
341     		self.graph.addLegend(["Simulation Data","Fit"])
342     	    except:
343     		pass
344     	else:
345     	    self.changeData()
