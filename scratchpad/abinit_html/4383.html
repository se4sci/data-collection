41     def angle_kpt(vector1,vector2):
42       dot_prod = N.dot(vector1,vector2.T)
43       arg = dot_prod/(N.linalg.norm(vector1)*N.linalg.norm(vector2))
44       if arg > 1: arg=1
45       if arg < -1: arg=-1
46       theta = N.arccos(arg)/(N.pi*180)
47       return theta

53         self.position_special = []
54         angle = 0

56         for ikpt in N.arange(nkpt):
57           if ikpt == 0:
58             self.special_kpoints.append(kpt[ikpt,:])
59             self.position_special.append(ikpt)
60           elif ikpt == 1:
61             vector2 = kpt[ikpt,:]-kpt[ikpt-1,:]
62           elif ikpt == nkpt-1:
63             self.special_kpoints.append(kpt[ikpt,:])
64             self.position_special.append(ikpt)
65           else:
66             vector1 = kpt[ikpt,:]-kpt[ikpt-1,:]
67             if N.allclose(kpt[ikpt-1,:],[3.0/8,3.0/8,3.0/4],1e-8):
68               self.special_kpoints.append(kpt[ikpt-1,:])
69               self.position_special.append(ikpt-1)
