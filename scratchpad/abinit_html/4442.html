1     #!/usr/bin/env python
2     """
3     Script for running the parallel tests of the test suite with different number of MPI nodes.
4     Note: This script is just a small wrapper around runtests.py. It receives a list of paral
5     tests, register them and run all of them in parallel using distinct subprocesses. 
6     The user is responsible for providing a list of tests that do not cause an excessive overload of the OS.
7     """
8     from __future__ import print_function, division, absolute_import #, unicode_literals
9     
10     import sys
11     import os
12     import subprocess
13     
14     script = os.path.join(os.path.dirname(__file__), "runtests.py")
15     
16     MPI_NPROCS_LIST = [1, 2, 4, 10]
17     
18     def str_examples():
19         examples = """
20           Usage example (assuming the script is executed within a build tree):
21           \n
22           runparal.py 1        => Run paral[1] with 1,2,4,10 MPI nodes
23           runparal.py 1:3      => To select both test 1 and 2 
24           runparal.py 1:3  4   => Same as above but add test number 4
25         """
26         return examples
27     
28     def show_examples_and_exit(err_msg=None, error_code=1):
29         "Display the usage of the script."
30         sys.stderr.write(str_examples())
31         if err_msg: sys.stderr.write("Fatal Error\n" + err_msg + "\n")
32         sys.exit(error_code)
33     
34     def main():
35         try:
36             args = sys.argv[1:]
37     
38             tests = []
39             for arg in args:
40                 if ":" in arg:
41                     # String defining a python range
42                     tokens = map(int, arg.split(":"))
43                     tests.extend(list(range(*tokens)))
44                 else:
45                     # Assume integer
46                     num = int(arg)
47                     tests.append(num)
48     
49             # Build the list of strings to pass to runtest.py
50             test_strings = []
51             for t in tests:
52                 test_strings.append("paral[" + str(t) + "]")
53     
54         except Exception, exc:
55             show_examples_and_exit(str(exc))
56     
57         # Lauch the tests in a subprocess (non blocking algorithm)
58         processes = []
59         for s in test_strings:
60             for mpi_nprocs in MPI_NPROCS_LIST:
61                 workdir = os.path.join(os.getcwd(), s + "_MPI_" + str(mpi_nprocs))
62                 cmd_args = [script, s, "-n", str(mpi_nprocs), "-w", workdir]
63                 #print "About to execute", args
64                 p = subprocess.Popen(cmd_args)
65                 processes.append(p)
66     
67         # Parent waits here.
68         for p in processes:
69             p.wait()
70     
71         retcode = max([p.returncode for p in processes])
72     
73         print("Final return code %s" % retcode)
74         return retcode 
75     
76     
77     if __name__ == "__main__":
78         sys.exit(main())
