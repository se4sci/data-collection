8     class lazy_property(object):
9         """
10         lazy_property descriptor
11     
12         Used as a decorator to create lazy attributes.
13         Lazy attributes are evaluated on first use.
14         """
15     
16         def __init__(self, func):
17             self.__func = func
18             from functools import wraps
19             wraps(self.__func)(self)
20     
21         def __get__(self, inst, inst_cls):
22             if inst is None:
23                 return self
24     
25             if not hasattr(inst, '__dict__'):
26                 raise AttributeError("'%s' object has no attribute '__dict__'"
27                                      % (inst_cls.__name__,))
28     
29             name = self.__name__
30             if name.startswith('__') and not name.endswith('__'):
31                 name = '_%s%s' % (inst_cls.__name__, name)
32     
33             value = self.__func(inst)
34             inst.__dict__[name] = value
35             return value
36     
37         @classmethod
38         def invalidate(cls, inst, name):
39             """Invalidate a lazy attribute.
40     
41             This obviously violates the lazy contract. A subclass of lazy
42             may however have a contract where invalidation is appropriate.
43             """
44             inst_cls = inst.__class__
45     
46             if not hasattr(inst, '__dict__'):
47                 raise AttributeError("'%s' object has no attribute '__dict__'"
48                                      % (inst_cls.__name__,))
49     
50             if name.startswith('__') and not name.endswith('__'):
51                 name = '_%s%s' % (inst_cls.__name__, name)
52     
53             if not isinstance(getattr(inst_cls, name), cls):
54                 raise AttributeError("'%s.%s' is not a %s attribute"
55                                      % (inst_cls.__name__, name, cls.__name__))
56     
57             if name in inst.__dict__:
58                 del inst.__dict__[name]

70             from subprocess import call

116     def pprint_table(table, out=sys.stdout, rstrip=False):
117         """
118         Prints out a table of data, padded for alignment
119         Each row must have the same number of columns.
120     
121         Args:
122             out:
123                 Output stream (file-like object)
124             table:
125                 The table to print. A list of lists.
126             rstrip:
127                 if true, trailing withespaces are removed from the entries.
128         """
129         def max_width_col(table, col_idx):
130             """Get the maximum width of the given column index"""
131             return max([len(row[col_idx]) for row in table])
132     
133         if rstrip:
134             for row_idx, row in enumerate(table):
135                 table[row_idx] = [c.rstrip() for c in row]
136     
137         col_paddings = []
138         ncols = len(table[0])
139         for i in range(ncols):
140             col_paddings.append(max_width_col(table, i))
141     
142         for row in table:
143             # left col
144             out.write( row[0].ljust(col_paddings[0] + 1) )
145             # rest of the cols
146             for i in range(1, len(row)):
147                 col = row[i].rjust(col_paddings[i] + 2)
148                 out.write(col)
149             out.write("\n")
