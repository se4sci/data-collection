7     import os
8     import re

11     # Files that will be checked.
12     re_srcfile = re.compile("\.([Ff]|[Ff]90|finc)$")
13     
14     def is_srcfile(dirpath, fname):
15       return re_srcfile.search(fname)
16     
17     # List of macros that should not be inlined with if statements or other instructions.
18     # These macros are defined in src/incs/abi_common.h
19     # 
20     MACRO_NAMES = [
21     "ABI_ALLOCATE",
22     "ABI_DEALLOCATE",
23     "ABI_STAT_ALLOCATE",
24     "ABI_STAT_DEALLOCATE",
25     "ABI_DATATYPE_ALLOCATE",
26     "ABI_DATATYPE_DEALLOCATE",
27     "ABI_STAT_DATATYPE_ALLOCATE",
28     "ABI_STAT_DATATYPE_DEALLOCATE",
29     "ABI_MALLOC",
30     "ABI_STAT_MALLOC",
31     "ABI_DT_MALLOC",
32     "ABI_STAT_DT_MALLOC",
33     "ABI_CALLOC",
34     "ABI_STAT_CALLOC",
35     "ABI_FREE",
36     "ABI_STAT_FREE",
37     "ABI_CHECK_ALLOC",
38     ]
39     
40     # Regular expressions for each macro.
41     regexps = dict()
42     
43     for name in MACRO_NAMES:
44       regexps[name] = re.compile(name + " ?\(.*\)")
45       #regexps[name] = re.compile(name + "\(.*\)") blanks between macro name and () are not permitted
46     
47     
48     def wrong_string(string):
49       "Return an empty string if input string does not contain inlined macros."
50       string = string.strip()
51       for macro_name in MACRO_NAMES:
52         pattern = regexps[macro_name]
53         if macro_name in string and not string.startswith("!"):
54           s = re.sub(pattern, "", string, count=0).strip()
55           if not s.startswith("!"): return s
56         else:
57           return ""
58     
59     def abinit_test_generator():
60       def test_func(abenv):
61          """Search for inlined CPP macros in ABINIT src files"""
62          top = abenv.apath_of("src")
63          try:
64            return main(top)
65          except Exception:
66            import sys
67            raise sys.exc_info()[1] # Reraise current exception (py2.4 compliant)
68       return {"test_func" : test_func}
69     
70     def main(top):
71       exit_status = 0
72       for dirpath, dirnames, files in os.walk(top):
73         for src in files:
74           if is_srcfile(dirpath, src):
75             fpath = os.path.join(dirpath,src)
76             with open(fpath, "rt") as fh:
77                 for lno, line in enumerate(fh):
78                   s = wrong_string(line)
79                   if s:
80                     print("(INLINED MACRO at %s:%d):  %s " % (src, lno+1, line))
81                     exit_status += 1
82     
83       if exit_status > 0:
84         err_msg = """
85       Please, avoid instructions like:
86     
87         if (allocated(arr)) ABI_DEALLOCATE(arr)
88     
89       When the code is compiled in profile mode, indeed, ABI_DEALLOCATE expands to 
90       the set of Fortran instructions:
91     
92         deallocate(arr)
93         call memocc_abi()
94     
95       These instructions MUST be placed inside an "if then" "end if" block.
96       This limitation can be lifted, but we need support at the level of the build system.
97       For the time being, one has to use the more verbose form:
98     
99         if (allocated(arr)) then 
100           ABI_DEALLOCATE(arr)
101         end if
102     
103       This is the list of macros that cannot be inlined:
104     
105       %(MACRO_NAMES)s
106     """ % globals()
107         print(err_msg)
108     
109       return exit_status
110
