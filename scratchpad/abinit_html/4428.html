24     def all_subclasses(cls):
25         """
26         Given a class `cls`, this recursive function returns a list with
27         all subclasses, subclasses of subclasses, and so on.
28         """
29         subclasses = cls.__subclasses__() 
30         return subclasses + [g for s in subclasses for g in all_subclasses(s)]

33     class EtsfObject(object):
34         """
35         Base class for netcdf Dimensions, Variables, Attributes.
36         Subclasses implement a `validate` method that receives a nc dataset 
37         and return a list of errors (strings).
38         """
39         def __str__(self):
40             return "<%s: %s>" % (self.__class__.__name__, self.name)

43     class EtsfDimension(EtsfObject):
44         """A dimension has a name, a type and, optionally, a list of allowed values."""
45         def __init__(self, name, xtype, allowed=None):
46             self.name = name
47             self.xtype = xtype
48             self.allowed = allowed
49     
50         def validate(self, ncdata):
51             """Validate the content in the nc dataset, return list of errors."""
52             errors = []
53             eapp = errors.append
54     
55             # Dimension should be present.
56             if self.name not in ncdata.dimensions:
57                 eapp("<%s> Not present in ncdata" % self.name)
58                 return errors
59     
60             # Check if value is allowed.
61             if self.allowed is not None:
62                 n = len(ncdata.dimensions[self.name])
63                 if n not in self.allowed:
64                     eapp.append("Value %s not in allowed %s" % (n, self.allowed))
65     
66             return errors

69     class EtsfAttribute(EtsfObject):
70         """A dimension has a name, a type and, optionally, a shape and list of allowed values."""
71         def __init__(self, name, xtype, shape=None, allowed=None):
72             self.name = name 
73             self.allowed = allowed
74     
75         def validate(self, ncdata):
76             """Validate the content in the nc dataset, return list of errors."""
77             errors = []
78             eapp = errors.append
79             nc_attrs = ncdata.ncattrs()
80     
81             # Dimension should be present.
82             if self.name not in nc_attrs:
83                 eapp("<%s> not present in ncdata" % self.name)
84     
85             # Check if value is allowed.
86             if self.allowed is not None:
87                 value = nc_attrs[self.name] 
88                 if value not in self.allowed:
89                     eapp("Value %s is not allowed" % str(value))
90     
91             return errors

94     class EtsfVariable(EtsfObject):
95         """
96         A variable has a name, a type and a list of dimensions.
97         The list of allowed values and the attributes that must be specified are optional.
98         """
99         # Stores all the instances we are gonna create.
100         all_variables = []
101     
102         def __init__(self, name, xtype, dimensions, allowed=None, reqattrs=None):
103             self.name = name
104             self.xtype = xtype
105             #if xtype == "char": assert shape
106             self.dimensions = dimensions
107             #self.shape = tuple(shape) if shape else ()
108             self.allowed = allowed
109             self.reqattrs = reqattrs
110             # TODO dtype
111     
112             self.ndim = len(self.dimensions)
113     
114             # Store the instance in cls.all_variables
115             self.__class__.all_variables.append(self)
116     
117         def validate(self, ncdata):
118             """Validate the content in the nc dataset, return list of errors."""
119             errors = []
120             eapp = errors.append
121     
122             # Variable should be present.
123             if self.name not in ncdata.variables:
124                 eapp("<%s> not present in ncdata" % self.name)
125                 return errors
126     
127             # If variable is present, check that dimension names agree with the specifications.
128             ncvar = ncdata.variables[self.name]
129     
130             # Number of dimensions
131             if self.ndim != ncvar.ndim:
132                 eapp("Has different ndim:\n\tFile: %s\n\tSpecs: %s" % (self.ndim, ncvar.ndim))
133                 return errors
134     
135             # Test shape
136             shape = np.array([len(ncdata.dimensions[dim.name]) for dim in self.dimensions])
137             if np.any(shape != ncvar.shape):
138                 eapp("Has different shape" % (shape, ncvar.shape))
139                 return errors
140     
141             # Test dimension names.
142             dim_names = np.array([dim.name for dim in self.dimensions])
143             ncdim_names = np.array(ncvar.dimensions)
144             #print("dim_names: ", dim_names); print("ncdim_names: ", ncdim_names)
145             if any(dim_names != ncdim_names):
146                 eapp("Wrong dimension names.\n\tFile: %s\n\tSpecs: %s" % (dim_names, ncdim_names))
147     
148             # Test allowed values.
149             if self.allowed is not None:
150                 assert not ncvar.shape 
151                 try:
152                     value = ncvar.getValue()[0] if not ncvar.shape else ncvar[:]
153                 except IndexError:
154                     value = ncvar.getValue() if not ncvar.shape else ncvar[:]
155     
156                 if value not in self.allowed:
157                     eapp("Value %s not in allowed %s" % (value, self.allowed))
158     
159             # Test attributes.
160             if self.reqattrs is not None:
161                 nc_attrs = ncvar.ncattrs()
162                 for attr in self.reqattrs:
163                     aname = attr.name
164                     if aname not in nc_attrs:
165                         eapp("Attribute %s is missing for variable: %s" % (aname, self.name))
166     
167             return errors

357     class EtsfGroup(object):
358         """"
359         This object is essentially a container of variables
360         A Group can contain other subgroups.
361         """
362         attributes = []
363         dimensions = []
364         variables = []
365         at_least_one_variable_in = []
366         subgroups = []
367     
368         @classmethod
369         def validate_file(cls, path):
370             ncdata = netCDF4.Dataset(path, mode="r")
371             errors = cls.validate(ncdata)
372             ncdata.close()
373             return errors
374     
375         @classmethod
376         def validate(cls, ncdata):
377             errors = []
378     
379             # Test attributes.
380             for attr in cls.attributes:
381                 errors += attr.validate(ncdata)
382     
383             # Test dimensions
384             for dim in cls.dimensions:
385                 errors += dim.validate(ncdata)
386     
387             # Test variables
388             for var in cls.variables:
389                 errors += var.validate(ncdata)
390     
391             # Test variables.
392             ok, elist = 0, []
393             for var in cls.at_least_one_variable_in:
394                 if var.name in ncdata.variables:
395                     ok += 1
396                 else:
397                     elist.append("<%s> not present in ncdata" % var.name)
398     
399             if not ok:
400                 errors.extend(elist)
401                                                                                                     
402             # Test subgroups.
403             for group in cls.subgroups:
404                 errors += group.validate(ncdata)
405     
406             return errors

553     class WavefunctionGroup(EtsfGroup):
554         """
555         Specification for files containing the wavefunctions
556     
557         Specification
558     
559         A ETSF NetCDF file "containing the wavefunctions" should contain at least the information needed to build 
560         the density from this file. Also, since the eigenvalues are intimately linked to eigenfunctions, 
561         it is expected that such a file contain eigenvalues. Of course, files might contain less information 
562         than the one required, but still follow the naming convention of ETSF. It might also contain more information, 
563         of the kind specified in other tables of the present document.
564     
565         A ETSF NetCDF file "containing the wavefunctions" should contain the following set of mandatory information:
566     
567         The three attributes defined in Mandatory attributes
568     
569         The following dimensions from Dimensions that cannot be split:
570     
571             character_string_length
572             number_of_cartesian_directions
573             number_of_vectors
574             real_or_complex_coefficients and/or real_or_complex_wavefunctions
575             number_of_symmetry_operations
576             number_of_reduced_dimensions
577     
578         The following dimensions from Dimensions that can be split:
579     
580             max_number_of_states
581             number_of_kpoints
582             number_of_spins
583             number_of_spinor_components
584     
585         In case of a real-space wavefunctions, the following dimensions from Dimensions that can be split:
586     
587             number_of_grid_points_vector1
588             number_of_grid_points_vector2

689         """
690         Validate the etsf variables declared in file `path`. 
691         Return list of errors.
692         """
693         ncdata = netCDF4.Dataset(path, mode="r")
694         evars, all_errors = [] , []
695         d = {}
696         for var in EtsfVariable.all_variables:
697            if var.name in ncdata.variables:
698                 elist = var.validate(ncdata)
699                 if elist:
700                     all_errors.append(elist)
701                     evars.append(var)
702         ncdata.close()
703     
704         if not all_errors: return all_errors
705     
706         # Print errors
707         try:
708             title = "=== ERRORS IN FILE %s ===" % os.path.relpath(path)
709         except Exception as exc:
710             # I've seen this kind of error.
711             logger.warning("relpath returned %s" % str(exc))
712             title = "=== ERRORS IN FILE %s ===" % path
713     
714         bar = "=" * len(title)
715         cprint(bar, "magenta")
716         cprint(title, "magenta")
717         cprint(bar, "magenta")
718     
719         for var, elist in zip(evars, all_errors):
720             cprint("%s: Found [%d] error(s):" % (var, len(elist)), "red")
721             for i, e in enumerate(elist): 
722                 print("[%d] " % i, e)
723     
724         return all_errors

787     def validate_ncfile(path):
788         """
789         This function validates the netcdf files produced by Abinit.
790     
791         Return:
792             List of strings with error messages.
793         """
794         # Every netcdf file produced by Abinit must be listed in this dictionary.
795         # that maps the file extension to the list of Groups contained in the file.
796         # Legacy files that do not have etsfio groups are tagged with None and ignored.
797         # New netcdf files introduced in Abinit should try to use the etsfio specs as much 
798         # as possible and extend the format to treat new cases.
799         # Remember that gmatteo is watching you so don't try to hack this code.
800         ext2groups = {
801             # GS files.
802             "GSR.nc": [CrystalGroup, KpointsGroup, StatesGroup],
803             "DEN.nc": [DensityGroup],
804             "EIG.nc": None,
805             "OUT.nc": None,
806             "WFK.nc": [WavefunctionGroup],
807             #"POT.nc": [PotentialGroup],
808             #"VH.nc": [PotentialGroup],
809             #"VXC.nc": [PotentialGroup],
810             "VHXC.nc": [PotentialGroup],
811             #"KDEN.nc": [DensityGroup],
812             "HIST.nc": None,
813             #"PAWDEN.nc": [DensityGroup],
814             #"ELF.nc": [DensityGroup],
815             #"ELF_DOWN.nc": [DensityGroup],
816             #"ELF_UP.nc": [DensityGroup],
817             "PSPS.nc": None,
818             #"STM.nc": [],
819             #"GDEN1.nc":
820             #"GDEN2.nc":
821             #"GDEN3.nc":
822             #"KDEN.nc":
823             #"LDEN.nc":
824     
825             # DFPT files.
826             "DDB.nc": [CrystalGroup],
827             "WFQ.nc": [WavefunctionGroup],
828             #"DEN1.nc": [DensityGroup],
829             #"POT1.nc": [PotentialGroup],
830             #"EIGR2D.nc": 
831             #"EIGI2D.nc":
832     
833             # Anaddb files
834             "anaddb.nc": [CrystalGroup],
835             "EC.nc": [CrystalGroup],
836             "PHBST.nc": [CrystalGroup],
837     
838             # GW files.
839             #"KSS.nc": [WavefunctionGroup],
840             "SCR.nc": [CrystalGroup, KpointsGroup, StatesGroup],
841             "SUS.nc": [CrystalGroup, KpointsGroup, StatesGroup],
842             "SIGRES.nc": [CrystalGroup, KpointsGroup, StatesGroup],
843             #"QP_DEN.nc": [DensityGroup],
844     
845             # BSE files.
846             "MDF.nc": [CrystalGroup, KpointsGroup, StatesGroup],
847         }
848     
849         groups = None
850         fname = os.path.basename(path)
851     
852         # DFPT files require regular expressions e.g. 
853         # _DS2_1WF6.nc, _DS2_DEN6, _DEN_POSITRON
854         re_1wf = re.compile("(\w+_)1WF(\d+)(\.nc)$")
855         re_1den = re.compile("(\w+_)DEN(\d+)(\.nc)$")
856         #re_1den = re.compile("(\w+_)POT(\d+)(\.nc)$")
857         if re_1wf.match(fname): groups = [WavefunctionGroup]
858         if re_1den.match(fname): groups = [DensityGroup]
859         if groups is not None: return validate_groups(path, groups)
860     
861         ext = fname.split("_")[-1]
862         try:
863             groups = ext2groups[ext]
864         except KeyError as exc:
865             errors = ["Unknown file extension in file %s" % fname]
866             print(errors)
867             return errors
868     
869         if groups is None: return []
870         return validate_groups(path, groups)
