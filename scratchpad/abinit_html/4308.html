6     def report_runtime(f):
7         """Decorator to print runtime of a function."""
8         def g(*args, **kwargs):
9             verbose = kwargs.get('verbose', False)
10             if i_am_master:
11                 with mpi_abort_if_exception():
12                     start = datetime.now()
13                     if verbose:
14                         print('Start on {}'.format(str(start).split('.')[0]))
15             comm.Barrier()
16             out = f(*args, **kwargs)
17             comm.Barrier()
18             if i_am_master:
19                 with mpi_abort_if_exception():
20                     end = datetime.now()
21                     runtime = end - start
22                     if verbose:
23                         print('End on {}'.format(str(end).split('.')[0]))
24                         print("Runtime: {} seconds ({:.1f} minutes)".format(runtime.seconds, runtime.seconds / 60.))
25             return out
26         g.__doc__ = f.__doc__
27         return g
