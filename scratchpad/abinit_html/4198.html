9     re_m4file  = re.compile("\.m4$")
10     re_hdrfile  = re.compile("\.h$")

12     re_acdef  = re.compile("AC_DEFINE\\(")
13     re_cppdef  = re.compile("^([ ]?)+#([ ]?)+define [0-9A-Z_]*")

16     #re_cppskip = re.compile("^#(include|define|undef|if 0|if 1|endif|error)")

18     re_cppdev  = re.compile("DEV_")

20     # Init CPP options
21     cpp_blocks = [["HAVE"],["CONFIG"],["H"]]
22     
23     cpp_explicit = [
24       "HAVE_TEST_TIME_PARTITIONING",
25       "i386",
26       "DFTI_CHECK",
27       "XC_MAJOR_VERSION",
28       "XC_MINOR_VERSION",

31     cpp_ignored = [
32       "HAVE_CONFIG_H",
33       "HAVE_IBM6",

37     cpp_keywords = ("#","endif","ifdef","ifndef","elif","if","else","defined","[<>=]=?[ ]*[0-9]*","\\|\\|","&&","\\(","\\)","!")

39     # Naming convention checker
40     def check_name(cpp_option,cpp_blocks):
41       opt = cpp_option.split("_")
42       ret = True
43       for i in range(len(opt)):
44         ret = (opt[i] in cpp_blocks[i])
45         if ( not ret ):
46           break
47       return ret

100       cpp_buildsys.sort()

103       for opt in cpp_buildsys:
104         tmp = opt.split("_")
105         for i in range(len(tmp)):
106           if ( i >= len(cpp_blocks) ):
107             cpp_blocks.append(list())
108           if ( not tmp[i] in cpp_blocks[i] ):
109             cpp_blocks[i].append(tmp[i])
