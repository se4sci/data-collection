27     from __future__ import print_function
28     import numpy as np
29     import netCDF4 as nc
30     
31     __version__ = '1.0.0'
32     
33     def merge_gkk_nc(out_fname, fnames):
34         """
35         Merge a list of GKK<i>.nc files containing different elements of the same qpoint.
36     
37         Arguments
38         ---------
39     
40         out_fname: Name for the merged file (will overwrite any existing file).
41         fnames: List of GKK<i>.nc files.
42     
43         """
44         if not fnames:
45             raise Exception('Empty list of files given for merge')
46     
47         fname0 = fnames[0]
48     
49         with nc.Dataset(out_fname, 'w') as dsout:
50             with nc.Dataset(fname0, 'r') as dsin:
51     
52                 nc_copy(dsin, dsout,
53                     except_dimensions=['number_of_atoms_for_gkk', 'number_of_cartesian_directions_for_gkk'],
54                     except_variables=['second_derivative_eigenenergies_actif'],
55                     )
56     
57                 q0 = dsin.variables['current_q_point'][...]
58                 natom = len(dsin.dimensions[u'number_of_atoms'])
59                 ncart = len(dsin.dimensions[u'number_of_cartesian_directions'])
60     
61             dsout.createDimension('number_of_atoms_for_gkk', natom)
62             dsout.createDimension('number_of_cartesian_directions_for_gkk', ncart)
63     
64             gkk = dsout.createVariable('second_derivative_eigenenergies_actif', np.dtype('float64'),
65                                         ('max_number_of_states', 'number_of_atoms_for_gkk',
66                                          'number_of_cartesian_directions_for_gkk', 'number_of_kpoints',
67                                          'product_mband_nsppol2'))
68     
69             for i, fname in enumerate(fnames):
70     
71                 iatom = i // ncart
72                 icart = i % ncart
73     
74                 with nc.Dataset(fname, 'r') as dsin:
75     
76                     # Check that the qpoints are the same
77                     q = dsin.variables['current_q_point'][...]
78                     if not all(np.isclose(q0, q)):
79                         raise Exception('Cannot merge GKK.nc at different q-points.')
80     
81                     gkki = dsin.variables[u'second_derivative_eigenenergies_actif'][:,0,0,...]
82                     gkk[:,iatom,icart,...] = gkki
83     
84     
85     def nc_copy(dsin, dsout, except_dimensions=None, except_variables=None):
86         """
87         Copy all dimensions and variable of one nc.Dataset instance into another.
88         """
89     
90         #Copy dimensions
91         for dname, dim in dsin.dimensions.iteritems():
92             if except_dimensions and dname in except_dimensions:
93                 continue
94             dsout.createDimension(dname, len(dim))
95     
96         #Copy variables
97         for vname, varin in dsin.variables.iteritems():
98             if except_variables and vname in except_variables:
99                 continue
100             outVar = dsout.createVariable(vname, varin.datatype, varin.dimensions)
101             outVar[...] = varin[...]
102

104     def interactive_merge_gkk_nc():
105         """Get inputs from the user and run merge_gkk_nc."""
106     
107         program_name = 'merge_gkk_nc'
108         description = """Merge several GKK<i>.nc files, belonging to the same q-point."""
109     
110         def get_user(s):
111             return raw_input(s.rstrip() + '\n').split('#')[0]
112     
113     
114         print(program_name)
115         print(len(program_name) * '-')
116         print(description + '\n')
117     
118         ui = get_user('Enter a name for the output file in which to merge (will overwrite any existing file):')
119         out_fname = str(ui)
120     
121         ui = get_user('Enter the number of files to merge:')
122         nfiles = int(ui)
123     
124         fnames = list()
125         for i in range(nfiles):
126             ui = get_user('Enter the name of file {}:'.format(i+1))
127             fname = str(ui)
128             fnames.append(fname)
129     
130         # Main execution
131         print('Executing...')
132         merge_gkk_nc(out_fname, fnames)
133     
134         print('All done.')
135
