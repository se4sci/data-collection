256           SUBROUTINE LSTSQR(NN,MM,F,A,P)
257     C     FITS THE FUNCTION F(X), DEFINED AT THE N POINTS X(I)
258     C     ( F(I)=F(X(I)) ), TO THE M FUNCTIONS P(I,X)
259     C     ( P(I,J)=P(I,X(J)) ), USING A LINEARIZED LEAST SQUARES
260     C     FITTING ROUTINE.  THE COEFFICIENT OF P(I,X) WILL BE
261     C     RETURNED AS A(I)
262           IMPLICIT DOUBLE PRECISION (A-H,O-Z)
263           PARAMETER (MAXN=50,MAXM=11)
264           DIMENSION X(MAXN),F(MAXN),A(MAXM),P(MAXM,MAXN)
265           DIMENSION G(MAXM,MAXN),AVG2(MAXM),B(MAXM),
266          1  C(MAXM,MAXM),SGP(MAXM),D(MAXM,MAXM)
267           M=MM
268           N=NN
269           RN=1D0/DBLE(N)
270     C     THE G'S LINEAR COMBINATIONS OF THE P'S, CHOSEN BY THE
271     C     GRAM-SCHMIDT ORTHOGONALIZATION PROCEDURE SO THAT
272     C     <G(M)*G(N)>=0 IF M.NE.N, WHERE
273     C     <H>=[H(X(1))+H(X(2))+H(X(3))+...+H(X(N))]/N
274     C     FOR ANY FUNCTION H(X)
275     C     CALCULATE THE ITH FUNCTION
276           DO 40 I=1,M
277              IM=I-1
278              IP=I+1
279              SG2=0D0
280              DO 10 K=IP,M
281     10         SGP(K)=0D0
282     C        AT THE JTH POINT
283              DO 30 J=1,N
284                SUM=0D0
285                DO 20 K=1,IM
286     20           SUM=SUM+C(I,K)*G(K,J)
287                G(I,J)=P(I,J)+SUM
288                SG2=SG2+G(I,J)*G(I,J)
289                DO 30 K=IP,M
290     30           SGP(K)=SGP(K)+P(K,J)*G(I,J)
291     C        AVG2(I)=<G(I)*G(I)>
292              AVG2(I)=RN*SG2
293     C        C(K,I)= -<P(K)*G(I)>/<G(I)*G(I)>
294              DO 40 K=IP,M
295     40         C(K,I)=-SGP(K)/SG2
296     C     SINCE <G(I)*G(J)>=0 FOR I.NE.J, IT'S TRIVIAL TO FIND
297     C     THE COEFFICIENTS FOR A LEAST SQUARES FIT OF F TO THE G'S
298           DO 60 I=1,M
299              SUM=0D0
300              DO 50 J=1,N
301     50         SUM=SUM+G(I,J)*F(J)
302     60       B(I)=RN*SUM/AVG2(I)
303     C     TO CONVERT THE B'S INTO A'S, WE FIRST NEED TO FIND THE
304     C     COEFFICIENTS FOR EXPANDING THE G'S IN TERMS OF THE P'S
305           DO 80 I=1,M
306              D(I,I)=1D0
307              IM=I-1
308              DO 80 K=1,IM
309                SUM=0D0
310                DO 70 L=K,IM
311     70           SUM=SUM+C(I,L)*D(L,K)
312     80         D(I,K)=SUM
313     C     FINALLY, WE CAN CHANGE THE B'S INTO A'S
314           DO 100 I=1,M
315              SUM=0D0
316              DO 90 J=I,M
317     90         SUM=SUM+B(J)*D(J,I)
318     100      A(I)=SUM
319           RETURN
320           END

323           SUBROUTINE  P L Y E V L (M,X0,A,X,N,P)
324     C     Evaluates the polynomial given by
325     C
326     C       p(x) = Sum [ a(i) (x-x0)^i ,{i,0,m}]
327     C
328     C       and its first N derivatives
329     C
330           IMPLICIT DOUBLE PRECISION (A-H,O-Z)
331     C     Note the dummy indexing on A and P:
332           DIMENSION A(0:1),P(0:1)
333           Y=X-X0
334     C     Zeroth order term (in Y)
335           IPROD=1
336           DO 100 J=0,N
337             P(J)=IPROD*A(M)
338     100     IPROD=IPROD*(M-J)
339           DO 200 I=M-1,0,-1
340             IPROD=1
341             DO 200 J=0,N
342               IF(IPROD.GT.0D0) P(J)=P(J)*Y+IPROD*A(I)
343     200       IPROD=IPROD*(I-J)
344           RETURN
345           END
346           real*8 function efun(v)
347           implicit real*8 (a-h,o-z)
348           PARAMETER (MAXN=50,MAXM=10,MAXIN=200)
349           common /ear/ a(0:maxm),earray(0:3),m,n
350           data two3 /0.6666666666666d0/
351           X=V**(-TWO3)
352     C
353     C       Call the polynomial to evaluate the energy and pressure:
354     C
355            CALL PLYEVL(M,0D0,A,X,1,EARRAY)
356            efun=EARRAY(0)
357            return
358            end
359           real*8 function defun(v)
360           PARAMETER (MAXN=50,MAXM=10,MAXIN=200)
361           implicit real*8 (a-h,o-z)
362           common /ear/ a(0:maxm),earray(0:3),m,n
363           data two3 /0.6666666666666d0/
364           X=V**(-TWO3)
365           defun=-TWO3*X*EARRAY(1)/v
366           return
367           end
368           FUNCTION DBRENT(AX,BX,CX,F,DF,TOL,XMIN)
369           implicit real*8 (a-h,o-z)
370           PARAMETER (ITMAX=1000,ZEPS=1.0E-12)
371           LOGICAL OK1,OK2
372           A=MIN(AX,CX)
373           B=MAX(AX,CX)
374           V=BX
375           W=V
376           X=V
377           E=0.
378           FX=F(X)
379           FV=FX
380           FW=FX
381           DX=DF(X)
382     c      print *,x,fx,dx
383           DV=DX
384           DW=DX
385           DO 11 ITER=1,ITMAX
386             XM=0.5*(A+B)
387             TOL1=TOL*ABS(X)+ZEPS
388             TOL2=2.*TOL1
389             IF(ABS(X-XM).LE.(TOL2-.5*(B-A))) GOTO 3
390             IF(ABS(E).GT.TOL1) THEN
391               D1=2.*(B-A)
392               D2=D1
393               IF(DW.NE.DX) D1=(W-X)*DX/(DX-DW)
394               IF(DV.NE.DX) D2=(V-X)*DX/(DX-DV)
395               U1=X+D1
396               U2=X+D2
397               OK1=((A-U1)*(U1-B).GT.0.).AND.(DX*D1.LE.0.)
398               OK2=((A-U2)*(U2-B).GT.0.).AND.(DX*D2.LE.0.)
399               OLDE=E
400               E=D
401               IF(.NOT.(OK1.OR.OK2))THEN
402                 GO TO 1
403               ELSE IF (OK1.AND.OK2)THEN
404                 IF(ABS(D1).LT.ABS(D2))THEN
405                   D=D1
406                 ELSE
407                   D=D2
408                 ENDIF
409               ELSE IF (OK1)THEN
410                 D=D1
411               ELSE
412                 D=D2
413               ENDIF
414               IF(ABS(D).GT.ABS(0.5*OLDE))GO TO 1
415               U=X+D
416               IF(U-A.LT.TOL2 .OR. B-U.LT.TOL2) D=SIGN(TOL1,XM-X)
417               GOTO 2
418             ENDIF
419     1       IF(DX.GE.0.) THEN
420               E=A-X
421             ELSE
422               E=B-X
423             ENDIF
424             D=0.5*E
425     2       IF(ABS(D).GE.TOL1) THEN
426               U=X+D
427               FU=F(U)
428             ELSE
429               U=X+SIGN(TOL1,D)
430               FU=F(U)
431               IF(FU.GT.FX)GO TO 3
432             ENDIF
433             DU=DF(U)
434     c      print *,x,fx,dx
435             IF(FU.LE.FX) THEN
436               IF(U.GE.X) THEN
437                 A=X
438               ELSE
439                 B=X
440               ENDIF
441               V=W
442               FV=FW
443               DV=DW
444               W=X
445               FW=FX
446               DW=DX
447               X=U
448               FX=FU
449               DX=DU
450             ELSE
451               IF(U.LT.X) THEN
452                 A=U
453               ELSE
454                 B=U
455               ENDIF
456               IF(FU.LE.FW .OR. W.EQ.X) THEN
457                 V=W
458                 FV=FW
459                 DV=DW
460                 W=U
461                 FW=FU
462                 DW=DU
463               ELSE IF(FU.LE.FV .OR. V.EQ.X .OR. V.EQ.W) THEN
464                 V=U
465                 FV=FU
466                 DV=DU
467               ENDIF
468             ENDIF
469     11    CONTINUE
470           print *,'DBRENT exceeded maximum iterations.'
471           print *,'ctrl+C to stop; any other key to continue...'
472           read *
473     c     PAUSE 'DBRENT exceeded maximum iterations.'
474     3     XMIN=X
475           DBRENT=FX
476           RETURN
477           END
