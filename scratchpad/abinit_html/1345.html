17     #  This script is to be used with the PHON code (or equivalent)
18     #  to calculate frozen phonon frequencies, free energies, etc...
19     #  It takes a DISP file of atomic displacements and an SPOSCAR file
20     #  with a with a supercell structure (VASP/PHON formats), and creates
21     #  the necessary lines for an abinit input file to calculate the
22     #  forces in displaced configurations.
23     #  See http://chianti.geol.ucl.ac.uk/~dario or
24     #  D. Alfe, Computer Physics Communications 180,2622-2633 (2009)
25     # 
26     #  NOTE: the symmetries in the present (1.28 8/2010) version of PHON
27     #  are not functioning properly in some cases. It is your own
28     #  responsibility to check it, and has nothing to do with ABINIT.
29     #
30     #  How to use:
31     #
32     #  1) run abinit for the relaxed structure with prtposcar 1.
33     #  this creates a reference XXX_POSCAR file, which you should rename
34     #  POSCAR.
35     #  2) run phon with LSUPER=.true.  or (e.g.)  phonopy -d --dim="2 2 3"
36     #     to create the SPOSCAR and DISP files
37     #  3) run this script (phondisp2abi.py).
38     #  4) copy script output to the abinit input file (removing duplicate
39     #  input variables etc...)
40     #  5) run abinit for each of the given datasets (and prtposcar 1 still)
41     #  6) concatenate the resulting XXX_FORCES files into one FORCES file
42     #  You also need to include the header lines for each displacement,
43     #  which are given by phondisp2abi.py in comments for each dataset
44     #  7) run phon again to get the desired phonons and properties.
45     #
46     #
47     
48     import re
49     import string
50     import numpy
51     import numpy.linalg
52     
53     #
54     #  convert PHON DISP and SPOSCAR files into ABINIT datasets with appropriately displaced atoms
55     #
56     
57     fp_disp = open('DISP')
58     lines_disp = fp_disp.readlines()
59     fp_sposcar = open('SPOSCAR')
60     lines_sposcar = fp_sposcar.readlines()
61     
62     # make unit cell input line
63     rprimd = numpy.zeros((3,3))
64     for idir  in range(3):
65     	line = lines_sposcar[2+idir]
66     	tokens = map(float,string.split(line))
67     	rprimd[0][idir] = tokens[0]
68     	rprimd[1][idir] = tokens[1]
69     	rprimd[2][idir] = tokens[2]
70     
71     
72     # get equilibirum positions
73     equilxred=[]
74     for line in lines_sposcar[7:]:
75     	equilxred.append(numpy.array(map(float,string.split(line))))
76     
77     # output unit cell input line
78     print "# Add this to the abinit input file to do the PHON displacements"
79     print "#   given in the DISP file, with respect to the supercell in SPOSCAR"
80     print "#"
81     print "# Remember the POSCAR files have sorted the atomic types so the positions"
82     print "#   and displacements are now type ordered (fix typat, spinat, etc!)"
83     print "#"
84     print "ndtset ", len(lines_disp)
85     print "# supercell lattice vectors "
86     print "acell 1 1 1 Angstr"
87     print "rprim"
88     print " %24.14f %24.14f %24.14f" % (rprimd[0][0], rprimd[1][0], rprimd[2][0]) 
89     print " %24.14f %24.14f %24.14f" % (rprimd[0][1], rprimd[1][1], rprimd[2][1]) 
90     print " %24.14f %24.14f %24.14f" % (rprimd[0][2], rprimd[1][2], rprimd[2][2]) 
91     
92     
93     idtset=1
94     # for each displacement,
95     for line in lines_disp:
96     	tokens = string.split(line)
97     # get displacement in reduced coordinates
98     	iatom = int(tokens[1])
99     	dispred = numpy.array(map(float,tokens[2:5]))
100     # add displacement to correct atom
101     	xred = list(equilxred)
102     	xred[iatom-1] = xred[iatom-1] + dispred
103     	
104     # output xred for this dataset
105     	print "# add the following line, without the #, to the FORCES file for this dtset, when concatenating"
106     	print "# %d %24.14f %24.14f %24.14f" % (iatom, dispred[0], dispred[1], dispred[2])
107     	print "xred%d" % (idtset,)
108     	for xred_1at in xred:
109     		print " %24.14f %24.14f %24.14f" % (xred_1at[0], xred_1at[1], xred_1at[2])
110     # increment dataset counter
111     	idtset=idtset+1
