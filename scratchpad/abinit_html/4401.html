55     class HasRegex(object):
56         """
57         Mixin class providing regular expressions used to analyze Fortran code.
58         Many regexs use `^` so we assume source lines have been already stripped.
59         """
60         # https://software.intel.com/en-us/fortran-compiler-18.0-developer-guide-and-reference-interface
61     
62         # A quoted expression.
63         RE_QUOTED = re.compile(r"(\".*?\")|('.*?')")
64     
65         RE_F90COMMENT = re.compile(r"\s*!(?P<value>.*)")
66         RE_OMP_SENTINEL = re.compile(r"\s*!\$OMP", re.I)
67     
68         RE_SEARCH_PROC = re.compile(r"(subroutine|function|program)", re.I)
69     
70         # We don't allow plain `end`.
71         # end subroutine name is mandatory
72         # Perhaps this is not standard-compliant
73         # but it makes the implementation more robust and the source code more readable.
74         RE_PROC_END = re.compile(r"^end\s*(?P<proc_type>subroutine|function|program)\s+(?P<name>\w*)", re.I)
75     
76         # PROGRAM [name] && END [PROGRAM [name]]
77         # NB: we enforce PROGRAM and NAME
78         RE_PROG_START = re.compile(r"^program\s+(?P<name>\w+)", re.I)
79         RE_PROG_END = re.compile(r"^end\s+program\s+(?P<name>\w+)", re.I)
80     
81         # MODULE <name> && END [MODULE [name]]
82         # NB: we enforce MODULE and NAME in END
83         RE_MOD_START= re.compile(r"^module\s+(?P<name>\w+)", re.I)
84         RE_MOD_END = re.compile(r"^end\s+module\s+(?P<name>\w+)", re.I)
85     
86         #[ MODULE ] PROCEDURE <procedure-name-list>
87         #re.compile(r"(module\s*|)procedure\s(?P<namelist>\w+)", re.I)
88     
89         # [<prefix>] <SUBROUTINE> <name> [(<args>)] [<suffix>]
90         # END [SUBROUTINE [name]]
91         # NB: we enforce name and name
92         RE_SUB_START = re.compile(r'(?P<prefix>(recursive|pure|elemental|\s)*)subroutine\s*(?P<name>\w+)', re.I)
93         RE_SUB_END = re.compile(r'^end\s+subroutine\s+(?P<name>\w+)', re.I)
94         RE_SUB_ARGS = re.compile(r"\((?P<args>.*?)\)")
95     
96         # [<prefix>] FUNCTION <name> ([<dummy-arg-list>]) [<suffix>]
97         # END [FUNCTION [name]]
98         RE_FUNC_START = re.compile(r"""
99     (?P<prefix>
100     (
101     recursive (.*) | pure (.*) | elemental (.*) |
102     logical | integer | integer(\s*\(.+\)\s*)? |
103     double\s+precision | real(\s*\(.+\))? | complex(\s*\(.+\))? |
104     character\s*\(\s*len=\w+\s*\) | type\s*\(\s*\w+\s*\)
105     )
106     \s+
107     )*
108     \s*function\s+(?P<name>\w+)\s*""",
109     re.I | re.VERBOSE)
110         #RE_FUNC_START = re.compile('^[ \t]*(([^!\'"\n]*?)function)',re.I)
111         #RE_FUNC_START = re.compile("^(?:(.+?)\s+)?function\s+(\w+)\s*(\([^()]*\))?(?=(?:.*result\s*\(\s*(\w+)\s*\))?)(?=(?:.*bind\s*\(\s*(.*)\s*\))?).*$", re.I)
112     
113         # Enforce function and name
114         RE_FUNC_END = re.compile(r"^end\s+function\s*(?P<name>\w+)", re.I)
115         #result_re = re.compile(r'result\s*\((.*?)\)', re.I)

117         # INTERFACE [generic-spec]

120         # END INTERFACE [generic-spec]

122         RE_INTERFACE_END = re.compile(r"^end\s+interface\s*(?P<name>\w*)", re.I)

124         # [if ()] call <name> [([<dummy-arg-list>])]
