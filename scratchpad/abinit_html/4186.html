9     def remove_text_in_parenthesis(s):
10         return re.sub(r'\([^)]*\)', '', s)

13     class FortypeParserError(Exception):
14         """Exceptions raised by FortypeParser."""
15     
16     
17     class FortypeParser(object):
18         """
19         Extracts the name of the variables defined in a Fortran datatype
20         Assumes plain F90 format without CPP macros.
21         """
22         Error = FortypeParserError
23     
24         # Name of the type ("type xxxx", "type :: xxxx" or "type, yyy :: xxxx"
25         re_name_type = re.compile('^[ \t]*type[ ]*(|.*::)[ ]*(?P<name>\w+)', re.IGNORECASE)
26     
27         # Detect "end type"
28         re_end_type = re.compile('^[ \t]*end[ ]+type[ ]*(?P<name>\w+)', re.IGNORECASE)
29     
30         def __init__(self, filepath, ftype_name):
31             """
32             Args:
33                 filepath:
34                     Fortran file with the declaraion of the datatype
35                 ftype_name:
36                     Name of the datatype
37             """
38             # Open the file and find the section with the type declaration.
39             indec, self.declaration = 0, []
40             with open(filepath) as fh:
41                 for line in fh:
42                     line = line.lower().strip()
43                     if not line or line.startswith("!"):
44                         continue
45     
46                     m = self.re_name_type.match(line)
47                     if m and m.groups()[1] == ftype_name:
48                         #print(m, m.groups())
49                         indec = 1
50     
51                     if indec:
52                         self.declaration.append(line)
53                         m = self.re_end_type.match(line)
54                         if m:
55                             #print(m.groups())
56                             if m.groups()[0] != ftype_name:
57                                 raise self.Error("Cannot find end to type declaration!")
58                             break
59     
60             if not self.declaration:
61                 raise self.Error("Cannot find declaration of Fortran type: %s" % ftype_name)
62     
63             # Here we assume abinit coding rules e.g. real :: var
64             # and we use "::" to find the section with the variables.
65             self.varnames, badlines = [], []
66             for dec in self.declaration[1:-1]:
67                 s = dec.find("::")
68                 if s == -1:
69                     badlines.append(dec)
70                 else:
71                     # Strip comments.
72                     dec = dec[s+2:]
73                     e = dec.find("!")
74                     if e != -1:
75                         dec = dec[:e]
76     
77                     # Dec is the string containing our variables.
78                     # Remove text inside (..) to avoid problems with arrays and split using ,
79                     # print("dec: ", dec)
80                     tokens = remove_text_in_parenthesis(dec)
81                     tokens = [s.strip() for s in tokens.split(",")]
82     
83                     self.varnames.extend(tokens)
84     
85             if badlines:
86                 print(badlines)
87                 err_msg = "You are not following abinit coding rules"
88                 raise ValueError(err_msg)
89     
90         def __str__(self):
91             return "\n".join(self.varnames)
92     
93         def json_dump(self, stream=sys.stdout):
94             import json
95             json.dump(self.varnames, stream, indent=4)
