38     # The documentation regexp.
39     re_index = re.compile(r"\<a\s+href\s*=\s*['\"](?P<file>var[a-z]+\.html)#(?P<var_link>[a-z0-9_]+)['\"]\s*\>\s*(?P<var_name>[a-z0-9_,\s]+)\s*\</a\>", re.IGNORECASE + re.MULTILINE)
40     re_title = re.compile(r"\<font\s+id\s*=\s*['\"]title['\"]\>\s*\<a\s+name\s*=\s*['\"](?P<var_id>[a-z0-9_]+)['\"]\>\s*(?P<var_name>[a-z0-9_,\s]+)\s*\</a\s*\>\s*\</font\s*\>", re.IGNORECASE + re.MULTILINE + re.DOTALL)
41     # The intagm routine regexp.
42     re_intagm = re.compile(r"^ *call +intagm *\(([a-zA-Z, 0-9():'%_*+-/]+) *&?\n? *&? *([a-zA-Z, 0-9():'%_*+-/]*)\) *$", re.MULTILINE + re.DOTALL + re.IGNORECASE)

44     def src_parse_file(arg, dirname, fnames):
45       (re_intagm, ab_vars) = arg
46       for f in fnames:
47         filename = os.path.join(dirname, f)
48         if (os.path.isfile(filename) and filename.endswith(".F90")):
49           with open(os.path.join(dirname, f), "rt") as tmpFile:
50             src_file = tmpFile.read()
51     
52           code = re_intagm.split(src_file)
53           for i in range((len(code) - 1) / 3):
54             if (re.match('^ *[^!] *call +intagm', code[i * 3]) is not None):
55               print(code[3 * i])
56               print(filename)
57               raise ValueError
58             (token, tread) = (code[i * 3 + 1] + code[i * 3 + 2]).split(",")[6:8]
59             re_set = re.compile(re.escape(token) + " *= *['\"]([a-z0-9][a-z0-9_]*)\s*['\"]", re.IGNORECASE + re.MULTILINE)
60             var = None
61             for j in range(i, -1, -1):
62               res = re_set.search(code[j * 3])
63               if (res is not None):
64                 var = res.group(1)
65                 break
66             if (var is None):
67               print(filename)
68               print(token)
69               raise ValueError
70             # Test if we don't leave_new after read of this variable.
71             if (re.match(r"^\s*if\s*\(\s*" + re.escape(tread) + \
72                          r"\s*==\s*1\s*\)\s*then\s*write.+" + \
73                          r"call\s+wrtout.+" + \
74                          r"call\s+leave_new.+" + \
75                          r"end\s*if", code[3 * i + 3], \
76                          re.IGNORECASE + re.MULTILINE + re.DOTALL) is None):
77               ab_vars[var] = [filename]

79     def doc_parse_index(filename, defs, ab_stat_links):
80       # Load the documentation into memory for later search.
81       with open(os.path.join(inp_dir, filename), "rt") as tmpFile:
82         doc_index = tmpFile.read()
83     
84       # Put all the matching elements into an iterator
85       keysIter = re_index.finditer(doc_index)
86     
87       # Consistency check for the links.
88       links = {}
89       for var in keysIter:
90         ele = var.groupdict()
91         names = ele['var_name'].split(',')
92         for name in names:
93           name = name.strip()
94           links[name] = [False]
95           ab_stat_links[0] += 1
96           if name in defs and \
97                (not(defs[name][2] == ele['var_link']) or not(defs[name][1] == ele['file'])):
98             print("Warning: variable '%11s' as a wrong link '%s#%s' in the HTML file '%s' (should be '%s#%s')." % \
99                 (name, ele['file'], ele['var_link'], filename, defs[name][1], defs[name][2]))
100             ab_stat_links[1] += 1
101       return links

103     def doc_parse_file(arg, dirname, fnames):
104       (re_title, ab_defs) = arg
105       for f in fnames:
106         filename = os.path.join(dirname, f)
107         if (os.path.isfile(filename) and f.startswith("var")):
108           with open(os.path.join(dirname, f), "r") as tmpFile:
109             doc_src = tmpFile.read()
110     
111           # Put all the matching elements into an iterator
112           keysIter = re_title.finditer(doc_src)
113     
114           # Store all the definitions.
115           for var in keysIter:
116             ele = var.groupdict()
117             names = ele['var_name'].split(',')
118             for name in names:
119               name = name.strip()
120               if name in ab_defs:
121                 print("Warning: variable '%11s' from '%s' has already a definition in '%s'." % \
122                       (name, f, ab_defs[name][1]))
123               ab_defs[name] = [False, f, ele['var_id']]
124             if (len(names) == 1) and not(ele['var_id'] == ele['var_name']):
125               print("Warning: variable '%11s' as a wrong name anchor '%s' in the HTML file '%s'." % \
126                     (ele['var_name'], ele['var_id'], f))
127               ab_defs[name][2] = name
