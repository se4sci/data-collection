34     env_ignore = ["DEFS"]
35     opt_ignore = [
36       "enable_config_file",
37       "fcflags_opt_.*",
38       "group_.*",
39       "prefix",
40       "with_config_file"]
41     val_ignore = [".*-fallback"]
42     
43     def is_ignored(keyword):
44       for opt in env_ignore + opt_ignore:
45         if ( "*" in opt ):
46           if ( re.match(opt,keyword) ):
47             return True
48         elif ( opt == keyword ):
49             return True
50       return False
51     
52     def is_ignored_val(keyword):
53       for val in val_ignore:
54         if ( "*" in val ):
55           if ( re.match(val,keyword) ):
56             return True
57         elif ( val == keyword ):
58             return True
59       return False

75     def main(home_dir=""):
76       from os.path import join as pj
77     
78       # Check if we are in the top of the ABINIT source tree
79       my_name = os.path.basename(__file__) + ".main"
80       if ( not os.path.exists(pj(home_dir,"configure.ac")) or
81            not os.path.exists(pj(home_dir, "src/98_main/abinit.F90")) ):
82         print("%s: You must be in the top of an ABINIT source tree." % my_name)
83         print("%s: Aborting now." % my_name)
84         sys.exit(1)
85     
86       # Init
87       re_env = re.compile("^#[A-Z][0-9A-Z_]*=")
88       re_opt = re.compile("^#[a-z][0-9a-z_]*=")
89     
90       # Extract environment variables from config file
91       cnf_env = MyConfigParser()
92       cnf_env.read(pj(home_dir, "config/specs/environment.conf"))
93       env_config = list()
94       for env in cnf_env.sections():
95         if cnf_env.get(env,"reset") == "no":
96           if not is_ignored(env): env_config.append(env)
97       env_config.sort()
98     
99       # Extract options from config file
100       cnf_opt = MyConfigParser()
101       cnf_opt.read(pj(home_dir, "config/specs/options.conf"))
102       opt_config = list()
103       opt_removed = list()
104       for opt in cnf_opt.sections():
105         tmp_sta = cnf_opt.get(opt,"status")
106         if ( tmp_sta == "removed" or tmp_sta == "dropped" ):
107           opt_removed.append(opt)
108         elif  ( "renamed" in tmp_sta ):
109           opt_removed.append(tmp_sta.split()[1])
110           if ( not is_ignored(opt) ):
111             opt_config.append(opt)
112         elif ( tmp_sta == "hidden" ):
113           opt_ignore.append(opt)
114         else:
115           if ( not is_ignored(opt) ):
116             opt_config.append(opt)
117       opt_config.sort()
118       opt_removed.sort()
119     
120       # Extract information from template
121       env_template = list()
122       opt_template = list()
123     
124       ac_fname = pj(home_dir, "doc/build/config-template.ac")
125       with open(ac_fname, "rt") as fh:
126           for line in fh:
127             if re_env.match(line):
128               tmp_env = re.sub("=.*","",line[1:-1])
129               if not is_ignored(tmp_env):
130                 if not tmp_env in env_template:
131                   env_template.append(tmp_env)
132             if re_opt.match(line):
133               tmp_opt = re.sub("=.*","",line[1:-1])
134               if not is_ignored(tmp_opt):
135                 if not tmp_opt in opt_template:
136                   opt_template.append(tmp_opt)
137     
138       env_template.sort()
139       opt_template.sort()
140     
141       # Check whether non-trivial option values are found in template
142       ac_fname = pj(home_dir,"doc/build/config-template.ac")
143     
144       with open(ac_fname, "rt") as fh:
145         tpl_data = fh.read()
146     
147       opt_values = dict()
148       for opt in cnf_opt.sections():
149         try:
150           tmp_values = cnf_opt.get(opt,"values").split()
151           if ( "no" in tmp_values ):
152             tmp_values.remove("no")
153           if ( "yes" in tmp_values ):
154             tmp_values.remove("yes")
155         except NoOptionError:
156           tmp_values = list()
157     
158         for val in tmp_values:
159           if ( (val[0] != "@") and (not is_ignored_val(val)) ):
160             if ( not re.search("\\* %s" % (val),tpl_data,re.MULTILINE) ):
161               if ( not opt in opt_values ):
162                 opt_values[opt] = list()
163               opt_values[opt].append(val)
164       opt_valkeys = list(opt_values.keys())
165       opt_valkeys.sort()
166     
167       # Compare environment and options
168       denv_config = [env for env in env_config if not env in env_template]
169       denv_template = [env for env in env_template if not env in env_config]
170       dopt_config = [opt for opt in opt_config if not opt in opt_template]
171       dopt_values = [opt for opt in opt_valkeys if opt not in dopt_config]
172       dopt_template = [opt for opt in opt_template if not opt in opt_config + opt_removed]
173       dopt_removed = [opt for opt in opt_removed if opt in opt_template]
174     
175       # Report any mismatch
176       nerr = ( len(denv_config) + len(denv_template) + len(dopt_config) + 
177              + len(dopt_values) + len(dopt_template) + len(dopt_removed) )
178     
179       if nerr > 0:
180         sys.stderr.write("%s: reporting environment and option mismatches\n\n" % (os.path.basename(sys.argv[0])))
181         sys.stderr.write("X: D=Documentation / I=Instance / R=Removed / U=Undefined\n\n")
182         sys.stderr.write("%s  %-48s  %-24s\n" % ("X","Option","Outdated file"))
183         sys.stderr.write("%s  %s  %s\n" % ("-","-" * 48,"-" * 24))
184     
185         for env in denv_config:
186           sys.stderr.write("%s  %-48s  %-24s\n" % ("I",env,"config-template.ac"))
187         for env in denv_template:
188           sys.stderr.write("%s  %-48s  %-24s\n" % ("U",env,"environment.conf"))
189         for opt in dopt_config:
190           sys.stderr.write("%s  %-48s  %-24s\n" % ("I",opt,"config-template.ac"))
191         for opt in dopt_values:
192           for val in opt_values[opt]:
193             sys.stderr.write("%s  %-48s  %-24s\n" % ("D","%s='%s'" % (opt,val),"config-template.ac"))
194         for opt in dopt_template:
195           sys.stderr.write("%s  %-48s  %-24s\n" % ("U",opt,"options.conf"))
196         for opt in dopt_removed:
197           sys.stderr.write("%s  %-48s  %-24s\n" % ("R",opt,"config-template.ac"))
198     
199         sys.stderr.write("\n")
200     
201       return nerr
