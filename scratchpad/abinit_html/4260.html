7     def moy (file_in) :
8     #-----------------------------------------------------------------|
9     	FILE_IN = open(file_in, 'r')
10     	neff= len(FILE_IN.readlines())
11     	FILE_IN.close()
12     	# calcul des moyennes et des fluctuations
13     	FILE_IN = open(file_in, 'r')
14     	sum = 0.
15     	sum2 = 0.
16     	for line in FILE_IN.readlines() :
17     		y = string.atof(line)
18     		sum = sum + y
19     		sum2 = sum2 + y * y
20     	amoy = sum / float(neff)
21     	fluc = math.sqrt(abs(sum2 / float(neff) - amoy * amoy))
22     	prec = 1.
23     	if abs(amoy) > prec :
24     			print '%6s = %8.1f +/ %8.4f ' % (file_in, amoy, fluc)
25     	else :
26     			print '%6s = %14.5e +/ %14.5e ' % (file_in, amoy, fluc)
27     	FILE_IN.close()

30     def lireNetcdf() :
31     #-----------------------------------------------------------------|
32     #Recherche les derniers fichiers *_HIST/*_OUT.nc cree
33     #Recherche le dernier fichier *_MOLDYN.nc cree
34     	lesfichiers = os.listdir(os.getcwd())
35     	f_fdr ='ok' ; fiche='' ; OUT_list = [] ; MOLDYN_list = []
36     	for fichier in lesfichiers :
37     		stats = os.stat(fichier)
38     		fic_tuple = time.localtime(stats[8]), fichier
39     		if fichier.find('_OUT.nc') == len(fichier)-7 and len(fichier)>7 :
40     			OUT_list.append(fic_tuple)
41     		if fichier.find('_MOLDYN.nc') == len(fichier)-10 and len(fichier)>10 :
42     			MOLDYN_list.append(fic_tuple)
43     	ok_OUT=-1;ok_MOLDYN=-1
44     	if len(OUT_list) > 0 :
45     		OUT_list.sort() ; OUT_list.reverse()
46     		fic_HIST=OUT_list[0][1].replace('_OUT.nc','_HIST.nc')
47     		if os.path.exists(fic_HIST) :
48     			fiche = fic_HIST
49     			ok_OUT=0
50     	elif ok_OUT==-1 and len(MOLDYN_list) > 0 :
51     		MOLDYN_list.sort() ; MOLDYN_list.reverse()
52     		fiche = MOLDYN_list[0][1]
53     		ok_MOLDYN=0
54     
55     	if ok_OUT == -1 and ok_MOLDYN == -1 :
56     		print "Vous n'avez pas de fichier netcdf pour diagnostics !"
57     		sys.exit(1)
58     
59     	ConvNetCdf_Ascii(fiche)
60     	return f_fdr

63     def ConvNetCdf_Ascii(file) :
64     	#Lit les fichiers mis au format netcdf et les transforme en ASCII
65             #Deux cas: fichier _MOLDYN.nc (old format) ou fichier _HIST/_OUT.nc (new format)
66     	
67             Ha_eV = 27.2113834e0   # Facteur de conversion 1 Hartree = 27,2113834 eV
68     	kb_eVK = 8.617342e-5   # Cste de Boltzman en eV/K  = (1.380658e-23)/(1.60217733e-19) #eV/K
69     	amu_emass = 1.660538782e-27/9.10938215e-31 # 1 atomic mass unit, in electronic mass
70     	fact = 29421.033e0     # Facteur de conversion  hartree/bohr^3 en GPa
71     
72             TypeFichier=0
73     	if file.find('_MOLDYN.nc') == len(file)-10 :
74     		TypeFichier=1
75     		fichier1=file
76     		fichier2=''
77     	elif file.find('_HIST.nc') == len(file)-8 :
78     		TypeFichier=2
79     		fichier1=file
80     		fichier2=file.replace('_HIST.nc','_OUT.nc')
81     	elif file.find('_OUT.nc') == len(file)-7 :
82     		TypeFichier=3
83     		fichier1=file.replace('_OUT.nc','_HIST.nc')
84     		fichier2=file
85     	if TypeFichier ==0 :
86     		print "Bug (formats des fichiers) !"
87     		sys.exit(1)
88     
89     	#Cas 1 ====================================================================
90     	if TypeFichier == 1 :
91     		ncfile = netcdf.netcdf_file(fichier1, 'r')
92     		Nbatomes= ncfile.dimensions['NbAtoms']
93     		DimTensor= ncfile.dimensions['DimTensor']
94     		Vol = ncfile.variables['Cell_Volume'].data[0]
95     		#Potential, kinetic and total energies
96     		E_pot= Ha_eV*ncfile.variables['E_pot'].data
97     		Nbtimes= E_pot.shape[0]
98     		EcrireFichierTemporaire('EPOT',[Nbtimes], E_pot)
99     		E_kin= Ha_eV*ncfile.variables['E_kin'].data
100     		EcrireFichierTemporaire('ECIN',[Nbtimes], E_kin)
101     		ETOT= E_kin + E_pot
102     		EcrireFichierTemporaire('ETOT',[Nbtimes], ETOT)
103     		#Temperature
104     		TEMPER = 2.* E_kin/(3. * kb_eVK * Nbatomes)
105     		EcrireFichierTemporaire('TEMPER',[Nbtimes], TEMPER)
106     		#Stress stensor
107     		Stress=  ncfile.variables['Stress'].data
108     		EcrireFichierTemporaire('Stress',[Nbtimes,DimTensor], Stress)
109     		#Pressure in GPa : -1/3 trace(stress tensor)  + (Nbatomes/V)*kb_eVK*T
110     		Pression = zeros(Nbtimes, float64)
111     		for i in range ( 0,  Nbtimes) :
112     			Pionique= (Nbatomes/Vol)* kb_eVK * TEMPER[i] /Ha_eV #Hartree/bohr^3
113     			Pression[i]= -(Stress[i][0]+ Stress[i][1] + Stress[i][2])/3. + Pionique
114     			Pression[i]= fact * Pression[i] #GPa
115     		EcrireFichierTemporaire('PRESS',[Nbtimes], Pression)
116     		ncfile.close()
117     	#Cas 2 or 3 ===============================================================
118     	if TypeFichier == 2 or TypeFichier ==3 :
119     		ncfile1 = netcdf.netcdf_file(fichier1, 'r')
120     		ncfile2 = netcdf.netcdf_file(fichier2, 'r')
121     		#Simulation cell data
122     		natom=ncfile1.dimensions['natom']
123     		strten_size=ncfile1.dimensions['six']
124     		typat=ncfile2.variables['typat'].data
125     		amu=ncfile2.variables['amu'].data
126     		mass=[]
127     		for typ in typat:
128     			mass.extend([amu_emass*amu[typ-1]])
129     		rprimd=ncfile1.variables['rprimd'].data[0]
130     		Vol=rprimd[0][0]*(rprimd[1][1]*rprimd[2][2]-rprimd[1][2]*rprimd[2][1]) \
131     		 +  rprimd[0][1]*(rprimd[1][2]*rprimd[2][0]-rprimd[1][0]*rprimd[2][2]) \
132     		 +  rprimd[0][2]*(rprimd[1][0]*rprimd[2][1]-rprimd[1][1]*rprimd[2][0])
133     		#Potential energy
134     		E_pot = Ha_eV*ncfile1.variables['etotal'].data
135     		Nbtimes =E_pot.shape[0] -1  # Temporarily !!!
136     		EcrireFichierTemporaire('EPOT',[Nbtimes], E_pot)
137     		#Kinetic energy from velocities and masses
138     		vel = ncfile1.variables['vel'].data
139     		E_kin=[]
140     		for itim in range(Nbtimes):
141     			jtim=itim+1 # Temporarily
142     			Ek=0.
143     			for iat in range(natom):
144     				Ek=Ek+mass[iat]*(vel[jtim][iat][0]**2+vel[jtim][iat][1]**2+vel[jtim][iat][2]**2)
145     			E_kin.extend([0.5*Ha_eV*Ek])
146     		EcrireFichierTemporaire('ECIN',[Nbtimes], E_kin)
147     		#Total energy
148     		ETOT = E_kin[0:Nbtimes] + E_pot[0:Nbtimes]
149     		EcrireFichierTemporaire('ETOT',[Nbtimes], ETOT)
150     		#Temperature
151     		TEMPER=[]
152     		for itim in range(Nbtimes):
153     			TEMPER.extend([2.*E_kin[itim]/(3.*kb_eVK*natom)])
154     		EcrireFichierTemporaire('TEMPER',[Nbtimes], TEMPER)
155     		#Stress stensor
156     		Stress=ncfile1.variables['strten'].data
157     		EcrireFichierTemporaire('Stress',[Nbtimes, strten_size], Stress)
158     		#Pressure
159     		Pression=[]
160     		for itim in range(Nbtimes):
161     			Pionique = (natom/Vol)*kb_eVK*TEMPER[itim]/Ha_eV
162     			PP=-(Stress[itim][0]+Stress[itim][1]+Stress[itim][2])/3. + Pionique
163      			Pression.extend([fact*PP])
164     		EcrireFichierTemporaire('PRESS',[Nbtimes], Pression)
165     		#Positions
166     		xred = ncfile1.variables['xred'].data
167             file_xred = open('XRED',"w")
168             for itim in range(Nbtimes):
169                 line="# TIME STEP %d \n" % (itim+1)
170                 file_xred.write(line)
171                 for ia in range(natom):
172                     line='%.10e  %.10e  %.10e \n' % (xred[itim][ia][0],xred[itim][ia][1],xred[itim][ia][2])
173                     file_xred.write(line)
174             file_xred.close
175          	xcart = ncfile1.variables['xcart'].data
176             file_xcart = open('XCART',"w")
177             for itim in range(Nbtimes):
178                 line="# TIME STEP %d \n" % (itim+1)
179                 file_xcart.write(line)
180                 for ia in range(natom):
181                     line='%.10e  %.10e  %.10e \n' % (xcart[itim][ia][0],xcart[itim][ia][1],xcart[itim][ia][2])
182                     file_xcart.write(line)
183             file_xcart.close
184     
185             ncfile1.close()
186             ncfile2.close()

188     def EcrireFichierTemporaire (nom_fic , dimensions,  data) :
189         file_out = open( nom_fic ,"w")
190         if (len (dimensions) == 1 ) :
191     	for i in range (0, dimensions[0]):
192     	    line = "%.10e\n"   % data[i]
193     	    file_out.write(line)
194         if (len (dimensions) == 2 ) :
195     	for i in range (0, dimensions[0]):
196     	    line = ' '
197     	    for j in range (0, dimensions[1]):
198     		lineC = "%.10e "   % data[i][j]
199     		line = line + lineC
200     	    line = line + "\n"
201     	    file_out.write(line)
202         file_out.close()
