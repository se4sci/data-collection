100     def get_parsers(options):
101         if not hasattr(options, "paths"):
102             raise ValueError("paths argument is missing")
103     
104         if os.path.isfile(options.paths[0]):
105             return [AbimemParser(path) for path in options.paths]
106     
107         else:
108             # Walk directory tree and find abimem files.
109             top = options.paths[0]
110             if len(options.paths) != 1:
111                raise ValueError("Expecting one argument with dirname, got %s" % len(options.paths))
112             if not os.path.isdir(top):
113                raise ValueError("Expecting existenting directory, got %s" % top)
114     
115             re_abimem = re.compile("^abimem_rank(\d+)\.mocc$")
116     
117             paths = []
118             for dirpath, dirnames, filenames in os.walk(top):
119                 for f in filenames:
120                     if not re_abimem.match(f): continue
121                     paths.append(os.path.join(dirpath, f))
122     
123             options.paths = paths
124             print("Will analyze %s abimem file(s)" % len(paths))
125             if not paths: sys.exit(1)
126             return [AbimemParser(path) for path in paths]
