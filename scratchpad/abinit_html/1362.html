22     def list_strings(arg):
23         """
24         Always return a list of strings, given a string or list of strings as
25         input.
26     
27         :Examples:
28     
29         >>> list_strings('A single string')
30         ['A single string']
31     
32         >>> list_strings(['A single string in a list'])
33         ['A single string in a list']
34     
35         >>> list_strings(['A','list','of','strings'])
36         ['A', 'list', 'of', 'strings']
37         """
38         if is_string(arg):
39             return [arg]
40         else:
41             return arg

44     class WildCard(object):
45         """
46         This object provides an easy-to-use interface for
47         filename matching with shell patterns (fnmatch).
48     
49         .. example:
50     
51         >>> w = WildCard("*.nc|*.pdf")
52         >>> w.filter(["foo.nc", "bar.pdf", "hello.txt"])
53         ['foo.nc', 'bar.pdf']
54     
55         >>> w.filter("foo.nc")
56         ['foo.nc']
57         """
58         def __init__(self, wildcard, sep="|"):
59             """
60             Args:
61                 wildcard:
62                     String of tokens separated by sep.
63                     Each token represents a pattern.
64                 sep:
65                     Separator for shell patterns.
66             """
67             self.pats = ["*"]
68             if wildcard:
69                 self.pats = wildcard.split(sep)
70     
71         def __str__(self):
72             return "<%s, patterns = %s>" % (self.__class__.__name__, self.pats)
73     
74         def filter(self, names):
75             """
76             Returns a list with the names matching the pattern.
77             """
78             names = list_strings(names)
79     
80             fnames = []
81             for f in names:
82                 for pat in self.pats:
83                     if fnmatch.fnmatch(f, pat):
84                         fnames.append(f)
85     
86             return fnames
87     
88         def match(self, name):
89             """
90             Returns True if name matches one of the patterns.
91             """
92             for pat in self.pats:
93                 if fnmatch.fnmatch(name, pat):
94                     return True
95     
96             return False

122     def mkrobodoc_files(top):
123         """
124         Generate the ROBODOC files in all the ABINIT directories
125         located within the top level directory top.
126     
127         Returns:
128             Exit status.
129         """
130         top = os.path.abspath(top)
131         # Select files with these extensions.
132         wildcard = WildCard("*.F90|*.finc")
133     
134         # Walk the directory tree starting from top
135         # Find the source files in the Abinit directories
136         # and add their name to the _dirname_ file used by robodoc
137         wrong_dirpaths = []
138     
139         for dirpath, dirnames, filenames in os.walk(top):
140             dirname = os.path.basename(dirpath)
141             if "dirname" in ("__pycache__", ): continue
142             robo_dfile = "_" + dirname  + "_"
143     
144             if robo_dfile not in filenames:
145                 # Not an abinit dir.
146                 wrong_dirpaths.append(os.path.abspath(dirpath))
147                 continue
148     
149             robo_dfile = os.path.abspath(os.path.join(dirpath, robo_dfile))
150             with open(robo_dfile, "r") as f:
151                 robotext = []
152                 for line in f:
153                     robotext.append(line.strip())
154                     if line.startswith("!! CHILDREN"): break
155                 else:
156                     raise ValueError("File %s does not have a valid '!! CHILDREN' section'" % robo_dfile)
157     
158             # Get source files, sort them in alphabetical order and generate new robodoc file.
159             src_files = wildcard.filter(os.listdir(dirpath))
160             if not src_files: continue
161             src_files = sorted(src_files)
162     
163             robotext +=  ["!! " + src_file for src_file in src_files]
164             robotext += ["!!", "!!***"]
165             #for line in robotext: print(line)
166     
167             # Write new robodoc file.
168             with open(robo_dfile, "w") as f:
169                 f.write("\n".join(robotext) + "\n")
170     
171         if wrong_dirpaths:
172             # Some dirs should not have a ROBODOC file.
173             # This code does not work if we have the same basename but oh well.
174             EXCLUDE_BASEDIRS = set([
175                 "src",
176                 "replacements",
177                 "01_interfaces_ext",
178                 "ptgroup_data",
179                 "incs",
180                 "libs",
181                 "mods",
182             ])
183     
184             # Remove dirs in EXCLUDE_BASEDIRS
185             wrong_dirpaths = [d for d in wrong_dirpaths if os.path.basename(d) not in EXCLUDE_BASEDIRS]
186     
187             if wrong_dirpaths:
188                 print("dirs_without_files")
189                 for i, dirpath in enumerate(wrong_dirpaths):
190                     print("%d) %s" % (i, dirpath))
191                     robodoc_dfile = os.path.join(dirpath, "_" + os.path.basename(dirpath) + "_")
192                     assert not os.path.isfile(robodoc_dfile)
193                     with open(robodoc_dfile, "w") as f:
194                         f.write(robodoc_dheader(dirpath))
195     
196         return len(wrong_dirpaths)
