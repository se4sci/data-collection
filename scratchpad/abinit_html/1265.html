21         def read_nc(self, fname=None):

23             fname = fname if fname else self.fname

25             super(FanFile, self).read_nc(fname)
26     
27             with nc.Dataset(fname, 'r') as root:
28     
29                 self.natom = len(root.dimensions['number_of_atoms'])
30                 self.nkpt = len(root.dimensions['number_of_kpoints'])
31                 self.nband = len(root.dimensions['max_number_of_states'])
32                 self.nsppol = len(root.dimensions['number_of_spins'])
33     
34                 # number_of_spins, number_of_kpoints, max_number_of_states
35                 self.occ = root.variables['occupations'][:,:,:]

37                 self.FAN = zeros((self.nkpt, self.nband, 3, self.natom,

40                 # product_mband_nsppol, number_of_atoms,  number_of_cartesian_directions,
41                 # number_of_atoms, number_of_cartesian_directions,
42                 # number_of_kpoints, product_mband_nsppol*2
43                 FANtmp = root.variables['second_derivative_eigenenergies_actif'][:,:,:,:,:,:,:]
44                 #FANtmp2 = zeros((self.nkpt,2*self.nband,3,self.natom,3,self.natom,self.nband))
45                 FANtmp2 = np.einsum('ijklmno->nomlkji', FANtmp)
46                 self.FAN.real[...] = FANtmp2[:, ::2, ...]
47                 self.FAN.imag[...] = FANtmp2[:, 1::2, ...]
48                 del FANtmp, FANtmp2

50                 # number_of_spins, number_of_kpoints, max_number_of_states   
51                 self.eigenvalues = root.variables['eigenvalues'][:,:,:]

54                 self.kpt = root.variables['reduced_coordinates_of_kpoints'][:,:]
55                 self.qred = root.variables['current_q_point'][:]
