89     def polynd_ab(x,y,degree,indices=None):
90         if degree < 1:
91             print 'ERROR: cannot find a polynomial going through two points with this degree (%s) ... exit' %degree
92             sys.exit()
93         if len(N.shape(x)) != 1 or len(N.shape(y)) != 1:
94             print 'ERROR: the array x and/or y is not 1D ... exit'
95             sys.exit()
96         if len(x) != len(y):
97             print 'ERROR: x and y arrays have different lengths ... exit'
98             sys.exit()
99         if indices == None:
100             indices = N.array([0,len(x)-1])
101         else:
102             if indices[0] < 0 or indices[1] >= len(x) or indices[0] == indices[1]:
103                 print 'ERROR: indices (0 <= indices[0]=%s < indices[1]=%s < len(x)=%s) are wrong ... exit' %(indices[0],indices[1],len(x))
104                 sys.exit()
105         if degree == 1:
106             pp = N.polyfit(N.array([x[indices[0]],x[indices[1]]]),N.array([y[indices[0]],y[indices[1]]]),degree)
107             return pp
108         x1 = x[indices[0]]
109         x2 = x[indices[1]]
110         y1 = y[indices[0]]
111         y2 = y[indices[1]]
112         xm = N.linspace(N.min(x),N.max(x),degree,endpoint=False)[1:]
113         prod1 = (x-x2)/(x1-x2)*y1
114         prod2 = (x-x1)/(x2-x1)*y2
115         coeff_list = list()
116         for ii in range(len(xm)):
117             prod1 = prod1*(x-xm[ii])/(x1-xm[ii])
118             prod2 = prod2*(x-xm[ii])/(x2-xm[ii])
119             prod_ii = (x-x2)*(x-x1)/((xm[ii]-x2)*(xm[ii]-x1))
120             for jj in range(len(xm)):
121                 if ii != jj:
122                     prod_ii = prod_ii*(x-xm[jj])/(xm[ii]-xm[jj])
123             coeff_list.append(prod_ii)
124         p1 = prod1 + prod2
125         newy = y - p1
126         A = N.vstack(N.array(coeff_list)).T
127         ym = N.linalg.lstsq(A,newy)[0]
128         xx = N.array([x1])
129         yy = N.array([y1])
130         for ii in range(len(xm)):
131             xx = N.append(xx,[xm[ii]])
132             yy = N.append(yy,[ym[ii]])
133         xx = N.append(xx,[x2])
134         yy = N.append(yy,[y2])
135         pp = N.polyfit(xx,yy,degree)
136         return pp

184     def smoothend(energy_pivots,polyfit_list,energy,delta_energy_ev=None):
185         method = 2
186         if delta_energy_ev == None:
187             if method == 1:
188                 delta_energy_ev = 0.05
189             elif method == 2:
190                 delta_energy_ev = 1.00
191         if method == 1:
192             xi = energy
193             if xi < energy_pivots[0]:
194                 print 'Error: energy should be larger than the first energy pivot ...'
195                 sys.exit()
196             if xi > energy_pivots[-1]:
197                 ii = len(energy_pivots)
198             else:
199                 ii = N.argwhere(energy_pivots>xi)[0]
200             fpi = N.polyval(N.polyder(polyfit_list[ii]),xi)
201             fi = N.polyval(polyfit_list[ii],xi)
202             if fpi == 0:
203                 print 'TODO, the derivative is 0 ... easy but lazy :-)'
204                 return
205             elif fpi > 0:
206                 fv = fi + delta_energy_ev
207             elif fpi < 0:
208                 fv = fi - delta_energy_ev
209             aa = fpi**2/(4*(fi-fv))
210             bb = fpi - 2*aa*xi
211             cc = fi - aa*xi**2 - bb*xi
212             xv = -bb/(2*aa)
213             new_energy_pivots = N.zeros(ii+2,N.float)
214             for jj in N.arange(ii):
215                 new_energy_pivots[jj] = energy_pivots[jj]
216             new_energy_pivots[-2] = energy
217             new_energy_pivots[-1] = xv
218             new_polyfit_list = list()
219             for jj in N.arange(ii+1):
220                 new_polyfit_list.append(polyfit_list[jj])
221             new_polyfit_list.append([aa,bb,cc])
222             new_polyfit_list.append([fv])
223             return new_energy_pivots,new_polyfit_list
224         if method == 2:
225             xi = energy

233             fpi = N.polyval(N.polyder(polyfit_list[ii]),xi)
234             fi = N.polyval(polyfit_list[ii],xi)
235             if fpi == 0:
236                 new_energy_pivots = N.zeros(ii+1,N.float)
237                 for jj in N.arange(ii):
238                     new_energy_pivots[jj] = energy_pivots[jj]

243                 new_polyfit_list.append([fi])
244                 return new_energy_pivots,new_polyfit_list
245             else:
246                 xv = xi + delta_energy_ev
247                 bb = fpi/(1.0-xi/xv)
248                 aa = -bb/(2.0*xv)
249                 cc = fi - aa*xi**2 - bb*xi
250                 pp = [aa,bb,cc]
251                 fv = N.polyval(pp,xv)
