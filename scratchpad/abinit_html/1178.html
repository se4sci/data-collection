61     def get_allowed_cpp_vars():
62         """
63         Inspect the libpaw header file, the autoconf macros and config.ac
64         Extract and return the set of allowed CPP options, used to check
65         the exclude_cpp_vars TEST_INFO section for possible typos.
66     
67         Based on ~abinit/abichecks/scripts/check-cpp-options.
68         """
69         import re
70         re_m4file = re.compile("\.m4$")
71         re_hdrfile = re.compile("\.h$")
72         re_acdef = re.compile("AC_DEFINE\\(")
73         re_cppdef = re.compile("^([ ]?)+#([ ]?)+define [0-9A-Z_]*")
74         
75         abidir = os.path.abspath("../")
76     
77         # Extract CPP options from the libPAW header files
78         cpp_libpaw = set()
79         for root, dirs, files in os.walk(os.path.join(abidir, "src/42_libpaw")):
80             for src in files:
81                 if not re_hdrfile.search(src): continue
82                 with open(os.path.join(root, src), "rt") as fh:
83                     for line in fh:
84                         if not re_cppdef.search(line): continue
85                         tmp_def = re.sub("^[# ]*define[ ]*([0-9A-Z_]*).*","\\1", line).strip()
86                         cpp_libpaw.add(tmp_def)
87     
88         # Extract CPP options from the build system
89         cpp_buildsys = set()
90         for root, dirs, files in os.walk(os.path.join(abidir, "config/m4")):
91             for src in files:
92                 if not re_m4file.search(src): continue
93                 with open(os.path.join(root, src), "rt") as fh:
94                     for line in fh:
95                         if not re_acdef.search(line): continue
96                         tmp_def = re.sub(".*AC_DEFINE\\([\\[]?([^\\],]*).*","\\1",line).strip()
97                         cpp_buildsys.add(tmp_def)
98     
99         with open(os.path.join(abidir, "configure.ac"), "rt") as fh:
100             for line in fh:
101                 if not re_acdef.search(line): continue
102                 tmp_def = re.sub(".*AC_DEFINE\\([\\[]?([^\\],]*).*","\\1",line).strip()
103                 cpp_buildsys.add(tmp_def)
104     
105         return cpp_buildsys.union(cpp_libpaw)
