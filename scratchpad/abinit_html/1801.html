24     /* 
25      * This function causes any buffered output on stream to be delivered to the file. 
26      * If stream is a null pointer, then fflush causes buffered output on all open 
27      * output streams to be flushed. It returns EOF if a write error occurs, or zero otherwise.
28      */
29     void 
30     FC_FUNC_(clib_fflush,CLIB_FFLUSH)
31       (int* ierr)
32     {
33       *ierr = fflush(NULL);
34     }

36     /* This function returns a string that is the value of the environment variable fname
37      * If the environment variable fname is not defined, the value is a null pointer. 
38      */
39     void 
40     FC_FUNC_(clib_getenv,CLIB_GETENV)
41       (int* ierr, STR_F_TYPE fname, STR_F_TYPE fvalue STR_ARG2)
42     {
43       char *cname, *cvalue;
44       TO_C_STR1(fname, cname);
45       cvalue = getenv (cname);
46       xfree(cname);
47     
48       *ierr = 1;
49       if (cvalue && strlen(cvalue) <= l2 ){
50         *ierr = 0;
51         TO_F_STR2(cvalue, fvalue);
52       } 
53     }

55     /*
56     *  Replacement for the F2003 intrinsic GET_ENVIRONMENT_VARIABLE.
57     *  This version can be called in Fortran code but does not support optional arguments.
58     *
59     * CALL GET_ENVIRONMENT_VARIABLE (NAME[,VALUE,LENGTH,STATUS,TRIM_NAME]) 
60     *   obtains the value of an environment variable.
61     *
62     *   NAME is a default character INTENT(IN) scalar that identifies the required environment
63     *          variable. The interpretation of case is processor dependent.
64     *   [VALUE] is a default character INTENT(OUT) scalar that is assigned the value of the environment variable.
65     *   [LENGTH]is a default integer INTENT(OUT) scalar. If the specified environment variable exists and has a value, 
66     *           LENGTH is set to the length (number of characters) of that value. Otherwise, LENGTH is set to 0.
67     *   [STATUS] is a default integer INTENT(OUT) scalar that indicates success or failure.
68     *   [TRIM_NAME] is a logical INTENT(IN) scalar that indicates whether trailing blanks
69     *               in NAME are considered significant.
70     */
71     
72     void 
73     FC_FUNC_(clib_get_environment_variable,CLIB_GET_ENVIRONMENT_VARIABLE)
74       (const STR_F_TYPE fname, STR_F_TYPE fvalue, int *length, int *status, const int *trim_name STR_ARG2)
75       /* name_len, value_len) */
76     {
77       char *cname, *cvalue;
78       int q_len;
79     
80       if (*trim_name)
81         TO_C_STR1(fname, cname)
82       else 
83         TO_C_STR1_NOTRIM(fname, cname)
84     
85       cvalue = getenv(cname);
86       q_len = (cvalue == NULL) ? 0 : strlen(cvalue);
87       xfree(cname);
88     
89       *length = (cvalue == NULL) ? 0 : q_len;
90     
91       if (cvalue == NULL)  
92         *status = 1;  /* $fname does not exist */
93     
94       else if (cvalue != NULL && l2 < q_len)  
95         *status = -1; /* Fortran string is too short */
96     
97       else
98         *status = 0;  /* OK */
99     
100       if (cvalue != NULL) TO_F_STR2(cvalue, fvalue)
101     }
