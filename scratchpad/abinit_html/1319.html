441         def verifStep(self):
442             if self.file1.isGoodFile():
443                 ni = self.sbni.value()
444                 nf = self.sbnf.value()
445                 if(ni>nf):
446                     self.showError("initial step must be inferior to final step")
447                     self.sbni.setValue(self.ni)
448                     self.sbnf.setValue(self.nf)
449                     return False
450                 if(nf==ni):
451                     self.showError("final step must be superior to initial step")
452                     self.sbni.setValue(self.ni)
453                     self.sbnf.setValue(self.nf)
454                     return False
455                 if((ni == self.ni)and(nf == self.nf)):
456                     return False
457                 return True
458             else:
459                 self.showError("Your NetCDF file is not correct")
460                 return False

535         def updateLabel(self):
536             E_tot = self.units['Energy'][1] * Math.average(self.file1.getE_Tot())
537             deviation = self.file1.getStandardDeviation(self.units['Energy'][1] *self.file1.getE_Tot(), E_tot)      
538             strETOT = self.displayData(E_tot,deviation)
539     
540             vol = self.units['Volume'][1] * Math.average(self.file1.getVol())
541             deviation = self.file1.getStandardDeviation(self.units['Volume'][1] *self.file1.getVol(), vol)        
542             strVol = self.displayData(vol,deviation)
543     
544             Temp = self.file1.getTemp()        
545             ATemp = Math.average(Temp) - self.units['Temperature'][1]        
546             deviation = Math.standard_deviation( Temp - self.units['Temperature'][1] , ATemp)
547             strTemp = self.displayData(ATemp,deviation)
548     
549             Press = Math.average( self.file1.getPress() ) * self.units['Pressure'][1]
550             deviation = Math.standard_deviation(self.units['Pressure'][1] * self.file1.getPress(), Press)
551             strPress = self.displayData(Press,deviation)
552     
553             self.lbl4.setText("Total Energy ("+str(self.units['Energy'][0])+"): "+strETOT)
554             self.lbl5.setText("Volume ("+str(self.units['Volume'][0])+"): "+strVol)
555             self.lbl6.setText("Temperature ("+str(self.units['Temperature'][0])+")  : "+strTemp)
556             self.lbl7.setText("Pressure ("+str(self.units['Pressure'][0])+")    : "+strPress)
