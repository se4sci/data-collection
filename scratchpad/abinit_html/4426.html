8     def number_of_cpus():
9         """
10         Number of virtual or physical CPUs on this system, i.e.
11         user/real as output by time(1) when called with an optimally scaling userspace-only program
12         Return -1 if ncpus cannot be detected
13         taken from:
14         http://stackoverflow.com/questions/1006289/how-to-find-out-the-number-of-cpus-in-python
15         """
16         import os,re,subprocess
17     
18         # Python 2.6+
19         try:
20             import multiprocessing
21             return multiprocessing.cpu_count()
22         except (ImportError, NotImplementedError):
23             pass
24     
25         # POSIX
26         try:
27             res = int(os.sysconf('SC_NPROCESSORS_ONLN'))
28             if res > 0: return res
29         except (AttributeError, ValueError):
30             pass
31     
32         # Windows
33         try:
34             res = int(os.environ['NUMBER_OF_PROCESSORS'])
35             if res > 0: return res
36         except (KeyError, ValueError):
37             pass
38     
39         # jython
40         try:
41             from java.lang import Runtime
42             runtime = Runtime.getRuntime()
43             res = runtime.availableProcessors()
44             if res > 0: return res
45         except ImportError:
46             pass
47     
48         # BSD
49         try:
50             sysctl = subprocess.Popen(['sysctl', '-n', 'hw.ncpu'], stdout=subprocess.PIPE)
51             scStdout = sysctl.communicate()[0]
52             res = int(scStdout)
53             if res > 0: return res
54         except (OSError, ValueError):
55             pass
56     
57         # Linux
58         try:
59             res = open('/proc/cpuinfo').read().count('processor\t:')
60             if res > 0: return res
61         except IOError:
62             pass
63     
64         # Solaris
65         try:
66             pseudoDevices = os.listdir('/devices/pseudo/')
67             expr = re.compile('^cpuid@[0-9]+$')
68             res = 0
69             for pd in pseudoDevices:
70                 if expr.match(pd) is not None:
71                     res += 1
72             if res > 0: return res
73         except OSError:
74             pass
75     
76         # Other UNIXes (heuristic)
77         try:
78             try:
79                 dmesg = open('/var/run/dmesg.boot').read()
80             except IOError:
81                 dmesgProcess = subprocess.Popen(['dmesg'], stdout=subprocess.PIPE)
82                 dmesg = dmesgProcess.communicate()[0]
83     
84             res = 0
85             while '\ncpu' + str(res) + ':' in dmesg:
86                 res += 1
87     
88             if res > 0: return res
89         except OSError:
90             pass
91     
92         return -1

100     class FileLock(object):
101         """ A file locking mechanism that has context-manager support so
102             you can use it in a with statement. This should be relatively cross
103             compatible as it doesn't rely on msvcrt or fcntl for the locking.
104             Taken from http://www.evanfosmark.com/2009/01/cross-platform-file-locking-support-in-python/
105         """
106         Error = FileLockException
107     
108         def __init__(self, file_name, timeout=10, delay=.05):
109             """ Prepare the file locker. Specify the file to lock and optionally
110                 the maximum timeout and the delay between each attempt to lock.
111             """
112             self.file_name = file_name
113             self.lockfile = os.path.abspath(file_name) + ".lock"
114             self.timeout = float(timeout)
115             self.delay = float(delay)
116             self.is_locked = False
117     
118             if (self.delay > self.timeout or
119                 self.delay   <= 0 or
120                 self.timeout <= 0):
121                 err_msg = "delay and timeout must be positive with delay <= timeout"
122                 raise ValueError(err_msg)
123     
124         @classmethod
125         def FakeLock(cls, file_name, timeout=10, delay=.05):
126             """Returns a fake lock file."""
127             fake = cls(file_name, timeout=timeout, delay=delay)
128     
129             def nop():
130                 """Monkey patch."""
131     
132             fake.acquire = nop
133             fake.release = nop
134             return fake
135     
136         def acquire(self):
137             """ Acquire the lock, if possible. If the lock is in use, it check again
138                 every `wait` seconds. It does this until it either gets the lock or
139                 exceeds `timeout` number of seconds, in which case it throws
140                 an exception.
141             """
142             start_time = time.time()
143             while True:
144                 try:
145                     self.fd = os.open(self.lockfile, os.O_CREAT|os.O_EXCL|os.O_RDWR)
146                     break
147                 except OSError as e:
148                     if e.errno != errno.EEXIST:
149                         raise
150                     if (time.time() - start_time) >= self.timeout:
151                         raise FileLockException("Timeout occured.")
152                     time.sleep(self.delay)
153     
154             self.is_locked = True
155     
156         def release(self):
157             """ Get rid of the lock by deleting the lockfile.
158                 When working in a `with` statement, this gets automatically
159                 called at the end.
160             """
161             if self.is_locked:
162                 os.close(self.fd)
163                 os.unlink(self.lockfile)
164                 self.is_locked = False
165     
166         def __enter__(self):
167             """ Activated when used in the with statement.
168                 Should automatically acquire a lock to be used in the with block.
169             """
170             if not self.is_locked: self.acquire()
171             return self
172     
173         def __exit__(self, type, value, traceback):
174             """ Activated at the end of the with statement.
175                 It automatically releases the lock if it isn't locked.
176             """
177             if self.is_locked: self.release()
178     
179         def __del__(self):
180             """ Make sure that the FileLock instance doesn't leave a lockfile
181                 lying around.
182             """
183             self.release()
