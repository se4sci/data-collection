55     def rflat(iterables):
56         """Iterator over all elements of a nested iterable. It's recursive!"""
57         for item in iterables:
58             if not hasattr(item, "__iter__"): 
59                 yield item
60             else: # iterable object.
61                 for it in rflat(item): yield it

63     def npymat2Fmat(npymat):
64         """Return a string with the F90 declaration of a numpy matrix."""
65                                                                         
66         shape = str(npymat.shape)
67         shape = shape.replace("(","(/").replace(")","/)")

69         fmat = npymat.T # From C to F ordering.
70         vect = ""
71         for idx, ele in enumerate(fmat.flat): 
72             #print ele

82         vect = "(/" + vect + "/)"
83         return " RESHAPE( %(vect)s, %(shape)s )" % locals()

94

107             for ridx, row in enumerate(self.rotation):

137         def _get_det(self):
138             """Return the determinant of a symmetry matrix mat[3,3]. It must be +-1"""
139             mat = self.rotation
140             det =   mat[0,0]* ( mat[1,1]*mat[2,2] - mat[1,2]*mat[2,1] )\
141                   - mat[0,1]* ( mat[1,0]*mat[2,2] - mat[1,2]*mat[2,0] )\
142                   + mat[0,2]* ( mat[1,0]*mat[2,1] - mat[1,1]*mat[2,0] )
143     
144             if abs(det) != 1: 
145                 raise RotationException("abs(det) must be 1 while it is " + str(abs(det)))
146             else:
147                 return det

159         def invert(self): 
160             """
161             Invert an orthogonal 3x3 matrix of INTEGER elements.
162             Note use of integer arithmetic. Raise RotationException if not invertible.
163             """
164     
165             det = self.det
166             mm = self.rotation
167             inv= N.matrix(N.zeros((3,3), N.int))
168     
169             inv[0,0] = mm[1,1] * mm[2,2] - mm[1,2] * mm[2,1]
170             inv[0,1] = mm[0,2] * mm[2,1] - mm[0,1] * mm[2,2] 
171             inv[0,2] = mm[0,1] * mm[1,2] - mm[0,2] * mm[1,1]
172     
173             inv[1,0] = mm[1,2] * mm[2,0] - mm[1,0] * mm[2,2]
174             inv[1,1] = mm[0,0] * mm[2,2] - mm[0,2] * mm[2,0]
175             inv[1,2] = mm[0,2] * mm[1,0] - mm[0,0] * mm[1,2] 
176     
177             inv[2,0] = mm[1,0] * mm[2,1] - mm[1,1] * mm[2,0]
178             inv[2,1] = mm[0,1] * mm[2,0] - mm[0,0] * mm[2,1] 
179             inv[2,2] = mm[0,0] * mm[1,1] - mm[0,1] * mm[1,0]
180     
181             # Make sure matrix is not singular
182             if det != 0: 
183                 return Rotation(inv/det)
184             else: 
185                 raise RotationException("Attempting to invert singular matrix")

295     def mk_classes(rotations):
296         """
297         Find the classes of a group. Return a list containing the indeces
298         of the operations in each class sorted in ascending order.
299                                                                            
300         A class is defined as the set of distinct items obtained by 
301         considering for each element, S, of the group all its conjugate 
302         X^-1 S X where X is one of the elements of the group.
303         """
304     
305         class_ids = list()
306         seen = [ 0 for i in range(len(rotations)) ]
307         nclass = -1 
308                                                                             
309         for idx, m in enumerate(rotations):
310             if seen[idx]: continue
311             seen[idx] = 1
312     
313             nclass += 1
314             class_ids.append([])
315             class_ids[nclass].append(idx)
316     
317             for x in rotations:
318                 new =  x.invert() * m * x
319                 idx_found = -1
320                 for idx_search, search in enumerate(rotations):
321                     if search == new:
322                         idx_found = idx_search
323                         break
324                 if (idx_found == -1): sys.stderr.write("idx_found == -1")
325                 if not seen[idx_found]: 
326                     seen[idx_found] = 1
327                     class_ids[nclass].append(idx_found)
328                                                                             
329         # Now sort the indeces.
330         sort_class_ids = list()
331         for ids in class_ids:
332             ids.sort()
333             sort_class_ids.append(ids)
334         return sort_class_ids

808     
809             # Read data from file and remove blank lines
810             lines = open(fname,"r").readlines()
811             self.text = [ line for line in lines if line.lstrip() ]
812     
813             # Extract the group name from the first string.
814             title = self.text.pop(0)
