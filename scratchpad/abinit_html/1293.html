3     def symmetrize_array(arr, indices_groups, axis=0):
4         """
5         Symmetrize an array by taking the average over each subset of indices.
6     
7         For example, given a list of indices_groups [[a1,a2,a3], [b1,b2,b3]],
8         the arrays (arr[a1], arr[a2], arr[a3]) will be averaged,
9         as well as the arrays (arr[b1], arr[b2], arr[b3]).
10     
11         Arguments
12         ---------
13     
14         arr: np.array
15             The array to be symmetrized.
16         indices_groups: 2D list.
17             A list of groups of indices
18         axis:
19             The dimension on which the indicies are taken.
20     
21         Returns
22         -------
23     
24         sym_arr: np.array
25             The symmetrized array, of the same shape as arr.
26         """
27     
28         sym_arr = np.copy(arr)
29     
30         indices = np.arange(arr.size).reshape(arr.shape)
31     
32         for group in indices_groups:
33     
34             arr_group = arr.take(group, axis=axis)
35             av_arr = np.average(arr_group, axis=axis) 
36     
37             for i in group:
38     
39                 ind = indices.take(i, axis=axis)
40                 sym_arr.put(ind, av_arr)
41     
42         return sym_arr
