32     server="http://localhost:8000"
33     
34     def rm_server(keyword):
35       if keyword.startswith(server):
36          return keyword[len(server):]
37     
38     # ---------------------------------------------------------------------------- #
39     
40     # Types of error :
41     #    Error: 403 Forbidden'
42     #    Error: 404 Not Found'
43     #    Error: 504 Gateway Time-out'
44     #    Error: 502 Bad Gateway'
45     #    Error: ReadTimeout:'
46     #    ConnectionError: ('Connection aborted.'
47     
48     no_error_list = [
49         { 'url'  : re.compile('(doi|aps).org'),
50           'info' : re.compile('^Redirected'),
51           'valid': re.compile('^403 Forbidden')
52         },
53     ]
54     
55     # ---------------------------------------------------------------------------- #
56     
57     #
58     # Main program
59     #
60     def main(filename,home_dir=""):
61       from os.path import join as pj
62     
63       # Check if we are in the top of the ABINIT source tree
64       my_name = os.path.basename(__file__) + ".main"
65       if ( not os.path.exists(pj(home_dir,"configure.ac")) or
66            not os.path.exists(pj(home_dir, "src/98_main/abinit.F90")) ):
67         print("%s: You must be in the top of an ABINIT source tree." % my_name)
68         print("%s: Aborting now." % my_name)
69         sys.exit(1)
70     
71       #
72       tree = etree.parse(filename)
73       
74       rc=0
75       for child in tree.xpath("/linkchecker/urldata"):
76       
77         for no_error in no_error_list:
78       
79           norc = True  # True : we consider that this is not an error
80       
81           url_rc = None
82           url = child.find('url')
83           url_rc = no_error['url'].search(url.text)
84           norc = norc and ( url_rc != None )
85       
86           info_rc = None
87           info=child.find('infos/info')
88           if info is not None:
89             try:
90              info_rc=no_error['info'].search(info.text)
91              norc = norc and ( info_rc != None )
92             except:
93              pass
94       
95           valid_rc = None
96           valid=child.find('valid').get("result")
97           valid_rc=no_error['valid'].search(valid)
98           norc = norc and ( valid_rc != None )
99       
100           if url_rc != None and info_rc != None and valid_rc != None:
101               break
102     
103           if not norc :
104               # found a true error... : reporting on bb
105               rc += 1
106               name=child.find('name')
107               parent=child.find('parent')
108               realurl=child.find('realurl')
109               print("{0:12} {1}".format('URL',url.text))
110               print("{0:12} {1}".format('Name',name.text))
111               print("{0:12} {1}, line {2}".format('Parent URL',rm_server(parent.text),parent.get('line')))
112               print("{0:12} {1}".format('Real URL',realurl.text))
113               print("{0:12} {1}".format('Result',valid))
114               print('--------------------')
115               break
116     
117       return rc
118

133

135       sys.exit(exit_status)
