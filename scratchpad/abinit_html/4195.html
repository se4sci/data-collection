29     def getstatusoutput(cmd):
30         """    Return (status, output) of executing cmd in a shell.
31     
32         Execute the string 'cmd' in a shell with 'check_output' and
33         return a 2-tuple (status, output). Universal newlines mode is used,
34         meaning that the result with be decoded to a string.
35     
36         A trailing newline is stripped from the output.
37         The exit status for the command can be interpreted
38         according to the rules for the function 'wait'. Example:
39         """
40         from subprocess import check_output, STDOUT, CalledProcessError
41         try:
42             data = check_output(cmd, shell=True, universal_newlines=True, stderr=STDOUT)
43             status = 0
44         except CalledProcessError as ex:
45             data = ex.output
46             status = ex.returncode
47         if data[-1:] == '\n':
48             data = data[:-1]
49         return status, data

55       from os.path import join as pj
56

60            not os.path.exists( pj(home_dir,"src/98_main/abinit.F90")) ):

62         print("%s: Aborting now." % my_name)

78           (ret,tmp) = getstatusoutput("diff  %s/%s %s/%s" % \

80           if ( ret != 0 ):
81             bex_diffs.append(ref_file)
82             sys.stdout.write(tmp)

84           bex_missing.append(ref_file)
85
