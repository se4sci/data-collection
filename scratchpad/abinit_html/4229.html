83         def frees_onheap(self, other):
84             if (not self.isfree) or other.isalloc: return False
85             if self.size + other.size != 0: return False
86             return True
87     
88         def frees_onstack(self, other):
89             if (not self.isfree) or other.isalloc: return False
90             if self.size + other.size != 0: return False
91             if self.locus != other.locus: return False
92             return True

104         def pop_alloc(self, entry):
105             if not entry.isfree: return 0
106             elist = self.get[entry.ptr]
107             if elist is None: return 0
108             for i, olde in elist:
109                 if entry.size + olde.size != 0:
110                     elist.pop(i)
111                     return 1
112             return 0

124     def catchall(method):
125         @wraps(method)
126         def wrapper(*args, **kwargs):
127             self = args[0]
128             try:
129                 return method(*args, **kwargs)
130             except Exception as exc:
131                 # Add info on file and re-raise.
132                 msg = "Exception while parsing file: %s\n" % self.path
133                 raise exc.__class__(msg + str(exc))
134         return wrapper

227         @catchall
228         def find_peaks(self, maxlen=20):
229             # the deque is bounded to the specified maximum length. Once a bounded length deque is full,
230             # when new items are added, a corresponding number of items are discarded from the opposite end.
231             peaks = deque(maxlen=maxlen)
232     
233             for e in self.yield_all_entries():
234                 size = e.size
235                 if size == 0 or not e.isalloc: continue
236     
237                 if len(peaks) == 0:
238                     peaks.append(e); continue
239     
240                 # TODO: Should remove redundant entries.
241                 if size > peaks[0].size:
242                     peaks.append(e)
243                     peaks = deque(sorted(peaks, key=lambda x: x.size), maxlen=maxlen)
244     
245             peaks = deque(sorted(peaks, key=lambda x: x.size, reverse=True), maxlen=maxlen)
246     
247             for peak in peaks:
248                 print(peak)
249     
250             return peaks
