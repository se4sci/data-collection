352     // Hamiltonian Initialization with precalculate coeff  (J=0 and U(i,j))
353     many_body_operator_generic<double> init_Hamiltonian( double *eps, int nflavor, double *U){
354         many_body_operator_generic<double> H; double coeff;
355       
356         for(int i = 0; i < nflavor; ++i)
357             for(int j = 0; j < nflavor; ++j){
358       
359         	    //cas diago principale:
360     	        if(i==j)
361     	            H += eps[i] * n("tot",i);
362         	    else{
363     	    	    coeff = double(U[i+j*nflavor])*0.5;
364     		        H += coeff * n("tot",i) * n("tot",j);
365         	    }
366             }
367         return H;
368     }

370     // Hamiltonian Initialization with precalculated coeff U(i,j,k,l) in "tot" notation
371     many_body_operator_generic<double> init_fullHamiltonian( double *eps, int nflavor, double *U){
372         many_body_operator_generic<double> H; double coeff;
373       
374         for(int i = 0; i < nflavor; ++i)
375             for(int j = 0; j < nflavor; ++j){
376           
377                 if(i==j)
378     	            H += eps[i] * n("tot",i) ; //Levels On diagonal Hamiltonian Matrix
379           
380                 for(int k = 0; k < nflavor; ++k)
381                     for(int l = 0; l < nflavor; ++l){ 
382                         //cas diago principale:
383     		            coeff = 0.5 * double( U[i+j*nflavor+k*nflavor*nflavor+l*nflavor*nflavor*nflavor] );
384     		            H += coeff * c_dag("tot",i)*c_dag("tot",j)*c("tot",l)*c("tot",k); //changing l <-> k order index to be in accordance with the Abinit definition
385     	            }
386             }
387         return H;
388     }

390     // Hamiltonian Initialization with precalculated coeff U(i,j,k,l) in "up" and "down" notation
391     many_body_operator_generic<double> init_fullHamiltonianUpDown( double *eps, int nflavor, double *U ){
392         many_body_operator_generic<double> H; double coeff;
393       
394         for(int i = 0; i < nflavor; ++i)
395             for(int j = 0; j < nflavor; ++j){
396           
397                 if(i==j)
398     	            H += eps[i] * ( n("up",i) + n("down",i) ); //Levels On diagonal Hamiltonian Matrix
399           
400                 for(int k = 0; k < nflavor; ++k)
401                     for(int l = 0; l < nflavor; ++l){
402                         coeff = 0.5 * double(U[i+j*nflavor+k*nflavor*nflavor+l*nflavor*nflavor*nflavor]);
403      
404     //            for(int s = 0; s < nspin; ++s)
405     //              for(int s1 = 0; s1 < nspin; ++s1){ 
406     		            H += coeff * c_dag("up",i)   * c_dag("down",j) * c("up",l)   * c("down",k);
407     		            H += coeff * c_dag("down",i) * c_dag("up",j)   * c("down",l) * c("up",k);
408     	        }
409         }
410         return H;
