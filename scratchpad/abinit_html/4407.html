168     class Message(object):
169         """
170         .. attributes:
171     
172             filepath:
173             lineno:
174             colno:
175             text:
176         """
177         def __init__(self, filepath, kind, text, lineno, colno, info):
178             self.filepath = filepath if filepath is not None else "Unknown"
179             self.kind = kind
180             self.text = text
181             self.lineno = int(lineno) if lineno is not None else 0
182             self.colno = int(colno) if colno is not None else 0
183             self.info = info
184     
185         @lazy_property
186         def filename(self):
187             return os.path.basename(self.filepath)
188     
189         @lazy_property
190         def dirname(self):
191             return os.path.basename(os.path.dirname(self.filepath))
192     
193         def __repr__(self):
194             return "<%s at %s +%d>" % (self.kind, self.filepath, self.lineno)
195     
196         def __str__(self):
197             return self.text

201         """
202     ../../../src/98_main/mrgscr.F90:1852:25:
203     
204             omega_new = CMPLX(-one,-one)
205                              1
206     Warning: Conversion from REAL(8) to default-kind COMPLEX(4) at (1) ... [-Wconversion]
207     
208     ../../../src/53_ffts/m_fft.F90:1606:47: Warning: Possible change of ... at (1) [-Wconversion]
209     
210     ../../../src/01_linalg_ext/m_linalg_interfaces.F90:84:14:
211     
212         character*1 :: TRANS
213                   1
214     Warning: Obsolescent feature: Old-style character length at (1)
215     
216     ../../../src/28_numeric_noabirule/interfaces_28_numeric_noabirule.F90:705:8:
217     
218     real*8, intent(inout) :: a(mesh)
219     1
220     Warning: GNU Extension: Nonstandard type declaration REAL*8 at (1)
221     
222     ../../../src/95_drive/eph.F90:320:54: Warning: Possible change of value in ... at (1) [-Wconversion]
223         """
224         def __init__(self, lines):
225             text = "\n".join(lines) #.encode("utf8")
226     
227             if len(lines) > 1:
228                 #print(lines[0].split(":"))
229                 filepath, lineno, colno, _ = lines[0].split(":")
230                 last = lines[-1]
231     
232                 if last.startswith("Warning: GNU Extension:"):
233                     kind = "GnuExtension"
234                 elif last.startswith("Warning: Obsolescent feature:"):
235                     kind = "Obsolescent"
236                 elif last.startswith("Warning: Same actual argument associated"):
237                     kind = "-Waliasing"
238                 elif last.startswith("Warning: Label"):
239                     kind = "Miscellaneous"
240                 else:
241                     kind = last.split()[-1].replace("[", "").replace("]", "")
242                     if kind not in GNU_WARNINGS:
243                         print("Don't know how to classify:", last, "\nSetting kind to `Miscellaneous`")
244                         kind = "Miscellaneous"
245     
246             else:
247                 tokens = lines[0].split()
248                 filepath, lineno, colno, _ = tokens[0].split(":")
249                 kind = tokens[-1].replace("[", "").replace("]", "")
250     
251             info = GNU_WARNINGS.get(kind)
252             if info is None:
253                 raise RuntimeError("Cannot find kind: %s in GNU_WARNINGS" % kind)
254     
255             super(self.__class__, self).__init__(filepath, kind, text, lineno, colno, info)

258     class WarningsParser(object):
259     
260         def __init__(self):
261             self.warns = []
262             self.filepaths = "Unknown"
263     
264         @classmethod
265         def from_compiler(cls, compiler):
266             for c in cls.__subclasses__():
267                 if c.compiler == compiler: return c()
268             raise ValueError("No Parser associated to compiler `%s`" % compiler)
269     
270         @abc.abstractmethod
271         def parse_lines(self, lines, filepath="Unknown"):
272             """Parse the file. Returns self."""
273     
274         def parse_file(self, filename):
275             self.filepath = os.path.abspath(filename)
276             with io.open(filename, "rt", encoding="utf8") as fh:
277                 return self.parse_lines(fh.readlines(), filepath=self.filepath)
278     
279         @property
280         def num_warns(self):
281             return len(self.warns)
282     
283         def __repr__(self):
284             return "<%s %s: num_warns: %s>" % (self.__class__.__name__, self.filepath, self.num_warns)
285     
286         def __str__(self):
287             return self.to_string()
288     
289         def to_string(self, verbose=0):
290             lines = ["<%s %s: num_warns: %s>" % (self.__class__.__name__, self.filepath, self.num_warns), ""]
291             lines.append(self.summarize())
292             return "\n".join(lines)
293     
294         def summarize(self):
295             """Return string with data in tabular form."""
296             counter = Counter(warn.kind for warn in self.warns)
297             table = [["Kind", "Count"]]
298             for kind, num in counter.most_common():
299                 table.append([str(kind), str(num)])
300     
301             try:
302                 from io import StringIO
303             except ImportError:
304                 from StringIO import StringIO
305             out = StringIO()
306             pprint_table(table, out=out)
307             return out.getvalue()

323         def fix(self):
324             from tools import Editor, user_wants_to_exit
325             import json
326             fixed = set()
327             if os.path.exists("fixed.json"):
328                 with open("fixed.json", "rt") as fh:
329                     fixed = set(json.load(fh)["fixed"])
330     
331             editor = Editor()
332             for i, warn in enumerate(self.warns):
333                 if repr(warn) in fixed:
334                     print("%s already fixed" % repr(warn))
335                     continue
336     
337                 print("Fixing %d/%d" % (i, len(self.warns)))
338                 print(warn)
339                 editor.edit_file(warn.filepath, lineno=warn.lineno)
340                 fixed.add(repr(warn))
341                 if user_wants_to_exit(): break
342     
343             with open("fixed.json", "wt") as fh:
344                 json.dump(list(fixed), fh)

374     class IfortRemark(Message):
375     
376         # Intel uses numbers to identify remarks and warnings.
377         # icc: command line remark #10010: option
378         RE_IFORT_REMARK = re.compile(r".+command line remark\s+#(?P<kind>\d+):\s*(?P<info>\w*)")
379     
380         def __init__(self, line):
381             m = self.RE_IFORT_REMARK.match(line)
382             if not m:
383                 raise ValueError("String does not match regex: %s" % line)
384     
385             super(self.__class__, self).__init__(filepath=None, kind=m.group("kind"), text=line,
386                 lineno=None, colno=None, info=m.group("info"))

389     class IfortWarning(Message):
390     
391         # ../../../src/17_libtetra_ext/m_tetrahedron.F90(1119): warning #6843: A dummy argument ... explicit value.   [TWEIGHT]
392         RE_IFORT_WARN = re.compile(r"(?P<filepath>.+)\((?P<lineno>\d+)\):\s+warning\s+#(?P<kind>\d+):\s*(?P<info>\w*)")
393     
394         def __init__(self, lines):
395             m = self.RE_IFORT_WARN.match(lines[0])
396             if not m:
397                 raise ValueError("String does not match regex: %s" % lines[0])
398     
399             super(self.__class__, self).__init__(
400                 filepath=m.group("filepath"), kind=m.group("kind"), text="\n".join(lines),
401                 lineno=m.group("lineno"), colno=None, info=m.group("info"))

404     class IfortParser(WarningsParser):
405         compiler = "ifort"
406     
407         def parse_lines(self, lines, filepath="Unknown"):
408             """
409             icc: command line remark #10010: option '-vec-report0' is deprecated and will be removed ...
410             """
411             buff = []
412             for line in lines:
413                 line = line.strip()
414     
415                 if "command line remark #" in line:
416                     self.warns.append(IfortRemark(line))
417                     continue
418     
419                 if "warning #" in line:
420                     if buff:
421                         self.warns.append(IfortWarning(buff))
422                     buff = [line]
423                 else:
424                     buff.append(line)
425     
426             if buff:
427                 self.warns.append(IfortWarning(buff))
428                 buff = []
429     
430             assert not buff
431             return self

434     def main():
435         import argparse
436         parser = argparse.ArgumentParser(formatter_class=argparse.RawDescriptionHelpFormatter)
437     
438         parser.add_argument('-v', '--verbose', default=0, action='count', # -vv --> verbose=2
439             help='verbose, can be supplied multiple times to increase verbosity.')
440     
441         parser.add_argument("filepath", type=str, help="File to parse.")
442         parser.add_argument("-c", "--compiler", type=str, default="gfortran",
443             help="Fortran Compiler. Allowed values in ['gfortran', 'ifort'], Default: gfortran")
444         parser.add_argument("-f", "--fix", action='store_true', default=False, help="Fix warnings in $EDITOR")
445     
446         options = parser.parse_args()
447     
448         #compiler = "gfortran" if "intel" not in options.path else "ifort"
449         parser = WarningsParser.from_compiler(options.compiler).parse_file(options.filepath)
450         print(parser.to_string(verbose=options.verbose))
451     
452         # Define set of ignored warnings.
453         if parser.compiler == "gfortran":
454             ignored_kinds = set([
455                 "-Wimplicit-interface",
456                 "-Wconversion",
457             ])
458     
459         elif parser.compiler == "ifort":
460             ignored_kinds = set([
461             ])
462     
463         else:
464             raise ValueError("Missing ignored_kinds for compiler %s" % options.compiler)
465     
466         retcode = 0
467         for warn in parser.warns:
468             if warn.kind in ignored_kinds: continue
469             if options.verbose: print(warn)
470             retcode += 1
471     
472         if options.fix:
473             parser.fix()
474     
475         return retcode
