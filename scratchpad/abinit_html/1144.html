95     def hideCycleIter(model, iter, window):
96         (strId,) = model.get(iter, 0)
97         if (strId is None):
98           return False
99         if (window.get_data("hideDefault").get_active()):
100           dtsets = window.get_data("dtsets")
101           hide = True
102           for i in range(dtsets.get_ndtset()):
103             (color,) = model.get(iter, 2 * i + 2)
104             hide = hide and (color != "blue")
105         else:
106           hide = False
107         return (not(hide) and re.match(window.get_data("filter"), strId) is not None)

109     def applyFilter(widget, window, entry):
110         patStr = entry.get_text()
111         if (patStr[-1] != "$"):
112             patStr += "$"
113         pat = re.compile(patStr)
114         if (pat is not None):
115             window.set_data("filter", pat)
116             window.get_data("treeview").get_model().get_model().refilter()

121     def withArg(argv, window):
122         fileSel = window.get_data("fileSel")
123         fileSel.set_filename(os.path.abspath(sys.argv[1]))
124         while (gtk.events_pending()):
125             gtk.main_iteration()
126         if (fileSel.get_filename() is not None):
127             loadData(fileSel, gtk.RESPONSE_ACCEPT, window)
128         else:
129             raise ValueError("No file '%s'." % os.path.abspath(sys.argv[1]))
130         return False

132     def XMLGetText(var):
133       if (var is None):
134         return ""
135     
136       if (var.text is not None):
137         text = var.text.strip().replace("\n", " ") + " "
138       else:
139         text = ""
140       for child in var.getchildren():
141         if (child.tag == "p"):
142             text += XMLGetText(child) + "\n"
143         elif (child.tag == "ul"):
144             text += "\n" + XMLGetText(child)
145         elif (child.tag == "li"):
146             text += "  * " + XMLGetText(child) + "\n"
147         elif (child.tag == "br"):
148             text += "\n"
149         else:
150             text += XMLGetText(child)
151       if (var.tail is not None):
152         text += var.tail.strip().replace("\n", " ") + " "
153       return re.sub("  *", " ", text)

155     def selectAtt(selection, window):
156         (model, iter) = selection.get_selected()
157         if (iter is None):
158             return
159         iSets = []
160         (att,) = model.get(iter, 0)
161         window.get_data("lblKeyword").set_markup("<b>" + att + "</b>")
162         window.get_data("lblMnemonic").set_markup("")
163         window.get_data("lblCategory").set_markup("")
164         window.get_data("lblType").set_markup("")
165         window.get_data("lblDefault").set_markup("")
166         window.get_data("description").set_text("")
167         varlist = window.get_data("varlist")
168         for var in varlist:
169           if (var.find("keyword").text == att):
170             window.get_data("lblMnemonic").set_markup(XMLGetText(var.find("mnemonics")))
171             window.get_data("lblCategory").set_markup(XMLGetText(var.find("categories")))
172             window.get_data("lblType").set_markup(XMLGetText(var.find("type")))
173             window.get_data("lblDefault").set_markup(XMLGetText(var.find("default")))
174             window.get_data("description").set_text(XMLGetText(var.find("description")))
175             break
