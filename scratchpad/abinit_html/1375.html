126             changed = []

180     class Suite(object):
181         """Information on one test suite"""
182         def __init__(self, suite_path):
183     
184             suite_path = os.path.abspath(suite_path)
185     
186             self.suite_path = os.path.abspath(suite_path)
187             self.name = os.path.basename(suite_path)
188     
189             module_name = os.path.join(suite_path, "__init__.py")
190             module = imp.load_source(module_name, os.path.join(suite_path, "__init__.py") )
191     
192             self.keywords = set(module.keywords)
193             self.need_cpp_vars = set(module.need_cpp_vars)
194     
195             # Divide tests into (active|disabled).
196             self.inp_paths = [p for p in module.inp_files if not p.startswith("-")]
197             self.disabled_inp_paths = [p[1:] for p in module.inp_files if p.startswith("-")]
198     
199             # Use absolute paths
200             self.inp_paths = [os.path.join(suite_path, "Input", p) for p in self.inp_paths]
201             self.disabled_inp_paths = [os.path.join(suite_path, "Input", p) for p in self.disabled_inp_paths]
202     
203             # True if the suite contains tests that should be executed with
204             # different numbers of MPI processes
205             self.is_multi_parallel = False
206             if hasattr(module, "is_multi_parallel"):
207                 self.is_multi_parallel = module.is_multi_parallel
208     
209             self.subsuites = {}
210             if hasattr(module, "subsuites"):
211                 subsuite_names = module.subsuites
212                 for k in subsuite_names: self.subsuites[k] = []
213     
214                 for name in subsuite_names:
215                     pattern = re.compile("-?t" + name + "_\d+\.in")
216                     for inp in module.inp_files:
217                         if pattern.match(inp):
218                             #print(inp, "--> subsuite: ", name)
219                             inp_path = os.path.join(suite_path, "Input", inp)
220                             self.subsuites[name].append(inp_path)
221     
222                 nfound = sum([len(paths) for paths in self.subsuites.values()])
223                 if nfound != len(module.inp_files):
224                     err_msg = ("At least one input_file does not belong to a subsuite, nfound = %s, __init__.nfiles = %s\n"
225                                % (nfound, len(module.inp_files)))
226     
227                     for inp in module.inp_files:
228                         for paths in self.subsuites.values():
229                             fnames = [os.path.basename(p) for p in paths]
230                             if inp in fnames: break
231                         else:
232                             err_msg += "%s not found\n" % inp
233     
234                     raise ValueError(err_msg)
235     
236                 # Remove disabled tests (if any).
237                 for sub_name, paths in self.subsuites.items():
238                     inp_paths = [p for p in paths if not p.endswith("-")]
239                     self.subsuites[sub_name] = inp_paths

299         def init_result_table(self):
300             """
301             Initialize a nested dictionary indexed by [suite_name][test.id].
302             The dictionary contains ALL the tests present in the database
303             and is used in testbot.py to store the results of the tests executed
304             by the buildbot slave. Values are initialized with {}.
305             """
306             res_table = {}
307             for suite_name in self.suite_names:
308                 res_table[suite_name] = {}
309     
310             for suite_name, suite in self.items():
311                 for test in suite:
312                     # test.id should be unique inside a suite. Check it once again.
313                     if test.id in res_table[suite_name]:
314                         other = res_table[suite_name][test.id]
315                         print("test\n:", test)
316                         print("other:\n", other)
317                         raise ValueError("Replicated test.id %s in suite %s" % (test.id, suite_name))
318     
319                     res_table[suite_name][test.id] = {}
320     
321             return res_table

609         def get_database(self, regenerate=False, with_pickle=False):
610             """
611             Return an instance of TestsDatabase initialized from an external pickle file.
612     
613             Args:
614                 regenerate:
615                     True to force the regeneration of the database
616                     and the writing of a new pickle file.
617                 with_pickle:
618                     Save the generated database in pickle format.
619             """
620             if regenerate or not os.path.exists(database_path):
621                 cprint("Regenerating database...", "yellow")
622                 database = self.build_database()
623     
624                 # Save the database in the cpickle file.
625                 # Use file locking mechanism to prevent IO from other processes.
626                 if with_pickle:
627                     print("Saving database to %s" % database_path)
628                     lock = FileLock(database_path)
629     
630                     lock.acquire()
631     
632                     with open(database_path, "wb") as fh:
633                         pickle.dump(database, fh, protocol=-1)
634     
635                     lock.release()
636     
637             else:
638                 cprint("Loading database from: %s" % database_path, "yellow")
639     
640                 # Read the database from the cpickle file.
641                 # Use file locking mechanism to prevent IO from other processes.
642                 lock = FileLock(database_path)
643     
644                 lock.acquire()
645     
646                 with open(database_path, "rb") as fh:

653         def _suite_args_parser(self, args=None):
654             """
655             Parse script arguments. Return a mapping suite_name --> [slice objects]
656             Three forms are possible
657             0)                               Run all tests.
658             1) v2[34:35] v1[12:] v5 v6[:45]  Select slices in the suites
659             2) v4- v5-                       Exclude suites
660             """
661             # Mapping (suite_name, subsuite_name) --> slice_obj
662             def all_tests():
663                 tuples = [(name, None) for name in self.suite_names]
664                 return dict.fromkeys(tuples, [slice(None),])
665     
666             if args is None or not args:
667                 # Run all tests.
668                 return all_tests()
669     
670             args = [s.replace(" ", "") for s in args]
671     
672             exclude_mode = any(arg.endswith("-") for arg in args)
673             if exclude_mode:
674                 d = all_tests()
675                 for arg in args:
676                     if arg.endswith("-"):
677                         arg = arg[:-1]
678                         if arg in self.suite_names:
679                             d.pop((arg, None))  # v4- --> Remove v4
680                         else:
681                             # TODO
682                             raise NotImplementedError("exclude_mode does not support subsuites")
683                             suite = self.suite_of_subsuite(arg)
684                             d.pop((suite.name, arg))  # gw1- --> skip tutorial/gw1
685     
686             else:
687                 import re
688                 re_slice = re.compile("^\[(\d*):(\d*)\]$")
689                 re_single = re.compile("^\[(\d*)\]$")
690     
691                 d = {}
692                 for arg in args:
693                     start_stop = slice(None)
694                     idx = arg.find("[")
695                     if idx != -1:
696                         arg, string = arg[:idx], arg[idx:]
697                         match = re_slice.search(string)
698                         if match:
699                             start, stop = match.group(1), match.group(2)
700                             #if not start: start = 1
701                             if not start: start = 0
702                             if not stop: stop = None
703                             if stop is None:
704                                 start_stop = slice(int(start), stop)
705                             else:
706                                 start_stop = slice(int(start), int(stop))
707                         else:
708                             match = re_single.search(string)
709                             if match:
710                                 start = int(match.group(1))
711                                 start_stop = slice(start, start+1)
712                             else:
713                                 raise ValueError("Wrong or unknown argument: %s" % arg)
714     
715                     if arg in self.suite_names:
716                         tp = (arg, None)
717                     elif arg in self.all_subsuite_names:
718                         suite = self.suite_of_subsuite(arg)
719                         tp = (suite.name, arg)
720                     else:
721                         raise ValueError("Wrong (suite_name|subsuite_name) : %s" % arg)
722     
723                     if tp not in d:
724                         d[tp] = [start_stop]
725                     else:
726                         d[tp].append(start_stop)
727     
728             return d

731                          authors=None, ivars=None, with_pickle=True, flat_list=False):
732             """
733             Main entry point for client code. Return an instance of AbinitTestSuite

735             Args:
736                 with_pickle: Save the generated database in pickle format.
737                 flat_list: True if a flat list of tests should be returned instead

763                     if suite_name in suites_without_slicing: slice_obj = None
764     
765                     tests = tests + database.get_test_suite(suite_name, subsuite_name=subsuite_name, slice_obj=slice_obj)
766

773                     exclude_keys = [k[:-1] for k in keys if k.endswith("-")]
774                     print("Extracting tests with keywords = %s, without keywords %s" % (with_keys, exclude_keys))
775     
776                     if "VASP" in with_keys:
777                         from pymods.tools import ascii_wasp

779                         print("Maybe you meant ABINIT!")
780                         sys.exit(1)

784                         print(ascii_scream())
785                         print("I really can't imagine how PGI could pass the ABINIT test suite!")
786                         sys.exit(1)

790                     exclude_authors = [a[:-1] for a in authors if a.endswith("-")]
791                     print("Extracting tests with authors = %s, without authors %s" % (with_authors, exclude_authors))
792     
793                 tests = tests.select_tests(with_keys= with_keys,
794                                            exclude_keys=exclude_keys,
795                                            with_authors=with_authors,
796                                            exclude_authors=exclude_authors,
797                                            ivars=ivars
798                                            )

800                 return tests

802                 # Build flat list of tests.

807                     #if hasattr(t, "tests"):
