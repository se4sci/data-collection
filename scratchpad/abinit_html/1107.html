29     Ab7Error ab7_kpoints_get_mp_k_grid(Ab7Symmetry *sym, int *nkpt, double **kpt,
30     				   double **wkpt, const int ngkpt[3],
31     				   const int nshiftk, const double *shiftk)
32     {
33       Ab7Error ab7_errno;
34       double kptrlatt[9], kptrlen;
35       double shiftk_[24];
36       int nshiftk_, i;
37     
38       nshiftk_ = nshiftk;
39       for (i = 0; i < nshiftk * 3; i++)
40         shiftk_[i] = shiftk[i];
41       KPT_CALL(binding_mp_k_1,BINDING_MP_K_1, sym, nkpt, ngkpt, kptrlatt, &kptrlen, &nshiftk_, shiftk_, &ab7_errno);
42       if (ab7_errno != AB7_NO_ERROR && ab7_errno != AB7_ERROR_SYM_BRAVAIS_XRED) return ab7_errno;
43       *kpt = g_malloc(sizeof(double) * (*nkpt) * 3);
44       *wkpt = g_malloc(sizeof(double) * (*nkpt));
45       KPT_CALL(binding_mp_k_2,BINDING_MP_K_2, sym, nkpt, *kpt, *wkpt, kptrlatt, &kptrlen, &nshiftk_, shiftk_, &ab7_errno);
46     
47       return ab7_errno;
48     }
49     Ab7Error ab7_kpoints_get_auto_k_grid(Ab7Symmetry *sym, int *nkpt, double **kpt,
50     				     double **wkpt, const double kptrlen)
51     {
52       Ab7Error ab7_errno;
53       double kptrlatt[9], kptrlen_;
54       double shiftk[24];
55       int nshiftk;
56     
57       kptrlen_ = kptrlen;
58       KPT_CALL(binding_auto_k_1,BINDING_AUTO_K_1, sym, nkpt, kptrlatt, &kptrlen_, &nshiftk, shiftk, &ab7_errno);
59       if (ab7_errno != AB7_NO_ERROR && ab7_errno != AB7_ERROR_SYM_BRAVAIS_XRED) return ab7_errno;
60       *kpt = g_malloc(sizeof(double) * (*nkpt) * 3);
61       *wkpt = g_malloc(sizeof(double) * (*nkpt));
62       KPT_CALL(binding_auto_k_2,BINDING_AUTO_K_2, sym, nkpt, *kpt, *wkpt, kptrlatt, &kptrlen_, &nshiftk, shiftk, &ab7_errno);
63     
64       return ab7_errno;
65     }
