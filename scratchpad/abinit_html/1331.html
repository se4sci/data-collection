26     class Canvas(FigureCanvas):
27     
28         def __init__(self, parent=None, width=6, height=4, dpi=100,x=0,y=0,pxlbl="",pylbl="",point = False,adjust=False):
29     
30             self.fig = Figure(figsize=(width, height), dpi=dpi,)
31     
32             FigureCanvas.__init__(self,self.fig)
33     
34             self.setPlot(x,y,pxlbl,pylbl,adjust=adjust)
35     
36     
37         def setPlot(self,x,y,xlbl,ylbl,point =False, adjust=False):
38             try :
39                 self.formater = ScalarFormatter(useOffset=True, useMathText=False, useLocale=None)
40                 self.formater.set_useOffset(0)
41             except:
42                 self.formater = ScalarFormatter(useOffset=True, useMathText=False)
43     
44             self.axes = self.fig.add_subplot(111)
45             self.axes.clear()
46     
47             if point :
48                 self.axes.plot(x,y,'.',markersize=15)
49             else :
50                 self.axes.plot(x,y)
51     
52             try:
53                 if (max(y)-min(y)) <10**-10:
54                     self.axes.set_ylim(min(y)-1,max(y)+1)
55             except:
56                 pass;
57             
58             self.axes.set_xlabel(xlbl)
59             self.axes.set_ylabel(ylbl)
60             self.axes.figure.set_facecolor('white')
61             self.axes.grid('on')
62     
63             self.axes.yaxis.set_major_formatter(self.formater)
64             self.axes.xaxis.set_major_formatter(self.formater)
65     
66             if adjust:
67                 self.adjust_x_lim(x,y)
68     
69             self.draw()
70     
71         def addLegend(self,plegend,markerscale=1):
72             self.axes.legend(plegend,loc=1,markerscale=markerscale)
73             self.draw()
74     
75         def addPlot(self,x,y,bar = False, point = False,marker='.',marker_size=25 ):
76             if bar == True:
77                 self.axes.bar(x, y, width=0.01)
78             elif point == True :
79                 self.axes.plot(x, y,marker,markersize=marker_size)
80             else:
81                 self.axes.plot(x,y)
82     
83             self.draw()
84     
85         def adjust_x_lim(self, x, y):
86             test = 0
87             xmin = 0
88             xmax = 0
89             for i in range(len(x)):
90                 if y[i] >= 0.0002:
91                     xmin = x[i]
92                     break
93             for i in range(len(x)):
94                 if y[i] <= 0.002 :
95                     test += 1
96                 if test > 60 :
97                     xmax = x[i-50]
98                     break
99             self.axes.set_xlim(xmin,xmax)
100             self.draw()
