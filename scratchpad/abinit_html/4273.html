237     def compute_dynmat(DDB):
238     # Retrive the amu for each atom
239       amu = zeros(DDB.natom)
240       for ii in N.arange(DDB.natom):
241         jj = DDB.typat[ii]
242         amu[ii] = DDB.amu[jj-1]
243     # Calcul of gprimd from rprimd
244       rprimd = DDB.rprim*DDB.acell
245       gprimd = N.linalg.inv(N.matrix(rprimd))
246     # Transform from 2nd-order matrix (non-cartesian coordinates, 
247     # masses not included, asr not included ) from DDB to
248     # dynamical matrix, in cartesian coordinates, asr not imposed.
249       IFC_cart = zeros((3,DDB.natom,3,DDB.natom),dtype=complex)
250       for ii in N.arange(DDB.natom):
251         for jj in N.arange(DDB.natom):
252           for dir1 in N.arange(3):
253             for dir2 in N.arange(3):
254               for dir3 in N.arange(3):
255                 for dir4 in N.arange(3):
256                   IFC_cart[dir1,ii,dir2,jj] += gprimd[dir1,dir3]*DDB.IFC[dir3,ii,dir4,jj] \
257                 *gprimd[dir2,dir4]
258     # Reduce the 4 dimensional IFC_cart matrice to 2 dimensional Dynamical matrice.
259       ipert1 = 0
260       Dyn_mat = zeros((3*DDB.natom,3*DDB.natom),dtype=complex)
261       while ipert1 < 3*DDB.natom:
262         for ii in N.arange(DDB.natom):
263           for dir1 in N.arange(3):
264             ipert2 = 0
265             while ipert2 < 3*DDB.natom:
266               for jj in N.arange(DDB.natom):
267                 for dir2 in N.arange(3):
268                   Dyn_mat[ipert1,ipert2] = IFC_cart[dir1,ii,dir2,jj]*(5.4857990965007152E-4)/ \
269                        N.sqrt(amu[ii]*amu[jj])
270                   ipert2 += 1
271             ipert1 += 1
272     # Hermitianize the dynamical matrix
273       dynmat = N.matrix(Dyn_mat)
274       dynmat = 0.5*(dynmat + dynmat.transpose().conjugate())
275     
276     # Solve the eigenvalue problem with linear algebra (Diagonalize the matrix)
277       [eigval,eigvect]=N.linalg.eigh(Dyn_mat)
278     
279     # Orthonormality relation 
280       ipert = 0
281       for ii in N.arange(DDB.natom):
282         for dir1 in N.arange(3):
283          eigvect[ipert] = (eigvect[ipert])*N.sqrt(5.4857990965007152E-4/amu[ii])
284          ipert += 1
285       kk = 0
286       for jj in eigval:
287         if jj < 0.0:
288           print "WARNING: An eigenvalue is negative with value: ",jj," ... but proceed with value 0.0"
289           eigval[kk] = 0.0
290           kk += 1
291         else:
292           kk += 1
293       omega = N.sqrt(eigval) #*5.4857990965007152E-4)
294     #  print "omega",omega
295     
296     # The acoustic phonon at Gamma should NOT contribute because they should be zero.
297     # Moreover with the translational invariance the ZPM will be 0 anyway for these
298     # modes but the FAN and DDW will have a non physical value. We should therefore 
299     # neglect these values.
300     #  if N.allclose(DDB.iqpt,[0.0,0.0,0.0]) == True:
301     #    omega[0] = 0.0
302     #    omega[1] = 0.0
303     #    omega[2] = 0.0
304     
305       return omega,eigvect,gprimd

310     def get_reduced_displ(natom,eigvect,omega,gprimd):
311       displ_FAN =  zeros((3,3),dtype=complex)
312       displ_DDW =  zeros((3,3),dtype=complex)
313       displ_red_FAN2 = zeros((3*natom,natom,natom,3,3),dtype=complex)
314       displ_red_DDW2 = zeros((3*natom,natom,natom,3,3),dtype=complex)
315       for imode in N.arange(3*natom): #Loop on perturbation (6 for 2 atoms)
316         if omega[imode].real > tol6:
317           for iatom1 in N.arange(natom):
318             for iatom2 in N.arange(natom):
319               for idir1 in N.arange(0,3):
320                 for idir2 in N.arange(0,3):
321                   displ_FAN[idir1,idir2] = eigvect[3*iatom2+idir2,imode].conj()\
322                      *eigvect[3*iatom1+idir1,imode]/(2.0*omega[imode].real)
323                   displ_DDW[idir1,idir2] = (eigvect[3*iatom2+idir2,imode].conj()\
324                      *eigvect[3*iatom2+idir1,imode]+eigvect[3*iatom1+idir2,imode].conj()\
325                      *eigvect[3*iatom1+idir1,imode])/(4.0*omega[imode].real)
326                   # Now switch to reduced coordinates in 2 steps (more efficient)
327               tmp_displ_FAN = zeros((3,3),dtype=complex)
328               tmp_displ_DDW = zeros((3,3),dtype=complex)
329               for idir1 in N.arange(3):
330                 for idir2 in N.arange(3):
331                   tmp_displ_FAN[:,idir1] = tmp_displ_FAN[:,idir1]+displ_FAN[:,idir2]*gprimd[idir2,idir1]
332                   tmp_displ_DDW[:,idir1] = tmp_displ_DDW[:,idir1]+displ_DDW[:,idir2]*gprimd[idir2,idir1]
333               displ_red_FAN = zeros((3,3),dtype=complex)
334               displ_red_DDW = zeros((3,3),dtype=complex)
335               for idir1 in N.arange(3):
336                 for idir2 in N.arange(3):
337                   displ_red_FAN[idir1,:] = displ_red_FAN[idir1,:] + tmp_displ_FAN[idir2,:]*gprimd[idir2,idir1]
338                   displ_red_DDW[idir1,:] = displ_red_DDW[idir1,:] + tmp_displ_DDW[idir2,:]*gprimd[idir2,idir1]
339     
340               displ_red_FAN2[imode,iatom1,iatom2,:,:] = displ_red_FAN[:,:]
341               displ_red_DDW2[imode,iatom1,iatom2,:,:] = displ_red_DDW[:,:]
342       return displ_red_FAN2,displ_red_DDW2

392     def get_bose(natom,omega,temp_info):
393       bose = N.array(zeros((3*natom,len(temp_info))))
394       for imode in N.arange(3*natom): #Loop on perturbation (6 for 2 atoms)
395         if omega[imode].real > tol6:
396           tt = 0
397           for T in temp_info:
398             if T < tol6:
399               bose[imode,tt] = 0.0
400             else:
401               bose[imode,tt] = 1.0/(N.exp(omega[imode].real/(kb_HaK*T))-1)
402             tt += 1
403       #print bose[:,0]
404       return bose

409     def gaussian(ifreq,omegatmp,gaussian_smearing):
410       x = (ifreq - omegatmp)/gaussian_smearing
411       gaussian = 1.0/(N.sqrt(N.pi)) * (1.0/gaussian_smearing) * N.exp(-(x**2))  
412       return gaussian

457     # Get reduced displacement (scaled with frequency)
458       displ_red_FAN2,displ_red_DDW2 = get_reduced_displ(EIGR2D.natom,eigvect,omega,gprimd)
459     # Einstein sum make the vector matrix multiplication ont the correct indices
460     # fan_corrQ and ddw_corrQ contains the ZPR on stern space.
461       fan_corrQ = N.einsum('ijklmn,olnkm->oij',EIGR2D.EIG2D,displ_red_FAN2)
462       ddw_corrQ = N.einsum('ijklmn,olnkm->oij',ddw_save,displ_red_DDW2)
463     
464       fan_corr = N.sum(fan_corrQ,axis=0)
465       ddw_corr = N.sum(ddw_corrQ,axis=0)
466     
467       print "Now compute active space ..."
468     
469     # Now computa active space
470     
471     #  FAN = N.einsum('ijklm,ijnom->ijklnom',GKK,N.conjugate(GKK))
472     #  fan_addQ = N.einsum('ijklmno,plnkm->ijop',FAN,displ_red_FAN2) 
473       
474     
475       temp = N.einsum('ijklm,nlokp->ijmnpo',GKK,displ_red_FAN2)
476       fan_addQ = N.einsum('ijklmn,ijmnk->ijkl',temp,N.conjugate(GKK)) 
477       temp = N.einsum('ijklm,nlokp->ijmnpo',ddw_save2,displ_red_DDW2)
478       ddw_addQ = N.einsum('ijklmn,ijmnk->ijkl',temp,N.conjugate(ddw_save2))
479     
480       if type == 1:
481         ddw_tmp = N.sum(ddw_addQ,axis=3)
482         occtmp = EIGR2D.occ[0,0,:]/2 # jband  
483         delta_E = N.einsum('ij,k->ijk',eig0[0,:,:].real,N.ones(EIGR2D.nband)) - \
484                   N.einsum('ij,k->ikj',eigq.EIG[0,:,:].real,N.ones(EIGR2D.nband)) - \
485                   N.einsum('ij,k->ijk',N.ones((EIGR2D.nkpt,EIGR2D.nband)),(2*occtmp-1))*smearing*1j # ikpt,iband,jband
486         delta_E_ddw = N.einsum('ij,k->ijk',eig0[0,:,:].real,N.ones(EIGR2D.nband)) - \
487                   N.einsum('ij,k->ikj',eig0[0,:,:].real,N.ones(EIGR2D.nband)) - \
488                   N.einsum('ij,k->ijk',N.ones((EIGR2D.nkpt,EIGR2D.nband)),(2*occtmp-1))*smearing*1j
489         ddw_add = N.einsum('ijk,ijk->ij',ddw_tmp,1.0/delta_E_ddw)
490         omegatmp = omega[:].real # imode
491         num1 = 1.0-occtmp # jband
492         deno1 = N.einsum('ijk,l->ijkl',delta_E,N.ones(3*EIGR2D.natom)) \
493               - N.einsum('ijk,l->ijkl',N.ones((EIGR2D.nkpt,EIGR2D.nband,EIGR2D.nband)),omegatmp) #ikpt,iband,jband,imode
494         div1 = N.einsum('i,jkil->lijk',num1,1.0/deno1) # (jband)/(ikpt,iband,jband,imode) ==> imode,jband,ikpt,iband
495         deno2 = N.einsum('ijk,l->ijkl',delta_E,N.ones(3*EIGR2D.natom)) \
496               + N.einsum('ijk,l->ijkl',N.ones((EIGR2D.nkpt,EIGR2D.nband,EIGR2D.nband)),omegatmp) #ikpt,iband,jband,imode
497         div2 = N.einsum('i,jkil->lijk',occtmp,1.0/deno2) # (jband)/(ikpt,iband,jband,imode) ==> imode,jband,ikpt,iband
498         fan_add = N.einsum('ijkl,lkij->ij',fan_addQ,div1+div2) # ikpt,iband,jband,imod
499     
500         print "Now compute generalized g2F Eliashberg electron-phonon spectral function ..." 
501      
502         fan_tmp = N.sum(fan_addQ,axis=2) # (ikpt, iband, imode)
503         ddw_tmp = N.sum(ddw_addQ,axis=2) # (ikpt, iband, imode)
504         g_kk = fan_tmp - ddw_tmp
505         
506         # Generalized Eliashberg function with Gaussian broadening
507         a2F =  zeros((len(energy),EIGR2D.nkpt,EIGR2D.nband),dtype=complex)
508         PDOS = zeros((len(energy)),dtype=complex)
509         ii = 0 
510         for ifreq in energy:
511           for imode in N.arange(3*EIGR2D.natom):
512             a2F[ii,:,:] += g_kk[:,:,imode]*gaussian(ifreq,omegatmp[imode],gaussian_smearing)
513             PDOS[ii] += gaussian(ifreq,omegatmp[imode],gaussian_smearing)
514           ii += 1
515     #    a2F_av = make_average(EIGR2D.nkpt,EIGR2D.nband,degen,a2F)
516     
517     # Correction from active space 
518       fan_corr += fan_add
519       ddw_corr += ddw_add
520       eigen_corr = (fan_corr[:,:] - ddw_corr[:,:])*wtq
521     # Index 0 contains total ZPR
522       total_corr[0,:,:] = eigen_corr[:,:]
523     # Index 1 contains the FAN correction
524       total_corr[1,:,:] = fan_corr[:,:]*wtq
525     # Index 2 contains the DDW correction
526       total_corr[2,:,:] = ddw_corr[:,:]*wtq
527     # Index 3-energy contains the Eliashberg functions
528       total_corr[3:len(energy)+3,:,:] = a2F*wtq 
529     
530       total_corr = make_average(EIGR2D.nkpt,EIGR2D.nband,degen,total_corr)
531     
532     # Index energy+4 contains the PDOS
533       total_corr[len(energy)+4:len(energy)*2+4,0,0] = PDOS*wtq
534     
535       return total_corr

540     def dynamic_zpm_temp(arguments,ddw_save,ddw_save2,type,temp_info,smearing,eig0,degen,energy,gaussian_smearing):
541       nbqpt,wtq,eigq_files,DDB_files,EIGR2D_files,GKK_files = arguments
542       GKKterm = system(directory='.',filename=GKK_files)
543       GKK = GKKterm.GKK
544       DDB = system(directory='.',filename=DDB_files)
545       EIGR2D = system(directory='.',filename=EIGR2D_files)
546       total_corr = zeros((5+2*len(energy),len(temp_info),EIGR2D.nkpt,EIGR2D.nband),dtype=complex)
547       eigq = system(directory='.',filename=eigq_files)
548     
549     # If the calculation is on a Homogenous q-point mesh
550     # retreve the weight of the q-point
551       if (wtq == 0):
552         wtq = EIGR2D.wtq
553         wtq = wtq[0]
554     
555     # Current Q-point calculated
556       print "Q-point: ",nbqpt," with wtq =",wtq," and reduced coord.",EIGR2D.iqpt
557       current = multiprocessing.current_process()
558       file_name = str('PYLOG_')+str(current.pid)
559       if os.path.isfile(file_name) :
560         with open(file_name,'a') as F:
561           F.write("Q-point: "+str(nbqpt)+" with wtq ="+str(wtq)+" and reduced coord."+str(EIGR2D.iqpt)+"\n")
562       else:
563         with open(file_name,'w') as F:
564           F.write("Q-point: "+str(nbqpt)+" with wtq ="+str(wtq)+" and reduced coord."+str(EIGR2D.iqpt)+"\n")
565     
566     
567     # Find phonon freq and eigendisplacement from _DDB
568       omega,eigvect,gprimd=compute_dynmat(DDB)
569     
570     # Compute the displacement = eigenvectors of the DDB. 
571     # Due to metric problem in reduce coordinate we have to work in cartesian
572     # but then go back to reduce because our EIGR2D matrix elements are in reduced coord.
573       fan_corr =  zeros((len(temp_info),EIGR2D.nkpt,EIGR2D.nband),dtype=complex)
574       ddw_corr = zeros((len(temp_info),EIGR2D.nkpt,EIGR2D.nband),dtype=complex)
575       fan_add = N.array(zeros((len(temp_info),EIGR2D.nkpt,EIGR2D.nband),dtype=complex))
576       ddw_add = N.array(zeros((len(temp_info),EIGR2D.nkpt,EIGR2D.nband),dtype=complex))
577     
578       bose = get_bose(EIGR2D.natom,omega,temp_info)
579     
580     # Get reduced displacement (scaled with frequency)
581       displ_red_FAN2,displ_red_DDW2 = get_reduced_displ(EIGR2D.natom,eigvect,omega,gprimd)
582     # Einstein sum make the vector matrix multiplication ont the correct indices
583       fan_corrQ = N.einsum('ijklmn,olnkm->oij',EIGR2D.EIG2D,displ_red_FAN2)
584       ddw_corrQ = N.einsum('ijklmn,olnkm->oij',ddw_save,displ_red_DDW2)
585     
586       fan_corr = N.einsum('ijk,il->ljk',fan_corrQ,2*bose+1.0)
587       ddw_corr = N.einsum('ijk,il->ljk',ddw_corrQ,2*bose+1.0)
588     
589       print "Now compute active space ..."
590     
591     # Now compute active space
592       temp = N.einsum('ijklm,nlokp->ijmnpo',GKK,displ_red_FAN2)
593       fan_addQ = N.einsum('ijklmn,ijmnk->ijkl',temp,N.conjugate(GKK))
594       temp = N.einsum('ijklm,nlokp->ijmnpo',ddw_save2,displ_red_DDW2)
595       ddw_addQ = N.einsum('ijklmn,ijmnk->ijkl',temp,N.conjugate(ddw_save2))
596     
597       if type == 1: 
598         occtmp = EIGR2D.occ[0,0,:]/2 # jband
599         delta_E_ddw = N.einsum('ij,k->ijk',eig0[0,:,:].real,N.ones(EIGR2D.nband)) - \
600                   N.einsum('ij,k->ikj',eig0[0,:,:].real,N.ones(EIGR2D.nband)) - \
601                   N.einsum('ij,k->ijk',N.ones((EIGR2D.nkpt,EIGR2D.nband)),(2*occtmp-1))*smearing*1j
602     
603         tmp = N.einsum('ijkl,lm->mijk',ddw_addQ,2*bose+1.0) # tmp,ikpt,iband,jband
604         ddw_add = N.einsum('ijkl,jkl->ijk',tmp,1.0/delta_E_ddw)
605         delta_E = N.einsum('ij,k->ijk',eig0[0,:,:].real,N.ones(EIGR2D.nband)) - \
606                   N.einsum('ij,k->ikj',eigq.EIG[0,:,:].real,N.ones(EIGR2D.nband)) - \
607                   N.einsum('ij,k->ijk',N.ones((EIGR2D.nkpt,EIGR2D.nband)),(2*occtmp-1))*smearing*1j # ikpt,iband,jband
608         omegatmp = omega[:].real # imode
609         num1 = N.einsum('ij,k->ijk',bose,N.ones(EIGR2D.nband)) +1.0 \
610               - N.einsum('ij,k->ijk',N.ones((3*EIGR2D.natom,len(temp_info))),occtmp) #imode,tmp,jband
611         deno1 = N.einsum('ijk,l->ijkl',delta_E,N.ones(3*EIGR2D.natom)) \
612               - N.einsum('ijk,l->ijkl',N.ones((EIGR2D.nkpt,EIGR2D.nband,EIGR2D.nband)),omegatmp) #ikpt,iband,jband,imode
613         div1 = N.einsum('ijk,lmki->ijklm',num1,1.0/deno1) # (imode,tmp,jband)/(ikpt,iband,jband,imode) ==> imode,tmp,jband,ikpt,iband
614         num2 = N.einsum('ij,k->ijk',bose,N.ones(EIGR2D.nband)) \
615               + N.einsum('ij,k->ijk',N.ones((3*EIGR2D.natom,len(temp_info))),occtmp) #imode,tmp,jband
616         deno2 = N.einsum('ijk,l->ijkl',delta_E,N.ones(3*EIGR2D.natom)) \
617               + N.einsum('ijk,l->ijkl',N.ones((EIGR2D.nkpt,EIGR2D.nband,EIGR2D.nband)),omegatmp) #ikpt,iband,jband,imode
618         div2 = N.einsum('ijk,lmki->ijklm',num2,1.0/deno2) # (imode,tmp,jband)/(ikpt,iband,jband,imode) ==> imode,tmp,jband,ikpt,iband
619         fan_add = N.einsum('ijkl,lmkij->mij',fan_addQ,div1+div2) # ikpt,iband,jband,imode
620     
621         print "Now compute generalized g2F Eliashberg electron-phonon spectral function ..."
622     
623         fan_tmp = N.sum(fan_addQ,axis=2) # (ikpt, iband, imode)
624         ddw_tmp = N.sum(ddw_addQ,axis=2) # (ikpt, iband, imode)
625         g_kk = fan_tmp - ddw_tmp
626     
627         # Generalized Eliashberg function with Gaussian broadening
628         a2F =  zeros((len(energy),EIGR2D.nkpt,EIGR2D.nband),dtype=complex)
629         PDOS = zeros((len(energy)),dtype=complex)
630         ii = 0
631         for ifreq in energy:
632           for imode in N.arange(3*EIGR2D.natom):
633             a2F[ii,:,:] += g_kk[:,:,imode]*gaussian(ifreq,omegatmp[imode],gaussian_smearing)
634             PDOS[ii] += gaussian(ifreq,omegatmp[imode],gaussian_smearing)
635           ii += 1
636     
637     
638     # Correction from active space 
639       fan_corr += fan_add
640       ddw_corr += ddw_add
641       eigen_corr = (fan_corr[:,:,:] - ddw_corr[:,:,:])*wtq
642     # Index 0 contains total ZPR
643       total_corr[0,:,:,:] = eigen_corr[:,:,:]
644     # Index 1 contains the FAN correction
645       total_corr[1,:,:,:] = fan_corr[:,:,:]*wtq
646     # Index 2 contains the DDW correction
647       total_corr[2,:,:,:] = ddw_corr[:,:,:]*wtq
648     # Index 3-energy contains the Eliashberg functions
649       total_corr[3:len(energy)+3,0,:,:] = a2F*wtq
650     
651       total_corr = make_average(EIGR2D.nkpt,EIGR2D.nband,degen,total_corr,temp=True)
652     
653     # Index energy+4 contains the PDOS
654       total_corr[len(energy)+4:len(energy)*2+4,0,0,0] = PDOS*wtq
655     
656       return total_corr

674       def __init__(self,arguments,ddw_save,ddw_save2,nb_cpus,type,temperature,temp_info,smearing,eig0,degen,energy,gaussian_smearing):
675         # Parallelize the work over cpus

677         partial_compute_wtq = partial(compute_wtq,type=type)

685           total = pool.imap(partial_dynamic_zpm,arguments)
686           self.total_corr = sum(total)

690                                        smearing=smearing,eig0=eig0,degen=degen,energy=energy,gaussian_smearing=gaussian_smearing)
691           total = pool.imap(partial_dynamic_zpm_temp,arguments)
692           self.total_corr = sum(total)
693     
694         pool.close()
695         pool.join()
