14     class DdbFileConverter(object):
15         """Class to read a DDB file in txt format and write it in netCDF format."""
16         # Note that in the netCDF file, it is written as
17         # E2D[iat,icart,jat,jcart]
18         # While it is read from the txt file as
19         # E2D[icart,iat,jcart,jat]

21         def read_nc(self, fname):
22             """Open the DDB.nc file and read it."""
23     
24             with nc.Dataset(fname, 'r') as root:
25     
26                 self.natom = len(root.dimensions['number_of_atoms'])
27                 self.ncart = len(root.dimensions['number_of_cartesian_directions'])  # 3
28                 self.ntypat = len(root.dimensions['number_of_atom_species'])
29                 #self.nkpt = len(root.dimensions['number_of_kpoints'])  # Not relevant
30                 #self.nband = len(root.dimensions['max_number_of_states'])  # Not even there
31                 #self.nsppol = len(root.dimensions['number_of_spins'])  # Not relevant
32     
33                 self.typat = root.variables['atom_species'][:self.natom]
34                 self.amu = root.variables['atomic_masses_amu'][:self.ntypat]
35                 self.rprim = root.variables['primitive_vectors'][:self.ncart,:self.ncart]
36                 self.xred = root.variables['reduced_atom_positions'][:self.natom,:self.ncart]
37                 self.qred = root.variables['q_point_reduced_coord'][:]
38     
39     
40                 # The d2E/dRdR' matrix
41                 self.E2D = np.zeros((self.natom, self.ncart, self.natom, self.ncart), dtype=np.complex)
42                 self.E2D.real = root.variables['second_derivative_of_energy'][:,:,:,:,0]
43                 self.E2D.imag = root.variables['second_derivative_of_energy'][:,:,:,:,1]
44                 self.E2D = np.einsum('aibj->bjai', self.E2D)  # Indicies are reversed when writing them from Fortran.
45     
46                 self.BECT = root.variables['born_effective_charge_tensor'][:self.ncart,:self.natom,:self.ncart]
47     
48         def write_nc(self, fname):
49             """Open withe a DDB.nc file."""
50     
51             with nc.Dataset(fname, 'w') as root:
52     
53                 root.createDimension('number_of_atoms', self.natom)
54                 root.createDimension('number_of_cartesian_directions', 3)
55                 root.createDimension('number_of_atom_species', self.ntypat)
56                 root.createDimension('cplex', 2)
57     
58                 data = root.createVariable('atom_species', 'i', ('number_of_atoms'))
59                 data[...] = self.typat[...]
60     
61                 data = root.createVariable('atomic_masses_amu', 'd', ('number_of_atom_species'))
62                 data[...] = self.amu[...]
63     
64                 data = root.createVariable('primitive_vectors', 'd', ('number_of_cartesian_directions', 'number_of_cartesian_directions'))
65                 data[...] = self.rprim[...]
66     
67                 data = root.createVariable('reduced_atom_positions', 'd', ('number_of_atoms', 'number_of_cartesian_directions'))
68                 data[...] = self.xred[...]
69     
70                 data = root.createVariable('q_point_reduced_coord', 'd', ('number_of_cartesian_directions'))
71                 data[...] = self.qred[...]
72     
73                 data = root.createVariable('born_effective_charge_tensor', 'd', ('number_of_cartesian_directions',
74                                                                                 'number_of_atoms', 'number_of_cartesian_directions'))
75                 data[...] = self.BECT[...]
76     
77                 # E2D dimensions must be swapped
78                 E2D_swap = np.zeros((self.natom, self.ncart, self.natom, self.ncart), dtype=np.complex)
79                 for iat in range(self.natom):
80                     for icart in range(self.ncart):
81                         for jat in range(self.natom):
82                             for jcart in range(self.ncart):
83                                 E2D_swap[iat, icart, jat, jcart] = self.E2D[icart, iat, jcart, jat]
84     
85                 data = root.createVariable('second_derivative_of_energy', 'd', ('number_of_atoms', 'number_of_cartesian_directions',
86                                                                                 'number_of_atoms', 'number_of_cartesian_directions',
87                                                                                 'cplex'))
88                 E2D_swap = np.einsum('aibj->bjai', E2D_swap)  # Indicies are reversed when writing them from Fortran.
89                 data[...,0] = E2D_swap.real
90                 data[...,1] = E2D_swap.imag

93         def read_txt(self, fname):
94           """Open the DDB file and read it."""
95           self.nkpt=0
96           self.rprim = zeros((3,3))
97           with open(fname,'r') as DDB:
98             Flag = 0
99             Flag2 = False
100             Flag3 = False
101             ikpt = 0
102             for line in DDB:
103               if line.find('natom') > -1:
104                 self.natom = np.int(line.split()[1])
105               if line.find('nkpt') > -1:
106                 self.nkpt = np.int(line.split()[1])
107                 self.kpt  = zeros((self.nkpt,3))
108               if line.find('ntypat') > -1:
109                 self.ntypat = np.int(line.split()[1])
110               if line.find('nband') > -1:
111                 self.nband = np.int(line.split()[1])
112               if line.find('acell') > -1:
113                 line = line.replace('D','E')
114                 tmp = line.split()
115                 self.acell = [np.float(tmp[1]),np.float(tmp[2]),np.float(tmp[3])]
116               if Flag2:
117                 line = line.replace('D','E')
118                 for ii in np.arange(3,self.ntypat):
119                   self.amu[ii] = np.float(line.split()[ii-3])
120                   Flag2 = False
121               if line.find('amu') > -1:
122                 line = line.replace('D','E')
123                 self.amu = zeros((self.ntypat))
124                 if self.ntypat > 3:
125                   for ii in np.arange(3):
126                     self.amu[ii] = np.float(line.split()[ii+1])
127                     Flag2 = True 
128                 else:
129                   for ii in np.arange(self.ntypat):
130                     self.amu[ii] = np.float(line.split()[ii+1])
131               if line.find(' kpt ') > -1:
132                 line = line.replace('D','E')
133                 tmp = line.split()
134                 self.kpt[0,0:3] = [float(tmp[1]),float(tmp[2]),float(tmp[3])]
135                 ikpt = 1
136                 continue
137               if ikpt < self.nkpt and ikpt > 0:
138                 line = line.replace('D','E')
139                 tmp = line.split()
140                 self.kpt[ikpt,0:3] = [float(tmp[0]),float(tmp[1]),float(tmp[2])]  
141                 ikpt += 1
142                 continue
143               if Flag == 2:
144                 line = line.replace('D','E')
145                 tmp = line.split()
146                 self.rprim[2,0:3] = [float(tmp[0]),float(tmp[1]),float(tmp[2])]
147                 Flag = 0
148               if Flag == 1:
149                 line = line.replace('D','E')
150                 tmp = line.split()
151                 self.rprim[1,0:3] = [float(tmp[0]),float(tmp[1]),float(tmp[2])]
152                 Flag = 2
153               if line.find('rprim') > -1:
154                 line = line.replace('D','E')
155                 tmp = line.split()
156                 self.rprim[0,0:3] = [float(tmp[1]),float(tmp[2]),float(tmp[3])]
157                 Flag = 1
158               if Flag3:
159                 line = line.replace('D','E')
160                 for ii in np.arange(12,self.natom): 
161                   self.typat[ii] = np.float(line.split()[ii-12]) 
162                 Flag3 = False 
163               if line.find(' typat') > -1:
164                 self.typat = zeros((self.natom))
165                 if self.natom > 12:
166                   for ii in np.arange(12):
167                     self.typat[ii] = np.float(line.split()[ii+1])
168                     Flag3 = True
169                 else:
170                   for ii in np.arange(self.natom):
171                     self.typat[ii] = np.float(line.split()[ii+1])
172               # Read the actual d2E/dRdR matrix
173               if Flag == 3:
174                 line = line.replace('D','E')
175                 tmp = line.split()
176                 if not tmp:
177                   break
178                 self.E2D[int(tmp[0])-1,int(tmp[1])-1,int(tmp[2])-1,int(tmp[3])-1] = \
179                   complex(float(tmp[4]),float(tmp[5]))
180               # Read the current Q-point
181               if line.find('qpt') > -1:
182                 line = line.replace('D','E')
183                 tmp = line.split()
184                 #self.qred = [np.float(tmp[1]),np.float(tmp[2]),np.float(tmp[3])]
185                 self.qred = np.array([np.float(tmp[1]),np.float(tmp[2]),np.float(tmp[3])])
186                 Flag = 3
187                 self.E2D = zeros((3,self.natom,3,self.natom),dtype=complex)
188     
189             self.ncart = 3
190             self.BECT = zeros((self.ncart,self.natom,self.ncart), dtype=float)
191             self.xred = zeros((self.natom, self.ncart), dtype=float)
192             for i in range(3):
193                 self.rprim[i,:] *= self.acell[i]
194             self.acell = np.ones(3, dtype=float)
