108     def gather_from_OUTCAR(filename):
109         """
110         Author: Nicholas Pike
111         Email : Nicholas.pike@smn.uio.no
112         
113         Purpose: Gather data from the output of a VASP calculation for the elastic tensor
114         
115         OUTPUT: Returns the system name, elastic tensor, ion mass, volume in (m^3), 
116                 and the nultiplicity of each atom in the unit cell
117         
118         """
119         system_name = ''
120         elasten     = np.empty(shape=(6,6))
121         ionmass     = []
122         vol         = 0.0
123         #open file and extract data
124         with open(filename,'r') as d:
125            for i, line in enumerate(d):
126               if 'POSCAR =' in line:
127                   data = linecache.getline(filename,i+1).split()
128                   system_name = data[2:] 
129               elif 'SYMMETRIZED ELASTIC MODULI (kBar)' in line:
130                   for j in range(6):
131                       data = linecache.getline(filename,i+j+4).split()
132                       elasten[j][:] = data[1:]
133               elif 'Mass of Ions in am' in line:
134                   data = linecache.getline(filename,i+2).split()
135                   for j in range(2,len(data)):
136                       ionmass = np.append(float(data[j]),ionmass)
137               elif 'volume of cell :' in line:
138                   data = linecache.getline(filename,i+1).split()
139                   vol  = float(data[4]) 
140               
141         atommult = []
142         for i in range(len(system_name)):
143             namelist = list(system_name[i])
144             number   = namelist[len(namelist)-1]
145             atommult = np.append(float(number),atommult)
146              
147         data_array = [system_name,elasten,ionmass,vol*ang_to_meter**3,atommult]
148         
149         return data_array

151     def gather_from_ANADDB(fileout,fileddb):
152         """
153         Author: Nicholas Pike
154         Email : Nicholas.pike@smn.uio.no
155         
156         Purpose: Gather data from the output of an ANADDB calculation for the elastic tensor
157         
158         OUTPUT: Returns the system name, elastic tensor, ion mass, volume in (m^3),

160         
161         """

163         ionmass     = []
164         vol         = 0.0
165         ntypat      = 0.0
166         typat       = []
167         
168         #open file and extract data
169         with open(fileout,'r') as d:
170            for i, line in enumerate(d):
171               if 'Elastic Tensor (relaxed ion) (unit:10^2GP):' in line:
172                   for j in range(6):
173                       data = linecache.getline(fileout,i+j+4).split()
174                       elasten[j][:] = data[:]
175               elif ' Unit cell volume ucvol= ' in line:
176                   data = linecache.getline(fileout,i+1).split()
177                   vol  = float(data[4]) 
178                            
179         with open(fileddb,'r') as d:
180             for i, line in enumerate(d):
181                 if 'amu' in line:
182                    data = linecache.getline(fileddb,i+1).split()
183                    ionmass = data[1:]
184                 elif 'ntypat' in line:
185                     data = linecache.getline(fileddb,i+1).split()
186                     ntypat = data[1]   
187                 elif 'typat' in line:
188                     data = linecache.getline(fileddb,i+1).split()
189                     typat = data[1:]                   
190         
191         #replace D with E
192         for i in range(len(ionmass)):
193             ionmass[i]=float(ionmass[i].replace('D','E'))
194                   
195         atommult = []
196         u,numtyat = np.unique(typat, return_counts= True)
197         for i in range(int(ntypat)):
198             atommult = np.append(numtyat[i],atommult)
199                  
200         data_array = ['',elasten*1000.0,ionmass,vol*ang_to_meter**3,atommult]
201         return data_array

203     def sound_velocities(data):
204         """
205         Author: Nicholas Pike
206         Email : Nicholas.pike@smn.uio.no
207         
208         Purpose: Calculate the longitudional, shear and average sound velocities
209         """
210         #declare variables
211         vl      = 0
212         vs      = 0
213         va      = 0
214         density = 0
215         B       = 0
216         BR      = 0
217         G       = 0
218         GR      = 0
219         YXX     = 0
220         YYY     = 0
221         YZZ     = 0
222         nuxy    = 0
223         nuyz    = 0
224         nuxz    = 0
225         
226         #calculate compliance tensor
227         s = np.linalg.inv(data[1])
228         
229         #calculate density
230         masstot = 0
231         for i in range(len(data[2])):
232             masstot +=data[2][i]*data[4][i] #data[2] is the mass data[4] is the multiplicity
233         density = masstot*amu_to_kg/data[3]/kgm3_to_cm3 #data[3] is the volume in g/cm3
234         
235         #Calculate bulk modulus and shear modulus
236         #note: data[1] is the elastic tensor
237         B  = 1.0/9.0*((data[1][0][0]+data[1][1][1]+data[1][2][2])+2.0*(data[1][0][1]+data[1][0][2]+data[1][1][2]))
238         BR = 1.0/((s[0,0]+s[1,1]+s[2,2])+2.0*(s[0,1]+s[1,2]+s[0,2]))
239         
240         G  = 1.0/15.0*((data[1][0][0]+data[1][1][1]+data[1][2][2])-(data[1][0][1]+data[1][0][2]+data[1][1][2])+3.0*(data[1][3][3]+data[1][4][4]+data[1][5][5]))
241         GR = 15.0/(4.0*(s[0,0]+s[1,1]+s[3,3])-4.0*(s[0,1]+s[1,2]-s[0,2])+3.0*(s[3,3]+s[4,4]+s[5,5])) 
242         
243         #calculate Youngs modulus
244         YXX = 1.0/s[0,0]
245         YYY = 1.0/s[1,1]
246         YZZ = 1.0/s[2,2]
247         
248         #calculate vl
249         if B + 4.0/3.0*G <= 0:
250             vl = 0
251         else:  
252         	vl  = np.sqrt((B+4.0/3.0*G)*kbar_to_GPa/(density/kgm3_to_cm3))*m_to_cm # convert kbar to GPa and then g/cm3 to kg /m3
253         
254         #calculate vs
255         if G <= 0:
256     	G = 0
257         else:
258             vs = np.sqrt(G*kbar_to_GPa/(density/kgm3_to_cm3))*m_to_cm # convert kbar to GPa and then g/cm3 to kg /m3
259         
260         #calculate va
261         va = (1.0/3.0*((1.0/vl**3)+(2.0/vs**3)))**(-1.0/3.0)
262         
263         #calculate poisson ratio
264         nuxy = -s[0,1]*YXX
265         nuyz = -s[1,2]*YYY
266         nuxz = -s[0,2]*YZZ
267         
268         #print results 
269         f1 = open('derived_elastic','a')
270         f1.write('Elastic Tensor\n')
271         f1.write('Direction XX\t  YY\t   ZZ\t    XY\t     YZ\t      ZX\n' )
272         f1.write('XX \t %1.4e %1.4e %1.4e %1.4e %1.4e %1.4e\n'%(data[1][0][0],data[1][0][1],data[1][0][2],data[1][0][3],data[1][0][4],data[1][0][5] ))
273         f1.write('YY \t %1.4e %1.4e %1.4e %1.4e %1.4e %1.4e\n'%(data[1][1][0],data[1][1][1],data[1][1][2],data[1][1][3],data[1][1][4],data[1][1][5] ))
274         f1.write('ZZ \t %1.4e %1.4e %1.4e %1.4e %1.4e %1.4e\n'%(data[1][2][0],data[1][2][1],data[1][2][2],data[1][2][3],data[1][2][4],data[1][2][5] ))
275         f1.write('XY \t %1.4e %1.4e %1.4e %1.4e %1.4e %1.4e\n'%(data[1][3][0],data[1][3][1],data[1][3][2],data[1][3][3],data[1][3][4],data[1][3][5] ))
276         f1.write('YZ \t %1.4e %1.4e %1.4e %1.4e %1.4e %1.4e\n'%(data[1][4][0],data[1][4][1],data[1][4][2],data[1][4][3],data[1][4][4],data[1][4][5] ))
277         f1.write('ZX \t %1.4e %1.4e %1.4e %1.4e %1.4e %1.4e\n'%(data[1][5][0],data[1][5][1],data[1][5][2],data[1][5][3],data[1][5][4],data[1][5][5] ))
278         f1.write('\n')
279         f1.write('Compliance Tensor\n')
280         f1.write('Direction XX\t  YY\t   ZZ\t    XY\t     YZ\t      ZX\n' )
281         f1.write('XX \t %1.4e %1.4e %1.4e %1.4e %1.4e %1.4e\n'%(s[0][0],s[0][1],s[0][2],s[0][3],s[0][4],s[0][5] ))
282         f1.write('YY \t %1.4e %1.4e %1.4e %1.4e %1.4e %1.4e\n'%(s[1][0],s[1][1],s[1][2],s[1][3],s[1][4],s[1][5] ))
283         f1.write('ZZ \t %1.4e %1.4e %1.4e %1.4e %1.4e %1.4e\n'%(s[2][0],s[2][1],s[2][2],s[2][3],s[2][4],s[2][5] ))
284         f1.write('XY \t %1.4e %1.4e %1.4e %1.4e %1.4e %1.4e\n'%(s[3][0],s[3][1],s[3][2],s[3][3],s[3][4],s[3][5] ))
285         f1.write('YZ \t %1.4e %1.4e %1.4e %1.4e %1.4e %1.4e\n'%(s[4][0],s[4][1],s[4][2],s[4][3],s[4][4],s[4][5] ))
286         f1.write('ZX \t %1.4e %1.4e %1.4e %1.4e %1.4e %1.4e\n'%(s[5][0],s[5][1],s[5][2],s[5][3],s[5][4],s[5][5] ))
287         f1.write('\n')
288         f1.write('Voigt Approximation:\n')
289         f1.write(' Bulk Modulus (kbar):    %4.2f\n'%B)
290         f1.write(' Shear Modulus (kbar):   %4.2f\n'%G)
291         f1.write(' Long Velocity (cm/s):   %4.2f\n'%vl)
292         f1.write(' Shear Velocity (cm/s):  %4.2f\n'%vs)
293         f1.write(' Avg. Velocity (cm/s):   %4.2f\n'%va)
294         f1.write('\n')
295         f1.write('Reuss Approximation:\n')
296         f1.write(' Bulk Modulus (kbar):    %4.2f\n'%BR)
297         f1.write(' Shear Modulus (kbar):   %4.2f\n'%GR)
298         f1.write('\n')
299         f1.write("Young's Modulus:\n")
300         f1.write(' Y_xx (kbar):           %4.2f\n' %YXX)
301         f1.write(' Y_yy (kbar):           %4.2f\n' %YYY)
302         f1.write(' Y_zz (kbar):           %4.2f\n' %YZZ)
303         f1.write('\n')
304         f1.write("Poisson's Ratio (from elastic tensor)\n")
305         f1.write(' nu_xy:                 %0.2f\n' %nuxy)
306         f1.write(' nu_yz:                 %0.2f\n' %nuyz)
307         f1.write(' nu_xz:                 %0.2f\n' %nuxz)
308         f1.write('\n')
309         f1.write('Additional information:\n')
310         f1.write(' Density (g/cm**3):      %4.4f\n'%density)
311         f1.write('\n')
312         f1.close()
313         
314         return vl, vs, va

316     def Debye_temperature(va,data):
317         """
318         Author: Nicholas Pike
319         Email : Nicholas.pike@smn.uio.no
320         
321         Purpose: Calculate the Debye Temperature
322         """
323         #declare variables
324         thetaD        = 0
325         natom         = 0
326         density       = 0
327         masstot       = 0
328         molarmass     = 0
329         atom_molecule = 0
330     
331         #calculate density
332         for i in range(len(data[2])):
333             masstot +=data[2][i]*data[4][i] #data[2] is the mass data[4] is the multiplicity
334         density = masstot*amu_to_kg/data[3]/kgm3_to_cm3 #data[3] is the volume in g/cm3
335         
336         #number of unit cells 
337         uc_count = 0
338         for i in range(len(data[4])):
339             uc_count = find_gcd(uc_count,data[4][i])
340         
341         #calculate molar mass
342         for i in range(len(data[2])):
343             molarmass += data[2][i]
344            
345         #calculate number of atoms 
346         for i in range(len(data[4])):
347             natom +=data[4][i]   
348         
349         #calculate number of atoms per molecule
350         atom_molecule = natom/uc_count
351     
352         #Calculate Debye temperature
353         thetaD = h/kb*((3.0*atom_molecule*Na*density)/(4.0*np.pi*molarmass))**(1.0/3.0)*va*natom**(-1.0/3.0)*m_to_cm
354         
355         #print results 
356         f1 = open('derived_elastic','a')
357         f1.write(' Debye Temp (K):         %4.2f\n'%thetaD)
358         f1.write('\n')
359         f1.close()
360         
361         return thetaD

364     def find_gcd(x, y):
365         """
366         Determines the gcd for an array of values.
367         """
368         while(y):
369             x, y = y, x % y
370         return x

372     def gruneisen(vs,vl):
373         """
374         Author: Nicholas Pike
375         Email : Nicholas.pike@smn.uio.no
376         
377         Purpose: Calculate the gruneisen parameter using poissons ratio
378         """
379         #declare variables
380         poisson = 0
381         grun    = 0
382         
383         #calculate poissons ratio
384         poisson = (1.0-2.0*(vs/vl)**2)/(2.0-2.0*(vs/vl)**2)
385         
386         #calculate gruneisen parameter
387         grun = 3.0/2.0*((1.0+poisson)/(2.0-3.0*poisson))
388         
389         #print results
390         f1 = open('derived_elastic','a')
391         f1.write(' Poisson Ratio (from sound) %0.4f\n'%poisson)
392         f1.write(' Gruneisen Param:           %4.4f\n'%grun)    
393         f1.write('\n')
394         f1.close()
395         
396         return grun

398     def thermal_cond(data,thetaD,grun):
399         """    
400         Author: Nicholas Pike
401         Email : Nicholas.pike@smn.uio.no
402         
403         Purpose: Calculate the thermal conductivity using Slack 
404                  J. Phys. Chem. Solids 34, 321 (1973)
405         
406         """
407         #declare variables
408         kappa   = 0.0
409         A       = 0
410         massav  = 0
411         natom   = 0
412         masstot = 0
413         temp    = 300.0
414             
415         #calculate A
416         A = 2.43E-8/(1.0-(0.514/grun)+(0.228/grun**2))
417         
418         #calculate the number of atoms
419         for i in range(len(data[4])):
420             natom +=data[4][i]
421         
422         #Calculate average atomic mass
423         for i in range(len(data[2])):
424             masstot += data[2][i]*data[4][i] #data[2] is the mass data[4] is the multiplicity
425         massav = masstot/natom
426         
427         #calculate kappa
428         kappa = A*massav*(data[3]/ang_to_meter**3)**(1.0/3.0)*thetaD**3/(grun**2*temp)
429         
430         #print results
431         f1 = open('derived_elastic','a')
432         f1.write(' Kappa @300K (W/cmK):    %2.4f\n'%kappa)
433         f1.close() 
434         
435         return kappa
