17     def browse_localpath(filepath):
18         """
19         Open filepath in the default browser.
20     
21         .. warning: This code is not portable since we should pass a url.
22         """
23         import webbrowser
24         if not filepath.startswith("file://"): filepath = "file://" + filepath
25     
26         try:
27             webbrowser.open(filepath)
28         except webbrowser.Error as exc:

33     def make_html_diff(fromfile, tofile):
34         import difflib
35     
36         with open(fromfile) as fh: fromlines = fh.readlines()
37         with open(tofile) as fh: tolines = fh.readlines()
38     
39         diff = difflib.HtmlDiff().make_file(fromlines, tolines, fromfile, tofile, context=True, numlines=3)
40     
41         # writelines because diff is a generator
42         _, tmp_path = tempfile.mkstemp(text=True, suffix=".html")
43         with open(tmp_path, "w") as fh: fh.writelines(diff)
44     
45         return tmp_path

48     def find_nextout(root="."):
49         """
50         Returns the path the next output file. Assumes Abinit conventions + gmatteo's convention 
51         for the extension i.e. run.abo, run.aboA ...
52         This function is the most complicated of the entire module!
53         """
54         from string import ascii_letters
55         root = os.path.abspath(root)
56                                                                                                                    
57         # Find all files with extensions `.abo?`
58         chars, found = [], False
59         for f in os.listdir(root):
60             head, ext = os.path.splitext(f)
61             if ext.startswith(".abo"):
62                 found = True
63                 if len(ext) > 4: chars.append(ext[-1])
64     
65         if not found:
66             # First run
67             fname = "run.abo"
68         else:
69             if not chars: 
70                 fname = "run.aboA"
71             else:
72                 last_ch = sorted(chars)[-1]
73                 if last_ch == "Z": raise RunTimeError("Got Z as last character, clean your directory!")
74                 print("last",last_ch)
75                 i = ascii_letters.index(last_ch)
76                 next_ch =  ascii_letters[i+1]
77                 fname = "run.abo" + next_ch
78                                                                                                                    
79         return os.path.join(root, fname)
