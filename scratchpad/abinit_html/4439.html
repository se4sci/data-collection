57         def xcopy(self, input=None):
58             '''
59             Converts the value of the input stream (or contents of input filename)
60             from xyaptu format to yaptu format, and invokes yaptu.copy
61             '''
62     
63             # Read the input
64             inf = input
65             try:
66                 inputText = inf.read()
67             except AttributeError:
68                 inf = open(input)
69                 if inf is None:
70                     raise ValueError("Can't open file (%s)" % input)
71                 inputText = inf.read()
72             try:
73                 inf.close()
74             except:
75                 pass
76     
77             # Translate (xyaptu) input to (yaptu) input, and call yaptu.copy()
78     
79             yinf = StringIO(self._x2y_translate(inputText))
80             self.copy(inf=yinf)
81             yinf.close()

83         def _x2y_translate(self, xStr):
84             ' Converts xyaptu markup in input string to yaptu delimeters '
85     
86             # Define regexps to match xml elements on.
87             # The variations (all except for py-expr, py-close) we look for are:
88             # <py-elem code="{python code}" /> |
89             # <py-elem code="{python code}">ignored text</py-elem> |
90             # <py-elem>{python code}</py-elem>
91     
92             # ${py-expr} | $py-expr | <py-expr code="pvkey" />
93             reExpr = re.compile(r'''
94               \$\{([^}]+)\} |  # ${py-expr}
95               \$([_\w]+) | # $py-expr
96               <py-expr\s+code\s*=\s*"([^"]*)"\s*/> |
97               <py-expr\s+code\s*=\s*"([^"]*)"\s*>[^<]*</py-expr> |
98               <py-expr\s*>([^<]*)</py-expr\s*>
99             ''', re.VERBOSE)
100     
101             # <py-line code="pvkeys=pageVars.keys()"/>
102             reLine = re.compile(r'''
103               <py-line\s+code\s*=\s*"([^"]*)"\s*/> |
104               <py-line\s+code\s*=\s*"([^"]*)"\s*>[^<]*</py-line> |
105               <py-line\s*>([^<]*)</py-line\s*>
106             ''', re.VERBOSE)
107     
108             # <py-open code="for k in pageVars.keys():" />
109             reOpen = re.compile(r'''
110               <py-open\s+code\s*=\s*"([^"]*)"\s*/> |
111               <py-open\s+code\s*=\s*"([^"]*)"\s*>[^<]*</py-open\s*> |
112               <py-open\s*>([^<]*)</py-open\s*>
113             ''', re.VERBOSE)
114     
115             # <py-clause code="else:" />
116             reClause = re.compile(r'''
117               <py-clause\s+code\s*=\s*"([^"]*)"\s*/> |
118               <py-clause\s+code\s*=\s*"([^"]*)"\s*>[^<]*</py-clause\s*> |
119               <py-clause\s*>([^<]*)</py-clause\s*>
120             ''', re.VERBOSE)
121     
122             # <py-close />
123             reClose = re.compile(r'''
124               <py-close\s*/> |
125               <py-close\s*>.*</py-close\s*>
126             ''', re.VERBOSE)
127     
128             # Call-back functions for re substitutions
129             # These must be in sync with what is expected in self.__init__
130             def rexpr(match,self=self):
131                 return '_:@%s@:_' % match.group(match.lastindex)
132             def rline(match,self=self):
133                 return '\n++yaptu %s #\n--yaptu \n' % match.group(match.lastindex)
134             def ropen(match,self=self):
135                 return '\n++yaptu %s \n' % match.group(match.lastindex)
136             def rclause(match,self=self):
137                 return '\n==yaptu %s \n' % match.group(match.lastindex)
138             def rclose(match,self=self):
139                 return '\n--yaptu \n'
140     
141             # Substitutions
142             xStr = reExpr.sub(rexpr, xStr)
143             xStr = reLine.sub(rline, xStr)
144             xStr = reOpen.sub(ropen, xStr)
145             xStr = reClause.sub(rclause, xStr)
146             xStr = reClose.sub(rclose, xStr)
147     
148             # When in debug mode, keep a copy of intermediate template format
149             if self.dbg:
150                 _sep = '====================\n'
151                 self.dbgOuf.write('%sIntermediate YAPTU format:\n%s\n%s' % (_sep, xStr, _sep))
152     
153             return xStr
154     
155         # Handle expressions that do not evaluate
156         def _handleBadExps(self, s):
157             ' Handle expressions that do not evaluate '
158             if self.dbg:
159                 self.dbgOuf.write('!!! ERROR: failed to evaluate expression: %s \n' % s)
160             return '***! %s !***' % s
161     
162         # Preprocess code
163         def _preProcess(self, s, why):
164             ' Preprocess embedded python statements and expressions '
165             return self._xmlDecode(s)
166         def _preProcessDbg(self, s, why):
167             ' Preprocess embedded python statements and expressions '
168             self.dbgOuf.write('!!! DBG: %s %s \n' % (s, why))
169             return self._xmlDecode(s)
170     
171         # Decode utility for XML/HTML special characters
172         _xmlCodes = [
173           ['"', '&quot;'],
174           ['>', '&gt;'],
175           ['<', '&lt;'],
176           ['&', '&amp;'],
177         ]
178         def _xmlDecode(self, s):
179             ' Returns the ASCII decoded version of the given HTML string. '
180             codes = self._xmlCodes
181             for code in codes:
182                 #s = string.replace(s, code[1], code[0])
183                 s = s.replace(code[1], code[0])
184             return s

326     __doc__ = """
327     Xyaptu is python-centric, in the sense that the XML tags offered reflect python constructs
328     (such as python expressions, statements, opening and closing blocks) and not particularly
329     constructs typically identified in web page templates. The advantage is simplicity, while
330     still keeping all the options open for the HTML or XML document designer.
331     
332     The primary requirements of xyaptu are:
333     
334     (a) expression evaluation, e.g. variable substitutions, function calls (b) loop over, and
335     format, a python data sequence (c) xyaptu mark-up must pass through an XML parser, to
336     naturally allow using XML tools of choice, such as XSLT, for generation of page templates
337     (d) but, since HTML is not XML, page templates need not otherwise be XML compliant. In
338     some future time xhtml may be the norm for web pages, but as yet eb design tools currently
339     in wide use do not produce XML compliant output. Thus, non-XML page templates, such as
340     HTML, must still be considered as valid input. (For the implementation, this implies that
341     xyaptu tags be matched using regular expressions, and not by parsing HTML or XML.) (e)
342     simplicity of use, with minimum learning and runtime overhead (f) separation of
343     presentation and application logic
344     
345     There are only 5 XML tags, to handle python statements (expression, line, block open,
346     block continuation clause, block close). Python expressions are also mapped to a 'pcdata'
347     token, to allow the use of python expressions also in places where tags are not allowed,
348     i.e. in attribute values. XML special characters (< > & ") must be encoded (< > & ") to be
349     used in python code. This, unfortunately, is unavoidable.
350     
351     Xyaptu may be run in debug mode, and debug output may be sent to either the specified
352     output filelike object, or to any writable filelike object. Please see the module
353     self-test for sample code. When in debug mode, the intermediate format of the template is
354     also copied to the debug stream (done in copier._x2y_translate). As a default behaviour,
355     expressions that do not evaluate are written out (surrounded with '! ' and ' !') to the
356     specified output stream, and, if in debug mode, an error message is written out to the
357     debug stream (which defaults to the output stream). To change this behaviour (and that of
358     debug in general) you would need to override the methods _handleBadExps and _preprocessDbg
359     in a sublcassed xyaptu.
360     
361     Mark-up syntax: A template may contain anything acceptable to targeted clients will
362     accept, plus the following xyaptu tags and 1 expression, to mark-up python expressions and
363     statements:
364     
365     <py-expr code="pvkey" /> -- expression <py-line code="pvkeys=pVars.keys()" /> -- line
366     statement <py-open code="if inSession:"/> -- block open <py-clause code="else:"/> -- block
367     continuation clause <py-close/> -- block close
368     
369     $pyvar | ${pyexpr} -- for simple interpolation of python variables, expressions, or
370     function calls (the second syntax option is for expressions that may contain spaces or
371     other characters not allowed in variable names)
372     
373     The advantage of pcdata tokens over XML tags is that they may be used anywhere, including
374     within XML attribute values, e.g.: <a href="http://host/$language/topic">topic</a>
375     
376     Alternate mark-up tag syntax: Because most web browsers by default do not display
377     attribute values, but they do show element values, as a convenience for those who like to
378     preview page templates in web browsers, an alternate tag syntax is provided. The five tag
379     examples above therefore become:
380     
381     <py-expr>pvkey -- expression <py-line>pvkeys=pVars.keys() -- line statement <py-open>if
382     inSession: -- block open <py-clause>else: -- block continuation clause <py-close># close
383     -- block close (content is ignored)
384     
385     Note that, in the case that a "code" attribute is specified, then _that_ code is executed,
386     and element content, if any, is ignored.
387     
388     Usage: (1) A runtime application prepares the document namespace in the form of a python
389     dictionary. (2) An xyaptu xcopier is initialised with this dictionary: from xyaptu import
390     xcopier xcp = xcopier(DNS) (3) The xcopy method of this xcopier instance may be called
391     with either the name of a page template file, or a filehandle, to instantiate the page
392     template within this namespace: xcp.xcopy( templateFileName | templateFileHandle ) For
393     page templates available as strings in memory, use StringIO: from cStringIO import
394     StringIO pageTemplateStream = StringIO(pageTemplateString) xcp.xcopy(pageTemplateStream)
395     (4) Output is by default sent to sys.stdout. A different output stream may be specified at
396     initialisation, as the value of an 'ouf' parameter: xcp = xcopier(DNS, ouf=myOutputStream)
397     (5) Debugging may be turned on by initialising xyaptu with a 'dbg=1' switch. Debug output
398     is sent to the specified output stream unless a separate stream is specified by the dbgOuf
399     parameter, also at initialisation time.
400     
401     For a full working example, see the module self-test source.
402     
403     Enhancements to consider:
404     
405     Add support for XML namespaces (py:expr, py:line, ...)
406     For each python statement (open, close, ...), yaptu adds an extra blank line. To be
407     'faithful' to the template source, it would be better if this is not so.
408     Do not process xyaptu mark-up when this is inside an XML comment in the source document
409     template
410     Add possibility to include xyaptu mark-up as verbatim document content, i.e. to be able to
411     write out ${pyexpr} as is.
