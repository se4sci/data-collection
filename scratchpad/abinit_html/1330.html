21     #--------------(AUTO)CORRELATION FUNCTION------------------------#
22     #----------------------------------------------------------------#
23     class Correlation:
24     
25        #-------------Constructor--------------#
26         def __init__(self, serie1,serie2=None):
27     
28             # This class aim is to calculate the (auto)correlation function
29             # the parameter is :
30             #    - pSerie witch is the serie  ([time,atom,[x,y,z]] )
31     
32             self.nbtime = len(serie1)               # Number of steps in the simulation
33             self.nbatom = len(serie1[0])               # Number of atoms in the simulation
34             self.CorrelationFunction = zeros(2*self.nbtime,complex) # (Auto)correlation function array
35             self.calculation(serie1,serie2)
36     
37        #------------Methods-----------------------#
38         def calculation(self, pserie1, pserie2):
39     
40             serie1 = pserie1
41     
42             if pserie2 == None:
43                 serie2 = pserie1
44             else:
45                 serie2 = pserie2
46     
47             for at in range(self.nbatom):
48     
49                 for j in range(3):
50     
51                     self.CorrelationFunction +=  ifft(conjugate(fft(serie1[:,at,j],2*self.nbtime,0))*fft(serie2[:,at,j],2*self.nbtime,0))
52     
53             #Normalisation by the therm 1/ ((nbtime - t) * 3 * N). N is the number of atoms, nbtime the number of step and t the step.
54             self.CorrelationFunction = self.CorrelationFunction.real[:self.nbtime] / ((self.nbtime-arange(self.nbtime)) * 3 * self.nbatom)
55     
56     
57         def getCorrelationFunction(self,normalize = False):
58             if len(self.CorrelationFunction ) >= 0 :
59                 if normalize:
60                     return self.CorrelationFunction/self.CorrelationFunction[0]
61                 else:
62                     return self.CorrelationFunction
63             else:
64                 return 0

70     #----------------------------------------------------------------#
71     class DOS:
72         def __init__(self, pdata, pres, pdtion):
73     
74             # This class aim is to calculate the density of phonon states (DOS),as being the Fourier transforms of the VACF
75             # the parameters are  :
76             #    - pdata witch is the array (1D) of the VACF
77             #    - pres witch is the resolution
78             #    - pdtion witch is ion time steps in atomic units of time
79             self.atu = 2.41884*10**-5          # one atomic time unit (in ps)
80             self.dtion = pdtion                   # ion time steps in atomic units of time
81             self.VACF = pdata                # Velocity autocorelation Function (array 1D)
82             self.dt = self.dtion * self.atu        # time step (in ps)
83             self.nbtime = len(pdata)        # Number of steps
84             self.times = self.dt * arange(self.nbtime)# 1D array with the times of the simulation
85             self.fact = 0.6582054674822119  # conversion factor ps => 1/meV
86     
87     
88             
89             #Calculation of sigma for the gaussian function
90             self.resolution = pres
91             self.sigma = 1.0/(1.5192669*self.resolution/(2.0*sqrt(2.0*log(2.0))))
92     
93     
94     
95         def getDOS(self):
96             #Return array (1D) phonons density of states (1/mev)
97             DOS = self.windowGaussian(self.VACF, self.times, self.sigma)
98             return fft(DOS,len(DOS),0).real[:len(self.VACF)] * self.dt * self.fact
99     
100     
101         def getFrequencies(self):
102             #Return array (1D) with the frenquencies (mev)
103             frequencies = 4.1356 * arange(self.nbtime)/(2 * self.nbtime*self.dt)
104             return frequencies

107         def windowGaussian(self, inputSeries, x, s = 50, x0 = 0):
108             # Return the input serie multiplying by the gaussian function
109             # The size of the output array is double that the input serie,
110             # because the final array is periodized function
111     
112     
113             gauss= exp(-0.5*( (x - x0)**2 / (2*s**2)))
114     
115     
116             # Creation of the final array:
117             res = zeros(2*len(inputSeries))
118     
119             # Multiplying the input serie by the gaussian:
120             win = gauss*inputSeries
121     
122             # Periodic function is creating :
123             res[:len(inputSeries)] = win
124             res[len(inputSeries):] = win[-1:-len(inputSeries)-1:-1]
125     
126     
127             #old algorithm:
128             #res = zeros((2*len(inputSeries) - 2,))
129             #win = gauss*inputSeries
130             #res[:len(inputSeries)] = win
131             #res[len(inputSeries):] = win[-2:0:-1]
132     
133             return res

137             return  rfft(self.VACF,2*len(self.VACF)-1,0).real * 2 * self.dt * self.fact

215     class RDF:
216     
217         def __init__(self, pfile, mode, atom1, atom2, box, pdr, pstep, No_Windows=False):
218     
219             # This class aim is to calculate the Radial Distrubution Function
220             # the parameters are :
221             #    - mode = 0 for normal RDF, 1 for its deconvolution
222             #    - pfile  is the output file (ascii/netcdf)
223             #    - atom1/2 give the number of the atom in typat
224             #    - box : give the number of box wide for the calculculation of g(r)
225             #    - pdr   : give the radial precision of the calculation
226             #    - pstep : give the self.increment for the step loop
227     
228             file = pfile
229     
230             pos = file.getXCart()          # position of atom (cartesian)
231     
232             inc = pstep                    # incrementation
233     
234             acell = file.getAcell()        # acell
235             a = acell[:,0]
236             b = acell[:,1]
237             c = acell[:,2]
238             a_max = max(acell[:,0])
239             b_max = max(acell[:,1])
240             c_max = max(acell[:,2])
241     
242             rprim = zeros((3,3))           # primitives vectors of the cell
243             rprim[0] = file.getRPrim()[0,0]
244             rprim[1] = file.getRPrim()[0,1]
245             rprim[2] = file.getRPrim()[0,2]
246     
247             rho = 1/file.getVol()        # density
248     
249             if box >= 0:
250     
251                 f = (1/2.+box)           # factor of maximum radius
252                 
253             else: # box < 0
254                 
255                 f = 3**0.5/2.
256     
257             rmax = f*min(a_max,b_max,c_max)
258     
259             deltaR = pdr                      # delta r  (Bohr)
260     
261             maxbin = int((rmax/deltaR))       # number of iteration
262     
263             typat = file.getTypat()           # Type of particule
264     
265             if atom1 == 0:
266                 indexAtom1 = array([i for i,x in enumerate(typat) if x == 1 and 2],dtype=int) + 1
267             else:
268                 indexAtom1 = array([i for i,x in enumerate(typat) if x == atom1],dtype=int) + 1 #get list of the index of typat1 (+1 for fortran)
269     
270                 
271             if atom2 == 0:
272                 indexAtom2 = array([i for i,x in enumerate(typat) if x == 1 and 2 ],dtype=int) + 1
273             else:
274                 indexAtom2 = array([i for i,x in enumerate(typat) if x == atom2],dtype=int) + 1 #get list of the index of typat2 (+1 for fortran)
275                     
276             nbtime =  len(pos)                    # final step
277     
278             self.r = arange(maxbin)*deltaR
279     
280             if mode == 0:
281     
282                 self.g = topo.pair_distribution(nbtime,pos,rprim,f,inc,a,b,c,deltaR,rho,indexAtom1,indexAtom2,self.r)
283     
284                 self.r += deltaR/2.
285     
286             else: #mode == 1
287     
288                 nba = len(pos[0])                   # number of atom
289     
290                 it = (int(nbtime/inc)+1)*nba*(nba-1)
291     
292                 self.m = topo.m_distribution(0,nbtime,pos,inc,it,a,b,c,indexAtom1,indexAtom2)
293     
294         def getR(self):
295             return self.r
296     
297         def getRDF(self):
298             return self.g[0]
299     
300         def getINT(self):
301             return self.g[1]
302     
303         def getDATA(self):
304             return self.m[2]

744         #-------------Constructor--------------#
745         def __init__(self, pfiles,pni,pWDData):
746             # This class aim is to determinate the deformations used for the calculation of elastic constant
747             # the parameters are :
748             #        -self.files
749             #        -self.ni
750     
751             self.files       = pfiles
752             self.ni          = pni
753             self.datasetWD   = pWDData #Dataset without deformation
754             self.deformation = {}
755             self.goodDeformation = True
756     
757             self.calculation()
758     
759         #------------Methods--------------------#
760         def calculation(self):
761             self.rprim0 = []
762             self.sigma = {}
763             self.deformation = {}
764     
765             if self.files['WD'].getIonMove()[0] in [1,6,7,8,9,12] :
766     
767                 self.files['WD'].setNi(self.ni)
768                 sigma = self.files['WD'].getStress()
769                 sigma = sum(sigma[:,:]) / len(sigma)
770                 self.deformation['WD'] = ['WD', 0 ,sigma]
771     
772                 self.rprim_0 = self.files['WD'].getRPrim()[0,:,:]
773     
774                 for i in range(len(self.files)-1):
775                     self.files[i].setNi(self.ni)
776                     sigma = self.files[i].getStress()
777                     sigma = sum(sigma[:,:]) / len(sigma)
778     
779                     rprim = self.files[i].getRPrim()[0,:,:]
780                     d = dot(inv(self.rprim_0),rprim)
781                     self.deform(i,d,sigma)
782                 return
783     
784             if self.files['WD'].getIonMove()[0] == 0 :
785                 if 1 == 1:
786                     sigma = self.files['WD'].getStress()[self.datasetWD]
787                     self.deformation['WD'] = ['WD', 0 ,sigma]
788                     self.rprim_0 = self.files['WD'].getRPrim()[self.datasetWD]
789     
790     
791                     self.nbdataset = self.files['WD'].getNbDataset()
792                     j = 0
793     
794                     for i in range(self.nbdataset) :
795     
796                         if i != self.datasetWD :
797                             sigma = self.files['WD'].getStress()[i]
798                             rprim = self.files['WD'].getRPrim()[i]
799                             d = dot(inv(self.rprim_0),rprim)
800                             self.deform(j,d,sigma)
801                             j += 1
802     
803                     for i in range(len(self.files)-1):
804                         self.nbdataset = self.files[i].getNbDataset()
805                         for j in range(self.nbdataset) :
806     
807                             sigma = self.files[i].getStress()[j]
808                             rprim = self.files[i].getRPrim()[j]
809                             d = dot(inv(self.rprim_0),rprim)
810                             self.deform(i,d,sigma)
811                     return
812             else :
813                 self.goodDeformation = False

823

825                     self.deformation[i] = ['tetra',d[0][0]-1,sigma]
826                     return

830                 if (d[0][0] == d[1][1] and abs(d[0][2]) <= 1e-10 and abs(d[1][2]) <= 1e-10):
831                     self.deformation[i] = ['orto',round(d[0][1],4),sigma]
832                     return

835                 if ( abs(d[0][1]) <= 1e-10 and  abs(d[0][2]) <= 1e-10 and  abs(d[0][1]) <= 1e-10 and  abs(d[1][0]) <= 1e-10):
836                     if (abs(d[1][0]) <= 1e-10 and  abs(d[1][2])<= 1e-10 and  abs(d[1][0]) <= 1e-10 ):
837                         self.deformation[i] = ['iso',d[0][0]-1,sigma]
838                 return

843             self.goodDeformation = False
