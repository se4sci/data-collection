136         def test_variables_in_tests(self):
137             """
138             Find variables that are not tested by comparing the database of variables
139             with the input file of the test.
140             Build dictionary with list of untested variables for the different codes.
141             Finally compare the new dictionary with the reference one and fail if they don't match.
142             """
143             # Build database with all input variables indexed by code name.
144             from abimkdocs.variables import get_codevars
145             codevars = get_codevars()
146     
147             # Build AbinitTestSuite object.
148             from doc import tests as tmod
149             tests = tmod.abitests.select_tests(suite_args=[], regenerate=True, flat_list=True)
150     
151             # Build conter for the different codes, keys are the varnames from the database.
152             from collections import Counter
153             count_code = {}
154             for code, d in codevars.items():
155                 count_code[code] = Counter({k: 0 for k in d})
156     
157             ierr = 0
158             doc_vnames = codevars["anaddb"].get_all_vnames(with_internal=False)
159             anaddb_f90vnames = self.get_anaddb_varnames_from_f90()
160             #print("anaddb_vames:", anaddb_f90vnames)
161             diff = anaddb_f90vnames - doc_vnames
162             if diff:
163                 ierr += 1
164                 print("The following variables are seen in anaddb F90 code but not in variables_anaddb.py")
165                 pprint(diff)
166     
167             diff = doc_vnames - anaddb_f90vnames
168             if diff:
169                 ierr += 1
170                 print("The following variables are seen in variables_anaddb.py but not in anaddb F90 code.")
171                 pprint(diff)
172     
173             doc_vnames = codevars["abinit"].get_all_vnames(with_internal=False)
174             abinit_f90vnames = self.get_abinit_varnames_from_f90()
175             #print("abinit_f90vnames:", abinit_f90vnames)
176     
177             diff = abinit_f90vnames - doc_vnames
178             if diff:
179                 ierr += 1
180                 print("The following variables are seen in abinit F90 code but not in variables_abinit.py")
181                 pprint(diff)
182     
183             diff = doc_vnames - abinit_f90vnames
184             if diff:
185                 ierr += 1
186                 print("The following variables are seen in variables_abinit.py but not in abinit F90 code")
187                 pprint(diff)
188             #assert 0
189     
190             # TODO: should parse chkvars and
191             black_list = set([
192                 "atompaw", "cut3d", "multibinit", "fftprof", "conducti", "mrgscr",
193                 "mrgddb", "mrggkk", "mrgdv", "band2eps", "ujdet", "fold2Bloch", "macroave",
194             ])
195             for test in tests:
196                 if test.executable in black_list: continue
197                 vnset = test.get_varname_set()
198                 count_code[test.executable].update(vnset)
199     
200             untested = OrderedDict()
201             for code, count in count_code.items():
202                 untested[code] = []
203                 # Add it if var is not tested and not internal.
204                 for vname, c in count.items():
205                     if c == 0 and not codevars[code][vname].is_internal:
206                         #print(code, vname)
207                         untested[code].append(vname)
208     
209             for code in sorted(untested.keys()):
210                 untested[code] = sorted(untested[code])
211                 if untested[code]:
212                     print("List of untested variables for code:", code)
213                     pprint(untested[code])
214             #assert 0
215     
216             ref_json_path = os.path.join(os.path.dirname(__file__), "untested_variables.json")
217             update_ref = False
218             if update_ref:
219                 with open(ref_json_path, "wt") as fh:
220                     json.dump(untested, fh, indent=4)
221             else:
222                 with open(ref_json_path, "rt") as fh:
223                     ref_untested = json.load(fh)
224     
225                 self.assertDictEqual(untested, ref_untested,
226                     msg="Detected mismatch between reference file and new list of untested variables.")
