47     Ab7Invars* ab7_invars_new_from_file(const char *filename)
48     {
49       int n, npsp;
50       Ab7Invars *dt, tmpDt;
51       char buf[8] = "        ";
52       
53       n = strlen(filename);
54       npsp = 0;
55     
56       VARS_CALL(new_from_file, NEW_FROM_FILE, &tmpDt, filename, &n, buf, &npsp);
57       if (tmpDt > 0)
58         {
59           dt = malloc(sizeof(Ab7Invars));
60           *dt = tmpDt;
61         }
62       else
63         dt = (Ab7Invars*)0;
64       return dt;
65     }
66     Ab7Invars* ab7_invars_new_from_file_with_pseudo(const char *filename, const char **pspfiles)
67     {
68       int n, npsp;
69       Ab7Invars *dt, tmpDt;
70       char *psp;
71       size_t i, len;
72       
73       n = strlen(filename);
74       npsp = 0;
75       if (pspfiles)
76         for (npsp = 0; pspfiles[npsp]; npsp++);
77     
78       if (npsp)
79         {
80           len = sizeof(char) * npsp * 264;
81           psp = malloc(len);
82           for (npsp = 0; pspfiles[npsp]; npsp++)
83             strncpy(psp + 264 * npsp, pspfiles[npsp], 264);
84           for (i = 0; i < len; i++)
85             if (psp[i] == '\0')
86               psp[i] = ' ';
87         }
88       
89       VARS_CALL(new_from_file, NEW_FROM_FILE, &tmpDt, filename, &n, psp, &npsp);
90       if (tmpDt > 0)
91         {
92           dt = malloc(sizeof(Ab7Invars));
93           *dt = tmpDt;
94         }
95       else
96         dt = (Ab7Invars*)0;
97       return dt;
98     }

132     {
133       Ab7Error ab7_errno;
134     
135       if (AB7_INVARS_TYPE(id) != _INT_SCALAR) return AB7_ERROR_INVARS_ATT;
136       VARS_CALL(get_integer, GET_INTEGER, dt, value, (int*)&id, &idtset, &ab7_errno);
