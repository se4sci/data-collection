26     /* - Function: void mtrace (void)
27      * When the mtrace function is called it looks for an environment variable named MALLOC_TRACE. 
28      * This variable is supposed to contain a valid file name. The user must have write access. 
29      * If the file already exists it is truncated. If the environment variable is not set or it does 
30      * not name a valid file which can be opened for writing nothing is done. The behavior of malloc etc. is not changed. 
31      * If the named file is successfully opened, mtrace installs special handlers for the functions malloc, realloc, and free 
32      * (see Hooks for Malloc). From then on, all uses of these functions are traced and protocolled into the file. 
33      * There is now of course a speed penalty for all calls to the traced functions so tracing should not be enabled during 
34      * normal use. 
35      * This function is a GNU extension and generally not available on other systems. The prototype can be found in mcheck.h. 
36      */
37     void 
38     FC_FUNC_(clib_mtrace, CLIB_MTRACE)
39       (int *ierr)
40     {
41     #ifdef HAVE_MCHECK_H
42       mtrace();
43       *ierr=0;
44     #else
45       *ierr=1;
46     #endif
47     }
48     
49     /* - Function: void muntrace (void)
50      * The muntrace function can be called after mtrace was used to enable tracing the malloc calls. 
51      * If no (successful) call of mtrace was made muntrace does nothing. 
52      * Otherwise it deinstalls the handlers for malloc, realloc, and free and then closes the protocol file. 
53      * No calls are protocolled anymore and the program runs again at full speed. 
54      * Please note that not only the application uses the traced functions, also libraries (including the C library itself) 
55      * use these functions.
56      * 
57      * It is no good idea to call muntrace before the program terminated. The libraries are informed about 
58      * the termination of the program only after the program returns from main or calls exit and so cannot 
59      * free the memory they use before this time. 
60      * 
61      * So the best thing one can do is to call mtrace as the very first function in the program and never call muntrace. 
62      * So the program traces almost all uses of the malloc functions (except those calls which are executed by 
63      * constructors of the program or used libraries).
64      * 
65      * This function is a GNU extension and generally not available on other systems. The prototype can be found in mcheck.h.
66      */
67     void 
68     FC_FUNC_(CLIB_muntrace,CLIB_MUNTRACE) 
69       (int* ierr)
70     {
71     #ifdef HAVE_MCHECK_H
72       muntrace();
73       *ierr=0;
74     #else
75       *ierr=1;
76     #endif
77     }

79     /*  
80      * You can ask malloc to check the consistency of dynamic memory by using the mcheck function. 
81      * This function is a GNU extension, declared in mcheck.h.
82      * Calling mcheck tells malloc to perform occasional consistency checks. These will catch things such as 
83      * writing past the end of a block that was allocated with malloc.
84      * It is too late to begin allocation checking once you have allocated anything with malloc. 
85      * So mcheck does nothing in that case. The function returns -1 if you call it too late, 
86      * and 0 otherwise (when it is successful).
87      * function prototype: mcheck (void (*abortfn) (enum mcheck_status status))
88      */
89     void 
90     FC_FUNC_(clib_mcheck,CLIB_MCHECK)
91       (int* ierr)
92     {
93     #ifdef HAVE_MCHECK_H
94       *ierr = mcheck (NULL);
95     #else
96       *ierr = 2;
97     #endif
98     }
