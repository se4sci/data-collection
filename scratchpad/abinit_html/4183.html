8       "HAVE_ALGO_LEVMAR":"HAVE_LEVMAR",
9       "HAVE_DFT_ATOMPAW":"HAVE_ATOMPAW",
10       "HAVE_DFT_BIGDFT":"HAVE_BIGDFT",
11       "HAVE_DFT_LIBXC":"HAVE_LIBXC",
12       "HAVE_DFT_WANNIER90":"HAVE_WANNIER90",
13       "HAVE_DFT_WANNIER90_V1":"HAVE_WANNIER90_V1",
14       "HAVE_MATH_GSL":"HAVE_GSL",
15       "HAVE_MATH_MLIB":"HAVE_LINALG_MLIB",
16       "HAVE_TIMER_PAPI":"HAVE_PAPI",
17       "HAVE_TRIO_ETSF_IO":"HAVE_ETSF_IO",
18       "HAVE_TRIO_NETCDF":"HAVE_NETCDF",
19       "HAVE_TRIO_NETCDF_MPI":"HAVE_NETCDF_MPI",
20       "HAVE_TRIO_PSML":"HAVE_PSML",

25     m4 = re.compile("\.m4$")
26     chdr = re.compile("\.[ch]$")
27     fortran = re.compile("\.([Ff]|[Ff]90|finc)$")
28     pynit = re.compile("__init__.py$")
29     tinp = re.compile("\.in$")
30     cppline = re.compile("^#")
31     cppkeys = ("define .*","include.*","ifdef","ifndef","elif","^if ","else","endif","defined","undef","!","&&","\|\|","\(","\)")
32     m4line = re.compile("AC_DEFINE")
33     tline = re.compile("need_cpp_vars")
34     
35     def fix_cppopts(top):
36       cppopts = dict()
37       for root,dirs,files in os.walk(top):
38         for src in files:
39           is_fort = False
40           if ( fortran.search(src) or chdr.search(src) ):
41             is_fort = True
42           is_m4 = False
43           if ( m4.search(src) ):
44             is_m4 = True
45           is_pynit = False
46           if ( pynit.search(src) ):
47             is_pynit = True
48           is_tinp = False
49           if ( tinp.search(src) ):
50             is_tinp = True
51           if ( is_fort or is_m4 or is_pynit or is_tinp ):
52             code = file(os.path.join(root,src),"r").readlines()
53             cmod = False
54     
55             for cidx in range(len(code)):
56               line = code[cidx]
57               lmod = False
58     
59               if ( is_fort and cppline.match(line) ):
60                 line = re.sub("^#","",line).strip()
61                 for kw in cppkeys:
62                   line = re.sub(kw,"",line)
63                 lmod = True
64               elif ( is_m4 and m4line.search(line) ):
65                 line = re.sub(".*AC_DEFINE","", line)
66                 line = re.sub("AC_DEFINE","", line)
67                 line = re.sub(",.*","",line)
68                 line = re.sub("[\(\[\]\)]","",line)
69                 lmod = True
70               elif ( is_pynit ):
71                 line = re.sub("[ \",]","",line)
72                 lmod = True
73               elif ( is_tinp and tline.search(line) ):
74                 line = re.sub(".*need_cpp_vars.*=[ ]*","",line)
75                 lmod = True
76     
77               if ( lmod ):
78                 line = line.split()
79                 for item in line:
80                   if ( item in renamed_keys ):
81                     cmod = True
82                     code[cidx] = re.sub(item,renamed_cppopts[item],code[cidx])
83     
84             if ( cmod ):
85               file(os.path.join(root,src),"w").write("".join(code))
86     
87       return 0
88     
89

91
