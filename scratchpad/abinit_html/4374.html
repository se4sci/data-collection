29         @classmethod
30         def from_file(cls,filename):
31             root = nc.Dataset(filename, 'r')
32             nsppol = len(root.dimensions['number_of_spins'])
33             nkpt = len(root.dimensions['number_of_kpoints'])
34             nval = len(root.dimensions['max_number_of_valence_bands'])
35             ncond = len(root.dimensions['max_number_of_conduction_bands'])
36             hsize = len(root.dimensions['total_number_of_transitions'])
37     
38             kpt = root.variables['reduced_coordinates_of_kpoints'][:,:] # Nkpt, 3
39             ham = root.variables['hamiltonian'][:,:,:] # Itp, It, [0,1]
40     
41             ham_c = np.vectorize(complex)(ham[:,:,0],ham[:,:,1])
42             vcks2t = root.variables['vcks2t'][:,:,:,:] # S, K, C, V
43     
44             lomo = root.variables['lomo'][:]
45             homo = root.variables['homo'][:]
46             lumo = root.variables['lumo'][:]
47             humo = root.variables['humo'][:]
48             diag = root.variables['diagonal'][:,:]
49             diag_c = np.vectorize(complex)(diag[:,0],diag[:,1])
50     
51             return cls(nsppol, nkpt, nval, ncond, hsize, ham_c, kpt, vcks2t, lomo, homo, lumo, humo, diag_c)

103         def get_filtered_kpt(self,xval=None,yval=None,zval=None):
104            
105            list_ikpt = []
106         
107            for ikpt in np.arange(self.nkpt):
108              if (  (xval is None or np.abs(self.kpt[ikpt,0]-xval) < 1e-8) 
109                and (yval is None or np.abs(self.kpt[ikpt,1]-yval) < 1e-8)
110                and (zval is None or np.abs(self.kpt[ikpt,2]-zval) < 1e-8)):
111                list_ikpt.append(ikpt)
112         
113            return list_ikpt
