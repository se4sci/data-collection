102         def changeMode(self):
103     
104             mode1 = self.sbm1.currentText()
105             mode2 = self.sbm2.currentText()
106             
107             self.marker_size = 2 #Set maker_size to 2 (except average)
108             self.markerscale = 5 #Set the relative size of legend marker vs original
109     
110             if self.sbrel.isChecked():
111                 reduced = True
112                 unit    = 1  # No Conversion 
113                 legend  = "" 
114             else:
115                 reduced = False
116                 unit   = self.units['Distance'][1]
117                 legend =  " ("+str(self.units['Distance'][0])+")"
118             option = 0
119     
120             if mode1 == 'Average':
121                 option = 1
122                 self.marker_size = 5 #set marker_size to 5 (better for visualization)
123                 self.markerscale = 2
124             elif mode1 == 'PBC (cubic only)':
125                 option = 2 
126     
127             for itype in range(1,self.ntypat+1):
128                 
129                 indexAtom = array([i for i,x in enumerate(self.typat) if x == itype],dtype=int)
130     
131                 self.pos_conv = Atom.convert_position(self.file,indexAtom,option=option,reduced=reduced)
132                 
133                 if mode2 == 'XY':
134                     self.x = self.pos_conv.getX()
135                     self.y = self.pos_conv.getY()
136                     self.legend_x = "x" + legend
137                     self.legend_y = "y" + legend
138     
139                 elif mode2 == 'XZ':
140                     self.x = self.pos_conv.getX()
141                     self.y = self.pos_conv.getZ()
142                     self.legend_x = "x" + legend
143                     self.legend_y = "z" + legend
144     
145                 else :
146                     self.x = self.pos_conv.getY()
147                     self.y = self.pos_conv.getZ()
148                     self.legend_x = "y" + legend
149                     self.legend_y = "z "+ legend
150                     
151                 if itype == 1:
152                     try: 
153                         self.graphPosition.updatePos(self.x*unit,self.y*unit,\
154                                                          self.legend_x,\
155                                                          self.legend_y,\
156                                                          marker_size=self.marker_size, name = self.name)
157                     except:
158                         self.graphPosition = Graph.graphic(self.x*unit,self.y*unit,\
159                                                                self.legend_x,\
160                                                                self.legend_y,\
161                                                                average=False,point = True,marker='.',\
162                                                                marker_size=self.marker_size,name = self.name)
163                 else:
164                     self.graphPosition.addPlot(self.x*unit,self.y*unit,\
165                                                    point=True,marker_size=self.marker_size)
166     
167             self.legend = []
168             for i in range(self.ntypat):
169                 self.legend.append(str(self.PTOE.getName(self.file.getZnucl()[i])))
170                 
171             self.graphPosition.addLegend(self.legend,markerscale=self.markerscale)
172             self.graphPosition.show()
