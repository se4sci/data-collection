27     from __future__ import print_function
28     import numpy as np
29     import netCDF4 as nc
30     
31     __version__ = '1.0.0'

33     def merge_ddb_nc(out_fname, fnames):
34         """
35         Merge a list of DDB.nc files containing different elements of the same qpoint.
36     
37         Arguments
38         ---------
39     
40         out_fname: Name for the merged file (will overwrite any existing file).
41         fnames: List of DDB.nc files.
42     
43         """
44         if not fnames:
45             raise Exception('Empty list of files given for merge')
46     
47         fname0 = fnames.pop(0)
48     
49         with nc.Dataset(out_fname, 'w') as dsout:
50             with nc.Dataset(fname0, 'r') as dsin:
51                 nc_copy(dsin, dsout)
52                 q0 = dsin.variables[u'q_point_reduced_coord'][...]
53     
54             for fname in fnames:
55                 with nc.Dataset(fname, 'r') as dsin:
56     
57                     # Check that the qpoints are the same
58                     q = dsin.variables[u'q_point_reduced_coord'][...]
59                     if not all(np.isclose(q0, q)):
60                         raise Exception('Cannot merge DDB.nc at different q-points.')
61     
62                     # Merge dynamical matrix
63                     dynmat = dsin.variables[u'second_derivative_of_energy'][...]
64                     dynmat_mask = dsin.variables[u'second_derivative_of_energy_mask'][...]
65     
66                     out_dynmat = dsin.variables[u'second_derivative_of_energy']
67                     out_dynmat_mask = dsin.variables[u'second_derivative_of_energy_mask']
68     
69                     ni,nj,nk,nl = dynmat_mask.shape
70     
71                     for i in range(ni):
72                      for j in range(nj):
73                       for k in range(nk):
74                        for l in range(nl):
75                         if dynmat_mask[i,j,k,l]:
76                             dsout.variables[u'second_derivative_of_energy'][i,j,k,l,:] = (
77                                 dynmat[i,j,k,l,:])
78     
79                             dsout.variables[u'second_derivative_of_energy_mask'][i,j,k,l] = (
80                                 dynmat_mask[i,j,k,l])
81     
82                     # Born effective charge tensor
83                     BECT = dsin.variables[u'born_effective_charge_tensor'][...]
84                     BECT_mask = dsin.variables[u'born_effective_charge_tensor_mask'][...]
85     
86                     ni,nj,nk = BECT_mask.shape
87     
88                     for i in range(ni):
89                      for j in range(nj):
90                       for k in range(nk):
91                         if BECT_mask[i,j,k]:
92                             dsout.variables[u'born_effective_charge_tensor'][i,j,k] = (
93                                 BECT[i,j,k])
94     
95                             dsout.variables[u'born_effective_charge_tensor_mask'][i,j,k] = (
96                                 BECT_mask[i,j,k])
97     
98     
99     def nc_copy(dsin, dsout):
100         """
101         Copy all dimensions and variable of one nc.Dataset instance into another.
102         """
103     
104         #Copy dimensions
105         for dname, dim in dsin.dimensions.iteritems():
106             dsout.createDimension(dname, len(dim))
107     
108         #Copy variables
109         for vname, varin in dsin.variables.iteritems():
110             outVar = dsout.createVariable(vname, varin.datatype, varin.dimensions)
111             outVar[...] = varin[...]

114     def interactive_merge_ddb_nc():
115         """Get inputs from the user and run merge_ddb_nc."""
116     
117         program_name = 'merge_ddb_nc'
118         description = """Merge several DDB.nc files, belonging to the same q-point."""
119     
120         def get_user(s):
121             return raw_input(s.rstrip() + '\n').split('#')[0]
122     
123     
124         print(program_name)
125         print(len(program_name) * '-')
126         print(description + '\n')
127     
128         ui = get_user('Enter a name for the output file in which to merge (will overwrite any existing file):')
129         out_fname = str(ui)
130     
131         ui = get_user('Enter the number of files to merge:')
132         nfiles = int(ui)
133     
134         fnames = list()
135         for i in range(nfiles):
136             ui = get_user('Enter the name of file {}:'.format(i+1))
137             fname = str(ui)
138             fnames.append(fname)
139     
140         # Main execution
141         print('Executing...')
142         merge_ddb_nc(out_fname, fnames)
143     
144         print('All done.')
145
