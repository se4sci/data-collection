98     def stream_has_colours(stream):
99         """
100         True if stream supports colours. Python cookbook, #475186
101         """
102         if not hasattr(stream, "isatty"):
103             return False
104     
105         if not stream.isatty():
106             return False  # auto color only on TTYs
107         try:
108             import curses
109             curses.setupterm()
110             return curses.tigetnum("colors") > 2
111         except:
112             return False  # guess false in case of error

115     def colored(text, color=None, on_color=None, attrs=None):
116         """Colorize text.
117     
118         Available text colors:
119             red, green, yellow, blue, magenta, cyan, white.
120     
121         Available text highlights:
122             on_red, on_green, on_yellow, on_blue, on_magenta, on_cyan, on_white.
123     
124         Available attributes:
125             bold, dark, underline, blink, reverse, concealed.
126     
127         Example:
128             colored('Hello, World!', 'red', 'on_grey', ['blue', 'blink'])
129             colored('Hello, World!', 'green')
130         """
131     
132         if __ISON and os.getenv('ANSI_COLORS_DISABLED') is None:
133             fmt_str = '\033[%dm%s'
134             if color is not None:
135                 text = fmt_str % (COLORS[color], text)
136     
137             if on_color is not None:
138                 text = fmt_str % (HIGHLIGHTS[on_color], text)
139     
140             if attrs is not None:
141                 for attr in attrs:
142                     text = fmt_str % (ATTRIBUTES[attr], text)
143     
144             text += RESET
145         return text

148     def cprint(text, color=None, on_color=None, attrs=None, **kwargs):
149         """Print colorize text.
150     
151         It accepts arguments of print function.
152         """
153         try:
154             print((colored(text, color, on_color, attrs)), **kwargs)
155         except TypeError:
156             # flush is not supported by py2.7
157             kwargs.pop("flush", None)
158             print((colored(text, color, on_color, attrs)), **kwargs)

161     def colored_map(text, cmap):
162         """
163         Return colorized text. cmap is a dict mapping tokens to color options.
164     
165         .. Example:
166     
167             colored_key("foo bar", {bar: "green"})
168             colored_key("foo bar", {bar: {"color": "green", "on_color": "on_red"}})
169         """
170         if not __ISON: return text
171         for key, v in cmap.items():
172             if isinstance(v, dict):
173                 text = text.replace(key, colored(key, **v))
174             else:
175                 text = text.replace(key, colored(key, color=v))
176         return text
177     
178     
179     def cprint_map(text, cmap, **kwargs):
180         """
181         Print colorize text. 
182         cmap is a dict mapping keys to color options. 
183         kwargs are passed to print function
184     
185         Example:
186             cprint_map("Hello world", {"Hello": "red"})
187         """
188         try:
189             print(colored_map(text, cmap), **kwargs)
190         except TypeError:
191             # flush is not supported by py2.7
192             kwargs.pop("flush", None)
193             print(colored_map(text, cmap), **kwargs)

196     def get_terminal_size():
197         """"
198         Return the size of the terminal as (nrow, ncols)
199         
200         Based on: 
201     
202             http://stackoverflow.com/questions/566746/how-to-get-console-window-width-in-python
203         """
204         try:
205             rc = os.popen('stty size', 'r').read().split()
206             return int(rc[0]), int(rc[1])
207         except:
208             pass
209     
210         env = os.environ
211         def ioctl_GWINSZ(fd):
212             try:
213                 import fcntl, termios, struct
214                 rc = struct.unpack('hh', fcntl.ioctl(fd, termios.TIOCGWINSZ, '1234'))
215                 return rc
216             except:
217                 return None
218     
219         rc = ioctl_GWINSZ(0) or ioctl_GWINSZ(1) or ioctl_GWINSZ(2)
220     
221         if not rc:
222             try:
223                 fd = os.open(os.ctermid(), os.O_RDONLY)
224                 rc = ioctl_GWINSZ(fd)
225                 os.close(fd)
226             except:
227                 pass
228     
229         if not rc:
230             rc = (env.get('LINES', 25), env.get('COLUMNS', 80))
231     
232         return int(rc[0]), int(rc[1])
