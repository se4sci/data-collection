20     def detectfile(filename,path): # type(filename) = type(path) = string
21         # method detectfile returns True if the specified file is found in the specified path
22         return filename in os.listdir(path)
23     
24     def clean(list): # type(list) = list of strings
25         # method clean removes character strings '\n' and '\r' and empty lines from a string list
26         # (the string list is usually obtained with the ".readlines()" method)
27         L = len(list)
28         for i in range(L):
29             list[L-1-i] = list[L-1-i].replace('\n','')
30             list[L-1-i] = list[L-1-i].replace('\r','')
31             if list[L-1-i].split() == []:
32                 list.pop(L-1-i)
