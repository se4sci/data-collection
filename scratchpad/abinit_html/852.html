98     /* Macros for memory checking and profiling
99      * TODO
100      * Add option to check automatically the exit status of allocate so that we can abort gracefully if oom.
101      * and abipy can detect the problem.
102      *
103      *   ABI_ALLOCATE   --> Allocate memory for intrinsic datatypes (real, integer, complex).
104      *   ABI_CALLOC     --> Clear alloc: same as ABI_ALLOCATE but initializes memory with zeros
105      *   ABI_DEALLOCATE --> Free memory allocated by ABI_ALLOCATE
106      *
107      * To allocate/deallocate arrays of user-defined type use:
108      *
109      *   ABI_DT_ALLOCATE
110      *   ABI_DT_DEALLOCATE
111      *
112      * Note for programmers using OpenMP:
113      *   Do not use stat=ABI_ALLOC_STAT_ABI inside an OpenMP parallel region
114      *   ABI_ALLOC_STAT_ABI is a global variable defined in m_profiling_abi.F90
115      *   and this can have a detrimental effect if the allocation is done inside an OpenMP parallel region.
116      *
117      #define HAVE_MEM_PROFILING
118     */
119     
120     #ifdef HAVE_MEM_PROFILING
121     
122     /* 
123      These macros are used to get the memory address of the objet and the memory allocated.
124     
125        - loc returns the address and is a language extension supported by gfortran and ifort.
126        - storage_size was introduced in F2003 
127     
128      Both loc and storage_size are polymorphic so one can use it with intrinsic types as well 
129      as user-defined datatypes.
130     */
131     #  define _LOC(x)  int(loc(x), kind=8) 
132     #  define _MEM(arr) product(int(shape(arr), kind=8)) * storage_size(arr, kind=8)/8
133     
134     /* and now the debugging macros */
135     #  define ABI_ALLOCATE(ARR, SIZE) \
136        allocate(ARR SIZE) NEWLINE \
137        call abimem_record(0, QUOTE(ARR), _LOC(ARR), "A", _MEM(ARR),  __FILE__, ABI_FUNC, __LINE__)
138     
139     #  define ABI_DEALLOCATE(ARR) \
140        call abimem_record(0, QUOTE(ARR), _LOC(ARR), "D", - _MEM(ARR), __FILE__, ABI_FUNC, __LINE__) NEWLINE \
141        deallocate(ARR) 
142     
143     #  define ABI_STAT_ALLOCATE(ARR,SIZE,ierr) \
144        allocate(ARR SIZE, stat=ierr) NEWLINE \
145        call abimem_record(0, QUOTE(ARR), _LOC(ARR), "A", _MEM(ARR),  __FILE__, ABI_FUNC, __LINE__)
146     
147     #  define ABI_DATATYPE_ALLOCATE(ARR,SIZE) \
148        allocate(ARR SIZE) NEWLINE \
149        call abimem_record(0, QUOTE(ARR), _LOC(ARR), "A", _MEM(ARR),  __FILE__, ABI_FUNC, __LINE__)
150     
151     #  define ABI_DATATYPE_DEALLOCATE(ARR)  \
152        call abimem_record(0, QUOTE(ARR), _LOC(ARR), "D", - _MEM(ARR), __FILE__, ABI_FUNC, __LINE__) NEWLINE \
153        deallocate(ARR) 
154     
155     #else
156     /* macros used in production */
157     #  define ABI_ALLOCATE(ARR,SIZE) allocate(ARR SIZE)
158     #  define ABI_DEALLOCATE(ARR)  deallocate(ARR)
159     #  define ABI_STAT_ALLOCATE(ARR,SIZE,ierr) allocate(ARR SIZE, stat=ierr)
160     #  define ABI_DATATYPE_ALLOCATE(ARR,SIZE)  allocate(ARR SIZE)
161     #  define ABI_DATATYPE_DEALLOCATE(ARR)   deallocate(ARR)
162     #endif
