5     def get_bose(omega, temperatures):
6         """
7         Get the Bose-Einstein occupations on a range of temperatures
8         for a given boson frequency omega (in Hartree).
9     
10         Arguments:
11             omega: float
12             temperatures: [ntemp]
13     
14         Returns: bose [ntemp]
15         """
16     
17         #if not temperatures:
18         #    ntemp = 1
19         #    temperatures = [0.]
20         #else:
21         ntemp = len(temperatures)
22         bose = zeros(ntemp)
23     
24         omega = np.float(omega)
25         if omega < tol6:
26             return bose
27     
28         upper_limit = 50.
29         
30         for tt, T in enumerate(temperatures):
31         
32             if T < tol6:
33                 continue
34         
35             x = omega / (kb_HaK * T)
36         
37             if x > upper_limit:
38                 continue
39         
40             bose[tt] = 1. / (np.exp(x) - 1)
41     
42         return bose
