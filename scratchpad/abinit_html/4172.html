26     def build_url(label, base, end):
27         """ Build a url from the label, a base, and an end. """
28         clean_label = re.sub(r'([ ]+_)|(_[ ]+)|([ ]+)', '_', label)
29         #print(clean_label)
30         variables = website.variables_code["abinit"]
31     
32         if clean_label in variables:
33             var = variables[clean_label]
34             return "/input_variables/%s/#%s" % (var.varset, var.name)
35     
36         if clean_label in website.bib_data.entries:
37             return "/bibliography/#%s" % clean_label
38     
39         return '%s%s%s' % (base, clean_label, end)

42     class WikiLinkExtension(Extension):
43     
44         def __init__(self, *args, **kwargs):
45             self.config = {
46                 'base_url': ['/', 'String to append to beginning or URL.'],
47                 'end_url': ['/', 'String to append to end of URL.'],

55             self.md = md

65     class WikiLinks(Pattern):
66         def __init__(self, pattern, config):
67             super(WikiLinks, self).__init__(pattern)
68             self.config = config
69     
70         def handleMatch(self, m):
71             token = m.group(2)
72             #base_url, end_url, html_class = self._getMeta()
73             #url = self.config['build_url'](token, base_url, end_url)
74             #page_rpath = "??"
75             #if hasattr(self.md, 'Meta') and "rpath" in self.md.Meta:
76             #    page_rpath = self.md.Meta["rpath"][0]
77             # Remove quotes (neeeded in py2.7 because mkdocs does not use pyyaml to parse meta).
78             #if "authors" in self.md.Meta:
79             #    print("authors", self.md.Meta["authors"])
80             website = Website.get()
81             page_rpath = None
82             try:
83                 page_rpath = self.md.Meta["rpath"][0].replace("'", "").replace('"', "")
84                 return website.get_wikilink(token, page_rpath)
85             except Exception as exc:
86                 #print("Meta", self.md.Meta)
87                 website.warn("Exception `%s:%s`\nwhile treating wikilink token: `%s` in `%s`" %
88                     (exc.__class__, str(exc), token, page_rpath))
89                 return ""
90     
91         def _getMeta(self):
92             """ Return meta data or config data. """
93             base_url = self.config['base_url']
94             end_url = self.config['end_url']
95             html_class = self.config['html_class']
96             if hasattr(self.md, 'Meta'):
97                 if 'wiki_base_url' in self.md.Meta:
98                     base_url = self.md.Meta['wiki_base_url'][0]
99                 if 'wiki_end_url' in self.md.Meta:
100                     end_url = self.md.Meta['wiki_end_url'][0]
101                 if 'wiki_html_class' in self.md.Meta:
102                     html_class = self.md.Meta['wiki_html_class'][0]
103             return base_url, end_url, html_class
