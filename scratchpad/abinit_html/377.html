98     static int dtsets_init(Dtsets *self, PyObject *args, PyObject *kwds)
99     {
100       const char* filename;
101       static char *kwlist[] = {"filename", NULL};
102       pthread_t parse_ab;
103       Ab7Invars *dt;
104     
105       if (! PyArg_ParseTupleAndKeywords(args, kwds, "s", kwlist, 
106     				    &filename))
107         return -1;
108     
109     
110       Py_BEGIN_ALLOW_THREADS
111         pthread_create(&parse_ab, NULL, _parse_ab, (void*)filename);
112       pthread_join(parse_ab, (void**)&dt);
113       DBG_printf("Parsing thread return %d.\n", _parse_error);
114       Py_END_ALLOW_THREADS
115         
116         if (dt == PTHREAD_CANCELED)
117           {
118     	PyErr_SetString(PyExc_RuntimeError, "Abinit thread cancelled.");
119     	return -1;
120           }
121     
122       switch (_parse_error)
123         {
124         case (1):
125           PyErr_SetString(PyExc_SyntaxError, _parse_message);
126           free(_parse_message);
127           return -1;
128         case (2):
129           PyErr_Format(PyExc_TypeError, "File '%s' is not an ABINIT file.", filename);
130           return -1;
131         case (3):
132           PyErr_SetString(PyExc_SystemExit, "Abinit called leave_new() routine.");
133           return -1;
134         default:
135           break;
136         }
137     
138       self->dt = dt;
139       DBG_printf("Parse OK -> %d.\n", *self->dt);
140     
141       Py_XDECREF(self->filename);
142       self->filename = PyString_FromString(filename);
143       
144       return 0;
145     }

174     static PyObject* dtsets_get(Dtsets *self, PyObject *args)
175     {
176       int idtset, id, type, rdInt, i, tmp;
177       size_t n;
178       double rdDbl;
179       Ab7Error error;
180       PyObject *tuple, *o;
181       const char *strId;
182       int ndims, dims[7];
183     #ifdef HAVE_NUMPY
184       npy_intp pydims[7];
185     #endif
186     
187       if (self->dt == NULL)
188         {
189           PyErr_SetString(PyExc_AttributeError, "dt");
190           return NULL;
191         }
192     
193       DBG_printf("dtsets.get called dt = %d.\n", *self->dt);
194     
195       /* Read the string id and get it from the dictionnary. */
196       if (!PyArg_ParseTuple(args, "si", &strId, &idtset))
197         return NULL;
198     
199       tuple = PyDict_GetItemString(DictIds, strId);
200       if (!tuple)
201         {
202           PyErr_BadArgument();
203           return NULL;
204         }
205       
206       o = PyTuple_GET_ITEM(tuple, 0);
207       id = (int) ((PyIntObject*)o)->ob_ival;
208       o = PyTuple_GET_ITEM(tuple, 1);
209       type = (int) ((PyIntObject*)o)->ob_ival;
210       DBG_printf("Get value for '%s' (%d), type %d.\n", strId, id, type);
211     
212       switch (type)
213         {
214         case (_INT_SCALAR):
215           error = ab7_invars_get_integer(self->dt, id, idtset, &rdInt);
216           if (error != AB7_NO_ERROR)
217     	{
218     	  PyErr_Format(PyExc_RuntimeError, "Can't get integer value from '%s'.", strId);
219     	  return NULL;
220     	}
221           o = PyInt_FromLong(rdInt);
222           Py_INCREF(o);
223           return (PyObject*)o;
224         case (_INT_ARRAY):
225         case (_DOUBLE_ARRAY):
226           /* We get the shape. */
227           error = ab7_invars_get_shape(self->dt, &ndims, dims, id, idtset);
228           DBG_printf("Shape is %d - %d %d %d %d %d %d %d\n",
229     		 ndims, dims[0], dims[1], dims[2], dims[3],
230     		 dims[4], dims[5], dims[6]);
231           if (error != AB7_NO_ERROR)
232     	{
233     	  PyErr_SetString(PyExc_RuntimeError, "Can't get shape of array.");
234     	  return NULL;
235     	}
236           n = 1;
237           for (i = 0; i < ndims; i++) n *= dims[i];
238           DBG_printf("Total size is %d.\n", (int)n);
239           if (n > 0)
240     	{
241     #ifdef HAVE_NUMPY
242     	  /* Copy and reverse dims into pydims. */
243     	  for (i = 0; i < ndims; i++)
244     	    pydims[i] = (npy_intp)dims[ndims - i - 1];
245     #else
246     	  /* Reverse only. */
247     	  for (i = 0; i < ndims / 2; i++)
248     	    {tmp = dims[ndims - i - 1]; dims[ndims - i - 1] = dims[i]; dims[i] = tmp;}
249     #endif
250     	  if (type == _INT_ARRAY)
251     	    {
252     #ifdef HAVE_NUMPY
253     	      o = PyArray_SimpleNew(ndims, pydims, PyArray_INT);
254     #else
255     	      o = PyArray_FromDims(ndims, dims, PyArray_INT);
256     #endif
257     	      DBG_printf("PyArray allocation OK (%d).\n", (int)n);
258     	      error = ab7_invars_get_integer_array(self->dt,
259     #ifdef HAVE_NUMPY
260     						   (int*)(PyArray_DATA(o)),
261     #else
262     						   (int*)((PyArrayObject*)o)->data,
263     #endif
264     						   n, id, idtset);
265     	      DBG_printf("Read %d.\n", error);
266     	    }
267     	  else
268     	    {
269     #ifdef HAVE_NUMPY
270     	      o = PyArray_SimpleNew(ndims, pydims, PyArray_DOUBLE);
271     #else
272     	      o = PyArray_FromDims(ndims, dims, PyArray_DOUBLE);
273     #endif
274     	      DBG_printf("PyArray allocation OK (%d).\n", (int)n);
275     	      error = ab7_invars_get_real_array(self->dt,
276     #ifdef HAVE_NUMPY
277     						(double*)(PyArray_DATA(o)),
278     #else
279     						(double*)((PyArrayObject*)o)->data,
280     #endif
281     						n, id, idtset);
282     	      DBG_printf("Read %d.\n", error);
283     	    }
284     	  if (error != AB7_NO_ERROR)
285     	    {
286     	      Py_DECREF(o);
287     	      PyErr_SetString(PyExc_RuntimeError, "Can't get values of array.");
288     	      return NULL;
289     	    }      
290     	}
291           else
292     	{
293     	  DBG_printf("Create null dimension array.\n");
294     	  dims[0] = 0;
295     	  if (type == _INT_ARRAY)
296     	    o = PyArray_FromDims(1, dims, PyArray_INT);
297     	  else
298     	    o = PyArray_FromDims(1, dims, PyArray_DOUBLE);
299     	  DBG_printf("OK.\n");
300     	}
301           Py_INCREF(o);
302           return (PyObject*)o;
303         case (_DOUBLE_SCALAR):
304           error = ab7_invars_get_real(self->dt, id, idtset, &rdDbl);
305           if (error != AB7_NO_ERROR)
306     	{
307     	  PyErr_SetString(PyExc_NotImplementedError, "Can't get number of datasets.");
308     	  return NULL;
309     	}
310           o = PyFloat_FromDouble(rdDbl);
311           Py_INCREF(o);
312           return (PyObject*)o;
313         }
314       
315       Py_RETURN_NONE;
316     }

411     void FC_FUNC(wrtout, WRTOUT)(int *unit,char message[500])
412     {
413       char *buf, *ptError, *ptInvars0, *ptInstrng, *ptSize, *pt;
414     
415       buf = my_strndup(message, 500);
416       pt = buf + 498;
417     /*   while (buf[0] == ' ') buf++; */
418       while (pt[0] == ' ' || pt[0] == '\0') pt--;
419       pt[1] = '\0';
420     
421       /* We analyse buf. If, it contains an error, we test if it is
422          about natom in inarvs0. If so, the file is not a valid ABINIT
423          file. On the contrary, we get the message and raise an error. */
424       ptError = strstr(buf, "ERROR");
425       ptInvars0 = strstr(buf, "Input natom must be defined");
426       ptInstrng = strstr(buf, "The occurence of a tab");
427       ptSize    = strstr(buf, "The size of your input file");
428       if (ptError && !ptInvars0 && !ptInstrng && !ptSize)
429         {
430           _parse_error = 1; /* Abinit file with errors. */
431           _parse_message = buf;
432           pthread_exit(NULL);
433         }
434       else if (ptError && (ptInvars0 || ptInstrng || ptSize))
435         {
436           _parse_error = 2; /* Not an abinit file. */
437           free(buf);
438           pthread_exit(NULL);
439         }
440       else
441         _parse_error = 0;
442       free(buf);
443     }

462     char *
463     my_strndup (char const *s, size_t n)
464     {
465       size_t len = my_strnlen (s, n);
466       char *new = malloc (len + 1);
467     
468       if (new == NULL)
469         return NULL;
470     
471       new[len] = '\0';
472       return memcpy (new, s, len);
473     }

480     /* Find the length of STRING, but scan at most MAXLEN characters.
481        If no '\0' terminator is found in that many characters, return MAXLEN.  */
482     
483     size_t
484     my_strnlen (const char *string, size_t maxlen)
485     {
486       const char *end = memchr (string, '\0', maxlen);
487       return end ? (size_t) (end - string) : maxlen;
488     }
