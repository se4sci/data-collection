20     def find_top_build_tree(start_path, with_abinit=True, ntrials=10):
21         """
22         Returns the absolute path of the ABINIT build tree.
23         Assume start_path is within the build tree.
24     
25         Raises:
26             `RuntimeError` if build tree is not found after ntrials attempts.
27         """
28         abs_path = os.path.abspath(start_path)
29     
30         trial = 0
31         while trial <= ntrials:
32             config_h = os.path.join(abs_path, "config.h")
33             abinit_bin = os.path.join(abs_path, "src", "98_main", "abinit")
34             # Check if we are in the top of the ABINIT source tree
35             if with_abinit:
36                 found = os.path.isfile(config_h) and os.path.isfile(abinit_bin)
37             else:
38                 found = os.path.isfile(config_h)
39     
40             if found:
41                 return abs_path
42             else:
43                 abs_path, tail = os.path.split(abs_path)
44                 trial += 1
45     
46         raise RuntimeError("Cannot find the ABINIT build tree after %s trials" % ntrials)

49     def get_array_from_path(path):
50         """Parse text file with data in tabular form, return numpy array."""
51         data = []
52         with open(path, "rt") as fh:
53             for l in fh:
54                 l = l.strip()
55                 if l.startswith("#") or not l: continue
56                 data.append(list(map(float, l.split())))
57     
58         return np.array(data).T.copy()

61     def get_axarray_fig_plt(ax_array, nrows=1, ncols=1, sharex=False, sharey=False,
62                             squeeze=True, subplot_kw=None, gridspec_kw=None,
63                             **fig_kw):
64         """
65         Helper function used in plot functions that accept an optional array of Axes
66         as argument. If ax_array is None, we build the `matplotlib` figure and
67         create the array of Axes by calling plt.subplots else we return the
68         current active figure.
69     
70         Returns:
71             ax: Array of :class:`Axes` objects
72             figure: matplotlib figure
73             plt: matplotlib pyplot module.
74         """
75         import matplotlib.pyplot as plt
76     
77         if ax_array is None:
78             fig, ax_array = plt.subplots(nrows=nrows, ncols=ncols, sharex=sharex,
79                                          sharey=sharey, squeeze=squeeze,
80                                          subplot_kw=subplot_kw,
81                                          gridspec_kw=gridspec_kw, **fig_kw)
82         else:
83             fig = plt.gcf()
84             ax_array = np.reshape(np.array(ax_array), (nrows, ncols))
85             if squeeze:
86                 if ax_array.size == 1:
87                     ax_array = ax_array[0]
88                 elif any(s == 1 for s in ax_array.shape):
89                     ax_array = ax_array.ravel()
90     
91         return ax_array, fig, plt

94     def compare_data(ref_data, out_data, title):
95         print("ref_data.shape:", ref_data.shape, "out_data.shape:", out_data.shape)
96         #if ref_data.shape == out_data.shape:
97         #    print(ref_data - out_data)
98     
99         nrows = min(len(ref_data), len(out_data)) - 1
100         axlist, fig, plt = get_axarray_fig_plt(None, nrows=nrows, ncols=1, sharex=True, sharey=False,
101                                                squeeze=True, subplot_kw=None, gridspec_kw=None)
102         axlist = np.array(axlist).ravel()
103     
104         # Assume file contains data in the form | x | f(x) | g(x) | ....
105         ref_xs = ref_data[0]
106         out_xs = out_data[0]
107     
108         for i, ax in enumerate(axlist):
109             ax.plot(ref_xs, ref_data[i+1], "--", label="ref" if i == 0 else None)
110             ax.plot(out_xs, out_data[i+1], ":", label="out" if i == 0 else None)
111             ax.grid(True)
112             if i == 0: ax.legend(loc="best", shadow=True)
113     
114         fig.suptitle(title)
115         plt.show()
