34     //define type for dynamical double format array 
35     typedef struct {
36       double *array;
37       size_t used;
38       size_t size;
39     } Array;
40     
41     void initArray(Array *a, size_t initialSize) {
42       a->array = (double *)malloc(initialSize * sizeof(double));
43       a->used = 0;
44       a->size = initialSize;
45     }
46     
47     void insertArray(Array *a, double element) {
48       if (a->used == a->size) {
49         a->size *= 2;
50         a->array = (double *)realloc(a->array, a->size * sizeof(double));
51       }
52       a->array[a->used++] = element;
53     }
54     
55     void freeArray(Array *a) {
56       free(a->array);
57       a->array = NULL;
58       a->used = a->size = 0;
59     }

61     void copyArraytoCArray(Array *l, double **a, size_t* size){
62       *size=0;
63       *a=(double *) malloc(sizeof(double)*l->used);
64       for(size_t i=0;i<l->used;i++){
65         (*a)[i]=l->array[i];
66         (*size)++;
67       }
68     }

107     // string to array of double.
108     void string2Array(char *input, double **farray, size_t *size) {
109       char *token = strtok(input, " \n\t");
110       Array tmp;
111       initArray(&tmp, 3);
112       while (token != NULL) {
113         insertArray(&tmp, strtod(token, NULL));
114         token = strtok(NULL, " ,\t\n");
115       }
116       copyArraytoCArray(&tmp, farray, size);
117       freeArray(&tmp);
118     }

157     void effpot_xml_getDimSystem(char *filename,int *natom,int *ntypat, int *nqpt, int *loc_nrpt,
158                                  int *tot_nrpt){
159       xmlDocPtr doc;
160       int i,iatom,irpt1,irpt2,iqpt,itypat,j,present;
161       xmlNodePtr cur,cur2;
162       xmlChar *key,*uri;
163       Array typat;
164     
165       initArray(&typat, 1);
166       *natom  = 0;  *nqpt  = 0; *loc_nrpt = 0; *tot_nrpt = 0;  *ntypat = 0;
167       iatom   = 0; irpt1   = 0; irpt2  = 0;  iqpt  = 0;  itypat = 0;
168       present = 0;
169       typat.array[0] = 0;
170     
171       doc = xmlParseFile(filename);
172       if (doc == NULL) printf(" error: could not parse file file.xml\n");
173     
174       cur = xmlDocGetRootElement(doc);
175       if (cur == NULL) {
176         fprintf(stderr," The document is empty \n");
177         xmlFreeDoc(doc);
178         return;
179       }
180       cur = cur->xmlChildrenNode;
181       while (cur != NULL) {
182         if ((!xmlStrcmp(cur->name, (const  xmlChar *) "atom"))) {
183           iatom++;
184           uri = xmlGetProp(cur, (const  xmlChar *) "mass");
185           insertArray(&typat,strtod(uri,NULL)); 
186           xmlFree(uri);
187         } 
188         if ((!xmlStrcmp(cur->name, (const  xmlChar *) "local_force_constant"))) {irpt1++;}
189         if ((!xmlStrcmp(cur->name, (const  xmlChar *) "total_force_constant"))) {irpt2++;}
190         if ((!xmlStrcmp(cur->name, (const  xmlChar *) "phonon"))) {
191           cur2 = cur->xmlChildrenNode;
192           while (cur2 != NULL) {
193             if ((!xmlStrcmp(cur2->name, (const  xmlChar *) "qpoint"))) {iqpt++;}
194             cur2 = cur2->next;
195           }
196         }
197         cur = cur->next;
198       }
199       for(i=0;i<typat.used;i++){
200         present = 0;
201         for(j=i+1;j<typat.used;j++){
202           if (typat.array[i] == typat.array[j] && typat.array[i] != 0){
203             present = 1;
204             break;
205           }
206         }
207         if(present==0 && typat.array[i] !=0){
208           itypat++;
209         }
210       }
211       xmlFreeDoc(doc);
212       freeArray(&typat);
213     
214       *natom  = iatom;
215       *nqpt   = iqpt;
216       *loc_nrpt   = irpt1;
217       *tot_nrpt   = irpt2;
218       *ntypat = itypat;
219     }
