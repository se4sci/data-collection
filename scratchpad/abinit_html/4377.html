174     def get_abinit_chars(tokens):
175     	"""Retrieve characters and frequencies for modes from anaddb output file"""
176     
177     #  find the correct occurence of Phonon which is followed by the gamma point freqs
178     	for ii in range(tokens.count('Phonon')):
179     		ind1 = tokens.index('Phonon')
180     		if tokens[ind1+1] != 'wavevector' or tokens[ind1+2] != '(reduced' :
181     			tokens.pop(ind1)
182     		else:
183     #  if gamma point
184     			if (float(tokens[ind1+5]) == 0.0 and \
185     				float(tokens[ind1+6]) == 0.0 and \
186     				float(tokens[ind1+7]) == 0.0 ) :
187     				tokens.pop(ind1)
188     			break
189     	abi_freq = []
190     	ind1=ind1+12
191     #	print tokens[ind1],tokens[ind1+1]
192     #	while re.compile('[-]?[0-9].[0-9]*[Ee]?[-]?[0-9]*').match(tokens[ind1]):
193     	while re.compile('[-]?[0-9]+.[0-9]*[Ee]?[-+]?[0-9]*').match(tokens[ind1]):
194     		abi_freq.append(float(tokens[ind1]))
195     		#print abi_freq
196     		ind1=ind1+1
197     
198     	for ii in range(tokens.count('Analysis')):
199     		ind1 = tokens.index('Analysis')
200     		if tokens[ind1+2] != 'degeneracies' or tokens[ind1+4] != 'characters':
201     			tokens.pop(ind1)
202     		else:
203     			break
204     	while not re.compile('Symmetry').match(tokens[ind1]):
205     		ind1=ind1+1
206     	imode=0
207     	abi_characters = []
208     	while not re.compile('=[=]*').match(tokens[ind1]):
209     		abi_characters.append([])
210     		while not re.compile('[0-9]').match(tokens[ind1]):
211     			ind1=ind1+1
212     		imode_file = int(tokens[ind1])-1
213     		if (imode_file != imode):
214     			print "Probably error : modes don't match ", imode, imode_file
215     		ind1=ind1+1
216     		#print "tokens now ", tokens[ind1:ind1+20]
217     		#print "abi_characters ", abi_characters
218     		degen = 1
219     		if not re.compile('degenerate').match(tokens[ind1]):
220     			while re.compile('[-]?[0-9]+\.[0-9]+').match(tokens[ind1]):
221     				charlist = int(float(tokens[ind1]))
222     				abi_characters[imode].append(charlist)
223     				ind1 = ind1+1
224     		else:
225     #  Find degeneracy of present mode
226     			while not  re.compile('[0-9]+').match(tokens[ind1]):
227     				ind1=ind1+1
228     			if re.compile('to').match(tokens[ind1+1]):
229     				ind1=ind1+2
230     			jmode = int(tokens[ind1])
231     			degen = jmode-imode
232     			
233     			ind1=ind1+1
234     			while not re.compile('[-]?[0-9]+\.[0-9]+').match(tokens[ind1]):
235     				ind1=ind1+1
236     			while re.compile('[-]?[0-9]+\.[0-9]+').match(tokens[ind1]):
237     				charlist = int(float(tokens[ind1]))
238     				abi_characters[imode].append(charlist)
239     				ind1 = ind1+1
240     			
241     			
242     # Duplicate characters for degenerate modes
243     		for ii in range(degen-1):
244     			abi_characters.append(abi_characters[imode])
245     		imode=imode+degen
246     
247     	#print "abi_characters = ", abi_characters,abi_freq
248     	return abi_characters,abi_freq

345     def find_ident(sym_elem):
346     	"""find the index of the identity operation in the symmetries"""
347     	for isym in range(len(sym_elem)):
348     		if sym_elem[isym] == [1,0,0,0,1,0,0,0,1]:
349     			return isym
350     	print "Error : ident not found"
351     	raise ValueError

353     def find_inv(sym_elem):
354     	"""find the index of the inverse operation in the symmetries, if it exists"""
355     	for isym in range(len(sym_elem)):
356     		if sym_elem[isym] == [-1,0,0,0,-1,0,0,0,-1]:
357     			return isym
358     #	print "Error : inv not found"
359     	raise ValueError

361     def find_princ_rotation(sym_elem,has_inv,iinv):
362     	"""Find the principal rotation of the group, as the highest order member which doesn't give -I"""
363     	morder = 0
364     	msym = 0
365     	for isym in range(len(sym_elem)):
366     		is_Sop = 0
367     		#if has_inv==1:
368     		#	if 
369     		# suppose that order is at most 6 for elements of sym
370     		tmpsym = [1,0,0,0,1,0,0,0,1]
371     		for iorder in range(1,7):
372     #			print "iorder= ", iorder
373     			tmpsym = sym_elem_prod(tmpsym,sym_elem[isym])
374     			if tmpsym == [-1,0,0,0,-1,0,0,0,-1]:
375     				is_Sop = 1
376     				break
377     			if tmpsym == [1,0,0,0,1,0,0,0,1]:
378     				break
379     		# if we found an S operation (axis+Inverse) dont count it
380     		#   should have a corresponding C operation
381     		#   the danger is that order(S) = 2*order(C) and we want the C
382     		#   to deterine the principal rotation
383     		if is_Sop == 1:
384     			break
385     		if iorder > morder:
386     			msym = isym
387     			morder = iorder
388     	return msym,morder

390     def sym_elem_prod (s1,s2):
391     	""" calculate (non commuting) product of s2*s1"""
392     	sym = []
393     	for ii in range(3):
394     		for jj in range(3):
395     			ss = 0
396     			for kk in range(3):
397     				ss = ss + s2[3*ii+kk]*s1[3*kk+jj]
398     			sym.append(ss)
399     	return sym
