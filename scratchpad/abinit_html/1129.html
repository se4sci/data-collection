12     class lazy_property(object):
13         """
14         lazy_property descriptor
15     
16         Used as a decorator to create lazy attributes.
17         Lazy attributes are evaluated on first use.
18         """
19     
20         def __init__(self, func):
21             self.__func = func
22             from functools import wraps
23             wraps(self.__func)(self)
24     
25         def __get__(self, inst, inst_cls):
26             if inst is None:
27                 return self
28     
29             if not hasattr(inst, '__dict__'):
30                 raise AttributeError("'%s' object has no attribute '__dict__'"
31                                      % (inst_cls.__name__,))
32     
33             name = self.__name__
34             if name.startswith('__') and not name.endswith('__'):
35                 name = '_%s%s' % (inst_cls.__name__, name)
36     
37             value = self.__func(inst)
38             inst.__dict__[name] = value
39             return value
40     
41         @classmethod
42         def invalidate(cls, inst, name):
43             """Invalidate a lazy attribute.
44     
45             This obviously violates the lazy contract. A subclass of lazy
46             may however have a contract where invalidation is appropriate.
47             """
48             inst_cls = inst.__class__
49     
50             if not hasattr(inst, '__dict__'):
51                 raise AttributeError("'%s' object has no attribute '__dict__'"
52                                      % (inst_cls.__name__,))
53     
54             if name.startswith('__') and not name.endswith('__'):
55                 name = '_%s%s' % (inst_cls.__name__, name)
56     
57             if not isinstance(getattr(inst_cls, name), cls):
58                 raise AttributeError("'%s.%s' is not a %s attribute"
59                                      % (inst_cls.__name__, name, cls.__name__))
60     
61             if name in inst.__dict__:
62                 del inst.__dict__[name]

75         """
76         Always return a list of strings, given a string or list of strings as input.

78         :Examples:
79     
80         >>> list_strings('A single string')

86         >>> list_strings(['A','list','of','strings'])

94     def splitall(path):
95         """Return list with all components of a path."""
96         allparts = []
97         while True:
98             parts = os.path.split(path)
99             if parts[0] == path:  # sentinel for absolute paths
100                 allparts.insert(0, parts[0])
101                 break
102             elif parts[1] == path: # sentinel for relative paths
103                 allparts.insert(0, parts[1])
104                 break
105             else:
106                 path = parts[0]
107                 allparts.insert(0, parts[1])
108         return allparts

463             """
464             True if variable is an array whose shape depends on dimension name `dimname`.
465     
466             Args: dimname: String of :class:`Variable` object.
467             """
468             if not self.isarray: return False
469             if isinstance(dimname, Variable): dimname = dimname.name
470             # This test is not very robust and can fail.
471             # Assume no space between `[` and name (there should be a test for this...)
472             key = "[[%s]]" % dimname
473             for d in self.dimensions:
474                 if key in str(d): return True
475             return False

482         def get_parent_names(self):
483             """
484             Return set of strings with the name of the parents
485             i.e. the variables that are connected to this variable
486             (either because they are present in dimensions on in requires).
487             """
488             #if hasattr(self, ...
489             import re
490             parent_names = []
491             WIKILINK_RE = r'\[\[([\w0-9_ -]+)\]\]'
492             # TODO
493             #  parent = self[parent]
494             # KeyError: "'nzchempot'
495             #WIKILINK_RE = r'\[\[([^\[]+)\]\]'
496             if isinstance(self.dimensions, (list, tuple)):
497                 for dim in self.dimensions:
498                     dim = str(dim)
499                     m = re.match(WIKILINK_RE, dim)
500                     if m:
501                         parent_names.append(m.group(1))
502     
503             if self.requires is not None:
504                 parent_names.extend([m.group(1) for m in re.finditer(WIKILINK_RE, self.requires) if m])
505     
506             # Convert to set and remove possibile self-reference.
507             parent_names = set(parent_names)
508             parent_names.discard(self.name)
509             return parent_names

538         def to_abimarkdown(self, with_hr=True):
539             """
540             Return markdown string Can use Abinit markdown extensions.
541             """
542             lines = []; app = lines.append
543     
544             app("## **%s** \n\n" % self.name)
545             app("*Mnemonics:* %s  " % str(self.mnemonics))
546             if self.characteristics:
547                 app("*Characteristics:* %s  " % ", ".join(self.characteristics))
548             if self.topic2relevances:
549                 app("*Mentioned in topic(s):* %s  " % ", ".join("[[topic:%s]]" % k for k in self.topic2relevances))
550             app("*Variable type:* %s  " % str(self.vartype))
551             if self.dimensions:
552                 app("*Dimensions:* %s  " % self.format_dimensions(self.dimensions))
553             if self.commentdims:
554                 app("*Commentdims:* %s  " % self.commentdims)
555             app("*Default value:* %s  " % self.defaultval)
556             if self.commentdefault:
557                 app("*Comment:* %s  " % self.commentdefault)
558             if self.requires:
559                 app("*Only relevant if:* %s  " % str(self.requires))
560             if self.excludes:
561                 app("*The use of this variable forbids the use of:* %s  " % self.excludes)
562     
563             # Add links to tests.
564             if hasattr(self, "tests") and not self.is_internal:
565                 # Constitutes an usage report e.g.
566                 # Rarely used, in abinit tests [8/888], in tuto abinit tests [2/136].
567                 assert hasattr(self, "tests_info")
568                 tests_info = self.tests_info
569                 ratio_all = len(self.tests) / tests_info["num_all_tests"]
570                 frequency = "Rarely used"
571                 if ratio_all > 0.5:
572                     frequency = "Very frequently used"
573                 elif ratio_all > 0.01:
574                     frequency = "Moderately used"
575     
576                 info = "%s, [%d/%d] in all %s tests, [%d/%d] in %s tutorials" % (
577                     frequency, len(self.tests), tests_info["num_all_tests"], self.executable,
578                     tests_info["num_tests_in_tutorial"], tests_info["num_all_tutorial_tests"], self.executable)
579     
580                 # Use https://facelessuser.github.io/pymdown-extensions/extensions/details/
581                 # Truncate list of tests if we have more that `max_ntests` entries.
582                 count, max_ntests = 0, 20
583                 app('\n??? note "Test list (click to open). %s"' % info)
584                 tlist = sorted(self.tests, key=lambda t: t.suite_name)
585                 d = {}
586                 for suite_name, tests_in_suite in groupby(tlist, key=lambda t: t.suite_name):
587                     ipaths = [os.path.join(*splitall(t.inp_fname)[-4:]) for t in tests_in_suite]
588                     count += len(ipaths)
589                     d[suite_name] = ipaths
590     
591                 for suite_name, ipaths in d.items():
592                     if count > max_ntests: ipaths = ipaths[:min(3, len(ipaths))]
593                     s = "- " + suite_name + ":  " + ", ".join("[[%s|%s]]" % (p, os.path.basename(p)) for p in ipaths)
594                     if count > max_ntests: s += " ..."
595                     app("    " + s)
596                 app("\n\n")
597     
598             # Add text with description.
599             app(2 * "\n")
600             app(self.text)
601             if with_hr: app("* * *" + 2*"\n")
602     
603             return "\n".join(lines)

605         def validate(self):
606             """Validate variable. Raises ValueError if not valid."""
607             errors = []
608             eapp = errors.append
609     
610             try:
611                 svar = str(self)
612             except Exception as exc:
613                 svar = "Unknown"
614                 eapp(str(exc))
615     
616             if self.abivarname is None:
617                 eapp("Variable `%s` has no name" % svar)
618     
619             if self.vartype is None:
620                 eapp("Variable `%s` has no vartype" % svar)
621             elif not self.vartype in ("integer", "real", "string"):
622                 eapp("%s must have vartype in ['integer', 'real', 'string'].")
623     
624             if self.topics is None:
625                 eapp("%s does not have at least one topic and the associated relevance" % svar)
626     
627             for topic, relevances in self.topic2relevances.items():
628                 if topic not in ABI_TOPICS:
629                     eapp("%s delivers topic `%s` that does not belong to the allowed list" % (sname, topic))
630                 for relevance in relevances:
631                     if relevance not in ABI_RELEVANCES:
632                         eapp("%s delivers relevance `%s` that does not belong to the allowed list" % (sname, relevance))
633     
634     	# Compare the characteristics of this variable with the refs to detect possible typos.
635             if self.characteristics is not None:
636                 if not isinstance(self.characteristics, list):
637                     eapp("The field characteristics of %s is not a list" % svar)
638                 else:
639                     for cat in self.characteristics:
640                         if cat.replace("[[", "").replace("]]", "") not in ABI_CHARACTERISTICS:
641                             eapp("The characteristics %s of %s is not valid" % (cat, svar))
642     
643             if self.dimensions is None:
644                 eapp("%s does not have a dimension. If it is a *scalar*, it must be declared so." % svar)
645             else:
646                 if self.dimensions != "scalar":
647                     if not isinstance(self.dimensions, (list, ValueWithConditions)):
648                         eapp('The dimensions field of %s is not a list neither a valuewithconditions' % svar)
649     
650             if self.varset is None:
651                 eapp('`%s` does not have a varset' % svar)
652             #else:
653             #    if not isinstance(self.varset, str) or self.varset not in ref_varset:
654             #        print('The field varset of %s should be one of the valid varsets' % str(self))
655     
656             if len(self.name) > 20:
657                 eapp("Lenght of `%s` is longer than 20 characters." % svar)
658     
659             if errors:
660                 raise ValueError("\n".join(errors))

676     class Range(object):
677         """
678         Specifies a range (start:stop:step)
679         """
680         start = None
681         stop = None
682     
683         def __init__(self, start=None, stop=None):
684             self.start = start
685             self.stop = stop
686     
687         def isin(self, value):
688             """True if value is in range."""
689             isin = True
690             if self.start is not None:
691                 isin = isin and (self.start <= self.value)
692             if stop is not None:
693                 isin = isin and self.stop > self.value
694             return str(self)
695     
696         def __repr__(self):
697             # Add whitespace after `[` or before `]` to avoid [[[ and ]]] patterns
698             # that enter into conflict with wikiling syntax [[...]]
699             if self.start is not None and self.stop is not None:
700                 return "[ " + str(self.start) + " ... " + str(self.stop) + " ]"
701             if self.start is not None:
702                 return "[ " + str(self.start) + "; ->"
703             if self.stop is not None:
704                 return "<-;" + str(self.stop) + " ]"
705             else:
706                 return None

709     class ValueWithConditions(dict):
710         """
711         Used for variables whose value depends on a list of conditions.
712     
713         .. example:
714     
715             ValueWithConditions({'[[paral_kgb]]==1': '6', 'defaultval': 2}),
716     
717             Means that the variable is set to 6 if paral_kgb == 1 else 2
718         """
719         def __repr__(self):
720             s = ''
721             for key in self:
722                 if key != 'defaultval':
723                     s += str(self[key]) + ' if ' + str(key) + ',\n'
724             s += str(self["defaultval"]) + ' otherwise.\n'
725             return s

727         def __str__(self):
728             return self.__repr__()

733         Used for variables that can assume multiple values.
734         This is the equivalent to the X * Y syntax in the Abinit parser.

745                 return str(self.number) + " * " + str(self.value)
