23     def prof_main(main):
24         """
25         Decorator for profiling main programs.
26     
27         Profiling is activated by prepending the command line options
28         supported by the original main program with the one of following keywords:
29              [`prof`, `tracemalloc`, `traceopen`]
30     
31         Example:
32     
33             $ script.py arg --foo=1
34     
35         becomes
36     
37             $ script.py prof arg --foo=1
38     
39         `prof`: profiles the code with `cProfile`.
40             In this case the decorated main accepts two new arguments:
41     
42             prof_file: Name of the output file with profiling data
43                 If not given, a temporary file is created.
44             sortby: Profiling data are sorted according to this value.
45                 default is "time". See sort_stats.
46     
47         `tracemalloc`: uses the tracemalloc module (py>3.4) to trace memory allocations
48     
49         `traceopen`: prints the list of open files before exiting (require `psutil` module)
50         """
51         from functools import wraps
52         @wraps(main)
53         def wrapper(*args, **kwargs):
54             import sys
55             do_prof, do_tracemalloc, do_traceopen = 3 * [False]
56             if len(sys.argv) > 1:
57                 do_prof = sys.argv[1] == "prof"
58                 do_tracemalloc = sys.argv[1] == "tracemalloc"
59                 do_traceopen = sys.argv[1] == "traceopen"
60     
61             if do_prof or do_tracemalloc or do_traceopen: sys.argv.pop(1)
62     
63             if do_prof:
64                 print("Entering profiling mode...")
65                 import pstats, cProfile, tempfile
66                 prof_file = kwargs.pop("prof_file", None)
67                 if prof_file is None:
68                     _, prof_file = tempfile.mkstemp()
69                     print("Profiling data stored in %s" % prof_file)
70     
71                 sortby = kwargs.pop("sortby", "time")
72                 cProfile.runctx("main()", globals(), locals(), prof_file)
73                 s = pstats.Stats(prof_file)
74                 s.strip_dirs().sort_stats(sortby).print_stats()
75                 return 0
76     
77             elif do_tracemalloc:
78                 print("Entering tracemalloc mode...")
79                 # Requires py3.4
80                 try:
81                     import tracemalloc
82                 except ImportError:
83                     print("Error while trying to import tracemalloc (requires py3.4)")
84                     raise SystemExit(1)
85     
86                 tracemalloc.start()
87                 retcode = main(*args, **kwargs)
88                 snapshot = tracemalloc.take_snapshot()
89                 top_stats = snapshot.statistics('lineno')
90     
91                 n = min(len(top_stats), 20)
92                 print("[Top %d]" % n)
93                 for stat in top_stats[:20]:
94                     print(stat)
95     
96             elif do_traceopen:
97                 try:
98                     import psutil
99                 except ImportError:
100                     print("traceopen requires psutil module")
101                     raise SystemExit(1)
102                 import os
103                 p = psutil.Process(os.getpid())
104                 retcode = main(*args, **kwargs)
105                 print("open_files", p.open_files())
106     
107             else:
108                 retcode = main(*args, **kwargs)
109     
110             return retcode
111     
112         return wrapper

115     @prof_main
116     def main():
117         verbose = 1 if "-v" in sys.argv or "--verbose" in sys.argv else 0
118         strict = "-s" in sys.argv or "--strict" in sys.argv
119     
120         if "--no-colors" in sys.argv:
121             from tests.pymods import termcolor
122             termcolor.enable(False)
123             sys.argv.remove("--no-colors")
124     
125         if len(sys.argv) > 1 and sys.argv[1] == "validate":
126             if len(sys.argv) == 2:
127                 return HTMLValidator(verbose).validate_website("./site")
128             else:
129                 validator = HTMLValidator(verbose)
130                 retcode = 0
131                 for page in sys.argv[2:]:
132                     retcode += validator.validate_htmlpage(page)
133                 return retcode
134     
135         if "--help" in sys.argv or "-h" in sys.argv:
136             return mkdocs.__main__.cli()
137     
138         if len(sys.argv) > 1 and ("--help" not in sys.argv or "-h" not in sys.argv):
139             deploy = False
140             website = Website.build("./doc", deploy=deploy, verbose=verbose)
141     
142         if len(sys.argv) > 1 and sys.argv[1] in ("build", "serve", "gh-deploy"):
143             website.generate_markdown_files()
144     
145         if "--dry-run" in sys.argv: return 0
146         print("Invoking mkdocs.__main__ to build HTML from MD files. It could take a while...")
147         mkdocs_retcode = mkdocs.__main__.cli()
148         return mkdocs_retcode + len(website.warnings)
149     
150     
151     if __name__ == '__main__':
152         sys.exit(main())
