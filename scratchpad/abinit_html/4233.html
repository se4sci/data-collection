10     class TimeoutError(Exception):
11         """Exceptions raised by SubProcessWithTimeout."""
12     
13     
14     class SubProcessWithTimeout(object):
15         """
16         Based on from http://stackoverflow.com/questions/3876886/timeout-a-subprocess?rq=1
17         """
18         Error = TimeoutError
19     
20         def __init__(self, timeout, delay=.05):
21             self.timeout = float(timeout)
22             self.delay = float(delay)
23     
24             if self.delay > self.timeout or self.delay <= 0 or self.timeout <= 0:
25                 raise ValueError("delay and timeout must be positive with delay <= timeout")
26     
27         def run(self, args,
28                 bufsize=0, executable=None, stdin=None, stdout=None, stderr=None, preexec_fn=None,
29                 close_fds=False, shell=False, cwd=None, env=None, universal_newlines=False, startupinfo=None, creationflags=0):
30             """Same interface as Popen"""
31     
32             self.proc = subprocess.Popen(args,
33                                          bufsize, executable, stdin, stdout, stderr, preexec_fn,
34                                          close_fds, shell, cwd, env, universal_newlines, startupinfo, creationflags)
35     
36             try:
37                 return_code = self._wait_testcomplete()
38                 return self.proc, return_code
39             except TimeoutError:
40                 raise
41     
42         def _wait_testcomplete(self):
43             start = time.time()
44             while (time.time()-start) < self.timeout:
45                 if self.proc.poll() is not None:  # 0 just means successful exit
46                     return self.proc.returncode
47                 else:
48                     time.sleep(self.delay)
49             # The process may exit between the time we check and the
50             # time we send the signal.
51             try:
52                 os.kill(-self.proc.pid, signal.SIGTERM)
53             except OSError as e:
54                 # If it's not because the process no longer exists, something weird is wrong.
55                 if e.errno != errno.ESRCH: raise
56             time.sleep(1)
57     
58             if self.proc.poll() is None: # Still hasn't exited.
59                 try:
60                     os.kill(-self.proc.pid, signal.SIGKILL)
61                 except OSError as e:
62                     if e.errno != errno.ESRCH: raise
63     
64             raise TimeoutError("timed out waiting for test to complete")
