1     """
2     Definition List Extension for Python-Markdown
3     =============================================
4     
5     Added parsing of Figure captions to Python-Markdown.
6     
7     A simple example:
8     
9         ![](http://lorempixel.com/350/150/)
10         :   Lorem ipsum dolor sit amet, consectetur adipiscing elit.
11             Praesent at consequat magna, faucibus ornare eros. Nam et
12             mattis urna. Cras sodales, massa id gravida
13     
14     Outputs:
15     
16         <figure>
17             <img alt="" src="http://lorempixel.com/350/150/" />
18             <figcaption>
19                 <p>Lorem ipsum dolor sit amet, consectetur adipiscing elit.
20                 Praesent at consequat magna, faucibus ornare eros. Nam et
21                 mattis urna. Cras sodales, massa id gravida</p>
22             </figcaption>
23         </figure>
24     
25     
26     Copyright 2013 - [Helder Correia](http://heldercorreia.com)
27     
28     """
29     
30     from __future__ import unicode_literals
31     from markdown import Extension
32     from markdown.inlinepatterns import IMAGE_LINK_RE, IMAGE_REFERENCE_RE
33     from markdown.blockprocessors import BlockProcessor
34     from markdown.util import etree
35     import re
36     
37     import logging
38     logger = logging.getLogger('MARKDOWN')
39     
40     FIGURES = [IMAGE_LINK_RE, IMAGE_REFERENCE_RE]
41     
42     
43     class FigcaptionProcessor(BlockProcessor):
44         """ Process figure captions."""
45     
46         RE = re.compile(r'(^|\n)[ ]{0,3}:[ ]{1,3}(?P<caption>.*?)(\n|$)')
47         FIGURES_RE = re.compile('|'.join(f for f in FIGURES))
48         NO_INDENT_RE = re.compile(r'^[ ]{0,3}[^ :]')
49     
50         def test(self, parent, block):
51             return bool(self.RE.search(block))
52     
53         def run(self, parent, blocks):
54             # pop the entire block as a single string
55             raw_block = blocks.pop(0)
56     
57             # Get list of figure elements before the colon (:)
58             m = self.RE.search(raw_block)
59     
60             # Get elements
61             elements = raw_block[:m.start()]
62             test_elements = self.FIGURES_RE.search(elements)
63     
64             if not test_elements:
65                 # This is not a figure item.
66                 blocks.insert(0, raw_block)
67                 return False
68     
69             # Get caption
70             block = raw_block[m.end():]
71             no_indent = self.NO_INDENT_RE.match(block)
72     
73             if no_indent:
74                 caption, theRest = (block, None)
75             else:
76                 caption, theRest = self.detab(block)
77             if caption:
78                 caption = '%s\n%s' % (m.group('caption'), caption)
79             else:
80                 caption = m.group('caption')
81     
82             # Create figure
83             figure = etree.SubElement(parent, 'figure')
84             figure.text = elements
85     
86             # Add definition
87             self.parser.state.set('fig')
88             figcaption = etree.SubElement(figure, 'figcaption')
89             self.parser.parseBlocks(figcaption, [caption])
90             self.parser.state.reset()
91     
92             if theRest:
93                 blocks.insert(0, theRest)
94     
95     
96     class FigcaptionExtension(Extension):
97         """ Add definition lists to Markdown. """
98     
99         def extendMarkdown(self, md, md_globals):
100             """ Add an instance of FigcaptionProcessor to BlockParser. """
101     
102             # def_list = 'def_list' in md.registeredExtensions
103             md.parser.blockprocessors.add('figcaption',
104                                           FigcaptionProcessor(md.parser),
105                                           '<ulist')
106     
107     
108     def makeExtension(configs={}):
109         return FigcaptionExtension(configs=configs)
