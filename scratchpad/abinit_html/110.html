17     Ab7Symmetry* ab7_symmetry_new()
18     {
19       Ab7Symmetry *sym, tmpSym;
20       
21       SYM_CALL(new,NEW, &tmpSym);
22       if (tmpSym > 0)
23         {
24           sym = g_malloc(sizeof(Ab7Symmetry));
25           *sym = tmpSym;
26         }
27       else
28         sym = (Ab7Symmetry*)0;
29       return sym;
30     }

142     Ab7Error ab7_symmetry_get_bravais(Ab7Symmetry *sym, int bravais[3][3],
143     				  int *holohedry, int *center, int *nBravSym,
144     				  Ab7SymmetryMat **bravSym)
145     {
146       Ab7Error ab7_errno;
147       int bravais_[9];
148       int syms[3 * 3 * AB7_MAX_SYMMETRIES];
149       int i;
150     
151       SYM_CALL(get_bravais,GET_BRAVAIS, sym, bravais_, holohedry, center, nBravSym, syms, &ab7_errno);
152       if (ab7_errno != AB7_NO_ERROR) return ab7_errno;
153     
154       *bravSym = g_malloc(sizeof(Ab7SymmetryMat) * (*nBravSym));
155       for (i = 0; i < *nBravSym; i++)
156         {
157           (*bravSym)[i].mat[0][0] = syms[9 * i + 0];
158           (*bravSym)[i].mat[0][1] = syms[9 * i + 1];
159           (*bravSym)[i].mat[0][2] = syms[9 * i + 2];
160           (*bravSym)[i].mat[1][0] = syms[9 * i + 3];
161           (*bravSym)[i].mat[1][1] = syms[9 * i + 4];
162           (*bravSym)[i].mat[1][2] = syms[9 * i + 5];
163           (*bravSym)[i].mat[2][0] = syms[9 * i + 6];
164           (*bravSym)[i].mat[2][1] = syms[9 * i + 7];
165           (*bravSym)[i].mat[2][2] = syms[9 * i + 8];
166         }
167       bravais[0][0] = bravais_[0];
168       bravais[0][1] = bravais_[1];
169       bravais[0][2] = bravais_[2];
170       bravais[1][0] = bravais_[3];
171       bravais[1][1] = bravais_[4];
172       bravais[1][2] = bravais_[5];
173       bravais[2][0] = bravais_[6];
174       bravais[2][1] = bravais_[7];
175       bravais[2][2] = bravais_[8];
176     
177       return ab7_errno;
178     }
179     Ab7Error ab7_symmetry_get_matrices(Ab7Symmetry *sym, int *nSym,
180     				   Ab7SymmetryMat **syms,
181     				   Ab7SymmetryTrans **transNon,
182     				   int **symAfm)
183     {
184       Ab7Error ab7_errno;
185       int syms_[3 * 3 * AB7_MAX_SYMMETRIES];
186       int symAfm_[AB7_MAX_SYMMETRIES];
187       double trans[3 * AB7_MAX_SYMMETRIES];
188       int i;
189     
190       SYM_CALL(get_matrices,GET_MATRICES, sym, nSym, syms_, trans, symAfm_, &ab7_errno);
191       if (ab7_errno != AB7_NO_ERROR && ab7_errno != AB7_ERROR_SYM_BRAVAIS_XRED) return ab7_errno;
192     
193       if (syms)
194         *syms = g_malloc(sizeof(Ab7SymmetryMat) * (*nSym));
195       if (transNon)
196         *transNon = g_malloc(sizeof(Ab7SymmetryTrans) * (*nSym));
197       if (symAfm)
198         *symAfm = g_malloc(sizeof(int) * (*nSym));
199       for (i = 0; i < *nSym; i++)
200         {
201           if (syms)
202     	{
203     	  (*syms)[i].mat[0][0] = syms_[9 * i + 0];
204     	  (*syms)[i].mat[0][1] = syms_[9 * i + 1];
205     	  (*syms)[i].mat[0][2] = syms_[9 * i + 2];
206     	  (*syms)[i].mat[1][0] = syms_[9 * i + 3];
207     	  (*syms)[i].mat[1][1] = syms_[9 * i + 4];
208     	  (*syms)[i].mat[1][2] = syms_[9 * i + 5];
209     	  (*syms)[i].mat[2][0] = syms_[9 * i + 6];
210     	  (*syms)[i].mat[2][1] = syms_[9 * i + 7];
211     	  (*syms)[i].mat[2][2] = syms_[9 * i + 8];
212     	}
213           if (transNon)
214     	{
215     	  (*transNon)[i].vect[0] = trans[0];
216     	  (*transNon)[i].vect[1] = trans[1];
217     	  (*transNon)[i].vect[2] = trans[2];
218     	}
219           if (symAfm)
220     	(*symAfm)[i] = symAfm_[i];
221         }
222     
223       return ab7_errno;
224     }
225     Ab7Error ab7_symmetry_get_group(Ab7Symmetry *sym,
226     				char **spaceGroup, int *spaceGroupId,
227     				int *pointGroupMagn, double genAfm[3])
228     {
229       Ab7Error ab7_errno;
230       char name[15];
231       int i;
232     
233       SYM_CALL(get_group,GET_GROUP, sym, name, spaceGroupId,
234     	      pointGroupMagn, genAfm, &ab7_errno);
235       if (ab7_errno != AB7_NO_ERROR && ab7_errno != AB7_ERROR_SYM_BRAVAIS_XRED) return ab7_errno;
236     
237       *spaceGroup = g_malloc(sizeof(char) * 16);
238       for (i = 0; i < 15; i++)
239         (*spaceGroup)[i] = name[i];
240       for (i = 14; i >= 0; i--)
241         if ((*spaceGroup)[i] == ' ')
242           (*spaceGroup)[i] = '\0';
243         else
244           break;
245       (*spaceGroup)[15] = '\0';
246     
247       return ab7_errno;
248     }
249     Ab7Error ab7_symmetry_get_equivalent_atom(Ab7Symmetry *sym, int **equiv,
250     					  int *nSym, int iAtom)
251     {
252       Ab7Error ab7_errno;
253       int equiv_[4 * AB7_MAX_SYMMETRIES];
254       int i;
255     
256       SYM_CALL(get_equivalent_atom,GET_EQUIVALENT_ATOM, sym, equiv_, &iAtom, &ab7_errno);
257       if (ab7_errno != AB7_NO_ERROR && ab7_errno != AB7_ERROR_SYM_BRAVAIS_XRED) return ab7_errno;
258     
259       ab7_errno = ab7_symmetry_get_n_sym(sym, nSym);
260       if (ab7_errno != AB7_NO_ERROR) return ab7_errno;
261     
262       *equiv = g_malloc(sizeof(int) * 4 * (*nSym));
263       for (i = 0; i < *nSym; i++)
264         {
265           (*equiv)[i * 4 + 0] = equiv_[i * 4 + 0];
266           (*equiv)[i * 4 + 1] = equiv_[i * 4 + 1];
267           (*equiv)[i * 4 + 2] = equiv_[i * 4 + 2];
268           (*equiv)[i * 4 + 3] = equiv_[i * 4 + 3];
269         }
270     
271       return ab7_errno;
272     }
