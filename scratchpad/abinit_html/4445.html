50         string = string.lower().strip().replace('"',"").replace("'","")
51         if string == "yes":

53         elif string == "no":
54             return False
55         else:
56             raise ValueError("Cannot interpret string: %s" % string)

235         def run_tests_with_np(self, mpi_nprocs, suite_args=None, runmode="static"):

238             Returns: (nfailed, npassed, nexecuted)
239             """
240             omp_nthreads = max(self.omp_num_threads, 1)
241             py_nthreads = self.ncpus // (mpi_nprocs * omp_nthreads)
242             assert py_nthreads > 0

244             test_suite = abitests.select_tests(suite_args, keys=self.keywords, regenerate=False)
245     
246             # Create workdir.
247             workdir = "TestBot_MPI" + str(mpi_nprocs)
248             if self.has_openmp: workdir += "_OMP" + str(omp_nthreads)

251                 raise RuntimeError("%s already exists!" % workdir)

253                 os.mkdir(workdir)

255             # Run the tests.
256             if self.has_openmp:
257                 msg = "Running ntests = %s, MPI_nprocs = %s, OMP_nthreads %s, py_nthreads = %s..." % (

290         def run(self):
291             """
292             Run all the automatic tests depending on the environment and the options
293             specified in the testbot configuration file.
294             Return the number of failing tests (+ no. passed tests if this is the reference slave).
295             """
296             # If with_tdirs and without_tdirs are not given => execute all tests.
297             # else create a list of strings with the suites to (execute|exclude).
298             suite_args = None
299     
300             # XG130410 Crude hack, to avoid paral and mpiio test directories
301             # in case of enable_mpi=no in config/specs/testfarm.conf
302             if not self.has_mpi:
303                 suite_args = "paral- mpiio-".split()
304     
305             if self.with_tdirs:
306                 suite_args = self.with_tdirs
307             elif self.without_tdirs:
308                 # Append "-" to the string to signal that the suite should be excluded.
309                 suite_args = "- ".join([s for s in self.without_tdirs]) + "-"
310                 suite_args = suite_args.split()
311     
312             # Run tests with 1 processor
313             #print(suite_args)
314             #runmode = "static"
315             #nexecuted = 0
316     
317             if self.runmode == "static":
318                 # Old mode: run all available tests with 1 MPI node here,
319                 # then use MPI mode in mp_suites (paral/mpiio)
320                 np_list = [2, 4, 10, 24]
321                 nfailed, npassed, nexecuted = self.run_tests_with_np(1, suite_args=suite_args, runmode=self.runmode)
322             else:
323                 # New mode: run all available tests with 2 MPI procs here,
324                 # then enter the mp_suites with [4, 10, 24] CPUs
325                 np_list = [4, 10, 24]
326                 nfailed, npassed, nexecuted = self.run_tests_with_np(2, suite_args=suite_args, runmode=self.runmode)
327     
328             if self.has_mpi:
329                 # Run the parallel tests in the multi-parallel suites.
330                 mp_suites = abitests.multi_parallel_suites()
331                 mp_names = [suite.name for suite in mp_suites]
332     
333                 suite_args = mp_names
334     
335                 # Prune dirs.
336                 if self.with_tdirs:
337                     suite_args = [s for s in self.with_tdirs if s in mp_names]
338     
339                 elif self.without_tdirs:
340                     suite_args = [s for s in mp_names if s not in self.without_tdirs]
341     
342                 for np in np_list:
343                     if np > self.ncpus or not suite_args: continue
344                     print("Running multi-parallel tests with %s MPI processors, suite_args %s" % (np, suite_args))
345                     para_nfailed, para_npassed, para_nexec = self.run_tests_with_np(np, runmode="static", suite_args=suite_args)
346     
347                     # Accumulate the counters.
348                     nfailed += para_nfailed
349                     npassed += para_npassed
350                     nexecuted += para_nexec
351     
352             # TODO Collect tarball files.
353             #for fname in self.targz_fnames:
354             #    print("got targz file: ", fname)
355     
356             # Dump the table with the final results.
357             # The buildbot master will upload it and the info will be used
358             # to generate the HTML summary table
359             #for suite_name in self.summary_table:
360             #  print suite_name, self.summary_table.suite_status(suite_name)
361             from pymods.tools import pprint_table
362             table = self.summary.totable()
363             pprint_table(table)
364     
365             self.summary.json_dump("testbot_summary.json")
366     
367             # Empty list of tests (usually due to the use of with, without options)
368             # Create file to signal this condition and return 0
369             if nexecuted == 0:
370                 print("No file found")
371                 with open("__emptylist__", "w") as fh:
372                     fh.write("nfailed = %d, npassed = %d, nexecuted %d" % (nfailed, npassed, nexecuted))
373                     return 0
374     
375             # Status error depends on the builder type.
376             if self.type == "ref":
377                 return nfailed + npassed  # Reference slave (all the tests must pass).
378             else:
379                 return nfailed

382     class TestBotSummary(object):
383         """Stores the final results of the tests performed by TestBot."""
384     
385         _possible_status = ["failed", "passed", "succeeded", "skipped"]
386     
387         def __init__(self, res_table):
388             self.res_table = res_table
389             self.failed = []
390             self.passed = []
391     
392         @lazy__str__
393         def __str__(self): pass
394     
395         def _min_status(self, items):
396             indices = [self._possible_status.index(item) for item in items]
397             return self._possible_status[min(indices)]
398     
399         def totable(self):
400             """
401             Return a table (list of lists whose elements are strings).
402             The table has the format:
403                 - [ suite_name1, suite_name2, ... ]
404                 - [ stats1,      stats2,      ... ]
405     
406             where stats is given by "nfail/npass/nsucces"
407             """
408             def stats2string(stats):
409                 "helper function that returns (nfail/npass/nsucc/nskipped)"
410                 lst = [ str(stats[k]) for k in self._possible_status]
411                 return "/".join(lst)
412     
413             table = [self.suite_names()]
414             row = []
415             for suite_name in self:
416                 status, stats = self.status_of_suite(suite_name)
417                 row.append(stats2string(stats))
418             table.append(row)
419     
420             return table

422         def merge_results(self, test_suite, run_info):
423             #assert test_suite._executed
424             self.run_info = run_info
425     
426             for test in test_suite:
427                 d = self.res_table[test.suite_name][test.id]
428     
429                 if "status" not in d:
430                     # Entry is not initialized. Save the status of this test.
431                     d["status"] = test.status
432                     d["number_of_runs"] = 1
433                     d["run_etime"] = test.run_etime
434                     d["tot_etime"] = test.tot_etime
435                 else:
436                     # Handle the case where we have executed the same test but with a different setup.
437                     # Ordering is: failed < passed < succeeded
438                     # hence a test is marked as "failed" if at least one run failed.
439                     d["status"] = self._min_status([test.status, d["status"] ])
440                     d["number_of_runs"] += 1
441                     d["run_etime"] += test.run_etime
442                     d["tot_etime"] += test.tot_etime
443     
444             # Store the full_id of the tests that (failed|passed).
445             for test in test_suite.failed_tests():
446                 self.failed.append(test.full_id)
447     
448             for test in test_suite.passed_tests():
449                 self.passed.append(test.full_id)
450     
451         def __iter__(self):
452             """Iterate over the suite names in alphabetical order."""
453             for suite_name in self.suite_names(): yield suite_name
454     
455         def suite_names(self):
456             """List of suite names in alphabetical order."""
457             return sorted(list(self.res_table.keys()))
458     
459         def status_of_suite(self, suite_name):
460             """
461             suite_name : string with the name of the suite.
462             return (suite_status, stats) where
463               suite_status is one of the possile status in `_possible_status`
464               stats is a dictionary : {failed:1, passed:2, succeeded:0, skipped:0}
465             """
466     
467             # Initialize stats setting the keys to 0
468             stats = dict.fromkeys(self._possible_status, 0)
469     
470             for test_id, d in self.res_table[suite_name].items():
471                 if "status" not in d: d["status"] = "skipped"
472                 stats[d["status"]] += 1
473     
474             for status in self._possible_status:
475                 if stats[status] > 0:
476                     suite_status = status
477                     break
478             else:
479                 # Ignore error if suite is empty else raise.
480                 if not self.res_table[suite_name]:
481                     suite_status = "succeeded"
482                 else:
483                     raise RuntimeError("[%s] Wrong list of status values!" % suite_name)
484     
485             return suite_status, stats

487         def json_dump(self, fname):
488             """
489             Save self.res_table, self.failed, self.passed in json format.
490             The file will be transferred from the slave to the buildbot master.
491             """
492             def as_ascii(s):
493                 #return str(s.encode("ascii", "ignore"))
494                 return str(s)
495     
496             d = {}
497             # list of strings with the name of the tests that (failed|passed)
498             d[as_ascii("failed")] = self.failed
499             d[as_ascii("passed")] = self.passed
500             d[as_ascii("summary_table")] = self.totable()
501     
502             for suite_name in self:
503                 suite_name = as_ascii(suite_name)
504                 print("---", suite_name, self.res_table[suite_name])
505                 if suite_name in d:
506                     #raise KeyError("Cannot overwrite key %s" % suite_name)
507                     print("Warning: About to overwrite key %s" % suite_name)
508                 d[suite_name] = self.res_table[suite_name]
509     
510             import json
511             with open(fname, "wt") as fh:
512                 json.dump(d, fh)
