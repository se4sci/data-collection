249     def compute_dynmat(DDB):
250     # Retrive the amu for each atom
251       amu = zeros(DDB.natom)
252       for ii in N.arange(DDB.natom):
253         jj = DDB.typat[ii].astype(int)
254         amu[ii] = DDB.amu[jj-1]
255     # Calcul of gprimd from rprimd
256       rprimd = DDB.rprim*DDB.acell
257       gprimd = N.linalg.inv(N.matrix(rprimd))
258     # Transform from 2nd-order matrix (non-cartesian coordinates, 
259     # masses not included, asr not included ) from DDB to
260     # dynamical matrix, in cartesian coordinates, asr not imposed.
261       IFC_cart = zeros((3,DDB.natom,3,DDB.natom),dtype=complex)
262       for ii in N.arange(DDB.natom):
263         for jj in N.arange(DDB.natom):
264           for dir1 in N.arange(3):
265             for dir2 in N.arange(3):
266               for dir3 in N.arange(3):
267                 for dir4 in N.arange(3):
268                   IFC_cart[dir1,ii,dir2,jj] += gprimd[dir1,dir3]*DDB.IFC[dir3,ii,dir4,jj] \
269                 *gprimd[dir2,dir4]
270     # Reduce the 4 dimensional IFC_cart matrice to 2 dimensional Dynamical matrice.
271       ipert1 = 0
272       Dyn_mat = zeros((3*DDB.natom,3*DDB.natom),dtype=complex)
273       while ipert1 < 3*DDB.natom:
274         for ii in N.arange(DDB.natom):
275           for dir1 in N.arange(3):
276             ipert2 = 0
277             while ipert2 < 3*DDB.natom:
278               for jj in N.arange(DDB.natom):
279                 for dir2 in N.arange(3):
280                   Dyn_mat[ipert1,ipert2] = IFC_cart[dir1,ii,dir2,jj]*(5.4857990965007152E-4)/ \
281                        N.sqrt(amu[ii]*amu[jj])
282                   ipert2 += 1
283             ipert1 += 1
284     # Hermitianize the dynamical matrix
285       dynmat = N.matrix(Dyn_mat)
286       dynmat = 0.5*(dynmat + dynmat.transpose().conjugate())
287     
288     # Solve the eigenvalue problem with linear algebra (Diagonalize the matrix)
289       [eigval,eigvect]=N.linalg.eigh(Dyn_mat)
290     
291     # Orthonormality relation 
292       ipert = 0
293       for ii in N.arange(DDB.natom):
294         for dir1 in N.arange(3):
295          eigvect[ipert] = (eigvect[ipert])*N.sqrt(5.4857990965007152E-4/amu[ii])
296          ipert += 1
297       kk = 0
298       for jj in eigval:
299         if jj < 0.0:
300           print("WARNING: An eigenvalue is negative with value: ",jj," ... but proceed with value 0.0")
301           eigval[kk] = 0.0
302           kk += 1
303         else:
304           kk += 1
305       omega = N.sqrt(eigval) #*5.4857990965007152E-4)
306     #  print "omega",omega
307     
308     # The acoustic phonon at Gamma should NOT contribute because they should be zero.
309     # Moreover with the translational invariance the ZPM will be 0 anyway for these
310     # modes but the FAN and DDW will have a non physical value. We should therefore 
311     # neglect these values.
312     #  if N.allclose(DDB.iqpt,[0.0,0.0,0.0]) == True:
313     #    omega[0] = 0.0
314     #    omega[1] = 0.0
315     #    omega[2] = 0.0
316     
317       return omega,eigvect,gprimd

322     def get_reduced_displ(natom,eigvect,omega,gprimd):
323       displ_FAN =  zeros((3,3),dtype=complex)
324       displ_DDW =  zeros((3,3),dtype=complex)
325       displ_red_FAN2 = zeros((3*natom,natom,natom,3,3),dtype=complex)
326       displ_red_DDW2 = zeros((3*natom,natom,natom,3,3),dtype=complex)
327       for imode in N.arange(3*natom): #Loop on perturbation (6 for 2 atoms)
328         if omega[imode].real > tol6:
329           for iatom1 in N.arange(natom):
330             for iatom2 in N.arange(natom):
331               for idir1 in N.arange(0,3):
332                 for idir2 in N.arange(0,3):
333                   displ_FAN[idir1,idir2] = eigvect[3*iatom2+idir2,imode].conj()\
334                      *eigvect[3*iatom1+idir1,imode]/(2.0*omega[imode].real)
335                   displ_DDW[idir1,idir2] = (eigvect[3*iatom2+idir2,imode].conj()\
336                      *eigvect[3*iatom2+idir1,imode]+eigvect[3*iatom1+idir2,imode].conj()\
337                      *eigvect[3*iatom1+idir1,imode])/(4.0*omega[imode].real)
338                   # Now switch to reduced coordinates in 2 steps (more efficient)
339               tmp_displ_FAN = zeros((3,3),dtype=complex)
340               tmp_displ_DDW = zeros((3,3),dtype=complex)
341               for idir1 in N.arange(3):
342                 for idir2 in N.arange(3):
343                   tmp_displ_FAN[:,idir1] = tmp_displ_FAN[:,idir1]+displ_FAN[:,idir2]*gprimd[idir2,idir1]
344                   tmp_displ_DDW[:,idir1] = tmp_displ_DDW[:,idir1]+displ_DDW[:,idir2]*gprimd[idir2,idir1]
345               displ_red_FAN = zeros((3,3),dtype=complex)
346               displ_red_DDW = zeros((3,3),dtype=complex)
347               for idir1 in N.arange(3):
348                 for idir2 in N.arange(3):
349                   displ_red_FAN[idir1,:] = displ_red_FAN[idir1,:] + tmp_displ_FAN[idir2,:]*gprimd[idir2,idir1]
350                   displ_red_DDW[idir1,:] = displ_red_DDW[idir1,:] + tmp_displ_DDW[idir2,:]*gprimd[idir2,idir1]
351     
352               displ_red_FAN2[imode,iatom1,iatom2,:,:] = displ_red_FAN[:,:]
353               displ_red_DDW2[imode,iatom1,iatom2,:,:] = displ_red_DDW[:,:]
354       return displ_red_FAN2,displ_red_DDW2

407     def get_bose(natom,omega,temp_info):
408       bose = N.array(zeros((3*natom,len(temp_info))))
409       for imode in N.arange(3*natom): #Loop on perturbation (6 for 2 atoms)
410         if omega[imode].real > tol6:
411           tt = 0
412           for T in temp_info:
413             if T < tol6:
414               bose[imode,tt] = 0.0
415             else:
416               bose[imode,tt] = 1.0/(N.exp(omega[imode].real/(kb_HaK*T))-1)
417             tt += 1
418       #print bose[:,0]
419       return bose

424     def gaussian(ifreq,omegatmp,gaussian_smearing):
425       x = (ifreq - omegatmp)/gaussian_smearing
426       gaussian = 1.0/(N.sqrt(N.pi)) * (1.0/gaussian_smearing) * N.exp(-(x**2))  
427       return gaussian

429     def fermidirac(freq,center,smearing):
430       x = (freq-center)/smearing
431       value = 0.25/smearing/(N.cosh(x/2.0))**2
432       return value
433     
434     def lorentzian(freq,center,smearing):
435       x = (freq-center)/smearing
436       value = 1.0/(N.pi)/smearing/(x**2+1)
437       return value

452         nbqpt,wtq,eigq_files,DDB_files,EIGR2D_files = arguments
453     
454       DDB = system(directory='.',filename=DDB_files)
455       EIGR2D = system(directory='.',filename=EIGR2D_files)
456       ntemp = len(temp_info)
457       if type == 1 or type == 2:
458         total_corr = zeros((4+2*len(energy),ntemp,EIGR2D.nsppol,EIGR2D.nkpt,EIGR2D.nband),dtype=complex)
459       elif type == 3:
460         total_corr = zeros((4+len(energy),ntemp,EIGR2D.nsppol,EIGR2D.nkpt,EIGR2D.nband),dtype=complex)
461       eigq = system(directory='.',filename=eigq_files)
462     
463     # If the calculation is on a Homogenous q-point mesh
464     # retreve the weight of the q-point
465       if (wtq == 0):
466         wtq = EIGR2D.wtq
467         wtq = wtq[0]
468     
469     # Current Q-point calculated
470       print("Q-point: ",nbqpt," with wtq =",wtq," and reduced coord.",EIGR2D.iqpt)
471       current = multiprocessing.current_process()
472       file_name = str('PYLOG_')+str(current.pid)
473       if os.path.isfile(file_name) :
474         with open(file_name,'a') as F:
475           F.write("Q-point: "+str(nbqpt)+" with wtq ="+str(wtq)+" and reduced coord."+str(EIGR2D.iqpt)+"\n")
476       else:
477         with open(file_name,'w') as F:
478           F.write("Q-point: "+str(nbqpt)+" with wtq ="+str(wtq)+" and reduced coord."+str(EIGR2D.iqpt)+"\n")
479     
480     
481     # Find phonon freq and eigendisplacement from _DDB
482       omega,eigvect,gprimd=compute_dynmat(DDB)
483     
484     # Compute the displacement = eigenvectors of the DDB. 
485     # Due to metric problem in reduce coordinate we have to work in cartesian
486     # but then go back to reduce because our EIGR2D matrix elements are in reduced coord.
487       fan_corr =  zeros((ntemp,EIGR2D.nkpt,EIGR2D.nband,EIGR2D.nsppol),dtype=complex)
488       ddw_corr = zeros((ntemp,EIGR2D.nkpt,EIGR2D.nband,EIGR2D.nsppol),dtype=complex)
489       fan_add = N.array(zeros((ntemp,EIGR2D.nkpt,EIGR2D.nband,EIGR2D.nsppol),dtype=complex))
490       ddw_add = N.array(zeros((ntemp,EIGR2D.nkpt,EIGR2D.nband,EIGR2D.nsppol),dtype=complex))
491     
492       bose = get_bose(EIGR2D.natom,omega,temp_info)
493       # bose (imode, itemp)
494     
495     # Get reduced displacement (scaled with frequency)
496       displ_red_FAN2,displ_red_DDW2 = get_reduced_displ(EIGR2D.natom,eigvect,omega,gprimd)
497       # displ_red(mode,atom1,atom2,dir1,dir2)
498     
499     # Einstein sum make the vector matrix multiplication ont the correct indices
500       fan_corrQ = N.einsum('iojklmn,plnkm->pijo',EIGR2D.EIG2D_bis,displ_red_FAN2)
501       ddw_corrQ = N.einsum('iojklmn,plnkm->pijo',ddw_save,displ_red_DDW2)
502       # fan_corrQ(mode,kpt,band,spin)
503     
504       # Sum over the modes with bose + reshape to (itemp,ispin,ikpt,iband)
505       fan_corr = N.einsum('ijkl,im->mljk',fan_corrQ,2*bose+1.0)
506       ddw_corr = N.einsum('ijkl,im->mljk',ddw_corrQ,2*bose+1.0)
507     
508       omegatmp = omega[:].real # imode
509       if type == 3:
510         imag_fan_add = 0.0
511       elif type == 1 or type == 2:
512         print("Now compute active space ...")
513     
514     #   Now compute active space
515         # sum over atom1,dir1
516         temp = N.einsum('iqjklm,nlokp->ijmnpoq',GKK,displ_red_FAN2)
517         fan_addQ = N.einsum('ijklmnq,iqjmnk->ijklq',temp,N.conjugate(GKK))
518         # fan_addQ(nkpt,nband,nband,imode,ispin)
519         temp = N.einsum('iqjklm,nlokp->ijmnpoq',ddw_save2,displ_red_DDW2)
520         ddw_addQ = N.einsum('ijklmnq,iqjmnk->ijklq',temp,N.conjugate(ddw_save2))
521         # fan_addQ(nkpt,nband,nband,imode,ispin)
522     
523         occtmp = EIGR2D.nsppol*EIGR2D.occ[:,:,:]/2 # jband # should be 1 !
524         delta_E_ddw = N.einsum('lij,k->lijk',eig0[:,:,:].real,N.ones(EIGR2D.nband)) - \
525                   N.einsum('lij,k->likj',eig0[:,:,:].real,N.ones(EIGR2D.nband)) - \
526                   N.einsum('i,ljk->ljik',N.ones((EIGR2D.nband)),(2*occtmp-1))*smearing*1j # spin,ikpt,iband,jband
527     
528         ddw_tmp = N.einsum('ijkln,lm->mijkn',ddw_addQ,2*bose+1.0) # itemp,ikpt,iband,jband,ispin
529         ddw_add = N.einsum('ijklm,mjkl->imjk',ddw_tmp,1.0/delta_E_ddw) # temp,spin,ikpt,iband
530         delta_E = N.einsum('lij,k->lijk',eig0[:,:,:].real,N.ones(EIGR2D.nband)) - \
531                   N.einsum('lij,k->likj',eigq.EIG[:,:,:].real,N.ones(EIGR2D.nband)) # spin,ikpt,iband,jband
532         delta_E_sm = N.einsum('i,ljk->ljik',N.ones((EIGR2D.nband)),(2*occtmp-1))*smearing*1j # spin,ikpt,iband,jband
533         num1 = N.einsum('ij,mkl->mkijl',bose,N.ones((EIGR2D.nsppol,EIGR2D.nkpt,EIGR2D.nband))) +1.0 \
534               - N.einsum('ij,mkl->mkijl',N.ones((3*EIGR2D.natom,ntemp)),occtmp) # spin,k,mod,temp,band # bef was (imode,tmp,band)
535         deno1 = N.einsum('mijk,l->mijkl',delta_E,N.ones(3*EIGR2D.natom),dtype=N.complex)
536     
537         if type==1: # dynamic
538           deno1 -= N.einsum('mijk,l->mijkl',N.ones((EIGR2D.nsppol,EIGR2D.nkpt,EIGR2D.nband,EIGR2D.nband)),omegatmp) #spin,ikpt,iband,jband,imode
539     
540         imag_part1 = N.pi*gaussian(deno1,0.0,gaussian_smearing)
541         #imag_part1 = N.pi*fermidirac(deno1,0.0,gaussian_smearing)
542         #imag_part1 = N.pi*lorentzian(deno1,0.0,gaussian_smearing)
543         #imag_part1 = gaussian_smearing/(deno1*deno1 + gaussian_smearing*gaussian_smearing)
544         deno1 += N.einsum('mijk,l->mijkl',delta_E_sm,N.ones(3*EIGR2D.natom),dtype=N.complex)
545     
546         div1 = N.einsum('ijklm,ijnmk->iklmjn',num1,1.0/deno1) # (spin,k,mod,temp,jband)/(spin,ikpt,iband,jband,mode) => (ispin,imod,tmp,jband,ikpt,iband)
547     
548         num2 = N.einsum('ij,mkl->mkijl',bose,N.ones((EIGR2D.nsppol,EIGR2D.nkpt,EIGR2D.nband))) \
549               + N.einsum('ij,mkl->mkijl',N.ones((3*EIGR2D.natom,ntemp)),occtmp) #imode,tmp,jband
550         deno2 = N.einsum('mijk,l->mijkl',delta_E,N.ones((3*EIGR2D.natom),dtype=N.complex)) 
551     
552         if type==1: # dynamic
553           deno2 += N.einsum('mijk,l->mijkl',N.ones((EIGR2D.nsppol,EIGR2D.nkpt,EIGR2D.nband,EIGR2D.nband)),omegatmp) #spin,ikpt,iband,jband,imode
554         
555         imag_part2 = N.pi*gaussian(deno2,0.0,gaussian_smearing)
556         #imag_part2 = N.pi*fermidirac(deno2,0.0,gaussian_smearing)
557         #imag_part2 = N.pi*lorentzian(deno2,0.0,gaussian_smearing)
558         #imag_part2 = gaussian_smearing/(deno2*deno2 + gaussian_smearing*gaussian_smearing)
559         deno2 -= N.einsum('mijk,l->mijkl',delta_E_sm,N.ones(3*EIGR2D.natom),dtype=N.complex)
560     
561         div2 = N.einsum('ijklm,ijnmk->iklmjn',num2,1.0/deno2) # (spin,k,mod,temp,jband)/(spin,ikpt,iband,jband,mode) => (ispin,imod,tmp,jband,ikpt,iband)
562     
563         fan_add = N.einsum('ijklq,qlmkij->mqij',fan_addQ,div1+div2) # (k,iband,jband,imod,ispin) * (spin,imod,tmp,jband,ikpt,iband) => (temp,ispin,ikpt,iband)
564     
565     
566         imag_div1 = N.einsum('ijklm,ijnmk->iklmjn',num1,imag_part1) # (spin,k,mod,temp,jband)/(spin,ikpt,iband,jband,mode) => (ispin,imod,tmp,jband,ikpt,iband)
567         imag_div2 = N.einsum('ijklm,ijnmk->iklmjn',num2,imag_part2) # (spin,k,mod,temp,jband)/(spin,ikpt,iband,jband,mode) => (ispin,imod,tmp,jband,ikpt,iband)
568         imag_fan_add = N.einsum('ijklq,qlmkij->mqij',fan_addQ,imag_div1+imag_div2) # (k,iband,jband,imod,ispin) * (spin,imod,tmp,jband,ikpt,iband) => (temp,ispin,ikpt,iband)
569     
570         print("Now compute generalized g2F Eliashberg electron-phonon spectral function ...")
571     
572         fan_tmp = N.einsum('ijklm->mijl',fan_addQ) # (ispin,ikpt, iband, imode)
573         ddw_tmp = N.einsum('ijklm->mijl',ddw_addQ) # (ispin,ikpt, iband, imode)
574         g_kk = fan_tmp - ddw_tmp
575     
576         # Eliashberg function 
577         a2F =  zeros((len(energy),EIGR2D.nsppol,EIGR2D.nkpt,EIGR2D.nband),dtype=complex)
578         for ifreq,freq in enumerate(energy):
579           for imode in N.arange(3*EIGR2D.natom):
580             a2F[ifreq,:,:,:] += g_kk[:,:,:,imode]*gaussian(freq,omegatmp[imode],gaussian_smearing)
581         fan_corr += fan_add
582         ddw_corr += ddw_add
583     
584       # PDOS
585       PDOS = zeros((len(energy)),dtype=complex)
586       ii = 0
587       for ifreq,freq in enumerate(energy):
588         for imode in N.arange(3*EIGR2D.natom):
589           PDOS[ifreq] += gaussian(freq,omegatmp[imode],gaussian_smearing)
590       
591       # From the equations ddw_corr has no physical imaginary part
592       ddw_corr.imag = 0.0
593       
594       eigen_corr = (fan_corr[:,:,:,:] - ddw_corr[:,:,:,:])*wtq
595     
596       total_corr[0,:,:,:,:] = eigen_corr[:,:,:,:]
597       total_corr[1,:,:,:,:] = fan_corr[:,:,:,:]*wtq
598       total_corr[2,:,:,:,:] = ddw_corr[:,:,:,:]*wtq
599       total_corr[3,:,:,:,:] = imag_fan_add*wtq
600       total_corr[4:len(energy)+4,0,0,0,0] = PDOS
601     
602       if type == 1 or type == 2:
603         total_corr[4+len(energy):len(energy)*2+4,0,:,:,:] = a2F
604     
605       total_corr = make_average(EIGR2D.nsppol,EIGR2D.nkpt,EIGR2D.nband,degen,total_corr,temp=True)
606     
607       return total_corr
