1     #!/usr/bin/env python
2     """ Command line interface to difflib.py providing diffs in four formats:
3     
4     * ndiff:    lists every line and highlights interline changes.
5     * context:  highlights clusters of changes in a before/after format.
6     * unified:  highlights clusters of changes in an inline format.
7     * html:     generates side by side comparison with change highlights.
8     """
9     from __future__ import print_function, division, absolute_import #, unicode_literals
10     
11     import sys, os, time, difflib, optparse
12     
13     def main():
14          # Configure the option parser
15         usage = "usage: %prog [options] fromfile tofile"
16         parser = optparse.OptionParser(usage)
17     
18         parser.add_option("-c", action="store_true", default=False,
19                           help='Produce a context format diff (default)')
20     
21         parser.add_option("-u", action="store_true", default=False,
22                           help='Produce a unified format diff')
23     
24         hlp = 'Produce HTML side by side diff (can use -c and -l in conjunction)'
25         parser.add_option("-m", action="store_true", default=False, help=hlp)
26     
27         hlp = 'Produce HTML table of side by side diff (can use -c and -l in conjunction)'
28         parser.add_option("-t", action="store_true", default=False, help=hlp)
29     
30         parser.add_option("-n", action="store_true", default=False,
31                           help='Produce a ndiff format diff')
32         parser.add_option("-l", "--lines", type="int", default=3,
33                           help='Set number of context lines (default 3)')
34     
35         parser.add_option("-f", "--file", type="string", default="",
36                           help="Write diff to file FILE. stdout is used if not specified", metavar="FILE")
37     
38         (options, args) = parser.parse_args()
39     
40         if len(args) == 0:
41             parser.print_help()
42             sys.exit(1)
43     
44         if len(args) != 2:
45             parser.error("need to specify both a fromfile and tofile")
46     
47         n = options.lines
48         fromfile, tofile = args # as specified in the usage string
49     
50         # we're passing these as arguments to the diff function
51         fromdate = time.ctime(os.stat(fromfile).st_mtime)
52         todate = time.ctime(os.stat(tofile).st_mtime)
53         fromlines = open(fromfile, 'U').readlines()
54         tolines = open(tofile, 'U').readlines()
55     
56         if options.u:
57             diff = difflib.unified_diff(fromlines, tolines, fromfile, tofile,
58                                         fromdate, todate, n=n)
59         elif options.n:
60             diff = difflib.ndiff(fromlines, tolines)
61     
62         elif options.m:
63             diff = difflib.HtmlDiff().make_file(fromlines, tolines, fromfile,
64                                                 tofile, context=options.c,
65                                                 numlines=n)
66         elif options.t:
67             diff = difflib.HtmlDiff().make_table(fromlines, tolines, fromfile,
68                                                  tofile, context=options.c,
69                                                  numlines=n)
70         else:
71             diff = difflib.context_diff(fromlines, tolines, fromfile, tofile,
72                                         fromdate, todate, n=n)
73     
74         # writelines because diff is a generator
75         if options.file:
76             fh = open(options.file, "w")
77             fh.writelines(diff)
78             fh.close()
79         else:
80             sys.stdout.writelines(diff)
81     
82         sys.exit(0)
83     
84     if __name__ == '__main__':
85         main()
