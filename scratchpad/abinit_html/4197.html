2     from __future__ import unicode_literals, division, print_function, absolute_import

4     import re
5     import os
6     import sys
7     
8     re_markers = re.compile("^(<<<<<<< TREE|=======|>>>>>>> MERGE-SOURCE)$")
9     re_fbktop  = re.compile("fallbacks$")
10     re_fbkdir  = re.compile("(exports|sources|stamps)")
11     re_tmpdir  = re.compile("^tmp")
12     re_tmpfile = re.compile("\.(orig|rej)$")
13     re_rstfile = re.compile("\.rst$")
14     
15     # TODO: Should look at .bzrignore (.gitignore)
16     exclude_exts = set([
17     ".gz", ".tgz", ".png", ".nc", ".jar", ".xcf", ".pyc",
18     ".gif", ".pct", ".jpg", ".jpeg", ".gif", ".pdf", ".svg",
19     ".a", ".o", ".mod", ".cpkl", ".pickle", ".tar",
20     ".swp", ".swo", ".odt",
21     ])
22     
23     exclude_bins = set([
24       "abinit", "anaddb", "mrgddb", "aim", "fftprof", "mrgdv", "mrgddb", "mrggkk", "ujdet",
25       "band2eps", "bsepostproc", "cut3d", "fold2Bloch", "conducti", "ioprof", "lapackprof",
26       "macroave", "optic", "vdw_kernelgen", "vdw_kernelgen", "mrgscr",
27     ])
28     
29     def check_item(item):
30       "True if item has to be analyzed."
31       if re_tmpfile.search(item): return False
32       if re_rstfile.search(item): return False
33       if item in exclude_bins: return False
34     
35       # check extension
36       _, ext = os.path.splitext(item)
37       if ext and ext.lower() in exclude_exts: return False
38       return True
39     
40     def main(top):
41       retval = 0
42       for root,dirs,files in os.walk(top):
43     
44         # Ignore Makefiles
45         if "Makefile.am" in files: files.remove("Makefile.am")
46         if "Makefile.in" in files: files.remove("Makefile.in")
47         if "Makefile" in files: files.remove("Makefile")
48     
49         # Ignore Autotools subdirs
50         if "autom4te.cache" in dirs: dirs.remove("autom4te.cache")
51     
52         # Ignore Bazaar subdirs
53         if ".bzr" in dirs: dirs.remove(".bzr")
54     
55         # Ignore temporary dirs
56         garb_dirs = [item for item in dirs if re_tmpdir.match(item)]
57         for d in garb_dirs: dirs.remove(d)
58     
59         # Ignore installed fallbacks
60         if re_fbktop.search(root):
61           garb_dirs = [item for item in dirs if re_fbkdir.match(item)]
62           for d in garb_dirs: dirs.remove(d)
63     
64         # Display conflict markers found
65         for item in files:
66           path = os.path.join(root, item)
67           if not check_item(item): continue
68     
69           try:
70     
71               if sys.version_info >= (3, 0):
72                 #with open(os.path.join(root, item), "rt", encoding="utf-8", errors="ignore") as fh:
73                 #with open(os.path.join(root, item), "rt", encoding="utf-8") as fh:
74                 with open(os.path.join(root, item), "rt", encoding="ISO-8859-1") as fh:
75                   chk_data = fh.readlines()
76               else:
77                 with open(path, "r") as fh:
78                   chk_data = fh.readlines()
79     
80               chk_stat = False
81               for line in chk_data:
82                 if re_markers.match(line):
83                   chk_stat = True
84                   retval += 1
85                   break
86     
87               if chk_stat:
88                 sys.stderr.write("Found conflict markers in:\n" % path)
89     
90           except Exception as exc:
91             retval += 1
92             sys.stderr.write("Exception while testing: %s\n%s\n" % (path, str(exc)))
93     
94       return retval
95     
96     if __name__ == "__main__":
97     
98       if len(sys.argv) == 1: 
99         top = "."
100       else:
101         top = sys.argv[1]
