64     # Main program
65     #
66     def main(home_dir):
67       from os.path import join as pj
68     
69       # Check if we are in the top of the ABINIT source tree
70       my_name = os.path.basename(__file__) + ".main"
71       if ( not os.path.exists( pj(home_dir,"configure.ac") ) or
72            not os.path.exists( pj(home_dir,"src/98_main/abinit.F90")) ):
73         print("%s: You must be in the top of an ABINIT source tree." % my_name)
74         print("%s: Aborting now." % my_name)
75         sys.exit(1)
76     
77       # Init
78       cnf_bin = MyConfigParser()
79       cnf_fname = pj(home_dir,"config/specs/binaries.conf")
80       cnf_bin.read(cnf_fname)
81       bin_list = cnf_bin.sections()
82       bin_list.sort()
83       dep_order = dict()
84       lib_order = dict()
85       re_num = re.compile("[0-9][0-9]_")
86     
87       # Check order of dependencies and libraries
88       for prg in bin_list:
89         if cnf_bin.has_option(prg,"dependencies"):
90           bin_deps = cnf_bin.get(prg,"dependencies").split()
91         else:
92           bin_deps = list()
93         dep_old = 100
94         dep_new = 100
95         for dep in bin_deps:
96           if dep in dep_levels:
97             dep_new = dep_levels[dep]
98           else:
99             sys.stderr.write("%s: Error: unregistered dependency '%s'\n" %  (my_name,dep))
100             sys.exit(10)
101           if dep_new > dep_old:
102             if not prg in dep_order:
103               dep_order[prg] = list()
104             dep_order[prg].append(dep)
105           dep_old = dep_new
106     
107         if cnf_bin.has_option(prg,"libraries"):
108           bin_libs = cnf_bin.get(prg,"libraries").split()
109         else:
110           bin_libs = list()
111         lib_old = 100
112         lib_new = 100
113         for lib in bin_libs:
114           if re_num.match(lib):
115             lib_new = int(re.sub("_.*","",lib))
116             if lib_new > lib_old:
117               if not prg in lib_order:
118                 lib_order[prg] = list()
119               lib_order[prg].append(lib)
120           lib_old = lib_new
121     
122       # Report any disorder
123       nerr = len(dep_order) + len(lib_order)
124     
125       if nerr > 0 :
126         sys.stderr.write("%s: reporting disordered libraries\n\n" % (os.path.basename(sys.argv[0])))
127         sys.stderr.write("X: D=Dependency / L=Library\n\n")
128         sys.stderr.write("%s  %-24s  %-24s\n" % ("X","Program","Dependency/Library"))
129         sys.stderr.write("%s  %s  %s\n" % ("-","-" * 24,"-" * 24))
130     
131         dep_keys = list(dep_order.keys())
132         dep_keys.sort()
133         for prg in dep_keys:
134           for dep in dep_order[prg]:
135             sys.stderr.write("%s  %-24s  %-24s\n" % ("D",prg,dep))
136         lib_keys = list(lib_order.keys())
137         lib_keys.sort()
138         for prg in lib_keys:
139           for lib in lib_order[prg]:
140             sys.stderr.write("%s  %-24s  %-24s\n" % ("L",prg,lib))
141         sys.stderr.write("\n")
142     
143       return nerr
