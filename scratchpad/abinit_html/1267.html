23         def read_nc(self, fname=None):
24             """Open the GKK.nc file and read it."""
25             fname = fname if fname else self.fname
26     
27             super(GkkFile, self).read_nc(fname)
28     
29             with nc.Dataset(fname, 'r') as root:
30     
31                 self.natom = len(root.dimensions['number_of_atoms'])
32                 self.nkpt = len(root.dimensions['number_of_kpoints'])
33                 self.nband = len(root.dimensions['max_number_of_states'])
34                 self.nsppol = len(root.dimensions['number_of_spins'])
35     
36                 # number_of_spins, number_of_kpoints, max_number_of_states
37                 self.occ = root.variables['occupations'][:,:,:]
38     
39                 # number_of_spins, number_of_kpoints, max_number_of_states   
40                 self.eigenvalues = root.variables['eigenvalues'][:,:,:]
41     
42                 # number_of_kpoints, 3
43                 self.kpt = root.variables['reduced_coordinates_of_kpoints'][:,:]
44                 self.qred = root.variables['current_q_point'][:]
45                 self.wtq = root.variables['current_q_point_weight'][:]
46                 self.rprimd = root.variables['primitive_vectors'][:,:]
47     
48                 # nband, natom, ncart, nkpt, product_mband_nsppol*2 
49                 GKKtmp = root.variables['second_derivative_eigenenergies_actif'][:,:,:,:,:]
50                 GKKtmp2 = np.einsum('ijkno->nokji', GKKtmp)
51                 self.GKK = np.zeros((self.nkpt, self.nsppol*self.nband, 3, self.natom, self.nband), dtype=np.complex)
52                 self.GKK.real[...] = GKKtmp2[:, ::2, ...]
53                 self.GKK.imag[...] = GKKtmp2[:, 1::2, ...]
54     
55                 # Now the second band index "n" refers to the state at "k+q,n"
56                 self.GKK = np.reshape(self.GKK,(self.nkpt,self.nsppol,self.nband,3,self.natom,self.nband))
57     
58                 self.GKK_mode = None
59     
60         def get_gkk_squared(self):
61             """
62             Get squared values of gkk with  reordered indices.
63             Returns:
64                 gkk2[nkpt,nband,3,natom,3,natom,nband]
65             """
66     
67             # No spin polarization at the moment.
68     
69             gkk2 = zeros((self.nkpt, self.nband, 3, self.natom, 3, self.natom,
70                           self.nband), dtype=np.complex)
71     
72             # nkpt,nband,3,natom,3,natom,nband
73             gkk2 = einsum('ijklm,ijnom->ijklnom', self.GKK[:,0,...],
74                                                   self.GKK[:,0,...].conjugate())
75     
76             return gkk2
77     
78         def get_kpt_gkk_squared(self, ikpt):
79             """
80             Get squared values of gkk with  reordered indices for a ginven k-point.
81             Returns:
82                 gkk2[nband,3,natom,3,natom,nband]
83             """
84     
85             # No spin polarization at the moment.
86     
87             gkk2 = zeros((self.nband, 3, self.natom, 3, self.natom,
88                           self.nband), dtype=np.complex)
89     
90             # nkpt,nband,3,natom,3,natom,nband
91             gkk2 = einsum('jklm,jnom->jklnom', self.GKK[ikpt,0,...],
92                                                self.GKK[ikpt,0,...].conjugate())
93     
94             return gkk2

96         def get_gkk_mode(self, ddb):
97             """
98             Convert the gkk from the cartesian/atomic basis to the mode basis.
99     
100             The resulting gkk are aslo scaled by the root mean squared displacement
101             of the mode, that is sqrt(hbar/(M omega)).
102     
103             Arguments
104             ---------
105             ddb:
106                 DdbFile object.
107     
108             Returns
109             -------
110     
111             GKK_mode: [nkpt, nband, nband, nmode]
112             """
113             assert ddb.nmode == 3 * self.natom
114     
115             self.GKK_mode = np.zeros((self.nkpt, self.nband, self.nband, ddb.nmode), dtype=np.complex)
116     
117             polvec = ddb.get_reduced_displ()
118     
119             self.GKK_mode = np.einsum('kniam,oia->knmo', self.GKK[:,0,...], polvec)
120     
121             return self.GKK_mode

123         def get_gkk2_DW_mode(self, ddb):
124             """
125             Compute the squared gkk matrix elements for the Debye-Waller
126             self-energy.
127     
128             The resulting gkk are aslo scaled by the mean squared displacement
129             of the mode, that is hbar/(M omega).
130     
131             Note that, for a given q-point contribution,
132             this function must be called with the GKK at q=Gamma, and the DDB
133             and the given q-point.
134     
135             Arguments
136             ---------
137             ddb:
138                 DdbFile object.
139     
140             Returns
141             -------
142     
143             GKK2_DW_mode: [nkpt, nband, nband, nmode]
144             """
145             assert ddb.nmode == 3 * self.natom
146     
147     
148             self.GKK2_DW_mode = np.zeros((self.nkpt, self.nband, self.nband, ddb.nmode), dtype=np.complex)
149     
150             polvec = ddb.get_reduced_displ()
151     
152             g_r = np.einsum('kniam->knim', self.GKK[:,0,...])
153     
154             for imode in range(ddb.nmode):
155     
156                 g_l1 = np.einsum('kniam,ia->knam', self.GKK[:,0,...], polvec[imode,...]) 
157     
158                 g_l2 = np.einsum('knam,ia->knim', np.conj(g_l1), polvec[imode,...]) 
159     
160                 g2 = np.einsum('knim,knim->knm', g_l2, g_r)
161     
162                 self.GKK2_DW_mode[...,imode] = np.real(g2)
163     
164             return self.GKK2_DW_mode

167         @mpi_watch
168         def broadcast(self):
169             """Broadcast the data from master to all workers."""
170         
171             comm.Barrier()
172     
173             if rank == 0:
174                 dim = np.array([self.natom, self.nkpt, self.nband, self.nsppol], dtype=np.int)
175             else:
176                 dim = np.empty(4, dtype=np.int)
177     
178             comm.Bcast([dim, MPI.INT])
179     
180             if rank != 0:
181     
182                 self.natom, self.nkpt, self.nband, self.nsppol = dim[:]
183     
184                 self.occ = np.empty((self.nsppol, self.nkpt, self.nband), dtype=np.float)
185     
186                 self.GKK = np.empty((self.nkpt,self.nsppol,self.nband,3,self.natom,self.nband),
187                                     dtype=np.complex)
188     
189                 self.eigenvalues = np.empty((self.nsppol, self.nkpt, self.nband), dtype=np.float)
190     
191                 # number_of_kpoints, 3
192                 self.kpt = np.empty((self.nkpt, 3), dtype=np.float)
193                 self.qred = np.empty(3, dtype=np.float)
194                 self.wtq = np.empty(1, dtype=np.float)
195                 self.rprimd = np.empty((3, 3), dtype=np.float)
196     
197             comm.Bcast([self.occ, MPI.DOUBLE])
198             comm.Bcast([self.GKK, MPI.COMPLEX])
199             comm.Bcast([self.eigenvalues, MPI.DOUBLE])
200             comm.Bcast([self.kpt, MPI.DOUBLE])
201             comm.Bcast([self.qred, MPI.DOUBLE])
202             comm.Bcast([self.wtq, MPI.DOUBLE])
203             comm.Bcast([self.rprimd, MPI.DOUBLE])
