55     //Since cufft doesn't have a readable return error code, we added it
56     static inline char* cufftGetErrorString(cufftResult res)
57     {
58       char* message=(char*)malloc(25*sizeof(char));
59     
60       switch(res){
61       case CUFFT_SUCCESS :
62         sprintf(message,"CUFFT_SUCCESS");
63         break;
64       case CUFFT_INVALID_PLAN :
65         sprintf(message,"CUFFT_INVALID_PLAN");
66         break;
67       case CUFFT_ALLOC_FAILED :
68         sprintf(message,"CUFFT_ALLOC_FAILED");
69         break;
70       case CUFFT_INVALID_TYPE :
71         sprintf(message,"CUFFT_INVALID_TYPE");
72         break;
73       case CUFFT_INVALID_VALUE :
74         sprintf(message,"CUFFT_INVALID_VALUE");
75         break;
76       case CUFFT_INTERNAL_ERROR :
77         sprintf(message,"CUFFT_INTERNAL_ERROR");
78         break;
79       case CUFFT_EXEC_FAILED :
80         sprintf(message,"CUFFT_EXEC_FAILED");
81         break;
82       case CUFFT_SETUP_FAILED :
83         sprintf(message,"CUFFT_SETUP_FAILED");
84         break;
85       case CUFFT_INVALID_SIZE :
86         sprintf(message,"CUFFT_INVALID_SIZE");
87         break;
88       case CUFFT_UNALIGNED_DATA :
89         sprintf(message,"CUFFT_UNALIGNED_DATA");
90         break;
91       default:
92         sprintf(message,"CUFFT_UNKNOWN_ERROR");
93       }
94     
95       return message;
96     }
