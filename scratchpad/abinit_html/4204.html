1     #!/usr/bin/env python
2     "Looking for forbidden statements in ABINIT src files"
3     # ======================================================================
4     # == Python script checking if "forbidden" statements -- in terms of  ==
5     # == "abirules" -- are present in ABINIT source files:                ==
6     # ==                                                                  ==
7     # == 1- Access to standard output or standard error using explicit    ==
8     # ==    unit numbers (0, 6, 7), not using "std_out", "std_err"        ==
9     # ==    or "ab_out" pre-defined keywords;                             ==
10     # ==       wrtout.F90 and wrtout_myproc.F90 files are ignored.        ==
11     # ==                                                                  ==
12     # == 2- Use of standard error as output unit                          ==
13     # ==    (identified as "not recommended").                            ==
14     # ==                                                                  ==
15     # == 3- Use of "MPI_COMM_WORLD" MPI keyword instead of one of the     ==
16     # ==    defined communicators: xmpi_world, mpi_enreg(:)%world_comm.   ==
17     # ==                                                                  ==
18     # == 4- Explicit "allocate" or "deallocate" statements                ==
19     # ==    (only macros are allowed: ABI_ALLOCATE, ABI_DEALLOCATE,       ==
20     # ==     ABI_DATATYPE_ALLOCATE, ABI_DATATYPE_DEALLOCATE).             ==
21     # ==                                                                  ==
22     # == 5- "call" statement not placed at the start of the line          ==
23     # ==    (badly interpreted by abilint).                               ==
24     # ==                                                                  ==
25     # ==  Following lines will not generate an error:                     ==
26     # ==   Lines beginning with: "!", "if(debug)"or  "if(ab_dbg)"         ==
27     # ==                                                                  ==
28     # ==            M. Torrent - June 2011 - rev Feb 2012 - rev Apri 2014 ==
29     # ======================================================================
30     from __future__ import unicode_literals, division, print_function, absolute_import
31     
32     import os
33     import re
34     import sys
35     
36     #Activate/Deactivate tests individually
37     ACTIVATE_TEST1=True   # Forbidden write(6
38     ACTIVATE_TEST2=True   # Use of stderr
39     ACTIVATE_TEST3=True   # MPI_COMM_WORLD
40     ACTIVATE_TEST4=True   # Allocate/deallocate
41     ACTIVATE_TEST5=False  # Call at start of line
42     
43     #Forbidden write statements
44     WRITE_FORBIDDEN_LIST = [  # Note: use only lowercases
45     "print *,",
46     "print*,",
47     "write(*,",
48     "write(6,",
49     "write(06,",
50     "write(006,",
51     "wrtout(6,",
52     "wrtout(06,",
53     "wrtout(006,",
54     "flush(6)",
55     "flush(06)",
56     "flush(006)",
57     "flush_unit(6)",
58     "flush_unit(06)",
59     "flush_unit(006)",
60     "write(7,",
61     "write(07,",
62     "write(007,",
63     "wrtout(7,",
64     "wrtout(07,",
65     "wrtout(007,",
66     "flush(7)",
67     "flush(07)",
68     "flush(007)",
69     "flush_unit(7)",
70     "flush_unit(07)",
71     "flush_unit(007)",
72     "write(0,",
73     "write(00,",
74     "write(000,",
75     "wrtout(0,",
76     "wrtout(00,",
77     "wrtout(000,"
78     ]
79     
80     #Not recommended write statements
81     WRITE_NOTRECOMMENDED_LIST = [  # Note: use only lowercases
82     "write(std_err,",
83     "wrtout(std_err,"
84     ]
85     
86     #Ignored files when looking for forbidden write access
87     IGNORED_WRITE_FILES = [
88     "wrtout.F90",
89     "wrtout_myproc.F90",
90     "m_libpaw_tools.F90",
91     "defs.h" # Needed to still have a full CT-QMC standalone version
92     ]
93     
94     #Forbidden MPI_COMM_WORLD keywords
95     COMMWORLD_FORBIDDEN_LIST = [  # Note: use only lowercases
96     "mpi_comm_world"
97     ]
98     
99     #Ignored files when looking for forbidden MPI_COMM_WORLD
100     IGNORED_COMMWORLD_FILES = [
101     "m_xmpi.F90",
102     "m_profiling_abi.F90",
103     "m_libpaw_mpi.F90",
104     "m_malloc.F90",
105     "defs.h"
106     ]
107     
108     #Forbidden allocate/deallocate statements
109     ALLOCATE_FORBIDDEN_LIST = [  # Note: use only lowercases
110     "deallocate",                # "deallocate" must be put before "allocate"
111     "allocate"
112     ]
113     
114     #Ignored files when looking for forbidden allocate/deallocate
115     IGNORED_ALLOCATE_FILES = [
116     "abi_common.h",
117     "libpaw.h",
118     "libtetra.h",
119     "malloc.finc",
120     "defs.h"
121     ]
122     
123     #List of "call" statements
124     CALL_STATEMENT_LIST = [  # Note: use only lowercases
125     "call"
126     ]
127     
128     #Ignored files when looking for forbidden call
129     IGNORED_CALL_FILES = [  # Note: use only lowercases
130     "abi_common.h",
131     "libpaw.h",
132     "libtetra.h",
133     "defs.h"
134     ]
135     
136     #Lines beginning with the following statements will not generate warnings
137     NO_ERROR_LIST = [  # Note: use only lowercases
138     "!",
139     "if(debug)",
140     "if(ab_dbg)"
141     ]

144     def abinit_test_generator():
145       def test_func(abenv):
146          "Looking for forbidden statements in ABINIT src files"
147          top = abenv.apath_of("src")
148          try:
149            return main(top)
150          except Exception:
151            import sys
152            raise sys.exc_info()[1] # Reraise current exception (py2.4 compliant)
153       return {"test_func" : test_func}
154     
155     def main(top):
156       print()
157       print('---------------------------------------------------------------------')
158       print(' Looking for forbidden statements in ABINIT src files:               ')
159       if ACTIVATE_TEST1 or ACTIVATE_TEST2:
160         print('  - forbidden access to standard output/standard error               ')
161       if ACTIVATE_TEST3:
162         print('  - forbidden allocate/deallocate statements                         ')
163       if ACTIVATE_TEST4:
164         print('  - forbidden explicit MPI_COMM_WORLD communicator                   ')
165       if ACTIVATE_TEST5:
166         print('  - forbidden call statements not placed at the start of the line    ')
167       print('---------------------------------------------------------------------')
168     
169       re_srcfile = re.compile("\.([Ff]|[Ff]90|finc|h)$")
170     
171       #Initialize counters
172       file_total_count=0
173       stat_forbidden_write_count=0
174       file_forbidden_write_count=0
175       stat_notrecommended_write_count=0
176       file_notrecommended_write_count=0
177       stat_forbidden_commworld_count=0
178       file_forbidden_commworld_count=0
179       stat_forbidden_allocate_count=0
180       file_forbidden_allocate_count=0
181       stat_forbidden_call_count=0
182       file_forbidden_call_count=0
183     
184       #Loop over files in src folder
185       for (root, dirs, files) in os.walk(top):
186         for src in files:
187           if ( re_srcfile.search(src) ):
188             file_total_count +=1
189             filename=os.path.join(root,src)
190             with open(filename, "rt") as fh:
191               src_data = fh.readlines()
192     
193             #Loop over lines in the file
194             lineno=0
195             icount_forbidden_write=0
196             icount_notrecommended_write=0
197             icount_forbidden_commworld=0
198             icount_forbidden_allocate=0
199             icount_forbidden_call=0
200             for line_orig in src_data:
201               lineno += 1
202     
203               #Transform line to lower case + eliminate whitespaces
204               line_lower=line_orig.lower()
205               line = re.sub(" ","",line_lower)
206     
207               #Skip lines beginning with an authorized character
208               ignored=0
209               for strg in NO_ERROR_LIST:
210                 if line.find(strg) == 0: ignored=1
211               if ignored==0:
212     
213                 #Look for forbidden write statements
214                 if ACTIVATE_TEST1 and (not src in IGNORED_WRITE_FILES):
215                   for strg in WRITE_FORBIDDEN_LIST:
216                     if line.find(strg) != -1:
217                       print('  Error: %s, line %d: found \"%s\" !' % (filename,lineno,strg))
218                       icount_forbidden_write +=1
219     
220                 #Look for not recommended write statements
221                 if ACTIVATE_TEST2 and (not src in IGNORED_WRITE_FILES):
222                   for strg in WRITE_NOTRECOMMENDED_LIST:
223                     if line.find(strg) != -1:
224                       print('- Warning: %s, line %d: found \"%s\" !' % (filename,lineno,strg))
225                       icount_notrecommended_write +=1
226     
227                 #Look for forbidden MPI_COMM_WORLD statements
228                 if ACTIVATE_TEST3 and (not src in IGNORED_COMMWORLD_FILES):
229                   for strg in COMMWORLD_FORBIDDEN_LIST:
230                     if line.find(strg) != -1:
231                       print('  Error: %s, line %d: found \"%s\" !' % (filename,lineno,strg))
232                       icount_forbidden_commworld +=1
233     
234                 #Look for forbidden allocate/deallocate statements
235                 if ACTIVATE_TEST4 and (not src in IGNORED_ALLOCATE_FILES):
236                   ifound=0
237                   for strg in ALLOCATE_FORBIDDEN_LIST:
238                     ialloc=line.find(strg)
239                     if ifound==0 and ialloc != -1:
240                       ifound=1
241                       if not '_'+strg in line:
242                         if ialloc+len(strg)<len(line):
243                           if line[ialloc-4:ialloc] != "abi_" and line[ialloc+len(strg)] == "(":
244                             print('  Error: %s, line %d: found \"%s\" !' % (filename,lineno,strg))
245                             icount_forbidden_allocate +=1
246     
247                 #Look for forbidden call statements (not placed at the start of the line)
248                 if ACTIVATE_TEST5 and (not src in IGNORED_CALL_FILES):
249                   for strg in CALL_STATEMENT_LIST:
250                     icall=line_lower.find(" "+strg+" ")
251                     if icall > 0:
252                       line_before=re.sub(" ","",line_lower[:icall])
253                       line_after=re.sub(" ","",line_lower[icall+3:len(line_lower)-1])
254                       nothing_before=(len(line_before) == 0)
255                       comment_before=(line_before.find("!") != -1)
256                       label_before=(not (re.match('[0-9]+',line_before)==None))
257                       quote_before=((line_before.find("'") != -1) or (line_before.find('"') != -1)) 
258                       problem_before=((not nothing_before) and (not label_before) and (not quote_before))
259                       problem_after=(re.match('[^\;\:\>\<\=\+\-\*\/\!\,]*(\(.*[\)\&]+|)',line_after)==None)
260                       if (not comment_before) and (problem_before or problem_after):
261                         print('  Error: %s, line %d: found \"%s\" not placed at the start of the line !' \
262                               % (filename,lineno,strg))
263                         icount_forbidden_call +=1
264     
265             #Update counters
266             stat_forbidden_write_count +=icount_forbidden_write
267             stat_notrecommended_write_count +=icount_notrecommended_write
268             stat_forbidden_commworld_count +=icount_forbidden_commworld
269             stat_forbidden_allocate_count +=icount_forbidden_allocate
270             stat_forbidden_call_count +=icount_forbidden_call
271             if icount_forbidden_write>0: file_forbidden_write_count +=1
272             if icount_notrecommended_write>0: file_notrecommended_write_count +=1
273             if icount_forbidden_commworld>0: file_forbidden_commworld_count +=1
274             if icount_forbidden_allocate>0: file_forbidden_allocate_count +=1
275             if icount_forbidden_call>0: file_forbidden_call_count +=1
276     
277       #Print final message
278       print( '----------->')
279       print( '- There are %d F90 or header files in the complete set.' % (file_total_count))
280     
281       exit_status = ( stat_forbidden_write_count + stat_notrecommended_write_count + 
282                       stat_forbidden_commworld_count + stat_forbidden_allocate_count +
283                       stat_forbidden_call_count )
284     
285       if stat_forbidden_write_count==0 and stat_notrecommended_write_count==0 and \
286          stat_forbidden_commworld_count==0 and stat_forbidden_allocate_count==0 and \
287          stat_forbidden_call_count==0:
288         print('- No Error or Warning !')
289       else:
290     
291         if stat_forbidden_write_count>0 :
292           print()
293           print( '>>  %d error(s) (forbidden write statement(s)), appearing in %d different file(s) !' % \
294                 (stat_forbidden_write_count,file_forbidden_write_count))
295           print()
296           print( '- Please replace the forbidden statement(s) by allowed ones: ')
297           print( '-  "write(std_out,", "write(std_err," or "write(ab_out,".')
298           print( '-  Note that std_out redirects to the ABINIT log file, cleanly,')
299           print( '-  while ab_out redirects to the ABINIT output file, cleanly also,')
300           print( '-  while std_err redirects to number 0 unit.')
301     
302         if stat_notrecommended_write_count>0 :
303           print()
304           print( '>> %d warnings (not recommended write statement(s)), appearing in %d different file(s) !' % \
305                 (stat_notrecommended_write_count,file_notrecommended_write_count))
306           print()
307           print( '- Writing to the standard error is allowed in the following cases :')
308           print( '-  - Within debugging sections of ABINIT, not activated in production version of ABINIT ; ')
309           print( '-  - In case an error has been detected, causing stop of ABINIT.')
310           print( '- In other cases, writing to std_err is not recommended, and should be avoided.')
311     
312         if stat_forbidden_commworld_count>0 :
313           print()
314           print( '>> %d error(s) (forbidden MPI_COMM_WORLD statement(s)), appearing in %d different file(s) !' % \
315                 (stat_forbidden_commworld_count,file_forbidden_commworld_count))
316           print()
317           print( '- Please replace the MPI_COMM_WORLD forbidden statement(s) by allowed ones: ')
318           print( '-   "xmpi_world" or "mpi_enreg%world_comm".')
319           print( '-   (MPI_COMM_WORLD is not allowed because it may be redefined in some cases)')
320     
321         if stat_forbidden_allocate_count>0 :
322           print()
323           print( '>> %d error(s) (forbidden allocate/deallocate statement(s)), appearing in %d different file(s) !' % \
324                 (stat_forbidden_allocate_count,file_forbidden_allocate_count))
325           print()
326           print( '- Please replace the forbidden allocate statement(s) by the allowed macro used in abinit: ABI_ALLOCATE')
327           print( '- Please replace the forbidden deallocate statement(s) by the allowed macro used in abinit: ABI_DEALLOCATE')
328           print( '- You need to declare  #include "abi_common.h"  for ABINIT to know about ABI_ALLOCATE and ABI_DEALLOCATE')        
329           print( '- Moreover, ABI_ALLOCATE might need ABI_ALLOC_STAT_ABI to be defined, for which you need to declare  use m_profiling_abi ')
330           print( '-')
331           print( '-  Note that the syntax of the ABI_ALLOCATE macro is not exactly the same as the allocate statement:')
332           print( '-   - only one array to be allocated in each ABI_ALLOCATE')
333           print( '-   - separate the array from the parenthesis and size with a comma')
334           print( '-     (this is due to the poor functionalities of the C preprocessor)')
335           print( '-   - example : instead of allocate(arr1(3,N),arr2(20)), you should write the allocations separately using:')
336           print( '-              ABI_ALLOCATE(arr1,(3,N))')
337           print( '-              ABI_ALLOCATE(arr2,(20))')
338           print( '-  Note that the syntax of the ABI_DEALLOCATE macro is not exactly the same as the deallocate statement:')
339           print( '-   - only one array to be deallocated in each ABI_DEALLOCATE')
340           print( '-   - example : instead of deallocate(arr1,arr2), you should write the deallocations separately using:')
341           print( '-               ABI_DEALLOCATE(arr1)')
342           print( '-               ABI_DEALLOCATE(arr2)')
343     
344         if stat_forbidden_call_count>0 :
345           print()
346           print( '>> %d error(s) (badly placed call statement(s)), appearing in %d different file(s) !' % \
347                 (stat_forbidden_call_count,file_forbidden_call_count))
348           print()
349           print( '- Please place the CALL statement at the start of the line. ')
350           print( '-   Avoid : "statement 1 ; call subroutine()"')
351           print( '-   Avoid : "if (condition) call subroutine()"')
352     
353       return exit_status
354     
355     # ---------------------------------------------------------------------------
356     if __name__ == "__main__":
357     
358       if len(sys.argv) == 1: 
359         top = "../../../src"
360       else:
361         top = sys.argv[1] 
362     
363       exit_status = main(top) 
364       sys.exit(exit_status)
