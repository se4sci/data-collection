38         def get_kwargs(self, dirname, basename, **kwargs):

45         def run_compare_nc(self, function, key, refdir=None, nc_ref=None):
46             """
47             Run 'compute' by generating the arguments with 'function'
48             then compare the array 'key' in nc_output.
49     
50             key:
51                 Key to be compared in both nc_output.
52             function:
53                 Function generating the keyword arguments for 'compute'.
54                 Takes a directory as single argument.
55             refdir:
56                 directory for reference files
57             nc_ref:
58                 name of the netcdf file for comparison.
59                 Alternate argument of refdir.
60             """
61     
62             out = compute(**function(self.tmpdir))
63     
64             if nc_ref is not None:
65                 pass
66             else:
67                 if refdir is None:
68                     refdir = self.refdir
69                 nc_ref = out.nc_output.replace(self.tmpdir, refdir)
70     
71             self.check_reference_exists(nc_ref)
72             self.AssertClose(out.nc_output, nc_ref, key)

74         def generate_ref(self, function):
75             """
76             function:
77                 Function generating the keyword arguments for 'compute'.
78                 Takes a directory as single argument.
79             """
80             return compute(**function(self.refdir))
