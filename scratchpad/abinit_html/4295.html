43         def read_nc(self, fname=None):
44             """Open the DDB.nc file and read it."""
45             fname = fname if fname else self.fname
46     
47             super(DdbFile, self).read_nc(fname)
48     
49             with nc.Dataset(fname, 'r') as root:
50     
51                 self.natom = len(root.dimensions['number_of_atoms'])
52                 self.ncart = len(root.dimensions['number_of_cartesian_directions'])  # 3
53                 self.ntypat = len(root.dimensions['number_of_atom_species'])
54     
55                 self.typat = root.variables['atom_species'][:self.natom]
56                 self.amu = root.variables['atomic_masses_amu'][:self.ntypat]
57                 self.rprim = root.variables['primitive_vectors'][:self.ncart,:self.ncart]
58                 self.xred = root.variables['reduced_atom_positions'][:self.natom,:self.ncart]
59                 self.qred = root.variables['q_point_reduced_coord'][:]
60     
61                 # The d2E/dRdR' matrix
62                 self.E2D = np.zeros((self.natom, self.ncart, self.natom, self.ncart), dtype=np.complex)
63                 self.E2D.real = root.variables['second_derivative_of_energy'][:,:,:,:,0]
64                 self.E2D.imag = root.variables['second_derivative_of_energy'][:,:,:,:,1]
65                 self.E2D = np.einsum('aibj->bjai', self.E2D)  # Indicies are reversed when writing them from Fortran.
66     
67                 self.BECT = root.variables['born_effective_charge_tensor'][:self.ncart,:self.natom,:self.ncart]

69         def broadcast(self):
70             """Broadcast the data from master to all workers."""
71         
72             comm.Barrier()
73     
74             if rank == 0:
75                 dim = np.array([self.natom, self.ncart, self.ntypat], dtype=np.int)
76             else:
77                 dim = np.empty(3, dtype=np.int)
78     
79             comm.Bcast([dim, MPI.INT])
80     
81             if rank != 0:
82     
83                 self.natom, self.ncart, self.ntypat = dim[:]
84     
85                 self.typat = np.empty(self.natom, dtype='i')
86                 self.amu = np.empty(self.ntypat, dtype=np.float)
87                 rprim = np.empty((self.ncart, self.ncart), dtype=np.float)
88                 self.xred = np.empty((self.natom, self.ncart), dtype=np.float)
89                 self.qred = np.empty((self.ncart), dtype=np.float)
90     
91                 self.E2D = np.empty((self.natom, self.ncart, self.natom, self.ncart),
92                                     dtype=np.complex)
93     
94                 self.BECT = np.empty((self.ncart, self.natom, self.ncart), dtype=np.float)
95     
96             else:
97                 rprim = self.rprim
98     
99             comm.Bcast([self.typat, MPI.INT])
100             comm.Bcast([self.amu, MPI.DOUBLE])
101             comm.Bcast([self.xred, MPI.DOUBLE])
102             comm.Bcast([self.qred, MPI.DOUBLE])
103             comm.Bcast([self.E2D, MPI.COMPLEX])
104             comm.Bcast([self.BECT, MPI.DOUBLE])
105             comm.Bcast([rprim, MPI.DOUBLE])
106     
107             self.rprim = rprim

126         def get_mass_scaled_dynmat_cart(self):
127             """
128             Format the dynamical matrix in a 3Nx3N matrix,
129             scale with masses, and transform into Cartesian coordinates.
130             """
131             # Retrive the amu for each atom
132             amu = zeros(self.natom)
133             for ii in np.arange(self.natom):
134               jj = self.typat[ii]
135               amu[ii] = self.amu[jj-1]
136         
137             # Transform from 2nd-order matrix (non-cartesian coordinates, 
138             # masses not included, asr not included ) from self to
139             # dynamical matrix, in cartesian coordinates, asr not imposed.
140             E2D_cart = zeros((3,self.natom,3,self.natom),dtype=complex)
141             for ii in np.arange(self.natom):
142               for jj in np.arange(self.natom):
143                 for dir1 in np.arange(3):
144                   for dir2 in np.arange(3):
145                     for dir3 in np.arange(3):
146                       for dir4 in np.arange(3):
147                         E2D_cart[dir1,ii,dir2,jj] += (self.E2D[ii,dir3,jj,dir4] *
148                                 self.gprimd[dir1,dir3] * self.gprimd[dir2,dir4])
149     
150             # Reduce the 4 dimensional E2D_cart matrice to 2 dimensional Dynamical matrice
151             # with scaled masses.
152             Dyn_mat = zeros((3*self.natom,3*self.natom),dtype=complex)
153             for ii in np.arange(self.natom):
154               for dir1 in np.arange(3):
155                 ipert1 = ii * 3 + dir1
156                 for jj in np.arange(self.natom):
157                   for dir2 in np.arange(3):
158                     ipert2 = jj * 3 + dir2
159     
160                     Dyn_mat[ipert1,ipert2] = (E2D_cart[dir1,ii,dir2,jj] *
161                                               me_amu / np.sqrt(amu[ii]*amu[jj]))
162         
163             # Hermitianize the dynamical matrix
164             #dynmat = np.matrix(Dyn_mat)
165             dynmat = Dyn_mat
166             dynmat = 0.5 * (dynmat + dynmat.transpose().conjugate())
167     
168             return dynmat

171         def get_E2D_cart(self):
172             """
173             Transform the 2nd-order matrix from non-cartesian coordinates
174             to cartesian coordinates...and also swap atom and cartesian indicies.
175             """
176             # Transform from 2nd-order matrix (non-cartesian coordinates, 
177             # masses not included, asr not included ) from self to
178             # dynamical matrix, in cartesian coordinates, asr not imposed.
179             E2D_cart = zeros((3,self.natom,3,self.natom),dtype=complex)
180             for ii in np.arange(self.natom):
181               for jj in np.arange(self.natom):
182                 for dir1 in np.arange(3):
183                   for dir2 in np.arange(3):
184                     for dir3 in np.arange(3):
185                       for dir4 in np.arange(3):
186                         E2D_cart[dir1,ii,dir2,jj] += (self.E2D[ii,dir3,jj,dir4] *
187                                 self.gprimd[dir1,dir3] * self.gprimd[dir2,dir4])
188     
189             return E2D_cart

192         def compute_dynmat(self, asr=None, zero_negative=True):
193             """
194             Diagonalize the dynamical matrix.
195         
196             Returns:
197               omega: the frequencies, in Ha
198               eigvect: the eigenvectors, in reduced coord
199             """
200             asr = asr if asr is not None else self.asr
201         
202             # Retrive the amu for each atom
203             amu = zeros(self.natom)
204             for ii in np.arange(self.natom):
205               jj = self.typat[ii]
206               amu[ii] = self.amu[jj-1]
207         
208             dynmat = self.get_mass_scaled_dynmat_cart()
209         
210             # Diagonalize the matrix
211             eigval, eigvect = np.linalg.eigh(dynmat)
212         
213             # Scale the eigenvectors 
214             for ii in np.arange(self.natom):
215               for dir1 in np.arange(3):
216                 ipert = ii * 3 + dir1
217                 eigvect[ipert] = eigvect[ipert] * np.sqrt(me_amu / amu[ii])
218     
219             # Nullify imaginary frequencies
220             if zero_negative:
221                 for i, eig in enumerate(eigval):
222                   if eig < 0.0:
223                     warnings.warn("An eigenvalue is negative with value: {} ... but proceed with value 0.0".format(jj))
224                     eigval[i] = 0.0
225     
226             # Impose the accoustic sum rule
227             if asr and self.is_gamma:
228               eigval[0] = 0.0
229               eigval[1] = 0.0
230               eigval[2] = 0.0
231     
232             # Frequencies
233             self.omega = np.sqrt(np.abs(eigval)) * np.sign(eigval)
234             self.eigvect = eigvect
235         
236             return self.omega, self.eigvect

238         def get_reduced_displ(self):
239             """
240             Compute the mode eigenvectors, scaled by the mode displacements
241             Also transform from cartesian to reduced coordinates.
242     
243             Returns: polvec[nmode,3,natom]
244             """
245     
246             # Minimal value for omega (Ha)
247             omega_tolerance = 1e-5
248     
249             self.polvec = zeros((self.nmode,3,self.natom), dtype=complex)
250             xi_at = zeros(3, dtype=complex)
251     
252             omega, eigvect = self.compute_dynmat()
253     
254             for imode in range(self.nmode):
255                 # Skip mode with zero frequency (leave displacements null)
256                 if omega[imode].real < omega_tolerance:
257                   continue
258     
259                 z0 = 1. / np.sqrt(2.0 * omega[imode].real)
260     
261                 for iatom in np.arange(self.natom):
262                     for idir in range(3):
263                         xi_at[idir] = eigvect[3*iatom+idir,imode] * z0
264     
265                     for idir in range(3):
266                         for jdir in range(3):
267                             self.polvec[imode,idir,iatom] += xi_at[jdir] * self.gprimd[jdir,idir]
268     
269             return self.polvec

271         def get_reduced_displ_squared(self):
272             """
273             Compute the squared reduced displacements (scaled by phonon frequencies)
274             for the Fan and the DDW terms.
275             """
276             # Minimal value for omega (Ha)
277             omega_tolerance = 1e-5
278     
279             natom = self.natom
280             omega, eigvect = self.compute_dynmat()
281     
282             displ_FAN = zeros((3,3), dtype=complex)
283             displ_DDW = zeros((3,3), dtype=complex)
284             displ_red_FAN2 = zeros((3*natom,natom,natom,3,3), dtype=complex)
285             displ_red_DDW2 = zeros((3*natom,natom,natom,3,3), dtype=complex)
286     
287             for imode in np.arange(3*natom):
288     
289               # Skip mode with zero frequency (leave displacements null)
290               if omega[imode].real < omega_tolerance:
291                 continue
292     
293               for iatom1 in np.arange(natom):
294                 for iatom2 in np.arange(natom):
295                   for idir1 in np.arange(0,3):
296                     for idir2 in np.arange(0,3):
297     
298                       displ_FAN[idir1,idir2] = (
299                           eigvect[3*iatom2+idir2,imode].conj() *
300                           eigvect[3*iatom1+idir1,imode] / (2.0 * omega[imode].real)
301                           )
302     
303                       displ_DDW[idir1,idir2] = (
304                           eigvect[3*iatom2+idir2,imode].conj() *
305                           eigvect[3*iatom2+idir1,imode] +
306                           eigvect[3*iatom1+idir2,imode].conj() *
307                           eigvect[3*iatom1+idir1,imode]) / (4.0 * omega[imode].real)
308     
309                   # Now switch to reduced coordinates in 2 steps (more efficient)
310                   tmp_displ_FAN = zeros((3,3),dtype=complex)
311                   tmp_displ_DDW = zeros((3,3),dtype=complex)
312     
313                   for idir1 in np.arange(3):
314                     for idir2 in np.arange(3):
315     
316                       tmp_displ_FAN[:,idir1] = tmp_displ_FAN[:,idir1] + displ_FAN[:,idir2] * self.gprimd[idir2,idir1]
317                       tmp_displ_DDW[:,idir1] = tmp_displ_DDW[:,idir1] + displ_DDW[:,idir2] * self.gprimd[idir2,idir1]
318     
319                   displ_red_FAN = zeros((3,3),dtype=complex)
320                   displ_red_DDW = zeros((3,3),dtype=complex)
321                   for idir1 in np.arange(3):
322                     for idir2 in np.arange(3):
323                       displ_red_FAN[idir1,:] = displ_red_FAN[idir1,:] + tmp_displ_FAN[idir2,:] * self.gprimd[idir2,idir1]
324                       displ_red_DDW[idir1,:] = displ_red_DDW[idir1,:] + tmp_displ_DDW[idir2,:] * self.gprimd[idir2,idir1]
325         
326                   displ_red_FAN2[imode,iatom1,iatom2,:,:] = displ_red_FAN[:,:]
327                   displ_red_DDW2[imode,iatom1,iatom2,:,:] = displ_red_DDW[:,:]
328     
329             self.displ_red_FAN2 = displ_red_FAN2
330             self.displ_red_DDW2 = displ_red_DDW2
331         
332             return displ_red_FAN2, displ_red_DDW2

334         def get_bose(self, temperatures):
335             """
336             Get the Bose-Einstein occupations on a range of temperatures.
337             Returns: bose(3*natom, Ntemperatures)
338             """
339             if self.omega is None:
340                 self.compute_dynmat()
341     
342             bose = zeros((3*self.natom, len(temperatures)))
343     
344             #bose[:,:] = get_bose(self.omega, temperatures)
345             for imode, omega in enumerate(self.omega):
346                 bose[imode,:] = get_bose(omega, temperatures)
347     
348             return bose
