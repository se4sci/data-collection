1     #!/usr/bin/env python
2     """
3     Change libtetrabz source file so that they are compatible with the Abinit build system.
4     """
5     from __future__ import unicode_literals, division, print_function, absolute_import
6     
7     import os
8     
9     here = os.path.dirname(__file__)
10     f90_files = [f for f in os.listdir(here) if f.startswith("libtetrabz") and f.endswith(".F90")]
11     paths = [os.path.join(here, f) for f in f90_files]
12     
13     def abi_sanitize(line):
14         line = line.replace("__MPI", "HAVE_MPI")
15         # Bind(C) with optional args requires F2015
16         line = line.replace("BIND(C)", "")
17         return line
18     
19     print("Replacing __MPI with HAVE_MPI in the following routines:")
20     for path in paths:
21         print("\t", path)
22         lines = ["""
23     #if defined HAVE_CONFIG_H
24     #include "config.h"
25     #endif
26     
27     #include "abi_common.h"
28     """]
29         lines = []
30     
31         with open(path, "rt") as fh:
32             lines += [abi_sanitize(l) for l in fh]
33     
34         print(lines)
35         with open(path, "wt") as fh:
36             fh.writelines(lines)
