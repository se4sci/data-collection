95     def vararg_callback(option, opt_str, value, parser):
96         """Callback for an option with variable arguments"""
97         assert value is None
98         value = []
99     
100         def floatable(str):
101             try:
102                 float(str)
103                 return True
104             except ValueError:
105                 return False
106     
107         for arg in parser.rargs:
108             # stop on --foo like options
109             if arg[:2] == "--" and len(arg) > 2: break
110             # stop on -a, but not on -3 or -3.0
111             if arg[:1] == "-" and len(arg) > 1 and not floatable(arg): break
112             value.append(arg)
113     
114         del parser.rargs[:len(value)]
115         setattr(parser.values, option.dest, value)

133     def parse_stats(stats):
134         # TODO Use BaseTest class attribute
135         _possible_status = ["failed", "passed", "succeeded", "skipped", "disabled",]
136     
137         if "+" in stats:
138             stats = [s.strip() for s in stats.split("+")]
139         elif stats == "all":
140             stats = ["failed", "passed", "succeeded"]
141         elif stats.startswith("not_"):
142             not_stat = stats[4:]
143             assert not_stat in _possible_status
144             stats = [s for s in _possible_status if s != not_stat]
145         else:
146             # String
147             stats = [stats]
148     
149         for s in stats:
150             if s not in _possible_status:
151                 raise ValueError("%s is not a valid status" % s)
152     
153         return stats
