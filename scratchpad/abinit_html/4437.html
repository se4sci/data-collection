58     def fix_punctuation_marks(s):
59         """
60         Remove whitespaces before `,` and `;` that trigger a bug in ConfigParser
61         when the option spans multiple lines separated by `;`
62     
63         For instance:
64     
65             #%% files_to_test =
66             #%%   t93.out, tolnlines = 0, tolabs = 0.000e+00, tolrel = 0.000e+00 ;
67             #%%   t93.out_ep_SBK, tolnlines = 0, tolabs = 0.000e+00, tolrel = 0.000e+00
68     
69         is not treated properly by ConfigParser due to ` ;` at the end of the line and
70         only t93.out is stored in dictionary and tested by fldiff.
71         """
72         for mark in (",", ";"):
73             s = (mark + " ").join([tok.rstrip() for tok in s.split(mark)])
74         return s + "\n"

184     def rmrf(top, exclude_paths=None):
185         """
186         Recursively remove all files and directories contained in directory top.
187     
188         Args:
189             exclude_paths:
190                 list with the absolute paths that should be preserved
191     
192         Returns the list of files and the directories that have been removed.
193         """
194         exc_paths = []
195         if exclude_paths is not None:
196             if is_string(exclude_paths):
197                 exc_paths = [exclude_paths]
198             else:
199                 exc_paths = exclude_paths
200     
201         removed = []
202         for (root, dirs, files) in os.walk(top):
203             for f in files:
204                 file_path = os.path.join(root, f)
205                 if file_path not in exc_paths:
206                     os.unlink(file_path)
207                     removed.append(file_path)
208             for d in dirs:
209                 dir_path = os.path.join(root, d)
210                 if dir_path not in exc_paths:
211                     shutil.rmtree(dir_path)
212                     removed.append(dir_path)
213     
214         return removed

361     def _str2filestotest(string):
362         """
363         Parse the files_to_test section.
364         Returns a tuple of `FileToTest` objects.
365         """
366         if not string:
367             return []
368     
369         if ";" in string:
370             file_lines = [s for s in string.split(";") if s.strip()]
371         else:
372             file_lines = [string]
373     
374         files_to_test = []
375         for line in file_lines:
376             tokens = line.split(",")
377             d = {"name": tokens[0]}
378             for tok in tokens[1:]:
379                 k, v = [s.strip() for s in tok.split("=")]
380                 if k in d:
381                     err_msg = "Found multiple occurences of keyword %s" % k
382                     raise AbinitTestInfoParserError(err_msg)
383                 d[k] = v
384             files_to_test.append(FileToTest(d))
385     
386         return tuple(files_to_test)

463         """True if string start with a TEST_INFO section or option."""
464         from re import compile
465         re_ncpu = compile("^NCPU_(\d+)$")
466         s = string.strip()
467         idx = s.find("=")
468         if idx == -1: # might be a section.
469             if s.startswith("[") and s.endswith("]"):
470                 if s[1:-1] in TESTCNF_SECTIONS: return 1 # [files]...
471                 if re_ncpu.search(s[1:-1]): return 1    # [NCPU_1] ...
472         else:
473             if s[:idx].strip() in TESTCNF_KEYWORDS: return 2
474     
475         return 0

871         """Pre-process source code with ANSI CPP."""
872         Error = CPreProcessorError
873     
874         def __init__(self, includes=None, opts=None, bin="cpp", verbose=0):
875             self.includes = ["."]
876             if includes is not None: self.includes = includes
877             self.opts = ["-DHAVE_CONFIG_H"]
878             if opts is not None: self.opts = opts
879             self.bin, self.verbose = bin, verbose
880     
881         def process_file(self, filepath, remove_lhash=True):
882             """
883             Read source from filepath, call CPP wit the includes and the
884             options passed to the constructor.
885     
886             Returns:
887                 preprocessed text.
888             """
889             if self.bin is None:
890                 # No pre-processing, return raw string.
891                 with open(filepath, "r") as f:
892                     return f.read()
893     
894             cmd = [self.bin]
895             if self.opts: cmd += self.opts
896             cmd += ["-ansi"]
897             if self.includes: cmd += ["-I"+inc for inc in self.includes]
898             cmd += [filepath]
899             cmd = " ".join(cmd)
900             if self.verbose: print(cmd)
901     
902             from subprocess import Popen, PIPE
903             p = Popen(cmd, shell=True, stdout=PIPE, stderr=PIPE)
904             stdout, stderr = p.communicate()
905     
906             if p.returncode:
907                 raise self.Error("C-preprocessor returned %d\n stderr:\n%s" % (p.returncode, stderr))
908     
909             # Remove leading hash symbols added by CPP
910             if not remove_lhash:
911                 return stdout
912             else:
913                 return "\n".join([str(l) for l in stdout.splitlines() if not l.startswith("#")])

1138     def input_file_has_vars(fname, ivars, comment="#", mode="any"):
1139         """
1140         Primitive parser that searches for the occurrence of input variables in the input file fname
1141     
1142         Args:
1143             fname:
1144                 Input file
1145             ivars:
1146                 dictionary whose keys are strings with the input variables to search.
1147                 ivar[varname] can be either None or an integer
1148                 if ivar[varname] is None, we have a match if varname is present
1149                 if ivar[varname] is int, we have a match if varname is present and it has value int
1150     
1151             mode: "all" or "any"
1152     
1153             return:
1154                 (bool, d)
1155                 bool is True is the input file contains the specified variables
1156                 d is a dictionary with the matching lines (empty dict if no occurence).
1157         """
1158         # This algorithm is not very robust as it assumes that the variable and the line
1159         # are placed on the same line.
1160         with open(fname, "r") as fh:
1161             lines = []
1162             for line in fh:
1163                 line = line.lower().strip()
1164                 idx = line.find(comment)
1165                 if idx != -1: line = line[:idx]
1166                 lines.append(line)
1167     
1168         matches = {}
1169         for k in ivars:
1170             matches[k] = []
1171     
1172         items = ivars.items()
1173     
1174         re_ivars = {}
1175         for varname in ivars:
1176             re_ivars[varname] = re.compile(varname + "\d*\s*(\d+)\s*")
1177     
1178         nfound = 0
1179         for line in lines:
1180             for varname, varvalue in items:
1181                 re_match = re_ivars[varname].match(line)
1182                 #print("match")
1183                 if varvalue is None and varname in line:
1184                     nfound += 1
1185                     matches[varname].append(line)
1186                 elif re_match:
1187                     num = int(re_match.group(1))
1188                     if num == int(varvalue):
1189                         #print line
1190                         matches[varname].append(line)
1191                         nfound += 1
1192     
1193         if nfound == 0:
1194             return False, {}
1195     
1196         if mode == "all":
1197             return all(bool(v) for v in matches.values()), matches
1198         elif mode == "any":
1199             return any(bool(v) for v in matches.values()), matches
1200         else:
1201             raise ValueError("Wrong mode %s" % mode)

1884         def compute_nprocs(self, build_env, nprocs, runmode):

1896               2) The user asks for more MPI nodes than max_nprocs (this value is reported in the TEST_INFO section).
1897               3) We have a multiparallel test (e.g. paral/tA.in) and nprocs is not in in nprocs_to_test
1898               4) nprocs is in exclude_nprocs
1899             """

1901             errors = []
1902             eapp = errors.append
1903             for var in self.need_cpp_vars:
1904                 if not var.startswith("!") and var not in build_env.defined_cppvars:
1905                     eapp("Build environment does not define the CPP variable %s" % var)
1906                 elif var[1:] in build_env.defined_cppvars:
1907                     eapp("Build environment defines the CPP variable %s" % var[1:])
1908     
1909             # Remove this check to run the entire test suite in parallel
1910             #runmode ="dynamic"
1911     
1912             if runmode == "static":
1913                 if nprocs > self.max_nprocs:
1914                     eapp("nprocs: %s > max_nprocs: %s" % (nprocs, self.max_nprocs))
1915     
1916             elif runmode == "dynamic":
1917                 # Will select the minimum between max_nprocs and nprocs
1918                 pass
1919     
1920             else:
1921                 raise ValueError("Wrong runmode %s" % runmode)
1922     
1923             if self.nprocs_to_test and nprocs != self.nprocs_to_test[0]:
1924                 eapp("nprocs: %s != nprocs_to_test: %s" % (nprocs, self.nprocs_to_test[0]))

1926             if nprocs in self.exclude_nprocs:
1927                 eapp("nprocs: %s in exclude_nprocs: %s" % (nprocs, self.exclude_nprocs))

1929             err_msg = "\n".join(errors)

1931                 real_nprocs = 0
1932             else:
1933                 real_nprocs = min(self.max_nprocs, nprocs)
