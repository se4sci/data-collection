177         def has_eigenvalue(self,nsppol,isppol,kpoint,iband):
178             if self.nsppol != nsppol:
179                 return False
180             for ikpt in range(self.nkpt):
181                 if N.absolute(self.kpoints[ikpt,0]-kpoint[0]) < csts.TOLKPTS and \
182                    N.absolute(self.kpoints[ikpt,1]-kpoint[1]) < csts.TOLKPTS and \
183                    N.absolute(self.kpoints[ikpt,2]-kpoint[2]) < csts.TOLKPTS:
184                     if iband >= self.bd_indices[isppol,ikpt,0]-1 and iband < self.bd_indices[isppol,ikpt,1]:
185                         return True
186                     return False
187             return False

602             print 'ERROR: eigenvalue containers do not contain GW and DFT eigenvalues ... exiting now'

605             print 'ERROR: the number of spins/kpoints is not the same in the GW and DFT containers ... exiting now'

616                 valgwarray = N.append(valgwarray,csts.hartree2ev*gwec.eigenvalues[isppol,ikpt,ibdmin:vbm_index])
617                 conddftarray = N.append(conddftarray,csts.hartree2ev*dftec.eigenvalues[isppol,ikpt,vbm_index:ibdmax+1])
618                 condgwarray = N.append(condgwarray,csts.hartree2ev*gwec.eigenvalues[isppol,ikpt,vbm_index:ibdmax+1])
619         if energy_pivots == None:
620             if plot_figures == 1:

633                 P.xlabel('DFT eigenvalues (in eV)')
634                 P.ylabel('GW correction to the DFT eigenvalues (in eV)')

653                         tmpdftarray = N.append(tmpdftarray,[dftarray[ii]])
654                         tmpgwarray = N.append(tmpgwarray,[gwarray[ii]])
655             elif iinterval == len(polyfit_degrees)-1:
656                 emin = energy_pivots[-1]
657                 emax = None

660                         tmpdftarray = N.append(tmpdftarray,[dftarray[ii]])

663                 emin = energy_pivots[iinterval-1]
664                 emax = energy_pivots[iinterval]

667                         tmpdftarray = N.append(tmpdftarray,[dftarray[ii]])
