5     subroutine atom_sym(nbtime,position,a,b,c,indexAtom1,Pos1,Pos2,Pos3,size1,size2,nba1)
6     
7       !input 
8       integer :: size1,size2,nba1
9       integer, intent(in) :: nbtime
10       !f2py  optional  ,  depend(position)    ::  size1=len(position[:,1,1])
11       !f2py  optional  ,  depend(position)    ::  size2=len(position[1,:,1])
12       !f2py  optional  ,  depend(indexAtom1)  ::  nba1=len(indexAtom1)
13       real, intent(in)  ::  position(size1,size2,3)
14       integer, intent(in)  ::  indexAtom1(nba1)
15       real, intent(in) :: a(size1),b(size1),c(size1)
16       !output
17       real(8),intent(out)  ::  Pos1(nbtime*nba1),Pos2(nbtime*nba1),Pos3(nbtime*nba1)
18       !local variable
19       real(8) :: x,y,z
20       integer :: i,j,k
21     
22       k = 1
23     
24       do  i=1,nbtime
25          do j=1,nba1
26      
27            x = position(i,indexAtom1(j),1)
28            y = position(i,indexAtom1(j),2)
29            z = position(i,indexAtom1(j),3)
30     
31            if (x >= 0) then  !replace atoms in a box [0:1]
32              
33              Pos1(k) = x - a(i)*int(x/a(i))
34              
35            else
36              
37              Pos1(k) = x - a(i)*int(x/a(i)-1)
38              
39            endif
40            
41            if (y >= 0) then  !replace atoms in a box [0:1]
42              
43              Pos2(k) = y - b(i)*int(y/b(i))
44              
45            else
46              
47              Pos2(k) = y - b(i)*int(y/b(i)-1)
48              
49            endif
50            
51            if (z >= 0) then  !replace atoms in a box [0:1]
52              
53              Pos3(k) = z - c(i)*int(z/c(i))
54              
55            else
56              
57              Pos3(k) = z - c(i)*int(z/c(i)-1)
58              
59            endif
60             
61             k = k+1
62             
63           enddo
64         enddo
65     
66     end subroutine atom_sym

73     subroutine atom_ave(nbtime,position,indexAtom1,pos1,pos2,pos3,size1,size2,nba1)
74     
75       !input 
76       integer :: size1,size2,nba1
77       integer, intent(in) :: nbtime
78       !f2py  optional  ,  depend(position)    ::  size1=len(position[:,1,1])
79       !f2py  optional  ,  depend(position)    ::  size2=len(position[1,:,1])
80       !f2py  optional  ,  depend(indexAtom1)  ::  nba1=len(indexAtom1)
81       real, intent(in)  ::  position(size1,size2,3)
82       integer, intent(in)  ::  indexAtom1(nba1)
83       !output
84       real(8),intent(out)  ::  pos1(nba1),pos2(nba1),pos3(nba1)
85       !local variable
86       real :: x,y,z,xred,yred,zred
87       integer :: i,j
88     
89       do  i=1,nbtime
90          do j=1,nba1
91     
92             x = position(i,indexAtom1(j),1)
93             y = position(i,indexAtom1(j),2)
94             z = position(i,indexAtom1(j),3)
95     
96             pos1(j) = pos1(j) + x/nbtime
97             pos2(j) = pos2(j) + y/nbtime
98             pos3(j) = pos3(j) + z/nbtime
99     
100          enddo
101       enddo
102     
103     end subroutine atom_ave
