390         def get_iqpts(self, nqpt):
391             """Distribute the q-points indicies to be treated by each worker."""
392     
393             max_nqpt_per_worker = (
394                 nqpt // size + min(nqpt % size, 1))
395             n_active_workers = (
396                 nqpt // max_nqpt_per_worker
397                 + min(nqpt % max_nqpt_per_worker, 1))
398     
399             my_iqpts = list()
400     
401             for i in range(max_nqpt_per_worker):
402     
403                 iqpt = rank * max_nqpt_per_worker + i
404     
405                 if iqpt >= nqpt:
406                     break
407     
408                 my_iqpts.append(iqpt)
409     
410             return my_iqpts

576

588                 if len(active_ranks) > 1:
589                     for irank in active_ranks[1:]:
590                         partial = comm.recv(source=irank, tag=10000+irank)
591                         qred_p, omega_p = partial
592                         for qred, omega in zip(qred_p, omega_p):
593                             i += 1
594                             qred_all[i,...] = qred[...]
595                             omega_all[i,...] = omega[...]
596     
597             elif self.active_worker:
598                 comm.send(partial, dest=0, tag=10000+rank)
599                 return
600             else:
601                 return
602     
603             self.qred = qred_all
604             self.omega = omega_all

864         @master_only
865         def compute_zp_spectral_function(self):
866             """
867             Compute the spectral function of all quasiparticles in the
868             semi-static approximation, that is, the 'upper bands' contribution
869             to the self-energy is evaluated at the bare energy.
870     
871             The spectral function is evaluated on a frequency mesh 'omegase'
872             that is shifted by the bare energies, such that, what is retured is
873     
874                 A'_kn(omega) = A_kn(omega + E^0_kn)
875     
876             """
877             nomegase = self.nomegase
878             nkpt = self.nkpt
879             nband = self.nband
880     
881             self.spectral_function = np.zeros((nomegase, nkpt, nband), dtype=float)
882     
883             omega = np.einsum('kn,l->knl', np.ones((nkpt, nband)), self.omegase)
884     
885             self.spectral_function = (
886                 (1 / np.pi) * np.abs(self.self_energy.imag) /
887                 ((omega - self.self_energy.real) ** 2 + self.self_energy.imag ** 2)
888                 )
889     
890         @master_only
891         def compute_td_spectral_function(self):
892             """
893             Compute the spectral function of all quasiparticles in the
894             semi-static approximation, that is, the 'upper bands' contribution
895             to the self-energy is evaluated at the bare energy.
896     
897             The spectral function is evaluated on a frequency mesh 'omegase'
898             that is shifted by the bare energies, such that, what is retured is
899     
900                 A'_kn(omega) = A_kn(omega + E^0_kn)
901     
902             """
903             nomegase = self.nomegase
904             nkpt = self.nkpt
905             nband = self.nband
906             ntemp = self.ntemp
907     
908             self.spectral_function_T = np.zeros((nomegase, ntemp, nkpt, nband),
909                                                 dtype=float)
910     
911             omega = np.einsum('ijt,l->ijlt',
912                               np.ones((nkpt, nband, ntemp)), self.omegase)
913     
914             self.spectral_function_T = (
915                 (1 / np.pi) * np.abs(self.self_energy_T.imag) /
916                 ((omega - self.self_energy_T.real) ** 2
917                  + self.self_energy_T.imag ** 2)
918                 )
