6     def reduce_kpoints(subset, fname_in, fname_out):
7         """
8         Reduce the number of kpoints by selecting a subset of indices.
9         Read the file fname_in and write the reduced data in fname_out.
10         Use the name of fname_in to determine the type of file, with possible
11         extentions: EIG.nc, EIGR2D.nc, EIGI2D.nc, GKK.nc, FAN.nc
12         """
13         if fname_in.endswith('EIG.nc'):
14             return reduce_kpoints_eig(subset, fname_in, fname_out)
15         elif fname_in.endswith('EIGR2D.nc') or fname_in.endswith('EIGI2D.nc'):
16             return reduce_kpoints_eigr2d(subset, fname_in, fname_out)
17         elif fname_in.endswith('GKK.nc'):
18             return reduce_kpoints_gkk(subset, fname_in, fname_out)
19         elif fname_in.endswith('FAN.nc'):
20             return reduce_kpoints_fan(subset, fname_in, fname_out)
21         else:
22             raise Exception('Unrecognized file type for file {}'.format(fname_in))

117     def reduce_dim(subset, dname, varnames, fname_in, fname_out):
118         """
119         Copy a netcdf file from fname_in into fname_out,
120         but reduce the dimension 'dname' 
121         """
122     
123         newdim = len(subset)
124     
125         with nc.Dataset(fname_in, 'r') as dsin:
126             with nc.Dataset(fname_out, 'w') as dsout:
127                 nc_copy(dsin, dsout,
128                         except_dimensions=[dname],
129                         except_variables=varnames)
130     
131                 dsout.createDimension(dname, newdim)
132     
133                 for varname in varnames:
134     
135                     datatype = dsin.variables[varname].datatype
136                     dimensions = dsin.variables[varname].dimensions
137                     axis = dimensions.index(dname)
138     
139                     data = dsout.createVariable(varname, datatype, dimensions)
140                     data[...] = np.take(dsin.variables[varname][...], subset, axis)
