155         def to_string(self, verbose=0, with_dataframe=True, width=90):
156             """
157             String representation with verbosity level `verbose`.
158             Text is wrapped at `width` columns.
159             """
160             w = TextWrapper(initial_indent="\t", subsequent_indent="\t", width=width)
161             lines = []; app = lines.append
162             app("%s:\n\t%s\n" % (self.__class__.__name__, os.path.relpath(self.path)))
163             app("Use modules:\n%s\n" % w.fill(", ".join(mod.name for mod in self.all_used_mods)))
164             app("Use dir_levels:\n%s\n" % w.fill(", ".join(map(str, sorted(set(mod.dirlevel for mod in self.all_used_mods))))))
165             app("Includes:\n%s\n" % w.fill(", ".join(self.all_includes)))
166             app("Used by modules:\n%s\n" % w.fill(", ".join(mod.name for mod in self.all_usedby_mods)))
167             app("Used by dir_levels:\n%s\n" % w.fill(", ".join(map(str, sorted(set(mod.dirlevel for mod in self.all_usedby_mods))))))
168     
169             if verbose:
170                 for a in ["programs", "modules", "subroutines", "functions"]:
171                     plist = getattr(self, a)
172                     if not plist: continue
173                     app("Fortran file contains %d %s(s)\n" % (len(plist), a[:-1]))
174                     for p in plist:
175                         app(p.to_string(verbose=verbose, width=width))
176     
177             if with_dataframe:
178                 df = self.get_stats(as_dataframe=True)
179                 app(df.to_string())
180     
181             if verbose > 1:
182                 app(self.stree())
183     
184             return "\n".join(lines)

217         def find_public_entity(self, name, all_names=None):
218             """
219             Find and return the public procedure or datatype with `name`.
220             Return None if not found.
221             """
222             for a in ["modules", "programs", "subroutines", "functions"]:
223                 for p in getattr(self, a):
224                     for e in p.public_procedures:
225                         if e.name == name: return e
226                         if all_names is not None: all_names.append(e.name)
227                     if a == "modules":
228                         # Try also in datatypes and interfaces.
229                         for e in p.types:
230                             if e.name == name: return e
231                             if all_names is not None: all_names.append(e.name)
232                         for e in p.interfaces:
233                             if e.name == name: return e
234                             if all_names is not None: all_names.append(e.name)
235     
236             return None

409     class AbinitProject(object):
410         """
411         This object defines the main entry point for client code.
412         It contains a dictionary (fort_files) mapping the basename of the Fortran files to FortranFile instances.
413         Provides methods to traverse the DAG, print information about the source code
414         and generate the configuration files required by the build system.
415         """

446         def __init__(self, srcdir, processes=1, verbose=0):
447             # Find directories with abinit.src files inside srcdir
448             # and get list of files treated by the build system.
449             self.srcdir = os.path.abspath(srcdir)
450             self.dirpaths = self.get_dirpaths()
451             self.verbose = verbose
452     
453             # Get source files from abinit.src and build mapping basename --> FortranFile
454             start = time.time()
455             def filter_fortran(files):
456                 return [f for f in files if f.endswith(".f") or f.endswith(".F90")]
457     
458             import imp
459             name2path = OrderedDict()
460             for d in self.dirpaths:
461                 if os.path.basename(d) == "98_main":
462                     # Special treatment for programs (no abinit.src here)
463                     for basename in filter_fortran(os.listdir(d)):
464                         if basename in name2path:
465                             raise RuntimeError("Found two Fortran files with same basename `%s`" % basename)
466                         name2path[basename] = os.path.join(d, basename)
467                 else:
468                     # Get source files from abinit.src.
469                     abinit_src = os.path.join(d, "abinit.src")
470                     mod = imp.load_source(abinit_src, abinit_src)
471                     for basename in filter_fortran(mod.sources):
472                         if basename in self.IGNORED_FILES: continue
473                         if basename in name2path:
474                             raise RuntimeError("Found two Fortran files with same basename `%s`" % basename)
475                         name2path[basename] = os.path.join(d, basename)
476     
477             print("Using %d processes to analyze %d directories with %d files" % (processes, len(self.dirpaths), len(name2path)))
478     
479             self.fort_files = OrderedDict()
480             if processes == 1:
481                 for basename, path in name2path.items():
482                     self.fort_files[basename] = FortranFile.from_path(path, macros=self.MACROS, verbose=self.verbose)
483             else:
484                 from multiprocessing import Pool
485                 pool = Pool(processes=processes)
486                 results = pool.map(self._pool_f, list(name2path.items()))
487                 for basename, fort_file in results:
488                     self.fort_files[basename] = fort_file
489     
490             print("Parsing completed in %.2f [s]" % (time.time() - start))
491             self.correlate()
492

496             #sys.stderr = open(str(os.getpid()) + ".err", "w")
497             basename, path = item
498             return (basename, FortranFile.from_path(path, macros=self.MACROS, verbose=self.verbose))

500         def correlate(self):
501             print("Building dependency graph ...")
502             start = time.time()
503     
504             # def correlate()
505             # Build dependency graph
506             # TODO: check for cyclic dependencies. ?
507             for fort_file in self.fort_files.values():
508                 for use_name in fort_file.all_uses:
509                     if use_name in EXTERNAL_MODS: continue
510                     try:
511                         used_mod = self.all_modules[use_name]
512                     except KeyError:
513                         raise RuntimeError(("Fortran module `%s` used by `%s` not found in Abinit project.\n" +
514                                             "Add it to the EXTERNAL_MODS set if it's not a typo.") % (use_name, fort_file.path))
515                     fort_file.all_used_mods.append(used_mod)
516     
517                     # FIXME
518                     key = os.path.basename(used_mod.path)
519                     self.fort_files[key].all_usedby_mods.append(fort_file)
520     
521             pub_procs = self.all_public_procedures()
522             all_interfaces = self.get_all_interfaces()
523             miss = []
524             for fort_file in self.fort_files.values():
525                 for proc in fort_file.iter_procedures():
526                     for child_name in proc.children:
527                         try:
528                             pub_procs[child_name].parents.append(proc)
529                         except KeyError:
530                             # TODO: Could be in subroutine contains
531                             #if child_name in all_interfaces:
532                             #    print("Found in interfaces:", child_name)
533                             #else:
534                             miss.append(child_name)
535     
536             def is_internal(name):
537                 return not any(name.startswith(s) for s in
538                     set(("mpi_", "dfftw_", "mkl_", "papif_", "plasma_", "elpa", "etsf_io_", "blacs_",
539                          "libpaw_", "gsl_", "gpu_", "xc_", "bigdft_")))
540     
541             miss = filter(is_internal, miss)
542             from .check_linalg_calls import blas_routines, lapack_routines
543             from .regex import FORTRAN_INTRINSICS
544             if miss:
545                 miss = set(miss)
546                 # Remove blas and lapack routines.
547                 miss = miss.difference(blas_routines)
548                 miss = miss.difference(lapack_routines)
549                 # Remove pblas and scalalapack routines.
550                 miss = miss.difference("p" + n for n in blas_routines)
551                 miss = miss.difference("p" + n for n in lapack_routines)
552                 # Remove public interfaces.
553                 miss = miss.difference(self.get_all_interfaces().keys())
554                 # Remove intrincs.
555                 miss = miss.difference(FORTRAN_INTRINSICS)
556                 miss = sorted(miss)
557                 print("Cannot find %d callees. Use --verbose to show list." % len(miss))
558                 if self.verbose: pprint(miss)
559     
560             print("Graph completed in %.2f [s]" % (time.time() - start))

791         def find_allmods(self, head_path):
792             """
793             Traverse the *entire* graph starting from head_path.
794             Return full list of `Module` objects required by head_path.
795             """
796             head = self.fort_files[head_path]
797             allmods, queue, visited = set(), set(), set()
798             queue.add(head)
799             while queue:
800                 fort_file = queue.pop()
801                 for mod in fort_file.all_used_mods:
802                     if mod.basename in visited: continue
803                     visited.add(mod.basename)
804                     allmods.add(mod)
805                     queue.add(self.fort_files[mod.basename])
806     
807             return allmods

814             # NB: To treat `dependencies` in an automatic way, I would need either an
