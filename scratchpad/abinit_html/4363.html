336       def E_kinCalculation(self):
337           natom = self.getNatom() 
338           mass  = self.getMass()
339           vel   = self.getVel()
340           self.E_kin= np.zeros(self.Nbtime-1)
341           self.E_kin[:] = np.sum((vel[:,:,0]**2+vel[:,:,1]**2+vel[:,:,2]**2)[:,:] * mass[:] * 0.5,axis=1)

343       def mass_calculation(self):
344           self.mass=[]
345           typat = self.getTypat()
346           amu = self.getAmu()
347           for typ in typat:
348               self.mass.extend([self.amu_emass*amu[int(typ-1)]])

350       def temperature_calculation(self):
351           Nbstep = self.getNbStep()-1
352           E_kin = self.getE_kin()
353           self.temperature = np.zeros(Nbstep)
354           for itim in range(Nbstep):
355               self.temperature[itim] = 2.*E_kin[itim]*self.Ha_eV/(3.*self.kb_eVK*self.getNatom())
356     
357       def pressure_calculation(self):          
358           Nbstep  = self.getNbStep()-1
359           TEMPER  = self.getTemp()
360           Vol     = self.getVol()
361           Stress  = self.getStress()
362           natom   = self.getNatom()
363           
364           self.pressure= np.zeros(Nbstep)
365           for itim in range(Nbstep):
366               Pionique = (natom/Vol[itim])*self.kb_eVK*TEMPER[itim] * self.fact /self.Ha_eV
367               PP=-(Stress[itim][0]+Stress[itim][1]+Stress[itim][2])/3. + Pionique
368               self.pressure[itim] = PP

370       def volume_calculation(self):
371           Nbstep = self.getNbStep()
372           rprim  = self.getRPrim()
373     
374           self.volume =  np.zeros(Nbstep)
375           self.volume[:] = rprim[:,0,0]*(rprim[:,1,1]*rprim[:,2,2]-rprim[:,1,2]*rprim[:,2,1]) \
376               +  rprim[:,0,1]*(rprim[:,1,2]*rprim[:,2,0]-rprim[:,1,0]*rprim[:,2,2]) \
377               +  rprim[:,0,2]*(rprim[:,1,0]*rprim[:,2,1]-rprim[:,1,1]*rprim[:,2,0])
378     
379       def angles_calculation(self):
380           Nbstep = self.getNbStep()
381           rprim  = self.getRPrim() 
382           self.angles = np.zeros((Nbstep,3))
383           self.angles[:,0] = (np.arccos((rprim[:,0,0]*rprim[:,1,0]+rprim[:,0,1]*rprim[:,1,1]+rprim[:,0,2]*rprim[:,1,2])\
384                               /np.sqrt((rprim[:,0,0]**2+rprim[:,0,1]**2+rprim[:,0,2]**2)*(rprim[:,1,0]**2+rprim[:,1,1]**2+rprim[:,1,2]**2))))
385           self.angles[:,1]  = (np.arccos((rprim[:,0,0]*rprim[:,2,0]+rprim[:,0,1]*rprim[:,2,1]+rprim[:,0,2]*rprim[:,2,2])\
386                               /np.sqrt((rprim[:,0,0]**2+rprim[:,0,1]**2+rprim[:,0,2]**2)*(rprim[:,2,0]**2+rprim[:,2,1]**2+rprim[:,2,2]**2))))
387           self.angles[:,2] = (np.arccos((rprim[:,2,0]*rprim[:,1,0]+rprim[:,2,1]*rprim[:,1,1]+rprim[:,2,2]*rprim[:,1,2])\
388                               /np.sqrt((rprim[:,2,0]**2+rprim[:,2,1]**2+rprim[:,2,2]**2)*(rprim[:,1,0]**2+rprim[:,1,1]**2+rprim[:,1,2]**2))))
