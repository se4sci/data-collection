73     import netCDF4 as nc
74     import matplotlib.pyplot as plt
75     
76     def plot_td_self_energy(fname, ikpt=0, ieig=0):
77         """Plot the self-energy computed at several temperatures."""
78     
79         with nc.Dataset(fname, 'r') as ds:
80         
81             # nomega
82             omega = ds.variables['omegase'][...]
83         
84             # ntemp
85             temperatures = ds.variables['temperatures'][...]
86         
87             # nspin, nkpt, nband, nomega, ntemp
88             self_energy = ds.variables['self_energy_temperature_dependent'][...]
89         
90         for i, T in enumerate(temperatures):
91         
92             # Select one state
93             re = self_energy[0,ikpt,ieig,:,i,0]
94             im = self_energy[0,ikpt,ieig,:,i,1]
95             
96             plt.plot(omega, re, label='T={}K'.format(int(T)))
97     
98         plt.legend(loc='upper right')
99     
100         fig = plt.gcf()
101         plt.close()
102     
103         return fig
104
