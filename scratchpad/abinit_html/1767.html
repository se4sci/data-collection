69     #define perrorexit(s) { perror(s); exit(1); }
70     
71     static int kill_signal = SIGKILL;
72     static char *progname;
73     static char *commandname;
74     static int  time_to_run;
75     
76     static void usage()
77     {
78         fprintf(stderr, "usage: %s [-signal] time command...\n", progname);
79         exit(1);
80     }
81     
82     static void terminate(sig)
83     int     sig;
84     {
85         signal(kill_signal, SIG_DFL);
86         fprintf(stderr, "timeout.c: aborting command ``%s'' after %d seconds with signal %d\n",
87     	    commandname, time_to_run, kill_signal);
88         kill(0, kill_signal);
89     }
90     
91     int     main(argc, argv)
92     int     argc;
93     char  **argv;
94     {
95         /*int     time_to_run;*/
96         pid_t   pid;
97         pid_t   child_pid;
98         int     status;
99     
100         progname = argv[0];
101     
102         /*
103          * Parse JCL.
104          */
105         while (--argc && *++argv && **argv == '-')
106     	if ((kill_signal = atoi(*argv + 1)) <= 0)
107     	    usage();
108     
109         if (argc < 2 || (time_to_run = atoi(argv[0])) <= 0)
110     	usage();
111     
112         commandname = argv[1];
113     
114         /*
115          * Run the command and its watchdog in a separate process group so that
116          * both can be killed off with one signal.
117          */
118         setsid();
119         switch (child_pid = fork()) {
120         case -1:					/* error */
121     	perrorexit("timeout: fork");
122         case 0:					/* run controlled command */
123     	execvp(argv[1], argv + 1);
124     	perrorexit(argv[1]);
125         default:					/* become watchdog */
126     	(void) signal(SIGHUP, terminate);
127     	(void) signal(SIGINT, terminate);
128     	(void) signal(SIGQUIT, terminate);
129     	(void) signal(SIGTERM, terminate);
130     	(void) signal(SIGALRM, terminate);
131     	alarm(time_to_run);
132     	while ((pid = wait(&status)) != -1 && pid != child_pid)
133     	     /* void */ ;
134     	return (pid == child_pid ? WEXITSTATUS(status) : -1);
135         }
136     }
