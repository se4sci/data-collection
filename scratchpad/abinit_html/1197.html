5     subroutine average(data,size_data,average_value)
6     
7     ! input values:  
8       integer :: size_data
9       real(8),dimension(size_data),intent(in) :: data
10      !f2py optional , depend(data) :: size_data=len(data)
11       real(8),intent(out) :: average_value
12       average_value  = 0.0
13     
14       do i=1,size_data
15         average_value  = average_value + data(i)
16       end do
17     
18       average_value = average_value / size_data
19       
20     end subroutine average

24     !------STANDAR DEVIATION ROUTINE-----!
25     !------------------------------------!
26     subroutine standard_deviation(data,average_input,size_data,deviation)
27     !  input  values:    
28       integer  ::  size_data
29       real(8), dimension(size_data),intent(in)  ::  data
30       real(8), intent(in), optional  ::  average_input  
31       !f2py  optional  ,  depend(data)  ::  size_data=len(data)
32       real(8),intent(out)  ::  deviation
33       real(8) :: average_data = 0
34     
35       
36       if (average_input/=0) then
37         average_data = average_input
38       else
39         average_data = 0
40         call  average(data,size_data,average_data)
41       end if
42     
43       do  i=1,size_data
44         deviation  =  deviation  +  (data(i)  -  average_data)**2
45       end  do
46         
47       deviation  =  deviation/size_data
48       deviation  =  deviation**0.5
49     
50     end subroutine  standard_deviation

55     !------------------------------------!
56     subroutine mean_square_displacement(position,indexAtom1,msd,size2,size3,nba1)
57       integer :: size2,size3,nba1,nbtime
58       !input  values:    
59       !f2py  optional  ,  depend(position)    ::  size2=len(position[:,1,1])
60       !f2py  optional  ,  depend(position)    ::  size3=len(position[1,:,1])
61       !f2py  optional  ,  depend(indexAtom1)  ::  nba1=len(indexAtom1)
62       integer, intent(in)  ::  indexAtom1(nba1)
63       real, intent(in)  ::  position(size2,size3,3)
64       !output 
65       real(8),intent(out)  ::  msd((size2-1)/2)
66     
67       !local variable
68       real(8), dimension(3)  ::  p1=(/1,1,1/),p2=(/1,1,1/)
69       real(8) r
70       integer i,k,tau
71     
72       nbtime = size2
73     
74       msd(:) = 0
75       do tau=1,int((nbtime-1)/2) !loop on step
76         do i=1,nbtime-tau-1 !loop on step
77           do k=1,size(indexAtom1)
78             p1 = position(i,indexAtom1(k),:)
79             p2 = position(i+tau,indexAtom1(k),:) - p1(:)
80             r = p2(1)**2+p2(2)**2+p2(3)**2
81             msd(tau)= msd(tau) + r
82           end do
83         end do
84         msd(tau)= msd(tau) / (nbtime-tau-1)
85       end do
86       msd(:) = msd(:)/(size(indexAtom1))
87     end subroutine mean_square_displacement
