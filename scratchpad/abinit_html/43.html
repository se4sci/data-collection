15     void dim_screening(double *p, double *y, int m, int n, void *adata)
16     {
17      /* The function to be fitted evaluated at the values x
18       * p[m] are the parameters
19       * y[n] is the measurement vector (containing the y-values)
20       * adata[n] is the optional data supplied (fixed) this will contain
21       *  the z-values with adata[0],adata[2],...,adata[2*i] containing the
22       *  real coordinate and adata[1],adata[3],...,adata[2*i+1] containing 
23       *  the imaginary coordinate (0<=i<n) */
24     
25     register int ip,in,idx;
26     double *data;
27     register double realp,imagp;
28     int npoles = m/3;
29     
30       data = (double *)adata;
31     
32       /* There is always at least one pole so do assignment in first loop*/
33       for(in=0; in<n; ++in){
34     
35         realp = -data[2*in+1]*p[2]+p[1]*p[1]-data[2*in]*data[2*in];
36         imagp = data[2*in]*(2.0*data[2*in+1]-p[2]);
37     
38         y[in] = -p[0]*imagp/(realp*realp+imagp*imagp);
39       }
40     
41       if(npoles>1) { /* if there are more poles, add the rest*/
42         for(in=0; in<n; ++in){
43           for(ip=1; ip<npoles; ++ip){
44     
45             realp  = -data[2*in+1]*p[3*ip+2]+p[3*ip+1]*p[3*ip+1]-data[2*in]*data[2*in];
46             imagp  = data[2*in]*(2.0*data[2*in+1]-p[3*ip+2]);
47     
48             y[in] += -p[3*ip]*imagp/(realp*realp+imagp*imagp);
49     
50           }
51         }
52       }
53     /* end */
54     }

56     void dre_and_im_screening(double *p, double *y, int m, int n, void *adata)
57     {
58      /* The function to be fitted evaluated at the values x
59       * p[m] are the parameters
60       * y[n] is the measurement vector (containing the y-values)
61       * adata[n] is the optional data supplied (fixed) this will contain
62       *  the z-values with adata[0],adata[2],...,adata[2*i] containing the
63       *  real coordinate and adata[1],adata[3],...,adata[2*i+1] containing 
64       *  the imaginary coordinate (0<=i<n) */
65     
66     register int ip,in,idx;
67     double *data;
68     register double realp,imagp;
69     register int npoles = m/3;
70     
71       data = (double *)adata;
72     
73       /* There is always at least one pole so do assignment in first loop*/
74       for(in=0; in<n/2; ++in){
75     
76         realp = -data[2*in+1]*p[2]+p[1]*p[1]-data[2*in]*data[2*in];
77         imagp = data[2*in]*(2.0*data[2*in+1]-p[2]);
78     
79         y[in    ] = -p[0]*imagp/(realp*realp+imagp*imagp);
80         y[n/2+in] = -p[0]*realp/(realp*realp+imagp*imagp);
81       }
82     
83       if(npoles>1) { /* if there are more poles, add the rest*/
84         for(in=0; in<n/2; ++in){
85           for(ip=1; ip<npoles; ++ip){
86     
87             realp  = -data[2*in+1]*p[3*ip+2]+p[3*ip+1]*p[3*ip+1]-data[2*in]*data[2*in];
88             imagp  = data[2*in]*(2.0*data[2*in+1]-p[3*ip+2]);
89     
90             y[in    ] += -p[3*ip]*imagp/(realp*realp+imagp*imagp);
91             y[n/2+in] += -p[3*ip]*realp/(realp*realp+imagp*imagp);
92     
93           }
94         }
95       }
96     /* end */
97     }
