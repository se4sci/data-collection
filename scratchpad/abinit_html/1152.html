1     #!/usr/bin/env python
2     """
3     This script automates the steps needed to debug a MPI executable with the GNU debugger
4     Note that GNU gdb is not a parallel debugger hence this crude approach may not work properly.
5     """
6     from __future__ import print_function, division
7

9     import os
10     import argparse 
11     import tempfile
12     
13     from subprocess import Popen, PIPE

15     def str_examples():
16         examples = (
17           "\n"
18           "Usage example:\n\n" 
19              "pmdbg.py -n 2 abinit files_file  ==> (Try) to run Abinit in parallel under the control of gdb\n"
20              "                                      Use 2 MPI processes, read input from files_file "
21         )
22         return examples
23     
24     def show_examples_and_exit(err_msg=None, error_code=1):
25         """Display the usage of the script."""
26         sys.stderr.write(str_examples())
27         if err_msg: 
28             sys.stderr.write("Fatal Error\n" + err_msg + "\n")
29         sys.exit(error_code)
30     
31     
32     def main():
33         """mpirun -n# xterm gdb binary -command=file"""
34         parser = argparse.ArgumentParser(epilog=str_examples(),formatter_class=argparse.RawDescriptionHelpFormatter)
35     
36         #parser.add_argument('-v', '--verbose', default=0, action='count', # -vv --> verbose=2
37         #                     help='verbose, can be supplied multiple times to increase verbosity')
38     
39         parser.add_argument('-n', default=1, type=int, help='Number of MPI processes')
40     
41         parser.add_argument("args", nargs="+", help='executable stdin_file')
42     
43         # Parse the command line.
44         try:
45             options = parser.parse_args()
46         except Exception:
47             show_examples_and_exit(error_code=1)
48     
49         binary, stdin_fname= options.args[0], options.args[1]
50     
51         _, dbg_fname = tempfile.mkstemp()
52     
53         with open(dbg_fname, "w") as fh:
54             fh.write("run < %s" % stdin_fname)
55     
56         # mpirun -n# xterm gdb binary -command=file.
57         cmd = "mpirun -n %i xterm -e gdb %s -command=%s" % (options.n, binary, dbg_fname)
58         print(cmd)
59     
60         p = Popen(cmd, shell=True) #, cwd=cwd, env=env)
61         retcode = p.wait()
62     
63         try:
64             os.remove(dbg_fname)
65         except IOError:
66             pass
