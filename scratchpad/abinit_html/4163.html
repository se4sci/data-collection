9             self.config = {
10                 'base_path': ['.', 'Default location from which to evaluate ' \
11                     'relative paths for the include statement.'],
12                 'encoding': ['utf-8', 'Encoding of the files used by the include statement.']
13             }
14             for key, value in configs.items():
15                 self.setConfig(key, value)

17         def extendMarkdown(self, md, md_globals):
18             md.preprocessors.add(

22     class IncludePreprocessor(Preprocessor):
23         """
24         This provides an "include" function for Markdown, similar to that found in
25         LaTeX (also the C pre-processor and Fortran). The syntax is {% filename %},
26         which will be replaced by the contents of filename. Any such statements in
27         filename will also be replaced. This replacement is done prior to any other
28         Markdown processing. All file-names are evaluated relative to the location
29         from which Markdown is being called.
30         """
31         def __init__(self, md, config):
32             super(IncludePreprocessor, self).__init__(md)
33             self.base_path = config['base_path']
34             self.encoding = config['encoding']
35     
36         def run(self, lines):
37             from abimkdocs.website import Website
38             website = Website.get()
39             return website.preprocess_mdlines(lines)
40     
41     
42     def makeExtension(*args,**kwargs):
43         return MarkdownInclude(*args, **kwargs)
