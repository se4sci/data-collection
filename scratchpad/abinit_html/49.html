24     int fstrstr(FILE *fp, const char* Str, const int ScanLen)
25     { 
26      int ScanChar;
27      long curFP, bytes=0;
28      
29      if (ScanLen == 0)
30        return 1;  // null string always return TRUE.
31        
32      curFP = ftell(fp);
33      do {
34        ScanChar = fgetc(fp);
35        if ((char)ScanChar == Str[0])
36        {
37          int i;
38          for (i=1; i<ScanLen; i++)
39            if ((ScanChar = fgetc(fp)) != Str[i])
40     	 break;
41          if (i == ScanLen) // Success!
42            return 1;
43          else 
44            fseek(fp, -i, SEEK_CUR);
45        }
46      } while (ScanChar != EOF);
47      // Search again from the beginning of the file
48      fseek(fp, 0, SEEK_SET);
49      do {
50        ScanChar = fgetc(fp);
51        if ((char)ScanChar == Str[0])
52        {
53          int i;
54          for (i=1; i<ScanLen; i++)
55            if ((ScanChar = fgetc(fp)) != Str[i])
56     	 break;
57          if (i == ScanLen) // Success!
58            return 1;
59          else 
60            fseek(fp, -i, SEEK_CUR);
61        }
62      } while (++bytes < curFP);
63      // If search failed, the fp will return old position.
64      return 0;
65     }

67     int readData(char *filename)
68     {
69      FILE *fp;
70      int row, col, z;
71      int ScanLength = strlen(strSign);
72      int ScanLength2 = strlen(strSign2);
73      int InnerIntervalLength = strlen(strInnerInterval);
74      int OuterIntervalLength = strlen(strOuterInterval);
75       
76      if (!(fp = fopen(filename, "rb")))
77        return -1;
78     
79      // Now scan the buffer string:
80      fseek(fp, 0L, SEEK_SET);
81      if (!fstrstr(fp, strSign, ScanLength))
82        return -3;
83      if (!fstrstr(fp, strSign2, ScanLength2))
84        return -7;
85        
86      for (z=0; z<L; z++)
87      { 
88        if (!fstrstr(fp, strOuterInterval, OuterIntervalLength)) 
89          return -9;
90        for (row=0; row<M; row++)
91        {     
92          if (!fstrstr(fp, strInnerInterval, InnerIntervalLength)) // Bypass "kpt#...coord)" parts
93            return -5;
94          for (col=0; col<N; col++)
95            fscanf(fp, "%lf", &Data[row+M*col+z*M*N]);
96        }   
97      }
98     
99      fclose(fp);
100      return 1;
101     }

103     void mexFunction(int nlhs, mxArray *plhs[], int nrhs, const mxArray *prhs[])
104     {
105      char name[128];
106      int dims[3]={1, 1, 1}, i, status;
107      double *reDIMS; 
108      
109      if (nrhs <= 1)
110      {
111        mexPrintf("\nData Extractor\n"
112                  "Usage: mexData([M N L], ReadFileNameString, SignatureString, InnerIntervalString, OuterIntervalString, SubSignatureString);\n"
113                  "       mexData extracts data from ASCII code file which contains useful data as well as text infomation.\n\n"
114                  "       As for two-dimension data, mexData searches Signature_String which indicates the useful data will appear,\n"
115                  "       then searches the Inner_Interval_String to locate the accurate position of the useful data.\n"
116                  "       After the N columns data are read, mexData again searches Interval_String to prepare reading another row data.\n"
117                  "       When all M-row by N-column data are extracted, mexData returns a double float M-by-N matrix and finishes its work.\n\n"
118                  "       As for three-dimension M-by-N-by-L data, Outer_Interval_String is used to differentiate every L dimension.\n"
119                  "       mexData returns M-by-N-by-L matrix.\n\n"
120                  "       Modified at 7/30/2002, add OuterIntervalString, SubSignatureString.\n"
121                  "       Modified at 8/11/2002, correct L dimension disposition(use Data[row+M*col+z*M*N] instead of Data[L*row+M*L*col+z]).\n"
122                  "       Used by your own warranty, Mike.Zhang(mz24cn@hotmail.com), June 2002.\n\n");
123        return;
124      }
125      reDIMS = mxGetPr(prhs[0]);
126      dataDims = mxGetN(prhs[0]);
127      for (i=0; i<dataDims; i++)
128        dims[i] = (int)reDIMS[i];
129      M = dims[0]; 
130      N = dims[1];
131      L = dims[2];
132      data = mxCreateNumericArray(dataDims, dims, mxDOUBLE_CLASS, mxREAL);
133      Data = mxGetPr(data);
134      
135      if (nrhs >= 2)
136      {
137        mxGetString(prhs[1], name, 127);
138        if (nrhs >= 3)
139          mxGetString(prhs[2], strSign, 127);
140        if (nrhs >= 4)
141          mxGetString(prhs[3], strInnerInterval, 127);
142        if (nrhs >= 5)
143          mxGetString(prhs[4], strOuterInterval, 127);
144        if (nrhs >= 6)
145          mxGetString(prhs[5], strSign2, 127);
146      }
147      mexPrintf("Data Extractor\n"
148                "Read file              :      '%s'\n"
149                "Signature string       :      '%s'\n"
150                "InnerInterval string   :      '%s'\n"
151                "OuterInterval string   :      '%s'\n"
152                "2nd Signature string   :      '%s'\n"
153                "Output array dimensions:      %d x %d x %d\n", 
154                name, strSign, strInnerInterval, strOuterInterval, strSign2, M, N, L);
155      status = readData(name);
156      if (status == 1)
157        mexPrintf("Data extracted success.\n");
158      else if (status == -1)
159        mexPrintf("Error: Can't open file '%s'.\n", name);
160      else if (status == -3)
161        mexPrintf("Error: Can't find string '%s'.\n", strSign);
162      else if (status == -5)
163        mexPrintf("Error: Can't find string '%s'.\n", strInnerInterval);
164      else if (status == -7)
165        mexPrintf("Error: Can't find string '%s'.\n", strSign2);
166      else if (status == -9)
167        mexPrintf("Error: Can't find string '%s'.\n", strOuterInterval);
168        
169      plhs[0] = data;
170     }
