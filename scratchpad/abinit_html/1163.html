33     env_ignore = ["CFLAGS","CXXFLAGS","FCFLAGS","NVCC_*","fcflags_opt_*"]
34     
35     def is_ignored(keyword):
36       for env in env_ignore:
37         if ( "*" in env ):
38           if ( re.match(env,keyword) ):
39             return True
40         elif ( env == keyword ):
41             return True
42       return False

57     def main(home_dir):
58       from os.path import join as pj
59                                                                                
60       # Check if we are in the top of the ABINIT source tree
61       my_name = os.path.basename(__file__) + ".main"
62       if ( not os.path.exists( pj(home_dir,"configure.ac") ) or
63            not os.path.exists( pj(home_dir,"src/98_main/abinit.F90")) ):
64         print("%s: You must be in the top of an ABINIT source tree." % my_name)
65         print("%s: Aborting now." % my_name)
66         sys.exit(1)
67     
68       # Init
69       re_dbgflags = re.compile("(^-g|[^0-9A-Za-z]-g)")
70       re_optflags = re.compile("(-O[0-9]|-xO[0-9])")
71     
72       # Extract environment variables from config file
73       cnf_env = MyConfigParser()
74       cnf_env.read( pj(home_dir,"config/specs/environment.conf") )
75       env_config = list()
76       for env in cnf_env.sections():
77         if cnf_env.get(env,"reset") == "no":
78           if not is_ignored(env):
79             env_config.append(env)
80       env_config.sort()
81     
82       # Extract information from build example config file
83       bex_ok = True
84       cnf_bex = MyConfigParser()
85       cnf_bex.read( pj(home_dir,"config/specs/build-examples.conf") )
86       env_forbidden = dict()
87       for bot in cnf_bex.sections():
88         env_forbidden[bot] = list()
89         for var in cnf_bex.options(bot):
90           if var in env_config:
91             val = cnf_bex.get(bot,var)
92             if re_dbgflags.search(val):
93               env_forbidden[bot].append(("D",var))
94               bex_ok = False
95             if re_optflags.search(val):
96               env_forbidden[bot].append(("O",var))
97               bex_ok = False
98       env_fkeys = list(env_forbidden.keys())
99       env_fkeys.sort()
100     
101       # Report any match
102       my_exitcode = 0
103       if not bex_ok:
104         sys.stderr.write("%s: reporting use of forbidden flags\n\n" % (os.path.basename(sys.argv[0])))
105         sys.stderr.write("X: D=debug / O=Optimization\n\n")
106         sys.stderr.write("%s  %-24s  %-48s\n" % ("X","Variable","Bot"))
107         sys.stderr.write("%s  %s  %s\n" % ("-","-" * 24,"-" * 48))
108     
109         for bot in env_fkeys:
110           if len(env_forbidden[bot]) > 0:
111             my_exitcode = 1
112             for (tag,var) in env_forbidden[bot]:
113               sys.stderr.write("%s  %-24s  %-48s\n" % (tag,var,bot))
114     
115         sys.stderr.write("\n")
116     
117       return my_exitcode
