13     # Object: compare 2 output files from ABINIT line by line with arithmetic
14     # comparisons of floating point substrings
15     #
16     # Usage: fldiff [-context] [ -ignore | -include ] [ -ignoreP | -includeP ] [ -easy | -medium | -ridiculous ] file1 file2 [label]
17     #
18     # The first character of each line in both files indicates the mode of comparison;
19     # these first characters MUST be identical in corresponding lines.
20     # By default, a floating point comparison with a tolerance of 1.01d-10 is done
21     # on all floating point strings and a character comparison is done on all other
22     # strings, disregarding multiple spaces.
23     # In order for two numbers to be declared different, BOTH the absolute difference
24     # and the relative difference (difference divided by the sum of absolute values)
25     # must be bigger than the tolerance.
26     #
27     # Some special characters at the beginning of lines require a different handling:
28     # -	mark lines as same regardless to their content (i. e. ignore lines)
29     #     (can be be present in the 2 files or not, but must begin with -)
30     # _ mark lines as same regardless to their content
31     #     (must be present in the 2 files, but can begin with _ in only one of them)
32     # +	mark lines as different regardless to their content
33     # ,	handle as + with -include option and as - with -ignore option
34     # P	handle as + with -includeP option and as - with -ignoreP option
35     # %	floating point comparisons are done with a tolerance of 1.01e-2
36     # ;	floating point comparisons are done irrespective of signs
37     # :	ignore floating point numbers and do a characters comparison
38     # .	do a characters comparison, but do not count this line in the Summary
39     #
40     # Both files should have the same number of non - starting lines.
41     
42     # With -context option, save character strings for context and print it
43     # with line number when floating difference is found.
44     #
45     # The -ignore and -include options affects the treatment of the ',' 
46     # special character in the first column (see above) 
47     #
48     # The -ignoreP and -includeP options affects the treatment of the 'P'
49     # special character in the first column (see above)
50     #
51     # If -ridiculous   is specified, the default tolerance is set to 1.01e-2 
52     # If -easy   is specified, the default tolerance is set to 1.01e-5 
53     # If -medium is specified, the default tolerance is set to 1.01e-8 
54     # These modifications do not apply to the tolerance determined by the
55     # '%',and '.' first-column special signs
56     #
57     # If "label" is specified, it will appear in the last, summary line.

105     def num_sigdigits(string):
106         """
107         Returns the number of significant digits in a string representing 
108         a floating point number in the form 1.234 
109         (exponential format is not supported here)
110     
111         >>> num_sigdigits("3. ")
112         0
113         >>> num_sigdigits("-3.27")
114         2
115         """
116         string = string.rstrip()
117         idx = string.find(".")
118         if idx == -1 or idx == len(string): return 0
119         digits = string[idx+1:]
120         assert all(c.isdigit() for c in digits)
121         return len(digits)

336     class Diff(object):
337         """
338         Stores the results of a token comparison.
339     
340         The object evaluates to True in boolean context if the difference is significative.
341         Diffs can be ordered 
342     
343         A diff has a code that is a string that can contain the following characaters:
344             "s" for string difference
345             "a" for absolute difference
346             "r" for relative difference
347             "e" if an error occured (diff of objects of different class
348         """
349         POSSIBLE_CODES = ["e", "r", "a", "s", ]
350     
351         def __init__(self, token1, token2, diff_codes, msg, **kwargs):
352             self.token1, self.token2 = token1, token2
353             self.codes = diff_codes
354             self.msg = msg
355             self.extra = kwargs
356     
357             # None if string
358             self.abs_err = kwargs.get("abs_err", None)
359             self.rel_err = kwargs.get("rel_err", None)
360             if self.abs_err is not None: assert  self.rel_err is not None
361     
362         @classmethod
363         def make_counter(cls):
364             return OrderedDict([(k, 0) for k in cls.POSSIBLE_CODES])
365     
366         def __str__(self):
367             if self.is_numeric:
368                 return str([str(self.token1), str(self.token2), self.msg])
369             else:
370                 return self.msg
371     
372         def __bool__(self):
373             return bool(self.msg)
374     
375         __nonzero__ = __bool__
376     
377         # We want to order diffs so that we can output the most important one that occurs on a line
378         # Numerical differences are more important that string difference
379         # When comparing two numerical differences, the relative error is more important than the absolute one. 
380         def __eq__(self, other):
381             left, right = (self.rel_err, self.abs_err), (other.rel_err, other.abs_err)
382             return left == right
383     
384         def __gt__(self, other):
385             """Python2 supports 2 > None but this syntax raises TypeError in py3k."""
386             left, right = (self.rel_err, self.abs_err), (other.rel_err, other.abs_err)
387     
388             # Compute absolute values
389             try:
390                 left = tuple(map(abs, left))
391             except TypeError:
392                 # (None, None) > x is always False
393                 return False
394             
395             try:
396                 right = tuple(map(abs, right))
397             except TypeError:
398                 # (num1, num2) > (None, None) is always True
399                 return False
400     
401             # Now we can safely compare two tuples of absolute values
402             # TODO: Fix problem with strings
403             return left > right

406         def is_numeric(self):
407             """True if we have a diff of numbers."""
408             ans = self.abs_err is not None
409             if ans: assert self.rel_err is not None
410             return ans

438     class Tokenizer(object):
439         """
440         The `Tokenizer` receives two strings, looks at the character in the first column (they must 
441         be equal) and split the lines in a list of tokens depending on the algorithm specified by 
442         the character in the first column
443         """
444         _HANDLERS = {
445         #  "-":                         # -	mark lines as same regardless to their content (i. e. ignore lines)
446         #                               #     (can be be present in the 2 files or not, but must begin with -)
447         #  "_":                         # _ mark lines as same regardless to their content
448                                         #     (must be present in the 2 files, but can begin with _ in only one of them)
449         #  "+":                         # +	mark lines as different regardless to their content
450           "+": ignore_numbers,          # +	mark lines as different regardless to their content
451         #  ",":                         # ,	handle as + with -include option and as - with -ignore option
452         #  "P":                         # P	handle as + with -includeP option and as - with -ignoreP option
453         #  "%":                         # %	floating point comparisons are done with a tolerance of 1.01e-2
454         #  ";":                         # ;	floating point comparisons are done irrespective of signs
455           ":":  ignore_numbers,         # :	ignore floating point numbers and do a characters comparison
456           ".":  ignore_numbers,         # .	do a characters comparison, but do not count this line in the Summary
457         }
458     
459         def __init__(self, diff_opts):
460             self.diff_opts = diff_opts
461     
462         def make_diff(self, line1, opos1, line2, opos2):
463             """
464             Return sorted(diff_list), exceptions
465             """
466             exceptions = []
467             eapp = exceptions.append
468     
469             ch1, ch2 = line1[0], line2[0]
470             if ch1 != ch2: 
471                 msg = "Different chars in first column"
472                 eapp(DiffLineException(msg))
473     
474             # Select the function that will tokenize the lines.
475             func = self._HANDLERS.get(ch1, parse_numstrings)
476     
477             tokens1 = func(line1)
478             tokens2 = func(line2)
479     
480             # If we don't have the same number of tokens in line
481             # we restrict the comparison to the minimum set of tokens
482             # and we register the exception.
483             if len(tokens1) != len(tokens2):
484                 msg ="Different number of tokens"
485                 eapp(DifferentNumberOfTokens(msg))
486     
487             diff_list = []
488     
489             for tok1, tok2 in zip(tokens1, tokens2):
490                 diff = tok1.diff(tok2, self.diff_opts)
491                 if not diff: continue
492                 diff_list.append(diff)
493     
494             # Return **sorted** list of diffs and exceptions
495             return sorted(diff_list, reverse=True), exceptions

675             self.diffs_map = OrderedDict()
676             self.exceptions = []
677     
678             try:
679                 self.matcher = matcher = Matcher(file1, file2)
680                 #matcher.show()
681             except IOError:
682                 raise 
683     
684             # Add exceptions encountered by matcher.
685             self.exceptions.extend(matcher.exceptions)
686     
687             # Number of different lines.
688             nlines = 0
689             tokenizer = Tokenizer(self.diff_opts)
690     
691             for fline1, fline2 in matcher:
692                 line1, opos1 = fline1.string, fline1.opos
693                 line2, opos2 = fline2.string, fline1.opos
694     
695                 diffs, excs = tokenizer.make_diff(line1, opos1, line2, opos2)
696                 
697                 if excs:
698                     self.exceptions.extend(excs)
699     
700                 if diffs: 
701                     # Store the line in an OrderedDict indexed by (opos1, opos2)
702                     key = (fline1.opos, fline2.opos)
703                     self.diffs_map[key] = diffs
704                     nlines += 1
705     
706             return 1 if self.exceptions or nlines else 0
707     
708         def nlines_totnumdiffs(self):
709             """Return the number of lines with significant differences and the total number of diffs"""
710             nlines, tot_numdiffs = 0, 0
711     
712             count = Diff.make_counter()
713     
714             for (pos1, pos2), diff_list in self.diffs_map.items():
715                 if any(diff_list): nlines += 1
716                 sig_diffs = [d for d in diff_list if d]
717                 tot_numdiffs += len(sig_diffs)
718                 for diff in sig_diffs:
719                     for c in diff.codes: count[c] += 1
720     
721             return nlines, tot_numdiffs, count
722     
723         def get_max_absdiff(self):
724             return self._get_amax("abs_err")
725     
726         def get_max_reldiff(self):
727             return self._get_amax("rel_err")
728     
729         def _get_amax(self, key):
730             # Build flat list of significant numerical diffs (i.e. diffs that evaluate to True)
731             # Use standard decorate and sort algorithm to get the position of the max.
732             items = []
733             for i, (_, diff_list) in enumerate(self.diffs_map.items()):
734                 items.extend([(d, i) for d in diff_list if d and d.is_numeric])
735                                                                                                 
736             # Sort numerical diffs according to |d.key|
737             abserr_sort = sorted(items, key=lambda t: abs(getattr(t[0], key)), reverse=True)
738             #pprint([str(s) for s in abserr_sort])
739     
740             if len(abserr_sort): 
741                 # Return diff, index
742                 return abserr_sort[0] 
743             else:
744                 # Return None if list is empty. 
745                 # This happens when there's no significant difference. 
746                 return None, None

748         def write_results(self, out=sys.stdout, verbose=0):
749             """
750             Print a summary of the comparison.
751             """
752             summary = ["< %s\n> %s\n" % (self.file1, self.file2)]
753             app = summary.append
754     
755             app("Number of exceptions: %d" % len(self.exceptions))
756             for i, exc in enumerate(self.exceptions):
757                 app("Exception %d): %s" % (i, str(exc)))
758     
759             nlines, tot_numdiffs, count = self.nlines_totnumdiffs()
760             app("Number of different lines: %d" % nlines)
761             app("Total number of significant differences: %d" % tot_numdiffs)
762             app("Diff counters: %s" % dict2str(count))
763     
764             # Find the max absolute error and the max relative error.
765             max_absdiff, idx_amax = self.get_max_absdiff()
766             max_reldiff, idx_rmax = self.get_max_reldiff()
767     
768             # Print info on the max absolute error and the max relative error.
769             #if max_absdiff is not None:
770             #if max_reldiff is not None:
771             app("Max absolute difference at %s: %s" % (idx_amax, max_absdiff))
772             app("Max relative difference at %s: %s" % (idx_rmax, max_reldiff))
773     
774             summary = "\n".join(summary)
775     
776             out.write(summary)
777             html = HtmlDiffFile()
778             html.make_summary(summary)
779     
780             for i, ((pos1, pos2), diff_list) in enumerate(self.diffs_map.items()):
781                 # No printout if there's not significant diff for this line.
782                 if not any(diff_list): continue
783     
784                 # the header has the format: diff_index, (pos1, pos2), diff counter
785                 dcount = dict2str(Counter("".join(d.codes for d in diff_list)))
786                 header = "%d, (%d, %d), %s" % (i, pos1, pos2, dcount)
787     
788                 # Write lines.
789                 s1 = "< " + self.matcher.get_line1(pos1)
790                 s2 = "> " + self.matcher.get_line2(pos2)
791     
792                 out.write(header + "\n")
793                 out.write(s1)
794                 out.write(s2)
795     
796                 if not options.verbose:
797                     # Print the most important difference
798                     print(diff_list[0])
799                 else:
800                     print(str(diff_list))
801     
802                 html.add_lines_and_diff(i, header, s1, s2, diff_list, verbose=verbose)
803     
804             out.write("\n")
805     
806             #html.write(os.path.abspath("foo.html"))
807             html.browse()
