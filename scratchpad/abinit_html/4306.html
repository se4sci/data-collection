1     """MPI distribution of work load"""
2     import sys, traceback

4     from mpi4py import MPI

6     comm = MPI.COMM_WORLD
7     size = comm.Get_size()

12     @contextmanager
13     def mpi_abort_if_exception():
14         """Terminate all mpi process if an exception is raised."""
15         try:
16             yield
17         except:
18             exc_type, exc_value, exc_traceback = sys.exc_info()
19             traceback.print_exception(exc_type, exc_value, exc_traceback)
20             MPI.COMM_WORLD.Abort(1)
21     
22     def mpi_watch(f):
23         """Decorator. Terminate all mpi process if an exception is raised."""
24         def g(*args, **kwargs):
25             with mpi_abort_if_exception():
26                 return f(*args, **kwargs)
27         return g
28     
29     def master_only(f):
30         """Decorator. Let a function be executed only by master."""
31         def g(*args, **kwargs):
32             if i_am_master:
33                 with mpi_abort_if_exception():
34                     return f(*args, **kwargs)
35             return
36         return g
