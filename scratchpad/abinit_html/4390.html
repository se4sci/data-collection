24     def tokenize_file(infilename):
25     	"""Take input file name and return list of tokens, eliminating comments"""
26     #  reg expss
27     	import re
28     	import sys
29     	import fileinput
30     	import math
31     
32     #
33     #  open and parse input 
34     #
35     
36     	try:
37     		fp = open (infilename, 'r')
38     	except IOError:
39     		print "Error opening file"
40     		raise
41     
42     	lines = fp.readlines ()
43     
44     #
45     #  put all tokens into tokens and remove comments
46     #
47     	tokens = []
48     	for line in lines:
49     		tmp = re.split ('[ \t\n]*',line)
50     #		print "tmp = ", tmp
51     		for tok in tmp:
52     			if (tok != ''):
53     				if (re.compile('[#!][.]*').match(tok)):
54     					break
55     				tokens.append(tok)
56     #	print "tokens = ", tokens
57     
58     	fp.close()
59     
60     	return tokens
