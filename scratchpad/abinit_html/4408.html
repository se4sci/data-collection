28     @contextmanager
29     def cd(path):
30         """
31         A Fabric-inspired cd context that temporarily changes directory for
32         performing some tasks, and returns to the original working directory
33         afterwards. E.g.,
34     
35             with cd("/my/path/"):
36                 do_something()
37     
38         Args:
39             path: Path to cd to.
40         """
41         # Taken from monty.os
42         cwd = os.getcwd()
43         os.chdir(path)
44         try:
45             yield
46         finally:
47             os.chdir(cwd)

49     @task
50     def make(ctx, jobs="auto", clean=False):
51         """
52         Touch all modified files and recompile the code with -jNUM.
53         """
54         with cd(ABINIT_SRCDIR):
55             cmd = "./abisrc.py touch"
56             cprint("Executing: %s" % cmd, "yellow")
57             result = ctx.run(cmd, pty=True)
58             if not result.ok:
59                 cprint("`%s` failed. Aborting now!" % cmd, "red")
60                 return 1
61     
62         top = find_top_build_tree(".", with_abinit=False)
63         jobs = max(1, number_of_cpus() // 2) if jobs == "auto" else int(jobs)
64     
65         with cd(top):
66             if clean: ctx.run("make clean", pty=True)
67             #cmd = "make -j%d > make.log 2> make.stderr" % jobs
68             cmd = "make -j%d  > >(tee -a make.log) 2> >(tee -a make.stderr >&2)" % jobs
69             cprint("Executing: %s" % cmd, "yellow")
70             ctx.run(cmd, pty=True)
71             # TODO Check for errors in make.stderr

94     @task
95     def abichecks(ctx):
96         """Execute (some of the) abichecks scripts."""
97         import time
98         retcode = 0
99         with cd(ABINIT_ROOTDIR):
100             script_dir = os.path.join("abichecks", "scripts")
101             exclude = ["check-input-vars.py", "check-libpaw.py", "warningschk.py"]
102             for py_script in [f for f in os.listdir(script_dir) if f.endswith(".py")]:
103                 if py_script in exclude: continue
104                 py_script = os.path.join(script_dir, py_script)
105                 print("Running", py_script, "... ", end="")
106                 start = time.time()
107                 result = ctx.run(py_script, pty=True)
108                 msg, color = ("[OK]", "green") if result.ok else ("[FAILED]", "red")
109                 cprint("%s (%.2f s)" % (msg, time.time() - start), color=color)
110                 if not result.ok: retcode += 1
111     
112         if retcode != 0:
113             cprint("%d FAILED TESTS", "red")
114         else:
115             cprint("ALL TESTS OK", "green")
116     
117         return retcode
