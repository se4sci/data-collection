74         def displayGraph(self):
75             res = self.sbres.value()
76             pdos = Analysis.DOS(self.VACF,res,self.dtion)
77     
78             y = pdos.getDOS()
79             x = pdos.getFrequencies()
80             
81             try:
82                 self.GraphDOS.update(x,y,'E (meV)', "Phonons DOS (1/meV)",name = self.name, adjust=True)
83             except:
84                 self.GraphDOS = Graph.graphic(x,y,'E (meV)', "Phonons DOS (1/meV)", average=False, adjust=True,name = self.name)
85                 self.connect(self.GraphDOS, QtCore.SIGNAL("myCustomizedSignal()"), self.close)
86                 self.GraphDOS.show()
87     
88             if self.checkbox.isChecked():
89                 self.addSprectum()

92         def showSprectrum(self):
93     
94             if self.checkbox.isChecked():
