52     def mpicfg_parser(fname, defaults=None):
53         """Parse the configuration file with MPI options."""
54         logger.debug("Parsing [MPI] section in file : " + str(fname))
55     
56         parser = SafeConfigParser(defaults)
57         parser.read(fname)
58     
59         # Read variables needed to handle the job.
60         d = {}
61     
62         for key, tup in CFG_KEYWORDS.items():
63             line_parser = tup[0]
64             section = tup[2]
65             if section in parser.sections():
66                 try:
67                     d[key] = parser.get(section, key)
68                 except NoOptionError:
69                     # Section exists but option is not specified. Use default value.
70                     d[key] = tup[1]
71             else:
72                 # Section does not exist. Use default value.
73                 d[key] = tup[1]
74     
75             # Process the line
76             try:
77                 d[key] = line_parser(d[key])
78             except:
79                 raise ValueError("Wrong line: key = " + str(key) + " d[key] = " + str(d[key]) )
80     
81         return d

285         def run(self, mpi_nprocs, bin_path, stdin_fname, stdout_fname, stderr_fname, cwd=None):
286             """
287             Args:
288                 mpi_nprocs:
289                     Number of MPI nodes.
290                 bin_path:
291                     Path of the executable.
292                 stdin_fname:
293                     Input file
294                 stdout_fname:
295                     Output file
296                 stderr_fname:
297                     Error file
298                 cwd:
299                     cd to cwd before launching the job.
300     
301             Set self.retcode
302             """
303             env = os.environ.copy()
304             if self.has_ompenv: env.update(self.ompenv)
305     
306             # Build valgrind command line
307             valcmd = ""
308             if self.has_valgrind:
309                 valcmd = "valgrind --tool=%s " % self.valgrind_cmdline
310     
311             # Perf command
312             perf_cmd = ""
313             if self.has_perf:
314                 perf_cmd = "perf %s " % self.perf_command
315     
316             if self.has_mpirun or self.has_srun:
317                 args = [perf_cmd, self.mpirun_np, str(mpi_nprocs), " %s " % self.mpi_args,
318                         valcmd, bin_path, "<", stdin_fname, ">", stdout_fname, "2>", stderr_fname]
319     
320             elif self.has_poe:
321                 # example ${poe} abinit ${poe_args} -procs 4
322                 # no support for valgrind, debugger or perf here since poe uses a weird syntax for command line options.
323                 args = [self.poe, bin_path, self.poe_args, " -procs "+ str(mpi_nprocs),
324                         " <", stdin_fname, ">", stdout_fname, "2>", stderr_fname]
325             else:
326                 assert mpi_nprocs == 1
327                 args = [perf_cmd, valcmd, bin_path, "<", stdin_fname, ">", stdout_fname, "2>",stderr_fname]
328     
329             if self.has_debugger:
330                 # Use completely different syntax if we are running under the control of gdb.
331                 # mpirun -np 2 xterm -e gdb fftprof --command=dbg_file
332     
333                 # Get the working directory (warning: I assume that stderr_fname is an absolute path).
334                 workdir = os.path.dirname(stderr_fname)
335     
336                 dbg_filepath = os.path.join(workdir, "dbg_commands")
337     
338                 fh = open(dbg_filepath, "w")
339                 fh.write("run < %s" % stdin_fname) # Use dbg syntax
340                 fh.close()
341     
342                 if self.has_mpirun or self.has_srun:
343                     args = [self.mpirun_np, str(mpi_nprocs), "xterm -e gdb", bin_path, "--command=%s" % dbg_filepath]
344                 else:
345                     args = ["gdb", bin_path, "--command=%s" % dbg_filepath]
346     
347             cmd = " ".join(args)
348             #print(cmd)
349     
350             #if self.has_valgrind: print("Invoking valgrind:\n %s" % cmd)
351             logger.debug("About to execute command:\n" + cmd)
352     
353             start_time = time.time()
354             self.retcode = -1
355     
356             try:
357                 if self.has_timebomb:
358                     p, self.retcode = self.timebomb.run(cmd, shell=True, cwd=cwd, env=env)
359                 else:
360                     p = Popen(cmd, shell=True, cwd=cwd, env=env)
361                     self.retcode = p.wait()
362     
363                 run_etime = time.time() - start_time
364     
365                 if self.retcode != 0:
366                     exc = JobRunnerError(self.retcode, " ".join(args), run_etime)
367                     logger.debug(str(exc))
368                     self.exceptions.append(exc)
369     
370             except:
371                 run_etime = time.time() - start_time
372                 prev_errmsg = str(sys.exc_info()[1])
373                 exc = JobRunnerError(self.retcode, " ".join(args), run_etime, prev_errmsg=prev_errmsg)
374                 self.exceptions.append(exc)
375     
376             return run_etime

401         #==3851== HEAP SUMMARY:
402         #==3851==     in use at exit: 25,149 bytes in 13 blocks
403         #==3851==   total heap usage: 841 allocs, 828 frees, 579,777,815 bytes allocated
404         #==3851==
405         #==3851== LEAK SUMMARY:
406         #==3851==    definitely lost: 0 bytes in 0 blocks
407         #==3851==    indirectly lost: 0 bytes in 0 blocks
408         #==3851==      possibly lost: 0 bytes in 0 blocks
409         #==3851==    still reachable: 25,061 bytes in 12 blocks
410         #==3851==         suppressed: 88 bytes in 1 blocks
411         #==3851== Rerun with --leak-check=full to see details of leaked memory
412         #==3851==
413         #==3851== For counts of detected and suppressed errors, rerun with: -v
414         #==3851== Use --track-origins=yes to see where uninitialised values come from
415         #==3851== ERROR SUMMARY: 10000000 errors from 60 contexts (suppressed: 0 from 0)

417         def parse(self, filename):
418     
419             def fragile_parser(key, string):
420                 """
421                 Extract number from a line in the form: key number ignored_tokens
422                 """
423                 start = line.find(key)
424                 if start == -1: raise ValueError("Cannot find key %s in string %s" % (key, string))
425                 bytes_lost = int(string[start + len(key):].split()[0])
426                 return bytes_lost
427     
428             lost_bytes = 0
429             fh = open(filename, "r")
430     
431             for line in fh:
432                 if "LEAK SUMMARY:" in line: break
433             else:
434                 raise RuntimeError("Cannot find 'LEAK SUMMARY' section in valgrind stderr file")
435     
436             keys = [
437                 "definitely lost:",
438                 "indirectly lost:",
439                 "possibly lost:",
440             ]
441     
442             # Inspect the next len(keys) line (memleak section)
443             errors = {}
444             for key, line in zip(keys, fh):
445                 bytes = fragile_parser(key, line)
446                 if bytes:
447                     errors[key] = bytes
448     
449             # Get total number of errors.
450             key = "ERROR SUMMARY:"
451             for line in fh:
452                 if key in line:
453                     num_errors = fragile_parser(key, line)
454                     if num_errors: errors[key] = num_errors
455     
456             fh.close()
457     
458             self._error_report = ""
459             if errors: self._error_report = str(errors)

462     class TimeBomb(object):
463     
464         def __init__(self, timeout, delay=.05, exec_path=None):
465             self.timeout = int(timeout)
466             self.delay = float(delay)
467             self.exec_path = exec_path
468     
469         def run(self, args,
470                 bufsize=0, executable=None, stdin=None, stdout=None, stderr=None, preexec_fn=None,
471                 close_fds=False, shell=False, cwd=None, env=None, universal_newlines=False, startupinfo=None, creationflags=0):
472             """Same interface as Popen."""
473     
474             try:
475     
476                 if self.exec_path:
477                     #
478                     # timeout exec is available.
479                     #
480                     if self.timeout > 0.:
481                         logger.debug("Using timeout function: " + self.exec_path)
482                         if is_string(args):
483                             args = " ".join([self.exec_path, str(self.timeout), args])
484                         else:
485                             args = [self.exec_path, str(self.timeout)] + args
486     
487                     p = Popen(args,
488                               bufsize=bufsize, executable=executable, stdin=stdin, stdout=stdout, stderr=stderr, preexec_fn=preexec_fn,
489                               close_fds=close_fds, shell=shell, cwd=cwd, env=env, universal_newlines=universal_newlines, startupinfo=startupinfo,
490                               creationflags=creationflags)
491     
492                     ret_code = p.wait()
493     
494                 else:
495                     #
496                     # timeout exec is NOT available.
497                     #
498                     if self.timeout > 0.0:
499                         logger.debug("Using SubprocesswithTimeout and timeout_time : "+str(self.timeout))
500                         p = SubProcessWithTimeout(self.timeout, delay=self.delay)
501     
502                         p, ret_code = p.run(args,
503                             bufsize=bufsize, executable=executable, stdin=stdin, stdout=stdout, stderr=stderr, preexec_fn=preexec_fn,
504                             close_fds=close_fds, shell=shell, cwd=cwd, env=env, universal_newlines=universal_newlines, startupinfo=startupinfo,
505                             creationflags=creationflags)
506                     else:
507                         logger.debug("Using Popen (no timeout_time)")
508                         p = Popen(args,
509                                   bufsize=bufsize, executable=executable, stdin=stdin, stdout=stdout, stderr=stderr, preexec_fn=preexec_fn,
510                                   close_fds=close_fds, shell=shell, cwd=cwd, env=env, universal_newlines=universal_newlines, startupinfo=startupinfo,
511                                   creationflags=creationflags)
512     
513                         ret_code = p.wait()
514     
515                 return p, ret_code
516     
517             except:
518                 raise

521     class OMPEnvironment(dict):
522         """
523         OpenMP variables.
524         see https://computing.llnl.gov/tutorials/openMP/#EnvironmentVariables
525         """
526         _keys = [
527            "OMP_SCHEDULE",
528            "OMP_NUM_THREADS",
529            "OMP_DYNAMIC",
530            "OMP_PROC_BIND",
531            "OMP_NESTED",
532            "OMP_STACKSIZE",
533            "OMP_WAIT_POLICY",
534            "OMP_MAX_ACTIVE_LEVELS",
535            "OMP_THREAD_LIMIT",
536            "OMP_STACKSIZE",
537            "OMP_PROC_BIND",
538         ]
539     
540         def __init__(self, *args, **kwargs):
541             """
542             Constructor method inherited from dictionary:
543     
544             >>> OMPEnvironment(OMP_NUM_THREADS=1)
545             {'OMP_NUM_THREADS': '1'}
546     
547             To create an instance from the INI file fname, use:
548                OMPEnvironment.from_file(fname)
549             """
550             self.update(*args, **kwargs)
551     
552             err_msg = ""
553             for key, value in self.items():
554                 self[key] = str(value)
555                 if key not in OMPEnvironment._keys:
556                     err_msg += "unknown option %s" % key
557             if err_msg: raise ValueError(err_msg)
558     
559         @classmethod
560         def from_file(cls, fname, allow_empty=False):
561             """Initialize the object from file."""
562             parser = SafeConfigParser()
563             parser.read(fname)
564     
565             inst = OMPEnvironment()
566     
567             # Consistency check. Note that we only check if the option name is correct,
568             # we do not check whether the value is correct or not.
569             if "openmp" not in parser.sections():
570                 if not allow_empty:
571                     raise ValueError("%s does not contain any [openmp] section" % fname)
572                 return inst
573     
574             err_msg = ""
575             for key in parser.options("openmp"):
576                 if key.upper() not in OMPEnvironment._keys:
577                     err_msg += "unknown option %s, maybe a typo" % key
578             if err_msg:
579                 raise ValueError(err_msg)
580     
581             for key in OMPEnvironment._keys:
582                 try:
583                     inst[key] = str(parser.get("openmp", key))
584                 except NoOptionError:
585                     try:
586                         inst[key] = str(parser.get("openmp", key.lower()))
587                     except NoOptionError:
588                         pass
589     
590             if not allow_empty and not inst:
591                 raise ValueError("Refusing to return with an empty dict")
592     
593             return inst
