4     def nc_copy(dsin, dsout, except_dimensions=None, except_variables=None):
5         """
6         Copy all dimensions and variable of one nc.Dataset instance into another.
7         """
8     
9         #Copy dimensions
10         for dname, dim in dsin.dimensions.iteritems():
11             if except_dimensions and dname in except_dimensions:
12                 continue
13             dsout.createDimension(dname, len(dim))
14     
15         #Copy variables
16         for vname, varin in dsin.variables.iteritems():
17             if except_variables and vname in except_variables:
18                 continue
19             outVar = dsout.createVariable(vname, varin.datatype, varin.dimensions)
20             outVar[...] = varin[...]
