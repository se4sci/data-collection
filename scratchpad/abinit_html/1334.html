18     
19     class convert_position:
20     
21         def __init__(self,pfile,indexAtom,option=0,reduced=False):
22     
23             # Option:
24             #       0 return the positions of each atom (Default)  
25             #       1 return the average positions 
26             #       2 return the positions of each atom using periodic boundary condistions 
27             # Reduced :
28             #       True  return the result in reduced coordinates
29             #       False return the result in cartesian coordinates
30     
31             self.file = pfile
32             natom  = len(indexAtom)    # Number particule
33             
34             if (reduced):
35                 self.pos = self.file.getXRed()   # position of atom (cartesian)
36     
37             else:
38                 self.pos = self.file.getXCart()  # position of atom (reduced)
39     
40             nbtime = len(self.pos)        # number of step
41             acell = self.file.getAcell()  # acell
42     
43             if(reduced):
44                 a = ones(nbtime)          # Set a,b and c to 1
45                 b = ones(nbtime)          # Only for cubic (must be generalize)
46                 c = ones(nbtime)
47             else:
48                 a = acell[:,0]
49                 b = acell[:,1]
50                 c = acell[:,2]
51         
52     
53             self.result = [[],[],[]]        
54     
55             if option==0:
56                 self.result = posit.atom_std(nbtime,self.pos,indexAtom+1)
57             
58             if option==1:
59                 self.result = posit.atom_ave(nbtime,self.pos,indexAtom+1)
60     
61             if option==2:
62                 self.result = posit.atom_sym(nbtime,self.pos,a,b,c,indexAtom+1)
63     
64     
65         def getX(self):
66             # Return array with x postion of 1 type of particule
67             # This array is reshape by type x[ntime]
68             return self.result[0]
69     
70         def getY(self):
71             # Return array with y postion of 1 type of particule
72             # This array is reshape by type y[ntime]
73             return self.result[1]
74     
75         def getZ(self):
76             # Return array with z postion of 1 type of particule
77             # This array is reshape by type z[ntime]
78             return self.result[2]
