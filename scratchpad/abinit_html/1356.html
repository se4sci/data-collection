2     """
3     Produce the routines used to write the content of a Fortran array to a netcdf file
4     These tools are mainly used for debugging or for a quick analysis of data.

7     def dimension_shape_type_kind(ndim, type, kind):
8         """Return a string with the fortran dimension."""
9         if ndim:
10             dimension = ("(" + ndim*":,")[:-1] + ")"
11         else:
12             dimension = ""
13                                                         
14         shape = "(%d)" % ndim if ndim > 0 else ""
15                                                         
16         if kind:
17             type_kind = "%s(%s)" % (type, kind)
18         else:
19             type_kind = type
20             if type == "character":
21                 type_kind = "%s(len=%s)" % (type, kind)
22     
23         return dimension, shape, type_kind

25     def ncwrite(type, kind, ndim):
26         """
27         Generate the routine for the allocation of pointers/arrays
28     
29         Args:
30             type:
31                 Fortran intrinsic type (real, integer, complex, logical, character)
32             kind: 
33                 Kind of the type. if type is 'character' ...
34             ndim:
35                 integer giving the number of dimensions.
36         """
37         # Construct the name of the function
38         function = "farr_" + type + "_" + kind + str(ndim)
39     
40         dimension, shape, type_kind = dimension_shape_type_kind(ndim, type, kind)
41         
42         # Fortran type --> ETSF-IO flags.
43         xtype = {
44             ("integer", ""): "int",
45             ("real", "sp"): "sp",
46             ("real", "dp"): "dp",
47             ("complex", "spc"): "sp",
48             ("complex", "dpc"): "dp",
49             ("character", ""): "char",
50         }[(type, kind)]
51     
52         # Use c2r to convert from complex to real
53         # Multiply dims by 2 to take into account both real and imag part.
54         # This trick can cause a sigfault for large arrays since
55         # the compiler may store the result of c2r on the stack.
56         convert_farray = "farray"; convert_fact = 1
57         if type == "complex": 
58             convert_farray = "c2r(farray)"
59             convert_fact = 2
60     
61         # Template string 
62         template = """\
63     subroutine %(function)s(farray, varname, ncid)
64     
65      implicit none
66     
67     !Arguments ------------------------------------
68      integer,intent(in) :: ncid
69      character(len=*),intent(in) :: varname
70      %(type_kind)s,intent(in) :: farray%(dimension)s
71     
72     !Local variables-------------------------------
73     !scalars
74     #ifdef HAVE_NETCDF
75      integer :: dimval,ii,varid
76      character(len=nctk_slen) :: dimname,shape_str   
77     
78     ! *********************************************************************
79     
80     ! Define dimensions.
81     ! Use *private* names for dimensions to avoid possible name collisions.
82     ! if varname is "foo", we use "__foo_dim1__", __foo_dim2__" for name of the dimensions
83      NCF_CHECK(nctk_set_datamode(ncid))
84      
85      shape_str = ""
86      do ii=1,size(shape(farray))
87        dimval = size(farray, dim=ii)
88        if (ii == 1) dimval = size(farray, dim=ii) * %(convert_fact)s
89        write(dimname,"(3a,i0,a)")"__",trim(varname),"_dim",ii,"__"
90     
91        NCF_CHECK(nctk_def_dims(ncid, nctkdim_t(dimname, dimval)))
92        shape_str = trim(shape_str) // ", " // dimname 
93      end do
94     
95      ! Define the variable.
96      NCF_CHECK(nctk_def_arrays(ncid, nctkarr_t(varname, '%(xtype)s', shape_str)))
97      varid = nctk_idname(ncid, varname)
98     
99      ! Write data
100      NCF_CHECK(nctk_set_datamode(ncid))
101      NCF_CHECK(nf90_put_var(ncid, varid, %(convert_farray)s))
102     
103     #else 
104      MSG_ERROR("netcdf support is not activated.")
105     #endif
106     
107     end subroutine %(function)s
108     """
109         return template % locals()
110

112     def main():
113         all_ndims = range(1,2)
114     
115         # Table with types and kinds
116         type_and_kinds = [
117             #("integer", ("",)),
118             ("real", ("dp",)),
119             ("complex", ("dpc",)),
120             #("logical", ("",)),
121             #("character", ("",)),
122         ]
123     
124         # The subroutines we are gonna create
125         subs = [ 
126             ncwrite,
127         ]
128     
129         text = []
130         for sub in subs:
131             for ndim in all_ndims:
132                 for type, kinds in type_and_kinds:
133                     for kind in kinds:
134                         text.append(sub(type=type, kind=kind, ndim=ndim))
135     
136         header = "! NOTE: This code has been generated by genarray.py\n"
137         text = ("\n!" + 80*"-" +"\n").join(text)
138         print(header + text)
139     
140         return 0
