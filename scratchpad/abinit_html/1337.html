10     import sys,os,commands
11     import string, math, shutil
12     from scipy.io import netcdf
13     import numpy as np
14     
15     class writeHIST:
16     
17         def __init__(self,pfile,pname,pni,pnf):
18             self.namefile = pname
19             self.file     = pfile
20             self.ni       = pni
21             self.nf       = pnf
22             self.write()
23     
24         def write(self):
25             
26     #       Copy _OUT.nc file
27             name_out_nc = str(self.file.getNameFile(fullpath=True)).replace('_HIST','_OUT.nc')
28             self.new_out_nc = str(self.namefile).replace('_HIST','_OUT.nc')
29             shutil.copyfile(name_out_nc,self.new_out_nc)
30     
31     #       Creation of new HIST file
32             f = netcdf.netcdf_file(self.namefile, 'w')
33     
34     #       dimensions:
35             f.createDimension('natom',self.file.getNatom()) 
36             f.createDimension('xyz',3) 
37             f.createDimension('time',self.file.getNbStep())
38             f.createDimension('six', 6) 
39     
40     #       variables:
41             xcart = f.createVariable('xcart','d',('time','natom','xyz'))
42             xcart[:,:,:] = self.file.xcart[self.ni-1:self.nf,0]
43             xcart.units = "bohr"
44             xcart.mnemonics = "vectors (X) of atom positions in CARTesian coordinates" ;
45     
46             xred = f.createVariable('xred','d',('time','natom','xyz'))
47             xred[:,:,:] = self.file.xred[self.ni-1:self.nf,0]
48             xred.units = "dimensionless" ;
49             xred.mnemonics = "vectors (X) of atom positions in REDuced coordinates" ;
50     
51             fcart = f.createVariable('fcart','d',('time','natom','xyz'))
52             fcart[:,:,:] = self.file.fcart[self.ni-1:self.nf,0]
53             fcart.units = "Ha/bohr" ;
54             fcart.mnemonics = "atom Forces in CARTesian coordinates" ;
55     
56             fred = f.createVariable('fred','d',('time','natom','xyz'))
57             fred[:,:,:] = self.file.fred[self.ni-1:self.nf,0]
58             fred.units = "dimensionless" ;
59             fred.mnemonics = "atom Forces in REDuced coordinates" ;
60     
61             vel = f.createVariable('vel','d',('time','natom','xyz'))
62             vel[:,:,:] = self.file.vel[self.ni-1:self.nf,0]
63             vel.units = "bohr*Ha/hbar" ;
64             vel.mnemonics = "VELocity" ;
65     
66             acell = f.createVariable('acell','d',('time','xyz'))
67             acell[:,:] = self.file.acell[self.ni-1:self.nf,0]
68             acell.units = "bohr" ;
69             acell.mnemonics = "CELL lattice vector scaling" ;
70     
71             rprimd = f.createVariable('rprimd','d',('time','xyz','xyz'))
72             rprimd[:,:,:] = self.file.rprimd[self.ni-1:self.nf,0]
73             rprimd.units = "bohr" ;
74             rprimd.mnemonics = "Real space PRIMitive translations, Dimensional" ;
75     
76             etotal = f.createVariable('etotal','d',('time',))
77             etotal[:] = self.file.E_pot[self.ni-1:self.nf]
78             etotal.units = "Ha" ;
79             etotal.mnemonics = "TOTAL Energy" ;
80     
81             ekin = f.createVariable('ekin','d',('time',))
82             ekin[:] = self.file.E_kin_ion[self.ni-1:self.nf]
83             ekin.units = "Ha" ;
84             ekin.mnemonics = "Energy KINetic ionic" ;
85     
86             mdtime = f.createVariable('mdtime','d',('time',))
87             mdtime[:] = self.file.mdtime[self.ni-1:self.nf]
88             mdtime.units = "hbar/Ha" ;
89             mdtime.mnemonics = "Molecular Dynamics TIME" ;
90     
91             strten = f.createVariable('strten','d',('time','six'))
92             strten[:,:] = self.file.stress[self.ni-1:self.nf,0]
93             strten.units = "Ha/bohr^3" ;
94             strten.mnemonics = "STRess tensor" ;
95     
96             f.close()
